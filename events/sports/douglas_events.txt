#Douglas 101: Mountaineering Club
douglas.101 = {
	dlc = dlc_morgenrote

	type = country_event

	title = douglas.101.t

	desc = douglas.101.d

	flavor = douglas.101.f
	
	duration = 3

	left_icon = scope:douglas_president_scope
	right_icon = scope:douglas_mountaineer_scope

	icon = "gfx/interface/icons/event_icons/event_mountaineering.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
	}

	immediate = {
		random_scope_character = {
			limit = {
				has_role = general
				is_busy = no
				OR = {
					has_trait = douglas_mountaineer
					has_trait = explorer
					has_trait = mountain_commander
					has_trait = surveyor
				}
			}
			save_scope_as = douglas_president_scope
		}
		var:douglas_mountaineer_var = {
			save_scope_as = douglas_mountaineer_scope
		}
		change_global_variable = {
			name = douglas_num_alpine_club_members_global_var
			add = 1
		}
	}

	option = {	#
		name = douglas.101.g #The mountain is calling!
		default_option = yes
	}
}

#Douglas 102: Alpine Club
douglas.102 = {
	dlc = dlc_morgenrote

	type = country_event

	title = douglas.102.t

	desc = douglas.102.d

	flavor = douglas.102.f
	
	duration = 3

	event_image = {
		texture = "gfx/event_pictures/douglas_pictures/douglas_mountaineering_alpine_club.dds"
	}

	icon = "gfx/interface/icons/event_icons/event_mountaineering.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
	}

	immediate = {
	}

	option = {	#
		name = douglas.102.g #To the summit together!
		default_option = yes
		if = {
			limit = {
				NOT = {
					has_variable = douglas_alpine_club_founding_country_var
				}
			}
			custom_tooltip = douglas_alpine_club_board_member_tt
			if = {
				limit = {
					NOT = {
						has_global_variable = douglas_alpine_club_board_member_5_global_var
					}
					has_global_variable = douglas_alpine_club_board_member_4_global_var
				}
				ROOT = {
					country_definition = {
						set_global_variable = {
							name =  douglas_alpine_club_board_member_5_global_var
							value = ROOT
						}
					}
				}
			}
			else_if = {
				limit = {
					has_global_variable =  douglas_alpine_club_board_member_3_global_var
				}
				ROOT = {
					country_definition = {
						set_global_variable = {
							name =  douglas_alpine_club_board_member_4_global_var
							value = ROOT
						}
					}
				}
			}
			else_if = {
				limit = {
					has_global_variable =  douglas_alpine_club_board_member_2_global_var
				}
				ROOT = {
					country_definition = {
						set_global_variable = {
							name =  douglas_alpine_club_board_member_3_global_var
							value = ROOT
						}
					}
				}
			}
			else_if = {
				limit = {
					has_global_variable =  douglas_alpine_club_board_member_1_global_var
				}
				ROOT = {
					country_definition = {
						set_global_variable = {
							name =  douglas_alpine_club_board_member_2_global_var
							value = ROOT
						}
					}
				}
			}
			else_if = {
				limit = {
					NOT = {
						has_global_variable =  douglas_alpine_club_board_member_1_global_var
					}
				}
				ROOT = {
					country_definition = {
						set_global_variable = {
							name =  douglas_alpine_club_board_member_1_global_var
							value = ROOT
						}
					}
				}
			}
		}
	}
