#Douglas 101: Mountaineering Club
douglas.101 = {
	dlc = dlc_morgenrote

	type = country_event

	title = douglas.101.t

	desc = douglas.101.d

	flavor = douglas.101.f
	
	duration = 3

	left_icon = scope:douglas_president_scope
	right_icon = scope:douglas_mountaineer_scope

	icon = "gfx/interface/icons/event_icons/event_mountaineering.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
	}

	immediate = {
		random_scope_character = {
			limit = {
				has_role = general
				is_busy = no
				OR = {
					has_trait = douglas_mountaineer
					has_trait = douglas_mountaineer_veteran
					has_trait = explorer
					has_trait = mountain_commander
				}
			}
			save_scope_as = douglas_president_scope
		}
		var:douglas_mountaineer_var = {
			save_scope_as = douglas_mountaineer_scope
		}
		change_global_variable = {
			name = douglas_num_alpine_club_members_global_var
			add = 1
		}
	}

	option = {	#
		name = douglas.101.g #The mountain is calling!
		default_option = yes
	}
}

#Douglas 103: Mountain Troops Regiment
douglas.103 = {
	dlc = dlc_morgenrote

	type = country_event

	title = douglas.103.t
	desc = douglas.103.d
	flavor = douglas.103.f

	duration = 3

	gui_window = event_window_1char_tabloid

	left_icon = scope:douglas_mountain_veteran

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_mountaineering.dds"

	trigger = {
	}

	immediate = {
		random_character = {
			limit = {
				has_trait = douglas_mountaineer
				has_trait = douglas_mountaineer_veteran
				has_trait = explorer
				has_trait = mountain_commander
			}
			save_scope_as = douglas_mountain_veteran
		}
	}

	option = {
		name = douglas.103.g #They will carry his legacy into the future.
		mountaineer_experience_gain_effect = {
			AMOUNT = 10
		}
		if = {
			limit = {
				scope:douglas_mountain_veteran = {
					has_trait = douglas_mountaineer
				}
			}
			scope:douglas_mountain_veteran = {
				remove_trait = douglas_mountaineer
				add_trait = douglas_mountaineer_veteran
			}
		}
		else_if = {
			limit = {
				scope:douglas_mountain_veteran = {
					NOR = {
						has_trait = douglas_mountaineer
						has_trait = douglas_mountaineer_veteran
					}
				}
			}
			scope:douglas_mountain_veteran = {
				add_trait = douglas_mountaineer
			}
		}
		else_if = {
			limit = {
				scope:douglas_mountain_veteran = {
					has_trait = douglas_mountaineer_veteran
					NOT = {
						has_trait = mountain_commander
					}
				}
			}
			scope:douglas_mountain_veteran = {
				add_trait = mountain_commander
			}
		}
	}

	after = {
	}
}

#Douglas 211: The Summit
douglas.211 = {
	dlc = dlc_morgenrote

	type = country_event

	title = douglas.211.t
	desc = douglas.211.d
	flavor = douglas.211.f

	placement = scope:douglas_location_scope

	duration = 3

	event_image = { 
		texture = "gfx/event_pictures/douglas_pictures/douglas_mountaineering_summit.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_mountaineering.dds"

	trigger = {
	}

	immediate = {
		set_douglas_location_scope_effect = yes
	}

	option = {
		name = douglas.211.g #Let's enjoy the dawn!
		default_option = yes
		set_douglas_summit_conquered_effect = yes
		add_douglas_summit_mountaineer_experience_effect = yes
		add_modifier = {
			name = douglas_mountaineering_success_modifier
			years = 5
		}
		hidden_effect = {
			if = {
				limit = {
					OR = {
						elgar_has_great_composer = yes
						elgar_has_composer = yes
					}
					NOR = {
						has_variable = elgar_alpine_symphony_var
						douglas_climbs_3000_meter_mountain_trigger = yes # You don't get an Alpine Symphony when climbing a mere hill!
					}
				}
				trigger_event = {
					id = dubois.424
					days = { 7 14 }
				}
			}
		}	
	}

	after = {
		remove_variable = douglas_mountain_expedition_var
	}
}
