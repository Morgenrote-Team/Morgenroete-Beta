#Vikelas 10: International Olympic Committee

vikelas.10 = {
	dlc = dlc_morgenrote

	type = country_event

	title = vikelas.10.t
	desc = vikelas.10.d

	flavor = {
		desc = vikelas.10.president
		desc = vikelas.10.president_nation
		desc = vikelas.10.board
		desc = vikelas.10.board_nations
		desc = vikelas.10.members
		desc = vikelas.10.members_nations
		desc = vikelas.10.next_games
		desc = {
			triggered_desc = {
				trigger = {
					any_country = {
						has_modifier = vikelas_olympics_preparation_modifier
					}
				}
				desc = vikelas.10.next_games_nation
			}
			triggered_desc = {
				trigger = {
					NOR = {
						any_country = {
							has_modifier = vikelas_olympics_preparation_modifier
						}
						any_country = {
							has_modifier = vikelas_our_games_our_rules_modifier
						}
					}
				}
				desc = vikelas.10.next_games_nation_undecided
			}
			triggered_desc = {
				trigger = {
					any_country = {
						has_modifier = vikelas_our_games_our_rules_modifier
					}
				}
				desc = vikelas.10.next_games_greece
			}
		}
	}

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_olympic_games.dds"

	event_image = {
		texture = "gfx/event_pictures/vikelas_pictures/vikelas_ioc_board.dds"
	}

	immediate = {
		random_country = {
			limit = {
				has_variable = vikelas_ioc_founding_country_var
			}
			save_scope_as = vikelas_ioc_president_scope
		}
		random_country = {
			limit = {
				has_modifier = vikelas_olympics_preparation_modifier
			}
			save_scope_as = vikelas_next_games_country_scope
		}
		random_country = {
			limit = {
				has_modifier = vikelas_our_games_our_rules_modifier
			}
			save_scope_as = vikelas_next_games_greece_scope
		}
		vikelas_calculate_number_participants_effect = yes
		vikelas_ioc_board_members_effect = yes
	}

	#Show me the Olympic Medal Table.
	option = {
		name = vikelas.10.g
		trigger_event = {
			id = vikelas.11
			popup = yes
		}
	}

	#Bye
	option = {
		name = vikelas.10.h
		default_option = yes
		remove_variable = vikelas_ioc_statistics_var
		custom_tooltip = vikelas_leave_ioc_statistics_tt
	}
}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				total_urbanization >= 1000
				average_sol >= 5

					is_building_type = building_vikelas_sports
					level >= 1
				}
			}
			save_scope_as = vikelas_cricket_state
		}

	option = {	#Nice.
		name = vikelas.21.g
		default_option = yes
		vikelas_set_cricket_as_national_sport_effect = yes
		add_modifier = {
			name = vikelas_cricket_motherland_modifier
		}
		scope:vikelas_cricket_state = {
			activate_production_method = {
				building_type = building_vikelas_sports
				production_method = pm_vikelas_cricket_club
			}
		}

		ai_chance = {
			base = 1
			modifier = { #AI should keep old national sport
				trigger = {
					vikelas_is_sports_motherland_trigger = yes
				}
				add = -1
			}
		}
	}

	option = {	#No
		name = vikelas.21.h

#Vikelas 101: First IOC Board Election (hidden)

vikelas.101 = {
	dlc = dlc_morgenrote

	type = country_event

	hidden = yes

	#The Board consists of:
	#The Founder Country
	#The President Country (if President and Founder are not the same)
	#Four Greater Powers which are neither Founder nor President
	#Two Major Powers
	#Two Random IOC Members from Europe
	#Two Random IOC Members from America
	#Two Random IOC Members from Asia
	#One Random IOC Member from Africa
	#One Random IOC Member from Australia&Oceania
	#A total of 15 Board Members is possible

	immediate = {
		###First: We fire every Board Member for a new Board###
		every_country = {
			limit = {
				has_variable = vikelas_ioc_board_member_var
			}
			remove_variable = vikelas_ioc_board_member_var
		}
		###The Founder has always a seat on the Board###
		random_country = {
			limit = {
				has_variable = vikelas_ioc_founding_country_var
			}
			set_variable = vikelas_ioc_board_member_var
		}
		###Four Great Powers are part of the Board###
		random_country = {
			limit = {
				country_rank = rank_value:great_power
				NOR = {
					has_variable = vikelas_ioc_board_member_var
					has_variable = vikelas_ioc_founding_country_var
				}
			}
			set_variable = vikelas_ioc_board_member_var
			set_variable = vikelas_ioc_member_var
		}
		random_country = {
			limit = {
				country_rank = rank_value:great_power
				NOT = {
					has_variable = vikelas_ioc_board_member_var
				}
			}
			set_variable = vikelas_ioc_board_member_var
			set_variable = vikelas_ioc_member_var
		}
		random_country = {
			limit = {
				country_rank = rank_value:great_power
				NOT = {
					has_variable = vikelas_ioc_board_member_var
				}
			}
			set_variable = vikelas_ioc_board_member_var
			set_variable = vikelas_ioc_member_var
		}
		random_country = {
			limit = {
				country_rank = rank_value:great_power
				NOT = {
					has_variable = vikelas_ioc_board_member_var
				}
			}
			set_variable = vikelas_ioc_board_member_var
			set_variable = vikelas_ioc_member_var
		}
		###Two Major Powers are part of the Board
		random_country = {
			limit = {
				country_rank = rank_value:major_power
				NOT = {

	dlc = dlc_morgenrote

	type = country_event

	title = vikelas.318.t
	desc = vikelas.318.d
	flavor = vikelas.318.f

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/vikelas_pictures/vikelas_football_team.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_sports.dds"

	trigger = {
		vikelas_has_football_association_trigger = no
		OR = {
			any_scope_building = {
				filter = {
					is_building_type = building_vikelas_sports
				}
				has_active_production_method = pm_vikelas_football_club
				percent >= 0.35
			}
			any_scope_building = {
				is_building_type = building_vikelas_sports
				has_active_production_method = pm_vikelas_football_club
				count >= 4
			}
		}
	}

	immediate = {
		set_variable = vikelas_football_association_var
	}

	option = {
		name = vikelas.318.g #Found the Association.
		default_option = yes
	}
}

#Vikelas 391: Adapt National Sport
vikelas.391 = {
	dlc = dlc_morgenrote

	type = country_event

	title = vikelas.391.t

	desc = vikelas.391.d

	flavor = vikelas.391.f

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/vikelas_pictures/vikelas_croquet.dds"
	}

	icon = "gfx/interface/icons/event_icons/event_sports.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
	}

	immediate = {
	}

	option = {	#Cricket
		name = vikelas.391.g
		trigger = {
			has_variable = vikelas_cricket_country_var
		}
		default_option = yes
		vikelas_set_cricket_as_national_sport_effect = yes
	}

	option = {	#Baseball
		name = vikelas.391.h
		trigger = {
			has_variable = vikelas_baseball_country_var
		}
		vikelas_set_baseball_as_national_sport_effect = yes
	}

	option = {	#Football
		name = vikelas.391.i
		trigger = {
			has_variable = vikelas_football_country_var
		}
		vikelas_set_football_as_national_sport_effect = yes
	}

	option = {	#Basketball
		name = vikelas.391.j
		trigger = {
			has_variable = vikelas_basketball_country_var
		}
		vikelas_set_basketball_as_national_sport_effect = yes
	}

	after = {
		hidden_effect = {
			remove_variable = vikelas_adapt_national_sport_var
			if = {
				limit = {
					is_subject = yes
					overlord = {
						var:vikelas_national_sport_var ?= root.var:vikelas_national_sport_var
					}
				}
				add_modifier = {
					name = vikelas_common_national_sports_modifier
				}
			}
			if = {
				limit = {
					is_subject = yes
					NOT = {
						overlord = {
							var:vikelas_national_sport_var ?= root.var:vikelas_national_sport_var
						}
					}
				}
				remove_modifier = vikelas_common_national_sports_modifier
			}
		}
	}
}
