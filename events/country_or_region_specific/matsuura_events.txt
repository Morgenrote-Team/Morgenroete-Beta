#Matsuura 101: A young man wants to travel to Ezo
matsuura.101 = {
	dlc = dlc_morgenrote

	type = country_event

	title = matsuura.101.t
	desc = matsuura.101.d
	flavor = matsuura.101.f

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	gui_window = event_window_1char_tabloid

	left_icon = scope:character_1_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	placement = scope:location_scope

	trigger = {
		NOT = {
			has_global_variable = matsuura_travel_event_global_var
		}
		has_state_in_state_region = STATE_TOHOKU
		c:JAP ?= this
		year >= 1843
	}

	immediate = {
		set_global_variable = matsuura_travel_event_global_var
		random_scope_state = {
			limit = {
				state_region = s:STATE_TOHOKU
			}
			save_scope_as = location_scope
		}
		create_character = {
			template = mr_matsuura_takeshiro_character_template

			save_scope_as = character_1_scope

			on_created = {
				set_variable = mr_no_headgear_var
				place_character_in_void = 1200

				geologist_experience_set_up_effect = yes
				writer_experience_set_up_effect = yes
				painter_experience_set_up_effect = yes
			}
		}

		set_variable = {
			name = agassiz_new_geologist_var
			value = scope:character_1_scope
		}
		set_variable = {
			name = manzoni_new_writer_var
			value = scope:character_1_scope
		}
		set_variable = {
			name = klimt_new_painter_var
			value = scope:character_1_scope
		}
	}

	option = { #He could study the landscape there!
	default_option = yes
		name = matsuura.101.g

		scope:character_1_scope = {
			mr_make_character_geologist_effect = yes

			save_scope_as = agassiz_new_geologist_scope
		}
		dubois_set_new_geologist_as_geologist_effect = yes

		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_closed_borders
					value > neutral
				}
			}

			add_modifier = {
				name = matsuura_allowed_travels_modifier
				years = 2
				is_decaying = yes
			}
		}
	}

	option = { #I heard he is a poet, maybe he will be inspired by his travels?
		name = matsuura.101.h

		scope:character_1_scope = {
			mr_make_character_writer_effect = yes

			save_scope_as = manzoni_new_writer_scope
		}
		manzoni_set_new_writer_local_writer_effect = yes

		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_closed_borders
					value > neutral
				}
			}

			add_modifier = {
				name = matsuura_allowed_travels_modifier
				years = 2
				is_decaying = yes
			}
		}
	}

	option = { #Maybe he could create paintings of the region?
		name = matsuura.101.i

		scope:character_1_scope = {
			mr_make_character_painter_effect = yes

			save_scope_as = klimt_new_painter_scope
		}
		klimt_set_new_painter_as_painter_effect = yes

		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_closed_borders
					value > neutral
				}
			}

			add_modifier = {
				name = matsuura_allowed_travels_modifier
				years = 2
				is_decaying = yes
			}
		}
	}

	option = { #No one is allowed to leave.
		name = matsuura.101.j

		scope:character_1_scope = {
			free_character_from_void = no
		}
	}

	after = {
		remove_variable = agassiz_new_geologist_var
		remove_variable = klimt_new_painter_var
		remove_variable = manzoni_new_writer_var
	}
}

matsuura.102 = {
	dlc = dlc_morgenrote

	type = country_event

	title = matsuura.102.t
	desc = matsuura.102.d
	flavor = matsuura.102.f

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_academy.dds"

	gui_window = event_window_1char_tabloid

	left_icon = scope:character_1_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	placement = scope:location_scope

	trigger = {
		#Triggered by on_game_started_after_lobby
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_KYUSHU
			}
			save_scope_as = location_scope
		}
		create_character = {
			template = mr_jap_udagawa_yoan_character_template

			save_scope_as = character_1_scope

			on_created = {
				set_variable = mr_no_headgear_var
				place_character_in_void = 1200

				chemist_experience_set_up_effect = yes #sets up experience for the character (0 till 30 and then 10 Experience per year)
			}
		}

		set_variable = {
			name = mendelejew_new_chemist_var
			value = scope:character_1_scope
		}
	}

	option = { #What is this "Chemistry" he is talking about?
	default_option = yes
		name = matsuura.102.g

		scope:character_1_scope = {
			mr_make_character_chemist_effect = yes

			save_scope_as = mendelejew_new_chemist_scope
		}
		mendelejew_set_new_chemist_as_chemist_effect = yes

		ig:ig_landowners = {
			add_modifier = {
				name = matsuura_barbarian_ideas_modifier
				years = 2
				is_decaying = yes
			}
		}
	}

	option = { #Barbarian ideas!
		name = matsuura.102.h

		scope:character_1_scope = {
			free_character_from_void = no
		}
	}
}

#Matsuura 103: Nintendo
matsuura.103 = {
	dlc = dlc_morgenrote

	type = country_event

	title = matsuura.103.t
	desc = matsuura.103.d
	flavor = matsuura.103.f

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_industry.dds"

	event_image = {
		texture = "gfx/event_pictures/matsuura_pictures/matsuura_nintendo.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	placement = scope:location_scope

	trigger = {
		game_date >= 1888
		game_date < 1895
		ROOT = c:JAP
		NOT = {
			has_variable = matsuura_nintendo_var
		}
		any_scope_state = {
			state_region = s:STATE_KANSAI
			any_scope_building = {
				is_building_type = building_paper_mill
			}
		}
	}

	immediate = {
		set_variable = matsuura_nintendo_var
		random_scope_state = {
			limit = {
				state_region = s:STATE_KANSAI
			}
			save_scope_as = location_scope
		}
	}

	option = { #Are there any games that could be more popular than Hanafuda cards?
	default_option = yes
		name = matsuura.103.g
		scope:location_scope = {
			add_modifier = {
				name = matsuura_nintendo_modifier
				years = 5
			}
		}
	}
}
