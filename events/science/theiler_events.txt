#Theiler 10: The Birth of Microbiology
theiler.10 = {
	dlc = dlc_morgenrote

	type = country_event

	title = theiler.10.t

	desc = theiler.10.d

	flavor = theiler.10.f

	placement = ROOT

	event_image = { #Original https://commons.wikimedia.org/wiki/File:Beitrge_zur_Biologie_der_Pflanzen_(1870-2006)_(19740093274).jpg
		texture = "gfx/event_pictures/theiler_pictures/theiler_cohn.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_biology.dds"

	trigger = {
		has_technology_researched = theiler_microbiology_tech
		NOT = {
			has_global_variable = theiler_microbiology_global_var
		}
	}

	immediate = {
		set_global_variable = theiler_microbiology_global_var
		if = {
			limit ={
				NOT = {
					has_variable = theiler_biologist_var
				}
			}
			create_new_character_base_effect = { CHARACTER = biologist PROJECT = theiler }
			theiler_set_new_biologist_as_biologist_effect = yes
			var:theiler_biologist_var = {
				save_scope_as = theiler_biologist_scope
			}
		}
		if = {
			limit = {
				var:theiler_biologist_var = {
					NOT = { has_trait = theiler_microbiologist }
				}
			}
			var:theiler_biologist_var = {
				add_trait = theiler_microbiologist
			}
		}
		var:theiler_biologist_var = {
			set_variable = {
				name = theiler_microbiology_achievement_var
				value = flag:theiler_microbiology_achievement_loc
			}
			save_scope_as = theiler_biologist_scope
		}
	}

	option = { #A breakthrough in the search for microorganisms.
		name = theiler.10.g
		default_option = yes
		biologist_experience_gain_effect = {
			AMOUNT = 30
		}
		theiler_add_x_microorganisms_effect = {
			AMOUNT = 500
		}
	}

	option = { #A breakthrough for germ theory.
		name = theiler.10.h
		trigger = {
			panum_has_miasma_theory_trigger = yes
		}
		biologist_experience_gain_effect = {
			AMOUNT = 30
		}
		panum_add_x_germ_theory_clues_effect = {
			AMOUNT = 10
		}
	}
}

#Theiler 32: Cydia spreads
theiler.32 = {
	dlc = dlc_morgenrote

	type = country_event

	placement = scope:theiler_state_scope

	title = theiler.32.t

	desc = theiler.32.d

	flavor = theiler.32.f

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/theiler_pictures/theiler_cydia.dds"
	}

	icon = "gfx/interface/icons/event_icons/event_biology.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	cooldown = {
		years = 5
	}

	trigger = {
		NOT = {
			has_variable = theiler_cydia_lead_arsenate_var
		}
		any_scope_state = {
			NOT = {
				has_state_trait = theiler_state_trait_cydia
			}
			any_scope_building = {
				OR = {
					is_building_type = building_wheat_farm
					is_building_type = building_maize_farm
					is_building_type = building_rye_farm
				}
				OR = {
					has_active_production_method = pm_apple_orchards
					has_active_production_method = pm_citrus_orchards #normally the parasite would not attack these, but wheat farms should be affected, too.
				}
				level >= 3
			}
			any_neighbouring_state = {
				has_state_trait = theiler_state_trait_cydia
			}
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				any_scope_building = {
					OR = {
						is_building_type = building_wheat_farm
						is_building_type = building_maize_farm
						is_building_type = building_rye_farm
					}
					OR = {
						has_active_production_method = pm_apple_orchards
						has_active_production_method = pm_citrus_orchards #normally the parasite would not attack these, but wheat farms should be affected, too.
					}
					level >= 3
				}
				any_neighbouring_state = {
					has_state_trait = theiler_state_trait_cydia
				}
			}
			save_scope_as = theiler_state_scope
		}
	}

	option = { #My apples!
		name = theiler.32.g
		default_option = yes
		scope:theiler_state_scope = {
			state_region = {
				add_state_trait = theiler_state_trait_cydia
			}
		}
	}
}
