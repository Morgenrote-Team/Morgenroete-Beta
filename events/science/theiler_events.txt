#Theiler 801: Achromatic Microscope
theiler.801 = {
	dlc = dlc_morgenrote

	type = country_event

	title = theiler.801.t

	desc = theiler.801.d

	flavor = theiler.801.f
	
	duration = 3

	gui_window = event_window_2char

	left_icon = scope:gaudi_engineer_scope
	right_icon = scope:theiler_biologist_scope

	icon = "gfx/interface/icons/event_icons/event_biology.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		tesla_has_engineer = yes
		theiler_has_biologist_trigger = yes
	}

	immediate = {
		var:tesla_engineer_var = {
			save_scope_as = gaudi_engineer_scope
		}
		var:theiler_biologist_var = {
			save_scope_as = theiler_biologist_scope
		}
	}

	option = { #It is certainly useful for the doctor.
		name = theiler.801.g
		default_option = yes
		add_modifier = {
			name = theiler_achromatic_microscope_modifier
		}
	}
}
