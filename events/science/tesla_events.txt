#Tesla 223: A First Prototype
tesla.223 = {
	dlc = dlc_morgenrote

	type = country_event

    title = tesla.223.t
    desc = tesla.223.d
	flavor = tesla.223.f

	duration = 3

	event_image = {
		#Photo of Babbage Analytical Engine Plan from 1840 at the Computer History Museum
		# CC BY 4.0 by ArnoldReinhold
		# https://en.wikipedia.org/wiki/Analytical_engine#/media/File:Babbage_Analytical_Engine_Plan_1840_CHM.agr.jpg
		texture = "gfx/event_pictures/gaudi_pictures/gaudi_analytical_engine.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/machinists.dds"

	cooldown = {
		years = 30
	}

	trigger = {
		any_scope_building = {
			is_building_type = building_university
		}
	}

	immediate = {
		if = {
			limit = {
				any_scope_building = {
					is_building_type = building_university
					has_active_production_method = pm_dubois_university_mathematics
				}
			}
			random_scope_building = {
				limit = {
					is_building_type = building_university
					has_active_production_method = pm_dubois_university_mathematics
				}
				save_scope_as = tesla_university_scope
			}
		}
		else = {
			random_scope_building = {
				limit = {
					is_building_type = building_university
				}
				save_scope_as = tesla_university_scope
			}
		}
	}

	option = { #I hope he will finish more than a prototype...
		name = tesla.223.g

		scope:tesla_university_scope = {
			add_modifier = {
				name = tesla_prototype_mechanical_computer_modifier
				years = 5
			}
		}
	}
}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    duration = 3

	immediate = {
		company:mr_basic_telegraphy ?= {
    		executive = {
        		save_scope_as = tesla_telegraph_boss_scope
    		}
		}
	}

	option = {
		name = tesla.3001.g #The state will participate in this venture.
		default_option = yes
		if = {
			limit = {
				has_journal_entry = je_tesla_transatlantic_cable
			}
			je:je_tesla_transatlantic_cable = {
				add_modifier = {
					name = tesla_transatlantic_cable_expedition_cost_modifier
					multiplier = mr_project_cost_small_value
				}
			}
		}
		if = {
			limit = {
				has_journal_entry = je_tesla_suez_india_cable
			}
			je:je_tesla_suez_india_cable = {
				add_modifier = {
					name = tesla_suez_india_cable_expedition_cost_modifier
					multiplier = mr_project_cost_small_value
				}
			}
		}
		if = {
			limit = {
				has_journal_entry = je_tesla_transpacific_cable
			}
			je:je_tesla_transpacific_cable = {
				add_modifier = {
					name = tesla_transpacific_cable_expedition_cost_modifier
					multiplier = mr_project_cost_small_value
				}
			}
		}

		ai_chance = {
			base = 1
		}
	}

	option = {
		name = tesla.3001.h #The company wants to build the cable, so they should pay for it.
		add_tesla_submarine_confidence_2_decrease_effect = yes

		ai_chance = {
			base = 0
		}
	}
