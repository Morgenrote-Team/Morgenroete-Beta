#Tesla 223: A First Prototype
tesla.223 = {
	dlc = dlc_morgenrote

	type = country_event

    title = tesla.223.t
    desc = tesla.223.d
	flavor = tesla.223.f

	duration = 3

	event_image = {
		#Photo of Babbage Analytical Engine Plan from 1840 at the Computer History Museum
		# CC BY 4.0 by ArnoldReinhold
		# https://en.wikipedia.org/wiki/Analytical_engine#/media/File:Babbage_Analytical_Engine_Plan_1840_CHM.agr.jpg
		texture = "gfx/event_pictures/gaudi_pictures/gaudi_analytical_engine.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/machinists.dds"

	cooldown = {
		years = 30
	}

	trigger = {
		any_scope_building = {
			is_building_type = building_university
		}
	}

	immediate = {
		if = {
			limit = {
				any_scope_building = {
					is_building_type = building_university
					has_active_production_method = pm_dubois_university_mathematics
				}
			}
			random_scope_building = {
				limit = {
					is_building_type = building_university
					has_active_production_method = pm_dubois_university_mathematics
				}
				save_scope_as = tesla_university_scope
			}
		}
		else = {
			random_scope_building = {
				limit = {
					is_building_type = building_university
				}
				save_scope_as = tesla_university_scope
			}
		}
	}

	option = { #I hope he will finish more than a prototype...
		name = tesla.223.g

		scope:tesla_university_scope = {
			add_modifier = {
				name = tesla_prototype_mechanical_computer_modifier
				years = 5
			}
		}
	}
}

#Tesla 224: Feature Creep
tesla.224 = {
	dlc = dlc_morgenrote

	type = country_event

    title = tesla.224.t
    desc = tesla.224.d
	flavor = tesla.224.f

	duration = 3

	event_image = {
		#Photo of Babbage Analytical Engine Plan from 1840 at the Computer History Museum
		# CC BY 4.0 by ArnoldReinhold
		# https://en.wikipedia.org/wiki/Analytical_engine#/media/File:Babbage_Analytical_Engine_Plan_1840_CHM.agr.jpg
		texture = "gfx/event_pictures/gaudi_pictures/gaudi_analytical_engine.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/machinists.dds"

	cooldown = {
		years = 5
	}

	trigger = {
		je:je_gaudi_mechanical_computer_project = {
			"scripted_bar_progress(tesla_mechanical_computer_project_progress_bar)" >= 50
		}
	}

	immediate = {
	}

	option = { #Would you please just build that computer?
		name = tesla.224.g
		default_option = yes

		tesla_add_x_mechanical_computer_progress_effect = { AMOUNT = -50 }
	}

	option = { #Threaten to cut funding.
		name = tesla.224.h

		random_list = {
			5 = {
				tesla_add_x_mechanical_computer_progress_effect = { AMOUNT = 10 }
			}
			74 = {}
			20 = {
				trigger = {
					var:tesla_engineer_var ?= {
						has_trait = psychological_affliction
					}
				}
				var:tesla_engineer_var ?= {
					add_trait = psychological_affliction
				}
			}
			1 = {
				var:tesla_engineer_var ?= {
					retire_engineer_effect = yes
				}
			}
		}
	}
}

#Tesla 3001: Atlantic Telegraph Company

tesla.3001 = {
	dlc = dlc_morgenrote

	type = country_event

    title = tesla.3001.t
    desc = tesla.3001.d
    flavor = tesla.3001.f

    gui_window = event_window_1char_tabloid

	left_icon = scope:tesla_telegraph_boss_scope

	icon = "gfx/interface/icons/event_icons/event_telegraph.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

    duration = 3

	immediate = {
		random_scope_character = {
			limit = {
				has_role = executive
				is_company_type = company_type:mr_basic_telegraphy
			}
			save_scope_as = tesla_telegraph_boss_scope
		}
		set_variable = {
			name = pbi_var_je_objective_across_the_sea
			value = 0
		}
	}

	option = {
		name = tesla.3001.g #
		default_option = yes
		add_journal_entry = {
			type = je_tesla_transatlantic_cable
		}
		change_variable = {
			name = pbi_var_je_objective_across_the_sea
			add = 1
		}
	}

	after = {
		scope:rapanui_telegraph_boss_scope = {
			free_character_from_void = no
		}
	}
}
