	dlc = dlc_morgenrote

	type = country_event

	title = andersson.10.t

	desc = andersson.10.d

	flavor = andersson.10.f

	event_image = {
		texture = "gfx/event_pictures/andersson_pictures/andersson_piltdown.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_anthropology.dds"

	trigger = {
	}

	immediate = {
		andersson_update_anthropological_collection_effect = yes
		owner = {
			set_variable = andersson_institute_inauguration_happened_var
		}
		if = {
			limit = { has_variable = andersson_anthropologist_var }
			var:andersson_anthropologist_var ?= {
				set_variable = {
					name = andersson_anthropologist_achievement_inaugurated_institute_var
					value = flag:andersson_anthropologist_achievement_inaugurated_institute_loc
				}
			}
		}
		random_scope_character = {
			limit = {
				has_role_of_type = ruler
			}
			save_scope_as = andersson_ruler
		}
	}

	#option = {
	#	name = andersson.10.g
	#	default_option = yes
	#	dubois_add_x_academy_progress = {
	#		AMOUNT = 30
	#	}
	#}

	option = {
		name = lepsius.10.h #Have a nice day
		default_option = yes

			AMOUNT = 30

#Andersson 21: Vero Man
andersson.21 = {
	dlc = dlc_morgenrote

	type = country_event

	title = andersson.21.t
	desc = andersson.21.d
	flavor = andersson.21.f

	duration = 3

	placement = scope:andersson_cave_state_scope

	event_image = {
		texture = "gfx/event_pictures/andersson_pictures/andersson_neoanthropus.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_anthropology.dds"

	trigger = {
		has_game_rule = andersson_anthropology_historical_rule
		NOT = {
			has_global_variable = andersson_vero_global_var
		}
		any_scope_state = {
			state_region = s:STATE_FLORIDA
		}
		game_date >= 1915.1.1
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_FLORIDA
			}
			save_scope_as = andersson_cave_state_scope
		}
		scope:andersson_cave_state_scope = {
			state_region = {
				set_global_variable = {
					name = andersson_vero_global_var
					value = THIS
				}
			}
		}
		country_definition = {
			set_global_variable = {
				name = andersson_vero_country_global_var
				value = this
			}
		}
		set_global_variable = {
			name = andersson_vero_year_global_var
			value = year
		}
	}

	option = {
		name = andersson.21.g #Fascinating
		default_option = yes
		andersson_add_neoanthropus_fossil_effect = yes
	}

	after = {
		hidden_effect = {
			every_country = {
				limit = {
					is_player = yes
				}
				post_notification = andersson_vero_discovered_notification
			}
		}
	}
}
