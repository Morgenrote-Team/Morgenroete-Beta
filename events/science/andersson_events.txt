#Andersson 10: Institute of Anthropology & Ethnology
andersson.10 = {
	dlc = dlc_morgenrote

	type = country_event

	title = andersson.10.t

	desc = andersson.10.d

	flavor = andersson.10.f

	event_image = {
		texture = "gfx/event_pictures/andersson_pictures/andersson_piltdown.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_anthropology.dds"

	trigger = {
	}

	immediate = {
		andersson_update_anthropological_collection_effect = yes
		owner = {
			set_variable = andersson_institute_inauguration_happened_var
		}
		if = {
			limit = { has_variable = andersson_anthropologist_var }
			var:andersson_anthropologist_var ?= {
				set_variable = {
					name = andersson_anthropologist_achievement_inaugurated_institute_var
					value = flag:andersson_anthropologist_achievement_inaugurated_institute_loc
				}
			}
		}
		random_scope_character = {
			limit = {
				has_role_of_type = ruler
			}
			save_scope_as = andersson_ruler
		}
	}

	option = {
		name = andersson.10.g
		default_option = yes
		dubois_add_x_academy_progress = {
			AMOUNT = 30
		}
	}

	option = {
		name = lepsius.10.h #Have a nice day
		dubois_add_x_academy_progress = {
			AMOUNT = 30
		}
	}
}


#Andersson 21: Vero Man
andersson.21 = {
	dlc = dlc_morgenrote

	type = country_event

	title = andersson.21.t
	desc = andersson.21.d
	flavor = andersson.21.f

	duration = 3

	placement = scope:andersson_cave_state_scope

	event_image = {
		texture = "gfx/event_pictures/andersson_pictures/andersson_neoanthropus.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_anthropology.dds"

	trigger = {
		has_game_rule = andersson_anthropology_historical_rule
		NOT = {
			has_global_variable = andersson_vero_global_var
		}
		any_scope_state = {
			state_region = s:STATE_FLORIDA
			andersson_state_has_cave_trigger = no
		}
		game_date >= 1915.1.1
	}

	immediate = {
		set_variable = andersson_neoanthropus_cave_set_var
		random_scope_state = {
			limit = {
				state_region = s:STATE_FLORIDA
			}
			andersson_add_neoanthropus_cave_effect = yes
			save_scope_as = andersson_cave_state_scope
		}
		scope:andersson_cave_state_scope = {
			state_region = {
				set_global_variable = {
					name = andersson_vero_global_var
					value = THIS
				}
			}
		}
		country_definition = {
			set_global_variable = {
				name = andersson_vero_country_global_var
				value = this
			}
		}
		set_global_variable = {
			name = andersson_vero_year_global_var
			value = year
		}
	}

	option = {
		name = andersson.21.g #Fascinating
		default_option = yes
		andersson_add_neoanthropus_fossil_effect = yes
		scope:andersson_cave_state_scope = {
			state_region = {
				add_state_trait = andersson_state_trait_neoanthropus_cave
			}
		}
	}

	after = {
		hidden_effect = {
			every_country = {
				limit = {
					is_player = yes
				}
				post_notification = andersson_vero_discovered_notification
			}
		}
	}
}

			}
			desc = andersson.919.f_europe
		}
		triggered_desc = {
			trigger = {
				has_global_variable = andersson_origin_asia_vote
			}
			desc = andersson.919.f_asia
		}
		triggered_desc = {
			trigger = {
				has_global_variable = andersson_origin_africa_vote
			}
			desc = andersson.919.f_africa
		}
	}

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/andersson_pictures/andersson_origin_congress.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_anthropology.dds"

	immediate = {
		random_country = {
			limit = {
				has_modifier = andersson_congress_host_modifier
			}
			save_scope_as = andersson_congress_host_country
		}
		var:andersson_anthropologist_var = {
			save_scope_as = andersson_anthropologist_scope
		}
		if = {
			limit = {
				global_var:andersson_origin_europe > global_var:andersson_origin_asia
				global_var:andersson_origin_europe > global_var:andersson_origin_africa
			}
			set_global_variable = andersson_origin_europe_vote
		}
		else_if = {
			limit = {
				global_var:andersson_origin_asia > global_var:andersson_origin_europe
				global_var:andersson_origin_asia > global_var:andersson_origin_africa
			}
			set_global_variable = andersson_origin_asia_vote
		}
		else_if = {
			limit = {
				global_var:andersson_origin_africa > global_var:andersson_origin_asia
				global_var:andersson_origin_africa > global_var:andersson_origin_europe
			}
			set_global_variable = andersson_origin_africa_vote
		}
		else = {
			set_global_variable = andersson_origin_asia_vote
		}
	}

	option = {
		trigger = {
			has_global_variable = andersson_origin_europe_vote
		}
		name = andersson.919.g #As expected, Humanity evolved in Europe.
		dubois_small_add_technology = yes
		anthropologist_experience_gain_effect = {
			AMOUNT = "15"
		}
	}

	option = {
		trigger = {
			has_global_variable = andersson_origin_asia_vote
		}
		name = andersson.919.h #All evidences suggest that humanity evolved in Asia.
		lepsius_small_add_technology = yes
		dubois_medium_add_technology = yes
		anthropologist_experience_gain_effect = {
			AMOUNT = "25"
		}
	}

	option = {
		trigger = {
			has_global_variable = andersson_origin_africa_vote
		}
		name = andersson.919.i #'Out of Africa' is an approved thesis now.
		default_option = yes
		lepsius_large_add_technology = yes
		dubois_large_add_technology = yes
		anthropologist_experience_gain_effect = {
			AMOUNT = "35"
		}
	}

	after = {
		hidden_effect = {
			every_country = {
				limit = {
					NOT = {
						this = ROOT
					}
					has_modifier = andersson_congress_attend_modifier
				}
				trigger_event = {
					id = andersson.920
					popup = yes
				}
			}
			remove_global_variable = andersson_origin_europe
			remove_global_variable = andersson_origin_asia
			remove_global_variable = andersson_origin_africa
		}
	}
}
