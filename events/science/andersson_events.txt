#Andersson 101: Anthropological Site found (Cave Paintings)
andersson.101 = {
	dlc = dlc_morgenrote

	type = country_event

	title = andersson.101.t
	desc = andersson.101.d
	flavor = andersson.101.f

	duration = 3

	placement = scope:andersson_cave_state

	gui_window = event_window_1char_tabloid

	left_icon = scope:andersson_anthropologist_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_anthropology.dds"

	trigger = {
		any_scope_state = {
			andersson_state_has_cave_paintings_trigger = yes
			andersson_state_has_discovered_cave_paintings_no_tooltip_trigger = no
		}
	}

	immediate = {
		random_scope_state = {
			limit = {
				andersson_state_has_cave_paintings_trigger = yes
				andersson_state_has_discovered_cave_paintings_no_tooltip_trigger = no
			}
			save_scope_as = andersson_cave_paintings_state
		}
		
	}

	option = {
		name = andersson.101.g #Fascinating
		default_option = yes
		anthropologist_experience_gain_effect = { AMOUNT = 15 }
		scope:andersson_cave_paintings_state = {
			if = {
				limit = {
					state_region = {
						has_variable = andersson_cave_paintings_var
					}
				}
				state_region = {
					add_state_trait = andersson_state_trait_cave_paintings_unexplored
				}
			}
		}
	}
}

#Andersson 201: Cave Exploration Destinantion

andersson.201 = {
	dlc = dlc_morgenrote

	type = country_event

	title = andersson.201.t
	desc = andersson.201.d
	flavor = andersson.201.f

	gui_window = event_window_1char_tabloid

	left_icon = scope:andersson_anthropologist_scope

	icon = "gfx/interface/icons/event_icons/event_anthropology.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
	}

	immediate = {
		var:andersson_anthropologist_var ?= {
			save_scope_as = andersson_anthropologist_scope
		}
		if = {
			limit = {
				andersson_country_has_neoanthropus_cave_trigger = yes
			}
			random_scope_state = {
				limit = {
					andersson_state_has_neoanthropus_cave_trigger = yes
				}
				save_scope_as = andersson_neoanthropus_cave_scope
			}
		}
		if = {
			limit = {
				andersson_country_has_palaeanthropus_cave_trigger = yes
			}
			random_scope_state = {
				limit = {
					andersson_state_has_palaeanthropus_cave_trigger = yes
				}
				save_scope_as = andersson_palaeanthropus_cave_scope
			}
		}
		if = {
			limit = {
				andersson_country_has_pithecanthropus_cave_trigger = yes
			}
			random_scope_state = {
				limit = {
					andersson_state_has_pithecanthropus_cave_trigger = yes
				}
				save_scope_as = andersson_pithecanthropus_cave_scope
			}
		}
		if = {
			limit = {
				andersson_country_has_australopithecus_cave_trigger = yes
			}
			random_scope_state = {
				limit = {
					andersson_state_has_australopithecus_cave_trigger = yes
				}
				save_scope_as = andersson_australopithecus_cave_scope
			}
		}
		if = {
			limit = {
				andersson_country_has_unexplored_cave_paintings_trigger = yes
			}
			random_scope_state = {
				limit = {
					andersson_state_has_unexplored_cave_paintings_trigger = yes
				}
				save_scope_as = andersson_cave_paintings_scope
			}
		}
	}

	option = {	#Neoanthropus Cave
		name = andersson.201.h
		trigger = {
			andersson_country_has_neoanthropus_cave_trigger = yes
			NOT = {
				has_journal_entry = je_andersson_cave_exploration_neoanthropus
			}
			scope:andersson_anthropologist_scope = {
				is_busy = no
			}
		}
		scope:andersson_anthropologist_scope = {
			set_character_busy_and_immortal = yes
		}
		scope:andersson_neoanthropus_cave_scope = {
			add_modifier = {
				name = andersson_cave_exploration_modifier
			}
			set_variable = andersson_cave_exploration_var
		}
		
		ai_chance = {
			base = 1
			#Focus on Neoanthropus if the AI nation has less than 10 bones
			modifier = {
				trigger = {
					andersson_has_maximum_neoanthropus_bones_trigger = no
				}
				add = 100
			}
		}
	}

	option = {	#Palaeanthropus Cave
		name = andersson.201.i
		trigger = {
			andersson_country_has_palaeanthropus_cave_trigger = yes
			NOT = {
				has_journal_entry = je_andersson_cave_exploration_palaeanthropus
			}
			scope:andersson_anthropologist_scope = {
				is_busy = no
			}
		}
		scope:andersson_anthropologist_scope = {
			set_character_busy_and_immortal = yes
		}
		scope:andersson_palaeanthropus_cave_scope = {
			add_modifier = {
				name = andersson_cave_exploration_modifier
			}
			set_variable = andersson_cave_exploration_var
		}
		
		ai_chance = {
			base = 1
			#Focus on Palaeanthropus if the AI nation has less than 10 bones
			modifier = {
				trigger = {
					andersson_has_maximum_palaeanthropus_bones_trigger = no
				}
				add = 100
			}
		}
	}

	option = {	#Pithecanthropus Cave
		name = andersson.201.j
		trigger = {
			andersson_country_has_pithecanthropus_cave_trigger = yes
			NOT = {

			modifier = {
				trigger = {
					andersson_has_maximum_pithecanthropus_bones_trigger = no
				}
				add = 100
			}
		}
	}

	option = {	#Australopithecus Cave
		name = andersson.201.k

				is_busy = no
			}
		}
		scope:andersson_anthropologist_scope = {
			set_character_busy_and_immortal = yes
		}
		scope:andersson_australopithecus_cave_scope = {
			add_modifier = {
				name = andersson_cave_exploration_modifier

		}
		
		ai_chance = {
			base = 1

				trigger = {
					andersson_has_maximum_australopithecus_bones_trigger = no
				}
				add = 100
			}
		}
	}

	option = {	#Mysterious Cave
		name = andersson.201.l
		trigger = {
			andersson_country_has_unexplored_cave_paintings_trigger = yes
			NOT = {
				has_journal_entry = je_andersson_cave_exploration_mysterious
			}

				is_busy = no
