#Pius 101: La Salette
pius.101 = {
	dlc = dlc_morgenrote

	type = country_event

    title = pius.101.t
    desc = pius.101.d
	flavor = pius.101.f

	placement = scope:pius_marian_apparition_state

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/pius_pictures/pius_maria.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_pope.dds"

	trigger = {
		NOT = {
			has_global_variable = pius_recent_marian_apparition_global_var
		}
	}

	immediate = {
		set_global_variable = {
			name = pius_recent_marian_apparition_global_var
			years = 5
		}
		set_variable = {
			name = pius_ongoing_marian_apparition_var
			years = 5
		}
		set_variable = pius_had_historical_marian_apparition_var
		random_scope_state = {
			limit = {
				state_region = s:STATE_RHONE
			}
			set_variable = {
				name = pius_marian_apparition_state_var
				years = 5
			}
			save_scope_as = pius_marian_apparition_state
		}
	}

	option = { #A Heavenly Warning.
		default_option = yes
		name = pius.101.g
		trigger = {
			OR = {
				country_has_state_religion = rel:catholic
				NOT = {
					has_law = law_type:law_state_atheism
				}
			}
		}
		add_modifier = {
			name = pius_approved_great_marian_apparition_modifier
			years = 7
		}
		scope:pius_marian_apparition_state = {
			add_loyalists_in_state = {
				strata = lower
				religion = rel:catholic
				value = medium_radicals
			}
		}
		hidden_effect = {
			trigger_event = {
				id = pius.121
				days = { 30 40 }
			}
		}

		ai_chance = {
			base = 75
			modifier = {
				trigger = {
					has_law = law_type:law_freedom_of_conscience
				}
				add = 25
			}
			modifier = {
				trigger = {
					has_law = law_type:law_total_separation
				}
				add = -25
			}
			modifier = {
				trigger = {
					has_law = law_type:law_total_separation
				}
				add = -25
			}
			modifier = {
				trigger = {
					ig:ig_intelligentsia = {
						is_in_government = yes
					}
				}
				add = -1000
			}
		}
	}

	option = { #Remain Rational.
		name = pius.101.h
		trigger = {
			OR = {
				NOT = {
					country_has_state_religion = rel:catholic
				}
				NOT = {
					has_law = law_type:law_state_religion
				}
			}
		}
		add_modifier = {
			name = pius_denied_great_marian_apparition_modifier
			years = 7
		}
		scope:pius_marian_apparition_state = {
			add_radicals_in_state = {
				strata = lower
				religion = rel:catholic
				value = large_radicals
			}
		}
		hidden_effect = {
			random_list = {
				85 = {
					trigger_event = {
						id = pius.111
						days = { 30 40 }
					}
				}

		NOT = {
			has_global_variable = pius_recent_marian_apparition_global_var
		}
		NOT = {
			has_variable = pius_had_random_marian_apparition_var
		}
		global_var:number_of_random_marian_apparitions < 5
		any_scope_state = {
			total_urbanization >= 500
			is_incorporated = yes
			is_capital = no
			religion_percent_state = {
				target = rel:catholic
