#Pius 101: La Salette
pius.101 = {
	dlc = dlc_morgenrote

	type = country_event

    title = pius.101.t
    desc = pius.101.d
	flavor = pius.101.f

	placement = scope:pius_marian_apparition_state

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/pius_pictures/pius_maria.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_pope.dds"

	trigger = {
		NOT = {
			has_global_variable = pius_recent_marian_apparition_global_var
		}
	}

	immediate = {
		set_global_variable = {
			name = pius_recent_marian_apparition_global_var
			years = 5
		}
		set_variable = {
			name = pius_ongoing_marian_apparition_var
			years = 5
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_RHONE
			}
			set_variable = {
				name = pius_marian_apparition_state_var
				years = 5
			}
			save_scope_as = pius_marian_apparition_state
		}
	}

	option = { #A Heavenly Warning.
		default_option = yes
		name = pius.101.g
		add_modifier = {
			name = pius_approved_great_marian_apparition_modifier
			years = 7
		}
		scope:pius_marian_apparition_state = {
			add_loyalists_in_state = {
				strata = lower
				religion = rel:catholic
				value = medium_radicals
			}
		}
	}

	option = { #Remain Rational.
		name = pius.101.h
		add_modifier = {
			name = pius_denied_great_marian_apparition_modifier
			years = 7
		}
		scope:pius_marian_apparition_state = {
			add_radicals_in_state = {
				strata = lower
				religion = rel:catholic
				value = large_radicals
			}
		}
		hidden_effect = {
			ramdom_list = {
				85 = {
					trigger_event = {
						id = pius.111
						days = { 30 40 }
					}
				}
				15 = {}
			}
		}
	}
}

#Pius 105: The Apparition of 'RANDOM_SCOPE_STATE'
pius.105 = {
	dlc = dlc_morgenrote

	type = country_event

    title = pius.105.t
    desc = pius.105.d
	flavor = pius.105.f

	placement = scope:pius_marian_apparition_state

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/pius_pictures/pius_maria_2.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_pope.dds"

	trigger = {
		NOT = {
			has_global_variable = pius_recent_marian_apparition_global_var
		}
		NOT = {
			has_variable = pius_had_random_marian_apparition_var
		}
		global_var:number_of_random_marian_apparitions < 5
		any_scope_state = {
			total_urbanization >= 500
			is_incorporated = yes
			is_capital = no
			religion_percent_state = {
				target = rel:catholic
				value >= 0.35
			}
		}
	}

	immediate = {
		set_global_variable = {
			name = pius_recent_marian_apparition_global_var
			years = 5
		}
		set_variable = {
			name = pius_ongoing_marian_apparition_var
			years = 5
		}
		set_variable = pius_had_random_marian_apparition_var
		change_global_variable = {
			name = number_of_random_marian_apparitions
			add = 1
		}
		random_scope_state = {
			limit = {
				total_urbanization >= 500
				is_incorporated = yes
				is_capital = no
				religion_percent_state = {
					target = rel:catholic
					value >= 0.35
				}
			}
			set_variable = {
				name = pius_marian_apparition_state_var
				years = 5
			}
			save_scope_as = pius_marian_apparition_state
		}
	}

	option = { #A Heavenly Warning.
		default_option = yes
		name = pius.105.g
		add_modifier = {
			name = pius_approved_marian_apparition_modifier
			years = 5
		}
		scope:pius_marian_apparition_state = {
			add_loyalists_in_state = {
				strata = lower
				religion = rel:catholic
				value = small_radicals
			}
		}
	}

	option = { #Remain Rational.
		name = pius.105.h
		add_modifier = {
			name = pius_denied_marian_apparition_modifier
			years = 5
		}
		scope:pius_marian_apparition_state = {
			add_radicals_in_state = {
				strata = lower
				religion = rel:catholic
				value = medium_radicals
			}
		}
		hidden_effect = {
			ramdom_list = {
				85 = {

pius.111 = {
	dlc = dlc_morgenrote

	type = country_event

    title = pius.111.t
    desc = pius.111.d
	flavor = pius.111.f

	placement = scope:pius_marian_apparition_state

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/pius_pictures/pius_maria_2.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_pope.dds"

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				has_variable = pius_marian_apparition_state_var
			}
			save_scope_as = pius_marian_apparition_state
		}
	}

	option = { #Let them pray.
		default_option = yes
		name = pius.111.g
		if = {
			limit = {
				has_modifier = pius_denied_marian_apparition_modifier
			}
			remove_modifier = {
				name = pius_denied_marian_apparition_modifier
			}
		}
		if = {
			limit = {
				has_modifier = pius_denied_great_marian_apparition_modifier
			}
			remove_modifier = {
				name = pius_denied_great_marian_apparition_modifier
			}
		}
		add_modifier = {
			name = pius_undecided_politics_on_marian_apparition_modifier
		}
	}

	option = { #Observe quietly.
		name = pius.111.h
		custom_tooltip = pius.111.h_tt
		trigger = {
			OR = {
				has_law = law_type:law_autocracy
				has_law = law_type:law_secret_police
				has_law = law_type:law_single_party_state
			}
		}
		hidden_effect = {
			random_list = {
				3 = {}
				7 = {
					trigger_event = {
						id = pius.113
						days = { 20 40 }
					}
				}
			}
		}
	}

	option = { #Disperse them by force.
		name = pius.111.i
		change_infamy = 2
		custom_tooltip = pius.111.i_tt
		add_modifier = {
			name = pius_action_against_marian_procession_modifier
			years = 3
		}
		hidden_effect = {
			random_list = {
				1 = {}
				9 = {
					trigger_event = {
						id = pius.112
						days = { 5 10 }
					}
				}
			}
		}
	}
}
