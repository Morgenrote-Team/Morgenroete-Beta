#Pius 101: La Salette
pius.101 = {
	dlc = dlc_morgenrote

	type = country_event

    title = pius.101.t
    desc = pius.101.d
	flavor = pius.101.f

	placement = scope:pius_marian_apparition_state

	duration = 3

	event_image = {
		texture = "gfx/event_pictures/pius_pictures/pius_maria.dds"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_pope.dds"

	trigger = {
	}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_RHONE
			}
			save_scope_as = pius_marian_apparition_state
		}
	}

	option = { #A Heavenly Warning.
		default_option = yes
		name = pius.101.g
		add_modifier = {
			name = pius_approved_great_marian_apparition_modifier
			years = 7
		}
		scope:pius_marian_apparition_state = {
			add_loyalists_in_state = {
				strata = lower
				religion = rel:catholic
				value = medium_radicals
			}
		}
	}

	option = { #Remain Rational.
		name = pius.101.h
		add_modifier = {
			name = pius_denied_great_marian_apparition_modifier
			years = 7
		}
		scope:pius_marian_apparition_state = {
			add_loyalists_in_state = {
				strata = lower
				religion = rel:catholic
				value = large_radicals
			}
		}
	}
}
