andersson_anthropological_research_progress_bar = {
	name = "andersson_anthropological_research_progress_bar"
	desc = "andersson_anthropological_research_progress_bar_desc"

	default = yes

	monthly_progress = {
		if = {
			limit = {
				andersson_has_active_anthropologist_no_tooltip_trigger = yes
			}
			add = {
				desc = "andersson_speleology_progress_active_anthropologist_tt"
				value = 1
			}

	}

	start_value = 0

### Speleology ###

andersson_speleology_progress_bar = {
	name = "andersson_speleology_progress_bar"
	desc = "andersson_speleology_progress_bar_desc"

	default = yes

	monthly_progress = {
		if = {
			limit = {
				andersson_has_active_anthropologist_no_tooltip_trigger = yes
			}
			add = {
				desc = "andersson_speleology_progress_anthropologist_tt"
				value = 1
			}
			if = {
				limit = {
					var:andersson_anthropologist_var ?= {
						has_variable = andersson_anthropologist_experience_var
					}
				}
				add = {
					desc = "andersson_speleology_progress_anthropologist_experience_tt"
					value = var:andersson_anthropologist_var.var:andersson_anthropologist_experience_var
					divide = 300
				}
			}
			if = {
				limit = {
					var:andersson_anthropologist_var ?= {
						has_trait = andersson_speleologist
					}
				}
				add = {
					desc = "andersson_speleology_progress_anthropologist_speleologist_trait_tt"
					value = 0.5
				}
			}
			if = {
				limit = {
					var:andersson_anthropologist_var ?= {
						has_trait = explorer
					}
				}
				add = {
					desc = "andersson_speleology_progress_anthropologist_explorer_trait_tt"
					value = 0.3
				}
			}
			if = {
				limit = {
					var:andersson_anthropologist_var ?= {
						OR = {
							has_trait = persistent
							has_trait = brave
						}
					}
				}
				add = {
					desc = "andersson_speleology_progress_anthropologist_positive_character_trait_tt"
					value = 0.2
				}
			}
			if = {
				limit = {
					var:andersson_anthropologist_var ?= {
						OR = {
							has_trait = hedonist
							has_trait = cautious
						}
					}
				}
				add = {
					desc = "andersson_speleology_progress_anthropologist_negative_character_trait_tt"
					value = -0.2
				}
			}
			if = {
				limit = {
					var:andersson_anthropologist_var ?= {
						OR = {
							has_trait = alcoholic
							has_trait = opium_addiction
							has_trait = cocaine_addiction
							has_trait = syphilis
							has_trait = psychological_affliction
							has_trait = senile
							has_trait = panum_cholera
							has_trait = panum_tuberculosis
							has_trait = panum_yellow_fever
						}
					}
				}
				add = {
					desc = "andersson_speleology_progress_anthropologist_negative_body_trait_tt"
					value = -0.5
				}
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 60
}
