douglas_mountain_expedition_progress_bar = {
	name = "douglas_mountain_expedition_progress_bar"
	desc = "douglas_mountain_expedition_progress_bar_desc"

	default_green = yes

	monthly_progress = {
		if = {
			limit = {
				douglas_has_mountaineer = yes
			}
			add = {
				desc = "douglas_mountain_expedition_progress_mountaineer_tt"
				value = 0.35
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						has_variable = douglas_mountaineer_experience_var
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_experience_tt"
					value = var:douglas_mountaineer_var.var:douglas_mountaineer_experience_var
					divide = 150
				}
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						has_trait = mountain_commander
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_mountain_commander_tt"
					value = 0.25
				}
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						has_trait = explorer
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_explorer_tt"
					value = 0.25
				}
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						has_trait = agassiz_geologist
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_geologist_tt"
					value = 0.15
				}
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						has_trait = surveyor
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_surveyor_tt"
					value = 0.15
				}
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						has_trait = persistent
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_persistent_tt"
					value = 0.15
				}
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						OR = {
							has_trait = charismatic
							has_trait = ambitious
							has_trait = imposing
							has_trait = meticulous
							has_trait = brave
						}
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_positive_trait_tt"
					value = 0.1
				}
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						OR = {
							has_trait = reckless
							has_trait = arrogant
							has_trait = wrathful
							has_trait = cruel
							has_trait = ruthless
						}
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_negative_trait_tt"
					value = -0.1
				}
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						OR = {
							has_trait = grifter
							has_trait = hedonist
						}
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_negative_skill_tt"
					value = -0.25
				}
			}
			if = {
				limit = {
					var:douglas_mountaineer_var ?= {
						OR = {
							has_trait = alcoholic
							has_trait = opium_addiction
							has_trait = cocaine_addiction
							has_trait = psychological_affliction
							has_trait = senile
							has_trait = sickly
							has_trait = kidney_stones
							has_trait = cancer
						}
					}
				}
				add = {
					desc = "douglas_mountain_expedition_progress_negative_condition_tt"
					value = -0.5
				}
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 10

}
