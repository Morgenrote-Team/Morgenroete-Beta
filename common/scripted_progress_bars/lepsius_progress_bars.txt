### Archaeological Excavation ###

lepsius_archaeological_excavation_progress_bar = {
	name = "lepsius_archaeological_excavation_progress_bar"
	desc = "lepsius_archaeological_excavation_progress_bar_desc"

	default_green = yes

	monthly_progress = {
		if = {
			limit = {
				lepsius_has_active_archaeologist = yes
			}
			add = {
				desc = "lepsius_archaeological_excavation_progress_archaeologist_tt"
				value = 5
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = lepsius_archaeologist_experienced
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_experienced_tt"
					value = 2
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = lepsius_archaeologist_famous
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_famous_tt"
					value = 5
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = persistent
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_persistent_tt"
					value = 1
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = meticulous
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_meticulous_tt"
					value = 1
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = engineer
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_engineer_tt"
					value = 1
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = surveyor
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_surveyor_tt"
					value = 1
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = master_bureaucrat
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_master_bureaucrat_tt"
					value = 1
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						OR = {
							has_trait = alcoholic
							has_trait = expensive_tastes
							has_trait = opium_addiction
							has_trait = cocaine_addiction
							has_trait = syphilis
							has_trait = psychological_affliction
						}
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_negative_trait_tt"
					value = -1
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = senile
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_senile_tt"
					value = -4
				}
			}
		}
		if = {
			limit = {
				has_variable = lepsius_excavation_teams
				var:lepsius_excavation_teams > 0
			}
			add = {
				desc = "lepsius_archaeological_excavation_progress_excavation_teams_tt"
				value = var:lepsius_excavation_teams
			}
		}
		if = {
			limit = {
				academics_num_humanities_universities > 0
			}
			add = {
				desc = "lepsius_archaeological_excavation_progress_humanities_universities_tt"
				value = academics_humanities_university_multiplier_for_excavation
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 5000
}
