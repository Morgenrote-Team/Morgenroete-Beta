gaudi_architectural_splendour_progress_bar = {
	name = "gaudi_architectural_splendour_progress_bar"
	desc = "gaudi_architectural_splendour_progress_bar_desc"

	default = yes

	monthly_progress = {
		if = {
			limit = {
				gaudi_has_architect = yes
			}
			add = {
				desc = "gaudi_architectural_splendour_progress_architect_tt"
				value = 1
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						has_variable = gaudi_architect_experience
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_experience_tt"
					value = var:gaudi_architect_var.var:gaudi_architect_experience
					divide = 500
				}
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						has_trait = engineer
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_engineer_tt"
					value = 1
				}
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						has_trait = verrier_mathematician
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_mathematician_tt"
					value = 0.5
				}
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						has_trait = surveyor
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_surveyor_tt"
					value = 0.5
				}
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						has_trait = master_bureaucrat

				}
			}

				limit = {
					var:gaudi_architect_var ?= {
						has_trait = ambitious
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_ambitious_tt"
					value = 0.35
				}
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						has_trait = persistent
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_persistent_tt"
					value = 0.35
				}
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						has_trait = meticulous
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_meticulous_tt"
					value = 0.25
				}
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						has_trait = innovative
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_innovative_tt"
					value = 0.25
				}
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						has_trait = charismatic
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_charismatic_tt"
					value = 0.25
				}
			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						OR = {
							has_trait = reckless
							has_trait = cruel
							has_trait = grifter
						}

				}
				add = {
					desc = "gaudi_architectural_splendour_progress_negative_trait_tt"
					value = -0.25

			}
			if = {
				limit = {
					var:gaudi_architect_var ?= {
						OR = {
							has_trait = alcoholic
							has_trait = opium_addiction
							has_trait = cocaine_addiction
							has_trait = psychological_affliction
							has_trait = senile
						}
					}
				}
				add = {
					desc = "gaudi_architectural_splendour_progress_negative_condition_tt"
					value = -0.5
				}
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 100
}
