			if = {

				limit = {
					var:lepsius_archaeologist_var = {

				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_surveyor_tt"
					value = 1

					var:lepsius_archaeologist_var = {
						has_trait = master_bureaucrat

					desc = "lepsius_archaeological_excavation_progress_archaeologist_master_bureaucrat_tt"

				}

						OR = {
							has_trait = alcoholic

				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_senile_tt"
					value = -4
				}
			}
		}
		if = {
			limit = {
				has_variable = lepsius_excavation_teams
				var:lepsius_excavation_teams > 0
			}
			add = {
				desc = "lepsius_archaeological_excavation_progress_excavation_teams_tt"
				value = var:lepsius_excavation_teams
			}
		}

			limit = {
				academics_num_humanities_universities > 0

			}

	min_value = 0

#Restoration Progress (Archaeologist)
lepsius_monument_restoration_progress_bar = {
	name = "lepsius_monument_restoration_progress_bar"
	desc = "lepsius_monument_restoration_progress_bar_desc"

	default = yes

	monthly_progress = {
		if = {
			limit = {
				lepsius_has_archaeologist = yes
				OR = {
					AND = {
						has_journal_entry = je_lepsius_ishtar_gate_transport
						has_variable = lepsius_ishtar_gate_minimum_var
					}
					AND = {
						has_journal_entry = je_lepsius_cleopatras_needle_transport
						has_variable = lepsius_cleopatras_needle_minimum_var
					}
					AND = {
						has_journal_entry = je_lepsius_piedras_negras_transport
						has_variable = lepsius_piedras_negras_minimum_var
					}
					AND = {
						has_journal_entry = je_lepsius_axum_obelisk_transport
						has_variable = lepsius_axum_obelisk_minimum_var
					}
				}
			}
			add = {
				desc = "lepsius_monument_restoration_progress_archaeologist_tt"
				value = 1
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var ?= {
						has_variable = lepsius_archaeologist_experience_var
					}
				}
				add = {
					desc = "lepsius_monument_restoration_progress_archaeologist_experience_tt"
					value = var:lepsius_archaeologist_var.var:lepsius_archaeologist_experience_var
					divide = 300
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = persistent
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_persistent_tt"
					value = 0.5
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = meticulous
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_meticulous_tt"
					value = 0.3
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = engineer
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_engineer_tt"
					value = 0.2
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						OR = {
							has_trait = alcoholic
							has_trait = expensive_tastes
							has_trait = opium_addiction
							has_trait = cocaine_addiction
							has_trait = syphilis
							has_trait = psychological_affliction
						}
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_negative_trait_tt"
					value = -0.5
				}
			}
			if = {
				limit = {
					var:lepsius_archaeologist_var = {
						has_trait = senile
					}
				}
				add = {
					desc = "lepsius_archaeological_excavation_progress_archaeologist_senile_tt"
					value = -0.8
