### Dinosaur Expedition/Excavation ###

dubois_paleontology_dinosaur_excavation_set_up_progress_bar = {
	name = "dubois_paleontology_dinosaur_excavation_set_up_progress_bar"
	desc = "dubois_paleontology_dinosaur_excavation_set_up_progress_bar_desc"
	default_green = yes

	monthly_progress = {
		#Only active if the Expedition wasn't established yet.
		if = {
			limit = {
				NOT = {
					je:je_dubois_paleontological_expedition ?= {
						has_modifier = dubois_established_paleontological_expedition_modifier
					}
				}
			}
			value = dubois_paleontology_monthly_progress_value
		}
	}

	start_value = 0
	min_value = 0
	max_value = 12
}
