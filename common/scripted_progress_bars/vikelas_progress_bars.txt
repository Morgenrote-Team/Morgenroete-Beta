vikelas_mechanic_progress_bar = {
	name = "vikelas_mechanic_progress_bar"
	desc = "vikelas_mechanic_progress_bar_desc"

	default_green = yes

	monthly_progress = {
		if = {
			limit = {
				has_variable = vikelas_mechanic_var
			}
			add = {
				desc = "vikelas_mechanic_progress_mechanic_tt"
				value = 1
			}
			if = {
				limit = {
					var:vikelas_mechanic_var ?= {
						has_variable = vikelas_mechanic_experience
					}
				}
				add = {
					desc = "vikelas_mechanic_progress_experience_tt"
					value = var:vikelas_mechanic_var.var:vikelas_mechanic_experience
					divide = 150
				}
			}
			if = {
				limit = {
					has_journal_entry = je_vikelas_mechanic_engine
					var:vikelas_mechanic_var ?= {
						has_trait = vikelas_engine_mechanic
					}
				}
				add = {
					desc = "vikelas_mechanic_progress_mechanic_engine_specialist_tt"
					add = 1
				}
			}
			if = {
				limit = {
					has_journal_entry = je_vikelas_mechanic_safety
					var:vikelas_mechanic_var ?= {
						has_trait = vikelas_safety_mechanic
					}
				}
				add = {
					desc = "vikelas_mechanic_progress_mechanic_safety_specialist_tt"
					add = 1
				}
			}
			if = {
				limit = {
					has_journal_entry = je_vikelas_mechanic_body
					var:vikelas_mechanic_var ?= {
						has_trait = vikelas_body_mechanic
					}
				}
				add = {
					desc = "vikelas_mechanic_progress_mechanic_body_specialist_tt"
					add = 1
				}
			}
			if = {
				limit = {
					var:vikelas_racer_var ?= {
						has_trait = tesla_mechanical_engineer
					}
				}
				add = {
					desc = "vikelas_mechanic_progress_racer_is_mechanic_tt"
					add = 0.3
				}
			}
			if = {
				limit = {
					any_scope_building = {
						is_building_type = building_vikelas_circuit
					}
				}
				add = {
					desc = "vikelas_mechanic_progress_circuit_tt"
					add = 0.2
				}
			}
			if = {
				limit = {
					var:vikelas_mechanic_var ?= {
						OR = {
							has_trait = alcoholic
							has_trait = expensive_tastes
							has_trait = opium_addiction
							has_trait = cocaine_addiction
							has_trait = syphilis
							has_trait = psychological_affliction
							has_trait = senile
							has_trait = panum_cholera
							has_trait = panum_tuberculosis
							has_trait = panum_yellow_fever
						}
					}
				}
				add = {
					desc = "vikelas_mechanic_progress_mechanic_negative_body_trait_tt"
					value = -0.5
				}
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 36
}
