### Medical System Expansion Project ###

panum_medical_system_project_progress_bar = {
	name = "panum_medical_system_project_progress_bar"
	desc = "panum_medical_system_project_progress_bar_desc"

	default_green = yes

	monthly_progress = {
		if = {
			limit = {
				has_variable = panum_physician_var
			}
			add = {
				desc = "panum_medical_system_project_progress_physician_tt"
				value = 1
			}
			if = {
				limit = {
					var:panum_physician_var ?= {
						has_variable = panum_physician_experience_var
					}
				}
				add = {
					desc = "panum_medical_system_project_progress_physician_experience_tt"
					value = var:panum_physician_var.var:panum_physician_experience_var
					divide = 300
				}
			}
			if = {
				limit = {
					var:panum_physician_var ?= {
						has_trait = master_bureaucrat
					}
				}
				add = {
					desc = "panum_medical_system_project_progress_physician_master_bureaucrat_trait_tt"
					value = 1
				}
			}
			if = {
				limit = {
					# Physician has Specialization
					panum_physician_has_specialization_trigger = yes
				}
				add = {
					desc = "panum_medical_system_project_progress_physician_specialization_tt"
					value = 0.5
				}
			}
			if = {
				limit = {
					var:panum_physician_var ?= {
						OR = {
							has_trait = inspirational_orator
							has_trait = demagogue
							has_trait = firebrand
						}
					}
				}
				add = {
					desc = "panum_medical_system_project_progress_physician_orator_trait_tt"
					value = 0.5
				}
			}
			if = {
				limit = {
					var:panum_physician_var ?= {
						OR = {
							has_trait = charismatic
							has_trait = persistent
							has_trait = meticulous
							has_trait = tactful
							has_trait = pious
							has_trait = imposing
							has_trait = honorable
						}
					}
				}
				add = {
					desc = "panum_medical_system_project_progress_physician_positive_character_trait_tt"
					value = 0.3
				}
			}
			if = {
				limit = {
					var:panum_physician_var ?= {
						OR = {
							has_trait = wrathful
							has_trait = arrogant
							has_trait = bigoted
							has_trait = cruel
						}
					}
				}
				add = {
					desc = "panum_medical_system_project_progress_physician_negative_character_trait_tt"
					value = -0.3
				}
			}
			if = {
				limit = {
					var:panum_physician_var ?= {
						OR = {
							has_trait = alcoholic
							has_trait = expensive_tastes
							has_trait = opium_addiction
							has_trait = cocaine_addiction
							has_trait = syphilis
							has_trait = psychological_affliction
							has_trait = senile
							has_trait = panum_cholera
							has_trait = panum_tuberculosis
							has_trait = panum_yellow_fever
						}
					}
				}
				add = {
					desc = "panum_medical_system_project_progress_physician_negative_body_trait_tt"
					value = -0.5
				}
			}
			#Total Levels of Humanities focused Universities
			if = {
				limit = {
					academics_humanities_university_multiplier_tiny >= 0
				}
				add = {
					desc = "panum_physician_project_progress_humanities_levels_tt"
					value = academics_humanities_university_multiplier_tiny
				}
			}
			add = mr_cheat_progress_value
		}
	}

	start_value = 0
	min_value = 0
	max_value = 30
}
