mendelejew_chemist_medicinal_plant_project_progress_bar = {
	name = "mendelejew_chemist_medicinal_plant_project_progress_bar"
	desc = "mendelejew_chemist_medicinal_plant_project_progress_bar_desc"

	default_bad = yes

	monthly_progress = {
		if = {
			limit = {
				has_variable = theiler_medicinal_plant_collection_done_var
				has_variable = mendelejew_chemist_var
				NOT = { has_variable = theiler_medicinal_plant_isolation_done_var }
			}
			add = {
				desc = "mendelejew_medicinal_plant_project_progress_medicinal_plants_collected_tt"
				value = 0.5
			}
			if = {
				limit = {
					var:mendelejew_chemist_var ?= {
						has_variable = mendelejew_chemist_experience_var
					}
				}
				add = {
					desc = "mendelejew_medicinal_plant_project_progress_chemist_experience_tt"
					value = var:mendelejew_chemist_var.var:mendelejew_chemist_experience_var
					divide = 300
				}
			}
			if = {
				limit = {
					var:mendelejew_chemist_var ?= {
						has_trait = mendelejew_chemist_inspector
					}
				}
				add = {
					desc = "mendelejew_medicinal_plant_project_progress_chemist_inspector_trait_tt"
					value = 1
				}
			}
			if = {
				limit = {
					var:mendelejew_chemist_var ?= {
						has_trait = meticulous
					}
				}
				add = {
					desc = "mendelejew_medicinal_plant_project_progress_chemist_meticulous_trait_tt"
					value = 0.25
				}
			}
			if = {
				limit = {
					var:mendelejew_chemist_var ?= {
						has_trait = persistent
					}
				}
				add = {
					desc = "mendelejew_medicinal_plant_project_progress_chemist_persistent_trait_tt"
					value = 0.1
				}
			}
			if = {
				limit = {
					var:mendelejew_chemist_var ?= {
						has_trait = ambitious
					}
				}
				add = {
					desc = "mendelejew_medicinal_plant_project_progress_chemist_ambitious_trait_tt"
					value = 0.1
				}
			}
			if = {
				limit = {
					var:mendelejew_chemist_var ?= {
						OR = {
							has_trait = alcoholic
							has_trait = expensive_tastes
							has_trait = opium_addiction
							has_trait = cocaine_addiction
							has_trait = syphilis
							has_trait = psychological_affliction
							has_trait = senile
							has_trait = panum_cholera
							has_trait = panum_tuberculosis
							has_trait = panum_yellow_fever
						}
					}
				}
				add = {
					desc = "mendelejew_medicinal_plant_project_progress_chemist_negative_body_trait_tt"
					value = -0.35
				}
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 15
}
