tesla_mechanical_computer_project_progress_bar = {
	name = "tesla_mechanical_computer_project_progress_bar"
	desc = "tesla_mechanical_computer_project_progress_bar_desc"

	default_green = yes

	monthly_progress = {
		add = {
			desc = "gaudi_mechanical_computer_project_progress_base_value_tt"
			value = 1
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
			}
			add = {
				desc = "gaudi_mechanical_computer_project_progress_mechanical_engineer_tt"
				value = 1
			}
			if = {
				limit = {
					var:tesla_engineer_var = {
						has_variable = tesla_mechanical_engineer_experience_var
					}
				}
				add = {
					desc = "gaudi_mechanical_computer_project_progress_mechanical_engineer_experience_tt"
					value = var:tesla_engineer_var.var:tesla_mechanical_engineer_experience_var
					divide = 300
				}
			}
			if = {
				limit = {
					tesla_has_mechanical_engineer_trigger = yes
					var:tesla_engineer_var = {
						has_trait = tesla_mechanical_engineer
					}
				}
				add = {
					desc = "gaudi_mechanical_computer_project_progress_mechanical_engineer_trait_tt"
					value = 1
				}
			}
			if = {
				limit = {
					tesla_has_mechanical_engineer_trigger = yes
					# Engineer is mathematician
					var:tesla_engineer_var = {
						has_trait = verrier_mathematician
					}
				}
				add = {
					desc = "gaudi_mechanical_computer_project_progress_mathematician_trait_tt"
					value = 1
				}
			}
			if = {
				limit = {
					tesla_has_mechanical_engineer_trigger = yes
					var:tesla_engineer_var = {
						has_trait = innovative
					}
				}
				add = {
					desc = "gaudi_mechanical_computer_project_progress_innovative_trait_tt"
					value = 0.5
				}
			}
			if = {
				limit = {
					tesla_has_mechanical_engineer_trigger = yes
					var:tesla_engineer_var = {
						has_trait = curie_physicist
					}
				}
				add = {
					desc = "gaudi_mechanical_computer_project_progress_physicist_trait_tt"
					value = 0.5
				}
			}
			if = {
				limit = {
					tesla_has_mechanical_engineer_trigger = yes
					var:tesla_engineer_var = {
						has_trait = meticulous
					}
				}
				add = {
					desc = "gaudi_mechanical_computer_project_progress_meticulous_trait_tt"
					value = 0.1
				}
			}
			if = {
				limit = {
					tesla_has_mechanical_engineer_trigger = yes
					var:tesla_engineer_var = {
						has_trait = persistent
					}
				}
				add = {
					desc = "gaudi_mechanical_computer_project_progress_persistent_trait_tt"
					value = 0.1
				}
			}
			if = {
				limit = {
					has_variable = tesla_engineer_var
					var:tesla_engineer_var = {
						OR = {
							has_trait = alcoholic
							has_trait = expensive_tastes
							has_trait = opium_addiction
							has_trait = cocaine_addiction
							has_trait = syphilis
							has_trait = psychological_affliction
						}
					}
				}
				add = {
					desc = "gaudi_mechanical_computer_project_progress_negative_trait_tt"
					value = -0.2
				}
			}
			if = {
				limit = {
					has_variable = tesla_engineer_var
					var:tesla_engineer_var = {
						has_trait = senile
					}
				}
				add = {
					desc = "gaudi_mechanical_computer_project_progress_senile_trait_tt"
					value = -3
				}
			}
		}
		if = {
			limit = {
				academics_num_mathematical_universities > 0
			}
			add = {
				desc = "gaudi_mechanical_computer_project_progress_specialized_universities_tt"
				value = academics_mathematical_university_multiplier_tiny
			}
		}
		if = {
			limit = {
				tesla_num_tooling_workshop_levels_value > 0
			}
			add = {
				desc = "gaudi_mechanical_computer_project_progress_tooling_workshop_levels_tt"
				value = tesla_num_tooling_workshop_levels_value
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 2000
}
