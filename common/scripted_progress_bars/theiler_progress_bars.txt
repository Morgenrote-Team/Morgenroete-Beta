theiler_easy_disease_project_progress_bar = {
	name = "theiler_easy_disease_project_progress_bar"
	desc = "theiler_easy_disease_project_progress_bar_desc"

	default_green = yes

	monthly_progress = {
		if = {
			limit = {
				has_variable = theiler_biologist_var
			}
			add = {
				desc = "theiler_disease_project_progress_biologist_tt"
				value = 0.5
			}
			if = {
				limit = {
					var:theiler_biologist_var ?= {
						has_variable = theiler_biologist_experience_var
					}
				}
				add = {
					desc = "theiler_disease_project_progress_biologist_experience_tt"
					value = var:theiler_biologist_var.var:theiler_biologist_experience_var
					divide = 300
				}
			}
			if = {
				limit = {
					var:theiler_biologist_var ?= {
						has_trait = theiler_microbiologist
					}
				}
				add = {
					desc = "theiler_disease_project_progress_biologist_microbiologist_trait_tt"
					value = 1
				}
			}
			if = {
				limit = {
					var:theiler_biologist_var ?= {
						has_trait = panum_physician
					}
				}
				add = {
					desc = "theiler_disease_project_progress_biologist_physician_trait_tt"
					value = 0.5
				}
			}
			if = {
				limit = {
					var:theiler_biologist_var ?= {
						has_trait = mendelejew_chemist
					}
				}
				add = {
					desc = "theiler_disease_project_progress_biologist_chemist_trait_tt"

			}
			if = {
				limit = {

				}
			}
			if = {
				limit = {
					var:theiler_biologist_var ?= {
						has_trait = innovative
					}
				}
				add = {
					desc = "theiler_disease_project_progress_biologist_innovative_trait_tt"
					value = 0.25
				}
			}
			if = {
				limit = {
					var:theiler_biologist_var ?= {
						has_trait = persistent
					}
				}
				add = {
					desc = "theiler_disease_project_progress_biologist_persistent_trait_tt"

				}
			}
			if = {
				limit = {
					var:theiler_biologist_var ?= {
						has_trait = ambitious
					}
				}
				add = {
					desc = "theiler_disease_project_progress_biologist_ambitious_trait_tt"

				}
			}
			if = {
				limit = {
					var:theiler_biologist_var ?= {
						OR = {
							has_trait = alcoholic
							has_trait = expensive_tastes
