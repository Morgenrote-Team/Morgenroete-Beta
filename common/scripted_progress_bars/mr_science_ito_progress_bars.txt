ito_botanical_expedition_medicinal_plants_progress_bar = {
	name = "ito_botanical_expedition_medicinal_plants_progress_bar"
	desc = "ito_botanical_expedition_medicinal_plants_progress_bar_desc"
	default_green = yes

	monthly_progress = {
		if = {
			limit = {
				ito_has_naturalist_no_tooltip_trigger = yes
			}
			add = {
				value = var:dubois_naturalist_var.var:dubois_naturalist_experience_var
				divide = 600
				desc = "ito_botanical_expedition_medicinal_plants_progress_bar_naturalist_experience_tt"
			}
			if = {
				limit = {
					var:dubois_naturalist_var = {
						has_trait = explorer
					}
				}
				add = {
					value = 0.1
					desc = "ito_botanical_expedition_medicinal_plants_progress_bar_naturalist_explorer_trait_tt"
				}
			}
			if = {
				limit = {
					var:dubois_naturalist_var = {
						has_trait = ito_botanist
					}
				}
				add = {
					value = 0.1
					desc = "ito_botanical_expedition_medicinal_plants_progress_bar_naturalist_botanist_trait_tt"
				}
			}
			if = {
				limit = {
					var:dubois_naturalist_var = {
						has_trait = persistent
					}
				}
				add = {
					value = 0.05
					desc = "ito_botanical_expedition_medicinal_plants_progress_bar_naturalist_persistent_trait_tt"
				}
			}
			if = {
				limit = {
					var:dubois_naturalist_var = {
						has_trait = meticulous
					}
				}
				add = {
					value = 0.05
					desc = "ito_botanical_expedition_medicinal_plants_progress_bar_naturalist_meticulous_trait_tt"
				}
			}
			if = {
				limit = {
					var:dubois_naturalist_var = {
						has_trait = senile
					}
				}
				add = {
					value = -0.2
					desc = "ito_botanical_expedition_medicinal_plants_progress_bar_naturalist_senile_trait_tt"
				}
			}
			if = {
				limit = {
					var:dubois_naturalist_var = {
						has_addiction = yes
					}
				}
				add = {
					value = -0.1
					desc = "ito_botanical_expedition_medicinal_plants_progress_bar_naturalist_addiction_trait_tt"
				}
			}
		}
		if = {
			limit = {
				academics_num_natural_universities >= 1
			}
			add = {
				value = academics_natural_university_multiplier_very_tiny_value
				desc = "ito_botanical_expedition_medicinal_plants_progress_bar_natural_history_universities_tt"
			}
		}
	}

	start_value = 0
	min_value = 0
	max_value = 10
}
