panum_physician = {
	type = skill
	texture = "gfx/interface/icons/character_trait_icons/panum_physician.dds"

	character_modifier = {
		character_popularity_add = 5
	}

	agitator_modifier = {
		political_movement_pop_attraction_mult = 0.05
	}

	country_modifier = {
		country_academics_pol_str_mult = 0.02
		country_society_tech_spread_mult = 0.05
	}

	interest_group_modifier = {
		interest_group_pop_attraction_mult = 0.02
	}

	possible = {
		owner ?= {
			has_technology_researched = medical_degrees
		}
		NOT = { age < define:NCharacters|ADULT_AGE }
	}

	weight = {
		value = 0.5
		if = {
			limit = {
				owner ?= {
					country_has_building_type_levels = {
						target = bt:building_university
						value  >= 10
					}
				}
			}
			add = 0.5
		}
	}

	replace = {
	}

	value = 2
}

###### PSYCHOLOGIST #############################################################

panum_psychologist = {
	type = personality
	texture = "gfx/interface/icons/character_trait_icons/panum_psychologist.dds"

	character_modifier = {
		character_popularity_add = 5
	}

	country_modifier = {
		country_panum_psychological_treatment_success_chance_mult = 0.1
	}

	agitator_modifier = {
		political_movement_pop_attraction_mult = 0.1
	}

	interest_group_modifier = {

	}

	possible = {
		owner ?= {
			has_technology_researched = psychoanalysis
		}
		NOT = { age < define:NCharacters|ADULT_AGE }
	}

	weight = {
		value = 0.2
		if = {
			limit = {
				is_interest_group_type = ig_intelligentsia
			}
			add = 0.3
		}
		if = {
			limit = {
				owner ?= {
					any_scope_building = {
						level >= 10
						has_active_production_method = pm_dubois_university_humanities
					}
				}
			}
			add = 0.5
		}
	}

	replace = {
		trait_child
	}

	value = 1
}
