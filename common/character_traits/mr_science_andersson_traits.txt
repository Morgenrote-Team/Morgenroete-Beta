###### ANTHROPOLOGIST ###########################################################

andersson_anthropologist = {
	type = skill
	texture = "gfx/interface/icons/character_trait_icons/andersson_anthropologist.dds"

	character_modifier = {
		character_popularity_add = 15
	}

	agitator_modifier = {
	}

	country_modifier = {
		country_academics_pol_str_mult = 0.025
		country_society_tech_research_speed_mult = 0.025
	}

	interest_group_modifier = {
		interest_group_pop_attraction_mult = 0.025
	}

	possible = {
		owner ?= {
			has_technology_researched = andersson_anthropology_tech
			any_scope_building = {
				is_building_type = building_university
			}
		}
		NOT = { age < define:NCharacters|ADULT_AGE }
	}

	weight = {
		value = 0.1
		if = {
			limit = {
				OR = {
					is_interest_group_type = ig_intelligentsia
					is_interest_group_type = ig_landowners
					is_interest_group_type = ig_industrialists
				}
			}
			add = 0.2
		}
		if = {
			limit = {
				owner ?= {
					any_scope_building = {
						level >= 10
						has_active_production_method = pm_dubois_university_humanities
					}
				}
			}
			add = 0.2
		}
	}

	replace = {
	}

	value = 2
}
