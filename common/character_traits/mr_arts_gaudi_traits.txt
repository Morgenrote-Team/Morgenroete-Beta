###ARCHITECT###

gaudi_architect = {
	type = skill
	texture = "gfx/interface/icons/character_trait_icons/gaudi_architect.dds"

	command_modifier = {
		character_battle_condition_dug_in_mult = 1
		character_battle_condition_logistics_secured_mult = 1
	}

	country_modifier = {
		country_construction_add = 10
	}

	agitator_modifier = {
	}

	interest_group_modifier = {
	}

	possible = {
		owner ?= { NOT = { is_country_type = decentralized } }
		NOT = { age < define:NCharacters|ADULT_AGE }
	}

	weight = {
		value = 0.2
		if = {
			limit = {
				owner ?= {
					any_scope_building = {
						level >= 10
						has_active_production_method = pm_artists_arts_academy_school
					}
				}
			}
			add = 0.2
		}
		if = {
			limit = {
				has_role_of_type = agitator
				interest_group ?= {
					is_interest_group_type = ig_trade_unions
				}
			}
			multiply = 2
		}
		if = {
			limit = {
				interest_group ?= {
					is_interest_group_type = ig_industrialists
				}
			}
			multiply = 2
		}
	}

	replace = {
	}

	value = 2
}
