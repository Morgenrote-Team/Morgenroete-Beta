###### ATHLETE ###########################################################

vikelas_athlete = {
	type = skill
	texture = "gfx/interface/icons/character_trait_icons/vikelas_athlete.dds"

	character_modifier = {
		character_popularity_add = 10
		character_health_add = 0.3
	}

	agitator_modifier = {
		political_movement_pop_attraction_mult = 0.025
	}

	command_modifier = {
		unit_morale_recovery_mult = 0.05
	}

	country_modifier = {
		country_influence_mult = 0.01
		country_vikelas_decorative_olympic_success_add = 50
	}

	interest_group_modifier = {
		interest_group_pop_attraction_mult = 0.05
	}

	possible = {
		owner ?= {
			has_technology_researched = vikelas_international_sports_tech
		}
		NOT = { age < define:NCharacters|ADULT_AGE }
	}

	weight = {
		value = 0.5
		if = {
			limit = {
				owner ?= {
					has_variable = vikelas_school_sport_var
				}
			}
			add = 1
		}
		if = {
			limit = {
				owner ?= {
					OR = {
						has_modifier = vikelas_olympics_preparation_modifier
						has_modifier = vikelas_olympics_hosted_modifier
					}
				}
			}
			add = 1
		}
		if = {
			limit = {
				owner ?= {
					has_variable = vikelas_ioc_member_var
				}
			}
			multiply = 2
		}
	}

	replace = {
	}

	value = 3
}
