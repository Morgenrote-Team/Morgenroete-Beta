#Olympics decision is removed
revive_olympic_games_decision = {
	is_shown = {
		always = no
	}

	possible = {
		this = c:GRE
		has_technology_researched = vikelas_international_sports_tech
		hidden_trigger = {
			NOT = { has_variable = revive_olympics_decision_taken }
		}
	}
	
	when_taken = {
		set_variable = {
			name = revive_olympics_decision_taken
			value = yes
		}
		add_modifier = {
			name = modifier_olympic_games
		}
		add_modifier = {
			name = modifier_olympic_games_bureaucracy
			days = short_modifier_time
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				bureaucracy > 0
			}

			add = 5
		}
		if = {
			limit = {
				OR = {
					is_diplomatic_play_committed_participant = yes
					is_at_war = yes
				}
			}

			subtract = 1000
		}	
	}
}

#Rome decision is removed
italy_move_capital_to_rome = {
	is_shown = {
		always = no
	}

	possible = {
		c:ITA ?= ROOT
		owns_entire_state_region = STATE_LAZIO
		NOT = { has_variable = moved_capital_to_rome }
		NOT = {
			capital = { state_region = s:STATE_LAZIO }
		}
	}
	when_taken = {
		set_variable = moved_capital_to_rome
		set_capital = STATE_LAZIO
		random_scope_state = {
			limit = {
				state_region = s:STATE_LAZIO
			}
			add_modifier = {
				name = modifier_restored_capital_to_rome
				days = normal_modifier_time
			}
		}
	}

	ai_chance = {
		value = 100
	}
}
