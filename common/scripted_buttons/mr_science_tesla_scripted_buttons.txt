######INFRASTRUCTURE PROJECT######

tesla_railway_project_new_priority_button = {
	name = "tesla_railway_project_new_priority_button"
	desc = "tesla_railway_project_new_priority_button_desc"

	possible = {
		custom_tooltip = {
			text = MR_ALREADY_PRIORITIZING_STATES_TRIGGER_TT
			NOT = {
				has_variable = mr_railway_state_prioritizing_window_active_var
			}
		}
	}

	effect = {
		clear_variable_list = states_for_railway_improvement_variable_list
		ordered_scope_state = {
			limit = {
				any_scope_building = {
					tesla_railway_valid_for_improvement_trigger = yes
				}
			}
			order_by = tesla_infrastructure_balance
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = states_for_railway_improvement_variable_list
					target = prev
				}
			}
		}
		#This is used to remove elements from the prioritized list from the new list
		if = {
			limit = { has_variable_list = prioritized_states_for_railway_improvement_variable_list }
			every_in_list = {
				variable = prioritized_states_for_railway_improvement_variable_list
				root = {
					remove_list_variable = {
						name = states_for_railway_improvement_variable_list
						target = prev
					}
				}
			}
		}
		#Activates the window
		set_variable = mr_railway_state_prioritizing_window_active_var
	}

	ai_chance = {
		value = 0 #It's just not worth for you, Mr AI
	}
}

######ADDITIONAL SUPPORT FOR THE CIVIL ENGINEER######

gaudi_additional_support_for_civil_engineer_button = {
	name = "gaudi_additional_support_for_civil_engineer_button"
	desc = "gaudi_additional_support_for_civil_engineer_button_desc"

	visible = { 
		NOT = { has_modifier = gaudi_additional_support_for_civil_engineer_modifier }
	}

	possible = {
	}

	effect = {
		add_modifier = {
			name = gaudi_additional_support_for_civil_engineer_modifier
		}
	}

	ai_chance = {
		value = 0
		if = {
			limit = {
				gold_reserves > 0
				weekly_net_fixed_income > 3000
			}
			add = 2000
		}
	}
}

je_tesla_switch_to_naval_research_project = {
	name = "je_tesla_switch_to_naval_research_project"
	desc = "je_tesla_switch_to_naval_research_project_desc"

	visible = {
		NOT = {
			je:je_gaudi_military_research_project = {
				has_variable = tesla_naval_research_focus_var
			}
		}
	}

	possible = {
		always = yes
	}

	effect = {
		custom_tooltip = {
			text = je_tesla_switch_to_naval_research_project_effect_tt

				set_variable = tesla_naval_research_focus_var
			}
		}
	}

	ai_chance = {
		value = 0 #It's just not worth for you, Mr AI
	}
}
