######STOP CURRENT ENGINEERING PROJECT######

mendelejew_stop_chemistry_project_button = {
	name = "mendelejew_stop_chemistry_project_button"
	desc = "mendelejew_stop_chemistry_project_button_desc"

	visible = {  always = yes }

	possible = {
		NOT = { has_variable = mendelejew_stop_chemist_improving_project_var }
	}

	effect = {
		set_variable = {
			name = mendelejew_stop_chemist_improving_project_var
			days = 7
		}
	}

	ai_chance = {
		value = 0 #It's just not worth for you, Mr AI
	}
}

###### CHEMICAL PLANTS IMPROVEMENT PROJECT ######

je_mendelejew_improve_fertilizer_project_change_list_button = {
	name = "je_mendelejew_improve_fertilizer_project_change_list_button"
	desc = "je_mendelejew_improve_fertilizer_project_change_list_button_desc"

	possible = {
		custom_tooltip = {
			text = MR_ALREADY_ADDING_NEW_BUILDINGS_TO_LIST_TRIGGER_TT
			NOT = {
				has_variable = mendelejew_improve_chemical_plants_building_choser_var
			}
		}
	}

	effect = {
		clear_variable_list = mendelejew_improve_chemical_plants_buildings_variable_list
		ordered_scope_building = {
			limit = {
				mendelejew_building_valid_for_chemical_plant_project_no_tooltip_trigger = yes
				NOT = {
					root = {
						is_target_in_variable_list = {
							name = mendelejew_improve_chemical_plants_buildings_chosen_variable_list
							target = prev
						}
					}
				}
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = mendelejew_improve_chemical_plants_buildings_variable_list
					target = prev
				}
			}
		}
		#Activates the window
		set_variable = mendelejew_improve_chemical_plants_building_choser_var
	}

	ai_chance = {
		value = 0 #It's just not worth for you, Mr AI
	}
}
