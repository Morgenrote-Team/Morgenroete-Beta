douglas_stop_mountaineering_project_button = {
	name = "douglas_stop_mountaineering_project_button"
	desc = "douglas_stop_mountaineering_project_button_desc"

	visible = {  always = yes }

	possible = {
		NOT = { has_variable = douglas_stop_mountaineering_project_var }
	}

	effect = {
		set_variable = douglas_stop_mountaineering_project_var
	}

	ai_chance = {
		value = 0 #It's just not worth for you, Mr AI
	}
}

douglas_mount_everest_physicist_button = {
	name = "douglas_mount_everest_physicist_button"
	desc = "douglas_mount_everest_physicist_button_desc"

	visible = {  always = no }

	possible = {
		curie_has_active_physicist = yes
		trigger_if = {
			limit = {
				has_variable = curie_physicist_var
			}
			curie_physicist_var = {
				has_trait = curie_meteorologist
			}
		}
	}

	effect = {
		set_variable = douglas_mount_everest_physicist_var
		trigger_event = {
			id = douglas.402
			popup = yes
		}
	}

	ai_chance = {
		value = 0 #It's just not worth for you, Mr AI
	}
}
