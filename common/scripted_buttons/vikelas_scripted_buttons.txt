vikelas_mechanic_boost_button = {
	name = "vikelas_mechanic_boost_button"
	desc = "vikelas_mechanic_boost_button_desc"

	visible = { 
		always = yes
	}

	possible = {
		NOT = {
			has_modifier = vikelas_mechanic_boost_modifier
		}
		custom_tooltip = {
			text = MR_PADDOCK_100_MECHANIC_INNOVATION_TRIGGER_TT
			var:vikelas_mechanical_innovation_var >= 100
		}
	}

	effect = {
		if = {
			limit = {
				is_player = yes
			}
			vikelas_add_x_mechanical_innovation_effect = { 
				AMOUNT = -100
			}
		}
		else = {
			vikelas_add_x_mechanical_innovation_effect = { 
				AMOUNT = -50
			}
		}
		add_modifier = {
			name = vikelas_mechanic_boost_modifier
		}
	}

	ai_chance = {
		value = 100
	}

}

vikelas_autoracing_found_automobile_club_button = {
	name = "vikelas_autoracing_found_automobile_club_button"
	desc = "vikelas_autoracing_found_automobile_club_button_desc"

	visible = { 
		NOT = {
			has_variable = vikelas_automobile_club_var
		}
	}

	possible = {
		has_technology_researched = combustion_engine
	}

	effect = {
		set_variable = vikelas_automobile_club_var
		hidden_effect = {
			create_new_character_base_effect = { CHARACTER = racer PROJECT = vikelas }
			vikelas_set_new_racer_as_racer_effect = yes
		}
		trigger_event = {
			id = vikelas.501
		}
	}

	ai_chance = {
		value = 100
	}
}

		}

		}
	}

	ai_chance = {
		value = 100
	}
}

vikelas_autoracing_found_grand_prix_button = {
	name = "vikelas_autoracing_found_grand_prix_button"
	desc = "vikelas_autoracing_found_grand_prix_button_desc"

	visible = { 
		NOT = {
			has_global_variable = vikelas_autoracing_grand_prix_founded_var
		}
		has_variable = vikelas_automobile_club_var
	}

	possible = {
		has_technology_researched = combustion_engine
		has_technology_researched = vikelas_international_sports_tech
		has_technology_researched = elgar_mass_culture_tech
		custom_tooltip = {
			text = vikelas_autoracing_found_grand_prix_tt
			any_country = {
				has_variable = vikelas_automobile_club_var
				has_technology_researched = vikelas_international_sports_tech
				count >= 5
			}
		}
	}

	effect = {
		set_global_variable = vikelas_autoracing_grand_prix_founded_var
		add_modifier = {
			name = vikelas_grand_prix_found_modifier
			years = 5
		}
		trigger_event = {
			id = vikelas.502
		}
	}

	ai_chance = {
		value = 100
	}
}

vikelas_autoracing_race_track_button = {
	name = "vikelas_autoracing_race_track_button"
	desc = "vikelas_autoracing_race_track_button_desc"

	visible = { 
		NOT = {
			any_scope_state = {
				has_modifier = vikelas_race_track_modifier
			}
		}
		has_variable = vikelas_automobile_club_var
		has_global_variable = vikelas_autoracing_grand_prix_founded_var
	}

	possible = {
		has_technology_researched = combustion_engine
		has_technology_researched = vikelas_international_sports_tech
		has_technology_researched = paved_roads
		has_variable = vikelas_racer_var
		any_scope_state = {
			is_incorporated = yes
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 15
			}
		}
		tesla_has_engineer = yes
	}

	effect = {
		random_scope_state = {
			limit = {
				is_incorporated = yes
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 15
				}
			}
			add_modifier = {
				name = vikelas_race_track_modifier
			}
		}
	}

	ai_chance = {
		value = 100
	}
}
