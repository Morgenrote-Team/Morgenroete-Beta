verrier_join_asteriods_conference_button = {
	name = "verrier_join_asteriods_conference_button"
	desc = "verrier_join_asteriods_conference_button_desc"
	
	visible = {
		NOT = {
			is_involved_in_journal_entry = je_mr_verrier_asteriods_conference
		}
	}
	possible = {
		has_technology_researched = verrier_astronomy_tech
		verrier_astronomical_observation = yes
		OR = {
			any_scope_character = {
				has_trait = verrier_astronomer
			}
			has_variable = verrier_astronomer_var
			var:verrier_astronomical_objects >= 50
		}
	}
	effect = {
		global_var:je_mr_verrier_asteriods_conference = {
			add_involved_country = prev
		}
	}
	ai_chance = {
		value = 100 # We want maximum AI participation
	}
}

verrier_support_ceres_button = {
	name = "verrier_support_ceres_button"
	desc = "verrier_support_ceres_button_desc"
	
	visible = {
		is_involved_in_journal_entry = je_mr_verrier_asteriods_conference
	}
	selected = {
		var:mr_conference_position_var ?= flag:mr_stance_pro_ceres
	}
	possible = {
		is_involved_in_journal_entry = je_mr_verrier_asteriods_conference
	}
	effect = {
		if = {
			limit = {
				var:mr_conference_position_var ?= flag:mr_stance_pro_ceres
			}
			custom_tooltip = {
				text = verrier_stop_supporting_ceres
				remove_variable = mr_conference_position_var
			}
		}
		else = {
			custom_tooltip = {
				text = verrier_start_supporting_ceres
				set_variable = {
					name = mr_conference_position_var
					value = flag:mr_stance_pro_ceres
				}
			}
		}
	}
	ai_chance = { # Inherited from verrier.31.g
		value = 40
		if = { #Want to defend prestige
			limit = {
				OR = {
					has_variable = verrier_discovered_ceres_variable
					has_variable = verrier_discovered_pallas_var
					has_variable = verrier_discovered_juno_var
					has_variable = verrier_discovered_vesta_var
				}
			}
			add = 100
		}
		if = { #No reason to reform if lesser objects
			limit = {
				global_var:verrier_total_astronomical_objects <= 2000
			}
			add = 25
		}
		if = { #No reason to reform if lesser objects
			limit = {
				global_var:verrier_total_astronomical_objects <= 2500
			}
			add = 20
		}
		if = { #If the player is the host and wants to preserve Ceres, it'll be 50:50 instead 60:40 against Ceres
			limit = {
				any_country = {
					is_player = yes
					has_modifier = verrier_congress_host_modifier
				}
			}
			add = 20
		}
	}
}

verrier_oppose_ceres_button = {
	name = "verrier_oppose_ceres_button"
	desc = "verrier_oppose_ceres_button_desc"
	
	visible = {
		is_involved_in_journal_entry = je_mr_verrier_asteriods_conference
	}
	selected = {
		var:mr_conference_position_var ?= flag:mr_stance_anti_ceres
	}
	possible = {
		is_involved_in_journal_entry = je_mr_verrier_asteriods_conference
	}
	effect = {
		if = {
			limit = {
				var:mr_conference_position_var ?= flag:mr_stance_anti_ceres
			}
			custom_tooltip = {
				text = verrier_stop_opposing_ceres
				remove_variable = mr_conference_position_var
			}
		}
		else = {
			custom_tooltip = {
				text = verrier_start_opposing_ceres
				set_variable = {
					name = mr_conference_position_var
					value = flag:mr_stance_anti_ceres
				}
			}
		}
	}
	ai_chance = { # Inherited from verrier.31.h
		base = 60
		if = { #Want to defend prestige and hurt others
			limit = {
				OR = {
					has_variable = verrier_discovered_uranus_variable
					has_variable = verrier_discovered_neptune_variable
				}
			}
			add = 100
		}
		if = { #Reason to reform if more objects
			limit = {
				global_var:verrier_total_astronomical_objects >= 3500
			}
			add = 10
		}
		if = { #No reason to reform if lesser objects
			limit = {
				global_var:verrier_total_astronomical_objects >= 4000
			}
			add = 20
		}
		if = { #Historically, Prussian academics lead the reform to introduce asteroids and demote Ceres, Vesta, Pallas and Juno
			limit = {
				ROOT = c:PRU
			}
			add = 100
		}
	}
}
