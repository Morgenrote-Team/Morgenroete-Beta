verrier_join_asteriods_conference_button = {
	name = "verrier_join_asteriods_conference_button"
	desc = "verrier_join_asteriods_conference_button_desc"
	
	visible = {
		NOT = {
			has_variable = verrier_left_asteroids_conference_var
		}
	}
	selected = {
		is_involved_in_journal_entry = je_mr_verrier_asteriods_conference
	}
	possible = {
		verrier_has_astronomer = yes
	}
	effect = {
		global_var:je_mr_verrier_asteriods_conference = {
			add_involved_country = prev
		}
	}
	ai_chance = {
		base = 0
		modifier = {
			trigger = { }
			add = 0
		}
	}
}

verrier_promote_minor_planets_button = {
	name = "verrier_promote_minor_planets_button"
	desc = "verrier_promote_minor_planets_button_desc"
	
	visible = {
		global_var:je_mr_verrier_asteriods_conference = {
			NOT = {
				has_variable = verrier_ceres_debate_extension
			}
		}
	}
	selected = {
		has_variable = verrier_supporting_minor_planets_var
	}
	possible = {
		is_involved_in_journal_entry = je_mr_verrier_asteriods_conference
	}
	effect = {
		if = {
			limit = {
				has_variable = verrier_supporting_minor_planets_var
			}
			custom_tooltip = {
				text = verrier_stop_supporting_minor_planets
				remove_variable = verrier_supporting_minor_planets_var
			}
		}
		else = {
			custom_tooltip = {
				text = verrier_start_supporting_minor_planets
				set_variable = verrier_supporting_minor_planets_var
			}
		}
	}
	
	ai_chance = {
		base = 0
		modifier = {
			trigger = { }
			add = 0
		}
	}
}
