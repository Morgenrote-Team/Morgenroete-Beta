curtiss_ballooning_club_subsidies_button = {
	name = "curtiss_ballooning_club_subsidies_button"
	desc = "curtiss_ballooning_club_subsidies_button_desc"

	visible = yes
	possible = { always = yes }
	
	cooldown = 3
	effect = {
		if = {
			limit = { has_variable = curtiss_ballooning_club_subsidies }
			clr_variable = curtiss_ballooning_club_subsidies
		}
		else = {
			set_variable = {
				name = curtiss_ballooning_club_subsidies
				value = yes
			}
		}
	}
	ai_chance = {
		base = 0

	# Balloonomania #
	
		modifier = {
			trigger = {
				NOT = { has_variable = curtiss_ballooning_club_subsidies }
				year <= 1850
				OR = {
					c:FRA ?= ROOT
					c:GBR ?= ROOT
				}
			}
			add = 15
		}

	# Great/Major Powers like to show off #

		modifier = {
			trigger = {
				country_rank >= rank_value:major_power
			}
			add = 10
		}	

	# Austerity #

		modifier = {
			trigger = {
				has_variable = curtiss_ballooning_club_subsidies
				curtiss_austerity_trigger = yes
			}
			add = 100	
		}
		modifier = {
			trigger = {
				NOT = { has_variable = curtiss_ballooning_club_subsidies }
				curtiss_austerity_trigger = yes
			}
			add = -100	
		}
	}
}
