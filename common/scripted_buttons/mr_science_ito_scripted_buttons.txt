###### FORESTRY IMPROVEMENT PROJECT ######

je_ito_improve_forestry_project_change_list_button = {
	name = "je_ito_improve_forestry_project_change_list_button"
	desc = "je_ito_improve_forestry_project_change_list_button_desc"

	possible = {
		custom_tooltip = {
			text = MR_ALREADY_ADDING_NEW_BUILDINGS_TO_LIST_TRIGGER_TT
			NOT = {
				has_variable = dubois_improve_forestry_project_building_choser_var
			}
		}
	}

	effect = {
		clear_variable_list = ito_improve_forestry_buildings_variable_list
		ordered_scope_building = {
			limit = {
				ito_building_valid_for_forestry_project_no_tooltip_trigger = yes
				NOT = {
					root = {
						is_target_in_variable_list = {
							name = ito_improve_forestry_buildings_chosen_variable_list
							target = prev
						}
					}
				}
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = ito_improve_forestry_buildings_variable_list
					target = prev
				}
			}
		}
		#Activates the window
		set_variable = dubois_improve_forestry_project_building_choser_var
	}

	ai_chance = {
		value = 0 #It's just not worth for you, Mr AI
	}
}
