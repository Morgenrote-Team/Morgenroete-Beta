######STOP CURRENT ENGINEERING PROJECT######

agassiz_stop_geologist_project_button = {
	name = "agassiz_stop_geologist_project_button"
	desc = "agassiz_stop_geologist_project_button_desc"

	visible = {  always = yes }

	possible = {
		NOT = { has_variable = agassiz_stop_geologist_improving_project_var }
	}

	effect = {
		set_variable = {
			name = agassiz_stop_geologist_improving_project_var
			days = 7
		}
	}

	ai_chance = {
		value = 0 #It's just not worth for you, Mr AI
	}
}

###### MINES IMPROVEMENT PROJECT ######

je_agassiz_improve_mines_project_change_list_button = {
	name = "je_agassiz_improve_mines_project_change_list_button"
	desc = "je_agassiz_improve_mines_project_change_list_button_desc"

	possible = {
		custom_tooltip = {
			text = MR_ALREADY_ADDING_NEW_BUILDINGS_TO_LIST_TRIGGER_TT
			NOT = {
				has_variable = agassiz_improve_mine_project_building_choser_var
			}
		}
	}

	effect = {
		clear_variable_list = agassiz_improve_mine_buildings_variable_list
		ordered_scope_building = {
			limit = {
				agassiz_mine_valid_for_improvement_trigger = yes
				NOT = {
					root = {
						is_target_in_variable_list = {
							name = agassiz_improve_mine_buildings_chosen_variable_list
							target = prev
						}
					}
				}
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = agassiz_improve_mine_buildings_variable_list
					target = prev
				}
			}
		}
		#Activates the window
		set_variable = agassiz_improve_mine_project_building_choser_var
	}

	ai_chance = {
		value = 0 #It's just not worth for you, Mr AI
	}
}
