###Architecture School###

#Trigger if an art academy with architecture school exists in the country
gaudi_has_architecture_school_trigger = {
	any_scope_state = {
		is_production_method_active = {
			building_type = building_art_academy
			production_method = pm_artists_arts_academy_school
		}
	}
}
gaudi_has_architecture_school_tooltip_trigger = {
	custom_tooltip = {
		text = gaudi_has_architecture_school_tooltip_trigger_tt
		any_scope_state = {
			is_production_method_active = {
				building_type = building_art_academy
				production_method = pm_artists_arts_academy_school
			}
		}
	}
}

###Architect Existence###

#Trigger if an architect exists
gaudi_has_architect = {
	custom_tooltip = {
		text = gaudi_has_architect_tt
		gaudi_has_architect_no_tooltip_trigger = yes
	}
}
gaudi_has_architect_no_tooltip_trigger = {
	has_variable = gaudi_architect_var
}

#Trigger if an unoccupied architect exists
gaudi_has_active_architect = {
	custom_tooltip = {
		text = gaudi_has_active_architect_tt
		gaudi_has_active_architect_no_tooltip_trigger = yes
	}
}
gaudi_has_active_architect_no_tooltip_trigger = {
	has_variable = gaudi_architect_var
	var:gaudi_architect_var ?= {
		is_busy = no
	}
}

#Trigger if a Great Architect exists
gaudi_has_great_architect = {
	custom_tooltip = {
		text = gaudi_has_great_architect_tt
		has_variable = gaudi_architect_var
		var:gaudi_architect_var = {
			has_trait = gaudi_great_architect
		}
	}
}

#Trigger if an unoccupied Great Architect exists
gaudi_has_active_great_architect = {
	custom_tooltip = {
		text = gaudi_has_active_great_architect_tt
		has_variable = gaudi_architect_var
		var:gaudi_architect_var ?= {
			is_busy = no
			has_trait = gaudi_great_architect
		}
	}
}

### Rating Triggers ###

#Triggers for the rating display
gaudi_architect_fantastic_rating_trigger = {
	gaudi_architect_rating_value > 11
}

gaudi_architect_very_good_rating_trigger = {
	gaudi_architect_rating_value > 7
}

gaudi_architect_good_rating_trigger = {
	gaudi_architect_rating_value > 3
}

gaudi_architect_mediocre_rating_trigger = {
	gaudi_architect_rating_value > 0
}
