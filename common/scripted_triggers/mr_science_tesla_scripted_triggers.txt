#Has an engineering Project
gaudi_no_engineering_project_active_trigger = {
	NOR = {
		has_modifier = gaudi_engineer_project_low_cost_modifier
		has_modifier = gaudi_engineer_project_medium_cost_modifier
		has_modifier = gaudi_engineer_project_high_cost_modifier
		has_modifier = gaudi_engineer_project_very_high_cost_modifier
	}
}

#Has an Engineer that is not busy
gaudi_has_active_engineer = {
	custom_tooltip = {
		text = gaudi_has_active_engineer_tt
		var:gaudi_engineer_var ?= {
			is_busy = no
		}
	}
}

#Valid buildings for construction sector improvement project
tesla_construction_sector_valid_for_improvement_trigger = {
	level > 0
	is_building_type = building_construction_sector
	NAND = {
		has_modifier = gaudi_new_construction_methods_modifier
		has_modifier = gaudi_faster_construction_methods_modifier
		has_modifier = gaudi_efficient_construction_methods_modifier
	}
}

#Valid buildings for railway (civil) improvement project
tesla_railway_valid_for_improvement_trigger = {
	level > 0
	is_building_type = building_railway
	NOT = { has_modifier = tesla_modernized_railway_lines_building_modifier }
}
#buildings that already have the (civil) improvement for railway
tesla_railway_invalid_for_improvement_trigger = {
	level > 0
	is_building_type = building_railway
	has_modifier = tesla_modernized_railway_lines_building_modifier
}

#states valid for irrigation improvement project
tesla_state_region_has_no_irrigation_system = {
	NOT = {
		has_state_trait = tesla_state_trait_irrigation
	}
}
#states valid for irrigation improvement project
tesla_state_region_has_irrigation_system = {
	has_state_trait = tesla_state_trait_irrigation
}

#Valid buildings for mechanical improvement project
tesla_building_valid_for_mechanical_improvement_trigger = {
	level > 0
	NOR = {
		is_building_group = bg_whaling
		is_building_group = bg_fishing
		is_building_group = bg_service
		is_building_group = bg_urban_facilities
		is_building_group = bg_power
		is_building_group = bg_government
		is_building_group = bg_infrastructure
		is_building_group = bg_military
		is_building_group = bg_subsistence_agriculture
		is_building_type = building_elgar_opera
	}
	NAND = {
		has_modifier = gaudi_building_throughput_add_modifier
		has_modifier = gaudi_building_production_mult_modifier
		has_modifier = gaudi_building_goods_input_mult_modifier
	}
}

#Valid Airport for improvement project
tesla_airport_valid_for_improvement_trigger = {
	level > 0
	is_building_type = building_airport
	NAND = {
		has_modifier = gaudi_runway_expansion_mult_modifier
		has_modifier = gaudi_state_of_the_art_planes_mult_modifier
		has_modifier = gaudi_efficient_motors_modifier
	}
}
