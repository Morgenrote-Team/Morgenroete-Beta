#This is used to prevent some historical characters from wearing hats
mr_no_hat_historic_person_trigger = {
	scope:character ?= {
		has_variable = mr_no_headgear_var
	}
}

mr_european_plus_clothes_trigger = {
    trigger_if = {
        limit = {
            WCR_is_active_trigger = yes
        }
        WCR_TPM_is_westernized_trigger = yes # Checks for both western countries and countries that have westernized
    }
    trigger_else = {
        OR = {
            european_clothes_trigger = yes
            scope:character ?= {
                OR = {
                    culture = cu:cherokee
                    culture = cu:muskogean
                    AND = { # Westernized Japan is caught by european_clothes_trigger but not Korea for some reason
                        culture = cu:korean
                        owner ?= {
                            has_variable = KOR_korean_empire_declared
                        }
                    }
                }
            }
            african_diaspora_clothes_trigger = yes
        }
    }
}

mr_academic_expedition_clothes_trigger = {
	scope:character ?= {
		is_busy = yes
		OR = {
			has_role = character_role_andersson_anthropologist
			AND = {
				has_role = character_role_verrier_astronomer
				has_variable = rapanui_character_on_pacific_expedition_var
			}
			AND = {
				has_role = character_role_ito_naturalist
				OR = {
					has_variable = dubois_naturalist_on_botanical_expedition_var
					has_variable = dubois_naturalist_on_safari_var
					has_variable = dubois_naturalist_on_zoological_expedition_var
					has_variable = rapanui_character_on_pacific_expedition_var
				}
			}
			AND = {
				has_role = character_role_dubois_paleontologist
				OR = {
					has_variable = rapanui_character_on_pacific_expedition_var
					has_variable = dubois_paleontologist_in_north_america_var
					has_variable = dubois_paleontologist_in_south_america_var
					has_variable = dubois_paleontologist_in_europe_var
					has_variable = dubois_paleontologist_in_africa_var
					has_variable = dubois_paleontologist_in_asia_var
					has_variable = dubois_paleontologist_in_australia_var
					has_variable = dubois_paleontologist_in_local_var
				}
			}
			AND = {
				has_role = character_role_lepsius_archaeologist
				OR = {
					has_variable = lepsius_archaeologist_in_egypt_var
					has_variable = lepsius_archaeologist_in_mesopotamia_var
					has_variable = lepsius_archaeologist_in_greece_var
					has_variable = lepsius_archaeologist_in_yucatan_var
					has_variable = lepsius_archaeologist_in_peru_var
				}
			}
			AND = {
				has_role = character_role_agassiz_geologist
				owner ?= {
					has_journal_entry = je_agassiz_volcano_expedition
				}
			}
		}
		NOT = { has_variable = mr_no_special_clothes_var }
	}
}

mr_academic_labcoat_clothes_trigger = {
	scope:character ?= {
		AND = {
			owner ?= {
				has_technology_researched = aniline

			OR = {
				has_role = character_role_mendelejew_chemist
				has_role = character_role_curie_physicist
				has_role = character_role_theiler_biologist
			}
		}
		AND = {
			has_role = character_role_panum_physician

				OR = {
					has_modifier = panum_medical_techniques_level_4_modifier

					has_modifier = panum_medical_techniques_level_6_modifier
					has_modifier = panum_medical_techniques_level_7_modifier
				}
			}
		}
	}
}
