#Boll Weevil Harvest Condition
theiler_state_can_have_boll_weevil_trigger = {
	game_date >= 1892.1.1 #Entered the USA in 1892
	NOT = {
		has_global_variable = theiler_recent_boll_weevil_plague_global_var
	}
	any_scope_state = {
		is_in_geographic_region = theiler_geographic_region_boll_weevil
		any_scope_building = {
			is_building_type = building_cotton_plantation
		}
		owner = {
			theiler_has_boll_weevil_prevention_trigger = no
		}
	}
}

theiler_has_boll_weevil_prevention_trigger = {
	has_variable = theiler_boll_weevil_prevention_var
}

#Codling Moth Harvest Condition
theiler_state_can_have_codling_moth_trigger = {
	NOT = {
		has_global_variable = theiler_recent_codling_moth_plague_global_var
	}
	any_scope_state = {
		is_in_geographic_region = theiler_geographic_region_codling_moth
		any_scope_building = {
			OR = {
				is_building_type = building_wheat_farm
				is_building_type = building_maize_farm
				is_building_type = building_rye_farm
			}
			OR = {
				has_active_production_method = pm_apple_orchards
				has_active_production_method = pm_citrus_orchards #normally the parasite would not attack these, but wheat farms should be affected, too.
			}
		}
		owner = {
			theiler_has_codling_moth_prevention_trigger = no
		}
	}
}

theiler_has_codling_moth_prevention_trigger = {
	has_variable = theiler_codling_moth_prevention_var
}
