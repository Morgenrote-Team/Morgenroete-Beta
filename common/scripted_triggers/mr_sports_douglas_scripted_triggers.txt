###Mountaineer###

#Trigger if an mountaineer exists
douglas_has_mountaineer = {
	custom_tooltip = {
		text = douglas_has_mountaineer_tt
		douglas_has_mountaineer_no_tooltip_trigger = yes
	}
}
douglas_has_mountaineer_no_tooltip_trigger = {
	has_variable = douglas_mountaineer_var
}

#Trigger if an unoccupied mountaineer exists
douglas_has_active_mountaineer = {
	custom_tooltip = {
		text = douglas_has_active_mountaineer_tt
		douglas_has_active_mountaineer_no_tooltip_trigger = yes
	}
}
douglas_mountaineer_can_be_available = {	# It is always worth cancelling training for something else
	custom_tooltip = {
		text = douglas_has_available_mountaineer_tt
		OR = {
			douglas_has_active_mountaineer_no_tooltip_trigger = yes
			has_variable = douglas_can_stop_active_mountaineering_project
		}
	}
}
douglas_has_training_mountaineer = {
	custom_tooltip = {
		text = douglas_has_training_mountaineer_tt
		has_journal_entry = je_douglas_mountaineering_training
	}
}
douglas_has_active_mountaineer_no_tooltip_trigger = {
	has_variable = douglas_mountaineer_var
	var:douglas_mountaineer_var ?= {
		is_busy = no
	}
}

#Trigger if the country has a vanilla mountain state trait
douglas_has_mountain_range_trigger = {
	custom_tooltip = {
		text = douglas_has_mountain_range_tt
		any_scope_state = {
			OR = {
				has_state_trait = state_trait_cascades
				has_state_trait = state_trait_sierra_nevada
				has_state_trait = state_trait_rocky_mountains
				has_state_trait = state_trait_sierra_madre_occidental
				has_state_trait = state_trait_sierra_madre_oriental
				has_state_trait = state_trait_central_american_cordilleras
				has_state_trait = state_trait_wasatch_range
				has_state_trait = state_trait_andes
				has_state_trait = state_trait_alps_mountains
				has_state_trait = state_trait_caucasus_mountains
				has_state_trait = state_trait_atlas_mountains
				has_state_trait = state_trait_zagros_mountains
				has_state_trait = state_trait_alborz_mountains
				has_state_trait = state_trait_baluchistan_mountains
				has_state_trait = state_trait_mount_lebanon
				has_state_trait = state_trait_himalayas_mountains
				has_state_trait = state_trait_great_dividing_range
			}
		}
	}
}

###Advanced Equipment###

douglas_has_advanced_equpiment_trigger = {
	OR = {
		AND = {
			douglas_climbs_level_4_mountain_trigger = yes
			has_modifier = douglas_mountaineering_equipment_level_5_modifier
		}
		AND = {
			douglas_climbs_level_3_mountain_trigger = yes
			OR = {
				has_modifier = douglas_mountaineering_equipment_level_4_modifier
				has_modifier = douglas_mountaineering_equipment_level_5_modifier
			}
		}
		AND = {
			douglas_climbs_level_2_mountain_trigger = yes
			OR = {
				has_modifier = douglas_mountaineering_equipment_level_3_modifier
				has_modifier = douglas_mountaineering_equipment_level_4_modifier
				has_modifier = douglas_mountaineering_equipment_level_5_modifier
			}
		}
		AND = {
			douglas_climbs_level_1_mountain_trigger = yes
			OR = {
				has_modifier = douglas_mountaineering_equipment_level_2_modifier
				has_modifier = douglas_mountaineering_equipment_level_3_modifier
				has_modifier = douglas_mountaineering_equipment_level_4_modifier
				has_modifier = douglas_mountaineering_equipment_level_5_modifier
			}
		}
	}
}

######TRIGGER FOR UNCONQUERED MOUNTAINS######

#Level 1 Summits (twelve mountains)
douglas_has_unconquered_level_1_summit_trigger = {
	OR = {
		AND = {
			NOT = {
				has_global_variable = douglas_mount_whitney_conqueror_global_var
			}
			any_scope_state = {
				state_region = s:STATE_CALIFORNIA
			}
		}
		AND = {
			NOT = {
				has_global_variable = douglas_citlaltepetl_conqueror_global_var
			}
			any_scope_state = {
				state_region = s:STATE_MEXICO
			}
		}
		AND = {
			NOT = {
				has_global_variable = douglas_popocatepetl_conqueror_global_var
			}
			any_scope_state = {
				state_region = s:STATE_MEXICO
			}
		}
		AND = {
			NOT = {
				has_global_variable = douglas_tajumulco_conqueror_global_var
			}
			any_scope_state = {
				state_region = s:STATE_GUATEMALA
			}
		}
		AND = {
			NOT = {
				has_global_variable = douglas_chimborazo_conqueror_global_var
			}
			any_scope_state = {
				state_region = s:STATE_ECUADOR
			}
		}
		AND = {
