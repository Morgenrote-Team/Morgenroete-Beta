##############################
# VIKELAS SCRPITED TRIGGERS  #
##############################

#Is this country a founding member of Grand Prix?
vikelas_is_grand_prix_country_trigger = {
	OR = {
		has_variable = vikelas_aiacr_founding_country_var
		has_variable = vikelas_aiacr_board_member_var
	}
}

#Is this country a country with a Grand Prix Race Track?
vikelas_has_grand_prix_race_track_trigger = {
	OR = {
		has_variable = vikelas_aiacr_founding_country_var #The founder will be considered a track owner, even without one
		any_scope_building = {
			is_building_type = building_vikelas_circuit
			has_active_production_method = pm_artists_arts_academy_conservatory
		}
	}
}

#Does the mechanic improve the race car?
vikelas_mechanic_improves_race_car_trigger = {
	OR = {
		has_journal_entry = je_vikelas_mechanic_engine
		has_journal_entry = je_vikelas_mechanic_safety
		has_journal_entry = je_vikelas_mechanic_body
	}
}

#Can this country participate at a Race?
vikelas_can_participate_autorace_trigger = {
	NOT = {
		has_variable = vikelas_racer_does_not_participate_var
	}
	has_variable = vikelas_automobile_club_var
	has_variable = vikelas_racer_var
}
