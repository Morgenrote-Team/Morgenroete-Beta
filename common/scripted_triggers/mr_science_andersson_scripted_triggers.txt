#Anthropologist

#Has active anthropologist
andersson_has_active_anthropologist = {
	custom_tooltip = {
		text = andersson_has_active_anthropologist_tt
		andersson_has_active_anthropologist_no_tooltip_trigger = yes
	}
}
andersson_has_active_anthropologist_no_tooltip_trigger = {
	trigger_if = {
		limit = { has_variable = andersson_anthropologist_var }
		var:andersson_anthropologist_var = {
			is_busy = no
		}
	}
	trigger_else = {
		always = no
	}
}

#Can find Human fossils

#Can find Palaeanthropus
andersson_can_find_palaeanthropus_trigger = {
	OR = {
		any_scope_building = {
			is_building_type = building_andersson_institute
			has_active_production_method = pm_andersson_camera
		}
		any_scope_building = {
			is_building_type = building_andersson_institute
			has_active_production_method = pm_andersson_stratigraphy
		}
		any_scope_building = {
			is_building_type = building_andersson_institute
			has_active_production_method = pm_andersson_x_rays
		}
	}
}

		}

### CAVE PAINTINGS ###

#Country has Cave Paintings
andersson_country_has_cave_paintings_trigger = {
	any_scope_state = {
		andersson_state_has_cave_paintings_trigger = yes
	}
}

#State has Cave Paintings
andersson_state_has_cave_paintings_trigger = {
	state_region = {
		has_variable = andersson_cave_paintings_var
	}
}

#State has discovered Cave Paintings
andersson_state_has_discovered_cave_paintings_no_tooltip_trigger = {
	OR = {
		has_state_trait = andersson_state_trait_cave_paintings
		has_state_trait = andersson_state_trait_cave_paintings_unexplored
	}
}

#Country has unexplored Cave Paintings
andersson_country_has_unexplored_cave_paintings_trigger = {
	any_scope_state = {
		andersson_state_has_unexplored_cave_paintings_trigger = yes
	}
}

#State has unexplored Cave Paintings
andersson_state_has_unexplored_cave_paintings_trigger = {
	state_region = {
		has_variable = andersson_cave_paintings_var
		has_state_trait = andersson_state_trait_cave_paintings_unexplored
	}
}

### CAVES ###

#Country has a Cave
andersson_country_has_cave_trigger = {
	any_scope_state = {
		andersson_state_has_cave_trigger = yes
	}
}

#State has a Cave
andersson_state_has_cave_trigger = {
	has_variable = andersson_cave_var
}

#Country has a Neoanthropus cave
andersson_country_has_neoanthropus_cave_trigger = {
	any_scope_state = {
		andersson_state_has_neoanthropus_cave_trigger = yes
	}
}

#State has a Neoanthropus cave
andersson_state_has_neoanthropus_cave_trigger = {
	has_variable = andersson_cave_var
	OR = {
		var:andersson_cave_var ?= flag:andersson_vero_cave_flag
		var:andersson_cave_var ?= flag:andersson_lagoa_santa_cave_flag
		var:andersson_cave_var ?= flag:andersson_cro_magnon_cave_flag
		var:andersson_cave_var ?= flag:andersson_boskop_cave_flag
		var:andersson_cave_var ?= flag:andersson_tianyuan_cave_flag
		var:andersson_cave_var ?= flag:andersson_talgai_cave_flag
	}
}

#Country has a Palaeanthropus cave
andersson_country_has_palaeanthropus_cave_trigger = {
	any_scope_state = {
		andersson_state_has_palaeanthropus_cave_trigger = yes
	}
}

#State has a Palaeanthropus cave
andersson_state_has_palaeanthropus_cave_trigger = {
	has_variable = andersson_cave_var
	OR = {
		var:andersson_cave_var ?= flag:andersson_neanderthal_cave_flag
		var:andersson_cave_var ?= flag:andersson_gibraltar_cave_flag
		var:andersson_cave_var ?= flag:andersson_spy_cave_flag
		var:andersson_cave_var ?= flag:andersson_krapina_cave_flag
		var:andersson_cave_var ?= flag:andersson_galilee_cave_flag
	}
}

#Country has a Pithecanthropus cave
andersson_country_has_pithecanthropus_cave_trigger = {
	any_scope_state = {
		andersson_state_has_pithecanthropus_cave_trigger = yes
	}
}

#State has a Pithecanthropus cave
andersson_state_has_pithecanthropus_cave_trigger = {
	has_variable = andersson_cave_var
	OR = {

		var:andersson_cave_var ?= flag:andersson_peking_cave_flag

#Country has a Australopithecus cave

	any_scope_state = {
		andersson_state_has_australopithecus_cave_trigger = yes

	has_variable = andersson_palaeanthropus_bone_var

	has_variable = andersson_pithecanthropus_bone_var
	var:andersson_pithecanthropus_bone_var >= 10
}

andersson_has_maximum_australopithecus_bones_trigger = {
	has_variable = andersson_australopithecus_bone_var

}

lepsius_has_human_fossil = {
	custom_tooltip = {
		text = lepsius_has_human_fossil_tt
		var:andersson_total_human_fossil >= 1
	}
}

lepsius_has_unknown_bone = {
	custom_tooltip = {
		text = lepsius_has_unknown_bone_tt
		lepsius_has_unknown_bone_no_tooltip_trigger = yes
	}
}
lepsius_has_unknown_bone_no_tooltip_trigger = {
	OR = {
		AND = {
			has_variable = lepsius_unknown_cromagnon_human_fossil
			var:lepsius_unknown_cromagnon_human_fossil >= 1
		}
		AND = {
			has_variable = lepsius_unknown_neanderthalensis_human_fossil
			var:lepsius_unknown_neanderthalensis_human_fossil >= 1
		}
		AND = {
			has_variable = lepsius_unknown_erectus_human_fossil
			var:lepsius_unknown_erectus_human_fossil >= 1
		}
		AND = {
			has_variable = lepsius_unknown_australopithecus_human_fossil
			var:lepsius_unknown_australopithecus_human_fossil >= 1
		}
		AND = {
			has_variable = lepsius_unknown_paranthropus_human_fossil
			var:lepsius_unknown_paranthropus_human_fossil >= 1
		}
		AND = {
			has_variable = lepsius_unknown_heidelbergensis_human_fossil
			var:lepsius_unknown_heidelbergensis_human_fossil >= 1
		}
	}
}
