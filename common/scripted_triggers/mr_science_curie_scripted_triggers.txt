######Physicist CHARACTER TRIGGERS######

curie_has_physicist = {
	custom_tooltip = {
		text = curie_has_physicist_tt
		curie_has_physicist_no_tooltip_trigger = yes
	}
}
curie_has_physicist_no_tooltip_trigger = {
	has_variable = curie_physicist_var
}

curie_has_active_physicist = {
	custom_tooltip = {
		text = curie_has_active_physicist_tt
		curie_has_active_physicist_no_tooltip_trigger = yes
	}
}
curie_has_active_physicist_no_tooltip_trigger = {
	has_variable = curie_physicist_var
	var:curie_physicist_var ?= {
		is_busy = no
	}
}

#Uranium state

curie_is_uranium_state = {
    any_scope_state = {
		any_scope_building = {
			is_building_type = building_uranium_mine
		}
	}
}

curie_has_uranium_state = {
	any_scope_state = {
		has_state_trait = curie_state_trait_uranium
	}
}

### Triggers for the rating display Physicist ###
curie_physicist_fantastic_rating_trigger = {
	curie_physicist_character_rating_value > 8
}

curie_physicist_very_good_rating_trigger = {
	curie_physicist_character_rating_value > 5
}

curie_physicist_good_rating_trigger = {
	curie_physicist_character_rating_value > 2
}

curie_physicist_mediocre_rating_trigger = {
	curie_physicist_character_rating_value > 0
}
