######Biologist CHARACTER TRIGGERS######

theiler_has_active_biologist_trigger = {
	custom_tooltip = {
		text = theiler_has_active_biologist_tt
		theiler_has_active_biologist_no_tooltip_trigger = yes
	}
}

theiler_has_biologist_trigger = {
	custom_tooltip = {
		text = theiler_has_biologist_tt
		has_variable = theiler_biologist_var
	}
}

theiler_has_active_biologist_no_tooltip_trigger = {
	has_variable = theiler_biologist_var
	var:theiler_biologist_var ?= {
		is_busy = no
	}
}

theiler_has_microbiologist_trigger = {
	custom_tooltip = {
		text = theiler_has_microbiologist_tt
		has_variable = theiler_biologist_var
		var:theiler_biologist_var ?= {
			has_trait = theiler_microbiologist
		}
	}
}

### Triggers to determine if a pathogen of a disease is possible ###

theiler_can_find_pathogen_trigger = {
	OR = {
		AND = {
			panum_has_cholera_description_trigger = yes
			theiler_has_cholera_pathogen_trigger = no
			theiler_has_achromatic_microscope_trigger = yes
			var:theiler_microorganisms_var >= 1000
		}
		AND = {
			panum_has_tuberculosis_description_trigger = yes
			theiler_has_tuberculosis_pathogen_trigger = no
			theiler_has_achromatic_microscope_trigger = yes
			var:theiler_microorganisms_var >= 2000
		}
		AND = {
			panum_has_smallpox_description_trigger = yes
			theiler_has_smallpox_pathogen_trigger = no
			theiler_has_electron_microscope_trigger = yes
			var:theiler_microorganisms_var >= 35000
		}
		AND = {
			panum_has_typhus_description_trigger = yes
			theiler_has_typhus_pathogen_trigger = no
			theiler_has_achromatic_microscope_trigger = yes
			var:theiler_microorganisms_var >= 20000
		}
		AND = {
			panum_has_yellow_fever_description_trigger = yes
			theiler_has_yellow_fever_pathogen_trigger = no
			theiler_has_electron_microscope_trigger = yes
			var:theiler_microorganisms_var >= 50000
		}
		AND = {
			panum_has_measles_description_trigger = yes
			theiler_has_measles_pathogen_trigger = no
			theiler_has_electron_microscope_trigger = yes
			var:theiler_microorganisms_var >= 100000
		}
		AND = {
			panum_has_scarlet_description_trigger = yes
			theiler_has_scarlet_pathogen_trigger = no
			theiler_has_achromatic_microscope_trigger = yes
			var:theiler_microorganisms_var >= 5000
		}
		AND = {
			panum_has_plague_description_trigger = yes
			theiler_has_plague_pathogen_trigger = no
			theiler_has_achromatic_microscope_trigger = yes
			var:theiler_microorganisms_var >= 10000
		}
		AND = {
			panum_has_diphteria_description_trigger = yes
			theiler_has_diphteria_pathogen_trigger = no
			theiler_has_achromatic_microscope_trigger = yes
			var:theiler_microorganisms_var >= 3000
		}
		AND = {
			panum_has_polio_description_trigger = yes
			theiler_has_polio_pathogen_trigger = no
			theiler_has_electron_microscope_trigger = yes
			var:theiler_microorganisms_var >= 75000
		}
	}
}

### Triggers to determine if a vaccine of a disease is possible ###

theiler_can_create_vaccine_trigger = {
	OR = {
		AND = {
			theiler_has_cholera_pathogen_trigger = yes
			theiler_has_cholera_vaccine_trigger = no
			#theiler_cholera_research_tech_requirement_trigger = yes
		}
		AND = {
			theiler_has_tuberculosis_pathogen_trigger = yes
			theiler_has_tuberculosis_vaccine_trigger = no
			#theiler_tuberculosis_research_tech_requirement_trigger = yes
		}
		AND = {
			theiler_has_smallpox_pathogen_trigger = yes
			theiler_has_smallpox_vaccine_trigger = no
			#theiler_smallpox_research_tech_requirement_trigger = yes
		}
		AND = {
			theiler_has_typhus_pathogen_trigger = yes
			theiler_has_typhus_vaccine_trigger = no
			theiler_typhus_research_tech_requirement_trigger = yes
		}
		AND = {
			theiler_has_yellow_fever_pathogen_trigger = yes
			theiler_has_yellow_fever_vaccine_trigger = no
			theiler_yellow_fever_research_tech_requirement_trigger = yes
		}
		AND = {
			theiler_has_measles_pathogen_trigger = yes
			theiler_has_measles_vaccine_trigger = no
			theiler_measles_research_tech_requirement_trigger = yes
		}
		#AND = { #There is no effective scarlet vaccine today!
		#	theiler_has_scarlet_pathogen_trigger = yes
		#	theiler_has_scarlet_vaccine_trigger = no
		#	theiler_scarlet_research_tech_requirement_trigger = yes
		#}
		AND = {
			theiler_has_plague_pathogen_trigger = yes
			theiler_has_plague_vaccine_trigger = no
			#theiler_plague_research_tech_requirement_trigger = yes
		}
		AND = {
			theiler_has_diphteria_pathogen_trigger = yes
			theiler_has_diphteria_vaccine_trigger = no
			theiler_diphteria_research_tech_requirement_trigger = yes
		}
		AND = {
			theiler_has_polio_pathogen_trigger = yes
			theiler_has_polio_vaccine_trigger = no
			theiler_polio_research_tech_requirement_trigger = yes
		}
	}
}
