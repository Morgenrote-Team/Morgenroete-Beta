###Geologist Triggers###

agassiz_has_active_geologist = {
	custom_tooltip = {
		text = dubois_has_active_geologist_tt
		agassiz_has_active_geologist_no_tooltip_trigger = yes
	}
}
agassiz_has_active_geologist_no_tooltip_trigger = {
	has_variable = agassiz_geologist_var
	var:agassiz_geologist_var ?= {
		is_busy = no
	}
}

agassiz_has_experienced_geologist_trigger = {
	custom_tooltip = {
		text = agassiz_has_experienced_geologist_trigger_tt
		agassiz_has_experienced_geologist_no_tooltip_trigger = yes
	}
}
agassiz_has_experienced_geologist_no_tooltip_trigger = {
	var:agassiz_geologist_var ?= {
		OR = {
			has_trait = agassiz_geologist_experienced
			has_trait = agassiz_geologist_famous
		}
	}
}

agassiz_has_famous_geologist_trigger = {
	custom_tooltip = {
		text = agassiz_has_famous_geologist_trigger_tt
		agassiz_has_famous_geologist_no_tooltip_trigger = yes
	}
}
agassiz_has_famous_geologist_no_tooltip_trigger = {
	var:agassiz_geologist_var ?= {
		has_trait = agassiz_geologist_famous
	}
}

agassiz_geologist_has_specialization_trigger = {
	has_variable = agassiz_geologist_var
	var:agassiz_geologist_var ?= {
		OR = {
			has_trait = agassiz_mineralogist
			has_trait = agassiz_geophysicist
			has_trait = agassiz_prospector
			has_trait = agassiz_oceanographer
			has_trait = agassiz_volcanologist
		}
	}
}

###Age of Earth Triggers###

agassiz_can_calculate_new_age_of_earth_trigger = {
	trigger_if = {
		limit = { has_global_variable = agassiz_global_age_of_earth_global_var }
		global_var:agassiz_global_age_of_earth_global_var < agassiz_country_age_of_earth_calculation_value
	}
	trigger_else = {
		always = no
	}
}

		has_journal_entry = je_agassiz_find_lead_project
		has_journal_entry = je_agassiz_find_sulfur_project
		has_journal_entry = je_agassiz_find_gold_project
		has_journal_entry = je_agassiz_find_uranium_project
		has_journal_entry = je_agassiz_find_oil_project
		has_journal_entry = je_agassiz_mid_atlantic_ridge
	}
}

### Triggers for Geology Automation ###

#Does this country use Geology Automation?
agassiz_is_geology_automated_trigger = {
	OR = {
		is_ai = yes
		has_variable = agassiz_geology_automation_var
	}
}

#Valid buildings for Mine improvement project
agassiz_mine_valid_for_improvement_trigger = {
	level > 0
	is_building_group = bg_mining
	NOT = {
		has_modifier = agassiz_building_production_mult_modifier
	}
}

#Valid buildings for Oil Rig improvement project
agassiz_oil_rig_valid_for_improvement_trigger = {
	level > 0
	is_building_type = building_oil_rig
	NOT = {
		has_modifier = agassiz_building_production_2_mult_modifier
	}
}
