### Historical Work Trigger for Historical Composers ###
# FLAG = Flag of the Masterpiece
elgar_has_composer_historical_masterpiece_country_trigger = {
	var:elgar_composer_var ?= {
		NOR = {
			var:elgar_masterpiece_1_var ?= flag:$FLAG$
			var:elgar_masterpiece_2_var ?= flag:$FLAG$
			var:elgar_masterpiece_3_var ?= flag:$FLAG$
		}
	}
}

### Culture Triggers ###

#Brazillian
elgar_country_or_composer_brazillian_character_trigger = {
	OR = {
		culture = {
			has_discrimination_trait = language_lusophone
			has_discrimination_trait = heritage_latin_american_settler
		}
		owner ?= {
			OR = {
				any_primary_culture = {
					has_discrimination_trait = language_lusophone
					has_discrimination_trait = heritage_latin_american_settler
				}
			}
		}
	}
}
