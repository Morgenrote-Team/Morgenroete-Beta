	OR = {
		has_state_trait = ito_state_trait_medicinal_plant_aloe_vera
		has_state_trait = ito_state_trait_medicinal_plant_atropa_belladonna
		has_state_trait = ito_state_trait_medicinal_plant_carapichae_ipecacuanha
		has_state_trait = ito_state_trait_medicinal_plant_cinchona_pubescens
		has_state_trait = ito_state_trait_medicinal_plant_digitalis_purpurea
		has_state_trait = ito_state_trait_medicinal_plant_ephedra_sinica
		has_state_trait = ito_state_trait_medicinal_plant_pilocarpus_pennatifolius
		has_state_trait = ito_state_trait_medicinal_plant_strophanthus_gratus
		has_state_trait = ito_state_trait_medicinal_plant_strychnos_nuxvomica
		has_state_trait = ito_state_trait_medicinal_plant_thymus_vulgaris
	}
}

ito_country_has_medicinal_plants_no_tooltip_trigger = {
	any_scope_state = {
		ito_state_has_medicinal_plants_no_tooltip_trigger = yes
	}
}

### Botanical Expedition Possible ###
ito_country_is_potential_botanical_expedition_target_no_tooltip_trigger = {
	any_scope_state = {
		ito_state_has_medicinal_plants_no_tooltip_trigger = yes

		ito_state_has_no_malaria_or_malaria_prevention_researched_trigger = yes
	}
}

ito_state_has_no_malaria_or_malaria_prevention_researched_trigger = {
	trigger_if = {
		limit = {
			NOT = {
				root ?= {
					has_technology_researched = malaria_prevention
				}
			}
		}
		NOR = {
			has_state_trait = state_trait_malaria
			has_state_trait = state_trait_severe_malaria
		}
	}
}

	OR = {

			has_state_trait = ito_state_trait_medicinal_plant_aloe_vera
		}

			has_diplomatic_pact = {
				who = ROOT
				type = botanical_license
			}
			any_scope_state = {
				has_state_trait = ito_state_trait_medicinal_plant_aloe_vera
			}
		}
	}
}

### Triggers for Improve Forestry Project ###
ito_country_has_building_valid_for_forestry_project_trigger = {
	custom_tooltip = {
		text = MR_NATURALIST_IMPROVE_FORESTRY_TRIGGER_TT
		any_scope_building = {
			ito_building_valid_for_forestry_project_no_tooltip_trigger = yes
		}
	}
}

ito_building_valid_for_forestry_project_no_tooltip_trigger = {
	level > 0
	is_building_type = building_logging_camp
	NAND = {
		has_modifier = dubois_building_throughput_add_modifier
		has_modifier = dubois_building_production_mult_modifier
		has_modifier = dubois_building_goods_input_mult_modifier
	}
}
