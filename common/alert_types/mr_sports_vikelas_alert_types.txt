	script_context = player_country
	texture = "gfx/interface/icons/alert_icons/vikelas_olympics_alert.dds"

	valid = {
		NOT = {
			has_global_variable = vikelas_ioc_founded_var
		}

		OR = {
			tesla_has_engineer = yes

			dubois_has_naturalist = yes
			dubois_has_paleontologist = yes
			lepsius_has_archaeologist = yes
			andersson_has_anthropologist = yes
		}
		is_at_war = no
	}

#	open_panel = tech_tree
	type = alert	# Time-limited option
	alert_group = morgenroete
}
