tesla_submarine_cable_possible_alert = {
	script_context = player_country
	texture = "gfx/interface/icons/alert_icons/gaudi_engineering_alert.dds"

	valid = {
		tesla_has_active_engineer_trigger = yes
		any_company = {
			is_company_type = company_type:mr_basic_telegraphy
		}
		OR = {
			AND = {
				NOT = {
					has_global_variable = tesla_transatlantic_cable_country_global_var
				}
				NOT = {
					has_global_variable = gaudi_transatlantic_radio_var
				}
				NOT = {
					has_modifier = tesla_transatlantic_cable_expedition_failed_modifier
				}
				investment_pool >= 5000000
				mr_gui_has_screw_frigate_tech_trigger = yes
			}
			AND = {
				NOT = {
					has_global_variable = tesla_suez_india_cable_country_global_var
				}
				has_global_variable = tesla_transatlantic_cable_country_global_var
				s:STATE_SINAI = {
					has_building = building_suez_canal
				}
				NOT = {
					has_modifier = tesla_suez_india_cable_expedition_failed_modifier
				}
				investment_pool >= 10000000
				mr_gui_has_ironclad_tech_tech_trigger = yes
			}
			AND = {
				NOT = {
					has_global_variable = tesla_transpacific_cable_country_global_var
				}
				has_global_variable = tesla_suez_india_cable_country_global_var
				NOT = {
					has_modifier = tesla_transpacific_cable_expedition_failed_modifier
				}
				investment_pool >= 20000000
				mr_gui_has_destroyer_tech_trigger = yes
			}
		}
	}

	type = important_action
	alert_group = morgenroete
}
