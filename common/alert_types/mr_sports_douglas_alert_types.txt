# texture = <path>					# the path is relative to the "game" folder
# valid = <trigger>					# an invalid alert will not be shown
# open_panel = <panel_name>[|<tab>]	# if specified, this panel will open when the alert is clicked. Cannot specify both open_panel and open_popup
									# an optional tab name can be specified after a | character
# open_popup = <popup_name>			# if specified, this popup will open when the alert is clicked. Cannot specify both open_panel and open_popup
# type = alert | important_action | angry_important_action	# if unspecified or "alert", this will be a standard alert. Otherwise, it will be shown inside the important actions alert (angry_important_action alerts are sorted first)
# alert_group = <alert_group_name>	# if specified, multiple alerts in the same group will be grouped in the important actions alert (if they are "important_action"). "important_actions with unspecified alert_group will never be grouped together

douglas_mountaineering_organization_possible_alert = {
	script_context = player_country
	texture = "gfx/interface/icons/alert_icons/douglas_mountaineering_alert.dds"

	valid = {
		OR = {
			AND = {
				douglas_has_active_mountaineer = yes
				douglas_has_training_mountaineer = no
				douglas_has_safe_mountain_trigger = yes
			}
			AND = {
				NOT = {
					has_modifier = douglas_mountaineering_regiment_modifier
				}
				mr_gui_has_defense_in_depth_tech_trigger = yes
				douglas_has_mountaineer = yes
				douglas_has_mountaineering_club_trigger = yes
				douglas_has_mountain_trigger = yes
				any_scope_character = {
					has_role_of_type = general
					is_busy = no
					OR = {
						has_trait = douglas_mountaineer
						has_trait = douglas_mountaineer_veteran
						has_trait = explorer
						has_trait = mountain_commander
					}
				}
				var:douglas_alpine_heritage_var >= 25
			}
			AND = {
				NOT = {
					has_modifier = douglas_mountaineering_equipment_level_5_modifier
				}
				douglas_mountaineer_can_be_available = yes	# This is worth cancelling training for.
				var:douglas_alpine_heritage_var >= 25
				#Level 1 Trigger: Geology
				mr_gui_has_geology_tech_trigger = yes
				#Level 2 Trigger: Logistics (Rope Group)
				trigger_if = {
					limit = {
						has_modifier = douglas_mountaineering_equipment_level_1_modifier
					}
					has_technology_researched = logistics
				}
				#Level 3 Trigger: Modern Nursing (Severe Injuries)
				trigger_if = {
					limit = {
						has_modifier = douglas_mountaineering_equipment_level_2_modifier
					}
					has_technology_researched = modern_nursing
				}
				#Level 4 Trigger: Seismology
				trigger_if = {
					limit = {
						has_modifier = douglas_mountaineering_equipment_level_3_modifier
					}
					has_technology_researched = agassiz_seismology_tech
				}
				#Level 5 Trigger: Chemical Warfare (Oxygen)
				trigger_if = {
					limit = {
						has_modifier = douglas_mountaineering_equipment_level_4_modifier
					}
					has_technology_researched = chemical_warfare
				}
			}
		}
	}

#	open_panel = tech_tree
	type = alert	# Restrictive conditions + High importance
	alert_group = morgenroete
}

douglas_mountaineering_club_possible_alert = {
	script_context = player_country
	texture = "gfx/interface/icons/alert_icons/douglas_mountaineering_alert.dds"

	valid = {
		OR = {
			AND = {
				NOT = {
					has_variable = douglas_mountaineering_club_var
				}
				mr_gui_has_geology_tech_trigger = yes
				any_scope_character = {
					has_role_of_type = general
					is_busy = no
					OR = {
						has_trait = douglas_mountaineer
						has_trait = douglas_mountaineer_veteran
						has_trait = explorer
						has_trait = mountain_commander
						has_trait = surveyor
					}
				}
			}
			AND = {
				NOT = {
					has_global_variable = douglas_alpine_club_founded_global_var
				}
				has_variable = douglas_mountaineering_club_var
				douglas_has_mountaineer = yes
				mr_gui_has_nationalism_tech_trigger = yes
				any_country = {
					has_variable = douglas_mountaineering_club_var
					count >= 6
				}
				var:douglas_mountaineer_var = {
					var:douglas_mountaineer_experience_var >= 50
					has_variable = douglas_num_mountain_ascents_var
					var:douglas_num_mountain_ascents_var >= 2
					has_variable = douglas_summits_conquered_var
					var:douglas_summits_conquered_var >= 2
				}
			}
			AND = {
				OR = {
					has_variable = douglas_alpine_club_founding_country_var
					has_variable = douglas_alpine_club_board_member_var
				}
				douglas_mountaineer_can_be_available = yes	# This is worth cancelling training for.
				any_country = {
					has_variable = douglas_alpine_club_founding_country_var
					NOT = {
						has_modifier = douglas_mountaineering_international_expedition_modifier
					}
					OR = {
						has_modifier = douglas_mountaineering_equipment_level_3_modifier
						has_modifier = douglas_mountaineering_equipment_level_4_modifier
						has_modifier = douglas_mountaineering_equipment_level_5_modifier
					}
					OR = {
						douglas_has_unconquered_level_2_summit_trigger = yes
						douglas_has_unconquered_level_3_summit_trigger = yes
						douglas_has_unconquered_level_4_summit_trigger = yes
					}
				}
				NOT = {
					has_modifier = douglas_mountaineering_international_expedition_setback_modifier
				}
			}
			AND = {
				has_variable = douglas_alpine_club_board_member_var
				douglas_mountaineer_can_be_available = yes	# This is worth cancelling training for.
				has_journal_entry = je_douglas_international_mountaineering_expedition_preparation
			}
		}
	}

#	open_panel = tech_tree
	type = alert	# Restrictive conditions + High importance
	alert_group = morgenroete
}
