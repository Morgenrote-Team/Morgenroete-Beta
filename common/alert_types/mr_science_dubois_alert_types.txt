dubois_paleontologist_unused_alert = {
	script_context = player_country
	texture = "gfx/interface/icons/alert_icons/dubois_dinosaur_bones_alert.dds"

	valid = {
		dubois_has_active_paleontologist_no_tooltip_trigger = yes
		OR = {
			#Paleontological Expedition
			AND = {
				NOR = {
					has_journal_entry = je_dubois_paleontological_expedition
					has_variable = dubois_paleontological_expedition_destination_selection_var
					has_modifier = dubois_had_expedition_modifier
				}
				OR = {
					dubois_has_formation = yes
					any_country = {
						has_diplomatic_pact = {
							who = ROOT
							type = paleontological_license
						}				
						dubois_has_formation = yes
					}
				}
			}
			#Mount Skeleton
			AND = {
				NOT = { has_variable = mr_mount_dinosaur_skeleton_button_var }
				dubois_paleontology_mounting_skeletons_possible_trigger = yes
			}
			#Remount Skeleton
			AND = {
				NOT = { has_variable = mr_remount_dinosaur_skeleton_button_var }
				dubois_has_imperfect_dinosaur_skeleton_no_tooltip_trigger = yes
			}
			#Dioramas
			dubois_paleontology_automation_cretaceous_scenery_possible_trigger = yes
			dubois_paleontology_automation_jurassic_scenery_possible_trigger = yes
			dubois_paleontology_automation_triassic_scenery_possible_trigger = yes
		}
	}

#	open_panel = tech_tree
	type = important_action
	alert_group = morgenroete
}

dubois_paleontological_team_possible_alert = {
	script_context = player_country
	texture = "gfx/interface/icons/alert_icons/dubois_dinosaur_bones_alert.dds"

	valid = {
		has_technology_researched = dubois_paleontology_tech
		OR = {
			AND = {
				OR = {
					NOT = { has_variable = dubois_total_paleontological_teams_var }
					var:dubois_total_paleontological_teams_var < 1
				}
				country_has_building_type_levels = {  
					target = bt:building_university  
					value  >= 3
				}
			}
			AND = {
				has_variable = dubois_total_paleontological_teams_var
				var:dubois_total_paleontological_teams_var = 1
				academics_num_natural_universities >= 10
			}
		}
	}

#	open_panel = tech_tree
	type = important_action
	alert_group = morgenroete
}

dubois_paleontological_collection_alert = {
	script_context = player_country
	texture = "gfx/interface/icons/alert_icons/dubois_dinosaur_bones_alert.dds"

	valid = {
		has_technology_researched = dubois_paleontology_tech
		NOT = {
			any_scope_state = {
				is_production_method_active = {
					building_type = building_dubois_zoo
					production_method = pm_paleontological_collection
				}
			}
		}
		var:dubois_fossil >= 100
		dubois_has_paleontologist_no_tooltip_trigger = yes
	}

#	open_panel = tech_tree
	type = important_action
	alert_group = morgenroete
}

dubois_extinction_congress_possible_alert = {
	script_context = player_country
	texture = "gfx/interface/icons/alert_icons/dubois_dinosaur_bones_alert.dds"

	valid = {
		has_global_variable = dubois_hippocampus_congress_hosted_var
		NOR = {
			has_global_variable = dubois_extinction_congress_hosted_var
			any_country = {
				has_modifier = andersson_congress_host_modifier
			}
		}
		has_technology_researched = dubois_paleontology_tech
		dubois_has_paleontologist_no_tooltip_trigger = yes
		global_var:dubois_global_fossil >= 550
		country_has_building_type_levels = {  
			target = bt:building_university  
			value  >= 20
		}
		any_scope_building = {
			is_building_type = building_dubois_zoo
		}
		any_country = {
			dubois_has_paleontologist_no_tooltip_trigger = yes
			count >= 15
		}
		dubois_has_dinosaur_skeleton_no_tooltip_trigger = yes
		any_country = {
			dubois_has_dinosaur_skeleton_no_tooltip_trigger = yes
			count >= 5
		}
	}

#	open_panel = tech_tree
	type = alert	# Time-limited option
	alert_group = morgenroete
}
