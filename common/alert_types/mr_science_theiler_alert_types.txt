# texture = <path>					# the path is relative to the "game" folder
# valid = <trigger>					# an invalid alert will not be shown
# open_panel = <panel_name>[|<tab>]	# if specified, this panel will open when the alert is clicked. Cannot specify both open_panel and open_popup
									# an optional tab name can be specified after a | character
# open_popup = <popup_name>			# if specified, this popup will open when the alert is clicked. Cannot specify both open_panel and open_popup
# type = alert | important_action | angry_important_action	# if unspecified or "alert", this will be a standard alert. Otherwise, it will be shown inside the important actions alert (angry_important_action alerts are sorted first)
# alert_group = <alert_group_name>	# if specified, multiple alerts in the same group will be grouped in the important actions alert (if they are "important_action"). "important_actions with unspecified alert_group will never be grouped together

theiler_biologist_unused_alert = {
	script_context = player_country
	texture = "gfx/interface/icons/invention_icons/theiler_microbiology_tech.dds"

	valid = {
		theiler_has_active_biologist_no_tooltip_trigger = yes
		OR = {
			#Agricultural Improvement
			AND = {
				any_scope_building = {
					theiler_building_valid_for_agriculture_project_no_tooltip_trigger = yes
				}
				NOR = {
					has_journal_entry = je_theiler_improve_agriculture_project
					has_variable = theiler_improve_agriculture_project_building_chooser_var
				}
			}
			#SIT (New World Screwworm Fly)
			AND = {
				NOT = {
					has_variable = theiler_screwworm_fly_prevention_var
				}
				has_technology_researched = theiler_cell_culture_tech
				has_technology_researched = verrier_nuclear_physics_tech
				has_technology_researched = curtiss_modern_aviation_tech
				any_scope_state = {
					is_in_geographic_region = theiler_geographic_region_screwworm_fly
				}
			}
			#Lead Arsenate (Potato Beetle/Codling Moth)
			AND = {
				NOR = {
					has_variable = theiler_codling_moth_prevention_var
					has_variable = theiler_potato_beetle_prevention_var
				}
				OR = {
					any_scope_state = {
						is_in_geographic_region = theiler_geographic_region_codling_moth
					}
					AND = {
						has_global_variable = theiler_potato_beetle_active_global_var
						any_scope_state = {
							is_in_geographic_region = theiler_geographic_region_potato_beetle
						}
					}
				}
				has_technology_researched = theiler_early_pesticides_tech
				mendelejew_has_chemist = yes
			}
			#Sodium Arsenite (Locust Swarm)
			AND = {
				theiler_has_locust_swarm_prevention_trigger = no
				any_scope_state = {
					is_vulnerable_to_locust_swarms = yes
				}
				mr_gui_has_modern_pesticides_tech_trigger = yes
				mendelejew_has_chemist = yes
			}
			#Calcium Arsenate (Boll Weevil)
			AND = {
				theiler_has_boll_weevil_prevention_trigger = no
				any_scope_state = {
					is_in_geographic_region = theiler_geographic_region_boll_weevil
				}
				mr_gui_has_modern_pesticides_tech_trigger = yes
				mr_gui_has_military_aviation_tech_trigger = yes
				mendelejew_has_chemist = yes
			}
			#Nicotine Sulfate (Tea Mosquito)
			AND = {
				theiler_has_tea_mosquito_prevention_trigger = no
				any_scope_state = {
					is_in_geographic_region = theiler_geographic_region_tea_mosquito
				}
				mr_gui_has_modern_pesticides_tech_trigger = yes
				mr_gui_has_malaria_prevention_tech_trigger = yes
				mendelejew_has_chemist = yes
			}
			#Grafting (Phylloxera)
			AND = {
				theiler_has_grape_phylloxera_prevention_trigger = no
				any_scope_state = {
					is_in_geographic_region = theiler_geographic_region_grape_phylloxera
				}
				mr_gui_has_genetics_tech_trigger = yes
			}
			#Create Vaccine
			AND = {
				NOT = { has_variable = theiler_biologist_action_active_var }
				has_technology_researched = theiler_microbiology_tech
				theiler_can_create_vaccine_trigger = yes
				theiler_has_x_microorganisms_no_tooltip_trigger = { AMOUNT = 10000 }
			}
		}
	}

#	open_panel = tech_tree
	type = alert	# High importance
	alert_group = morgenroete
}

	script_context = player_country
	texture = "gfx/interface/icons/invention_icons/theiler_microbiology_tech.dds"

	valid = {
		var:theiler_biologist_var ?= {
			NOT = {
				remaining_career_length = {
					role = character_role_theiler_biologist
					years = 5
				}
			}
		}
	}

	alert_group = morgenroete
}
