# script_context = <script context>	# see later for list of valid script contexts
# texture = <path>					# the path is relative to the "game" folder
# valid = <trigger>					# an invalid alert will not be shown
# open_panel = <panel_name>[|<tab>]	# if specified, this panel will open when the alert is clicked. Cannot specify both open_panel and open_popup
									# an optional tab name can be specified after a | character
# open_popup = <popup_name>			# if specified, this popup will open when the alert is clicked. Cannot specify both open_panel and open_popup
# type = alert | important_action | angry_important_action	# if unspecified or "alert", this will be a standard alert. Otherwise, it will be shown inside the important actions alert (angry_important_action alerts are sorted first)
# alert_group = <alert_group_name>	# if specified, multiple alerts in the same group will be grouped in the important actions alert (if they are "important_action"). "important_actions with unspecified alert_group will never be grouped together
#

panum_unused_physician_alert = {
	script_context = player_country
	texture = "gfx/interface/icons/alert_icons/panum_medicine_alert.dds"

	valid = {
		panum_has_active_physician_no_tooltip_trigger = yes
		OR = {
			AND = {
				NOT = {
					has_law = law_type:law_no_health_system
				}
				institution_investment_level = {
					institution	= institution_health_system
					value >= 1
				}
				OR = {
					any_scope_state = {
						any_scope_building = {
							is_building_type = building_government_administration
							level >= 5
							has_active_production_method = pm_panum_no_clinic
						}
					}
					any_scope_state = {
						any_scope_building = {
							is_building_type = building_government_administration
							level >= 20
							has_active_production_method = pm_panum_clinic
						}
					}
				}
			}
			AND = {
				trigger_if = {
					limit = { has_variable = panum_physician_advanced_medical_support_var }
					NOT = { var:panum_physician_advanced_medical_support_var > 2 }
				}
				trigger_else = {
					always = yes
				}
				var:panum_physician_var ?= {
					OR = {
						has_trait = panum_military_physician
						has_trait = panum_red_cross
					}
				}
			}
			AND = {
				NOR = {
					has_variable = panum_physician_action_active_var
					has_modifier = panum_medical_techniques_level_7_modifier
					has_law = law_type:law_no_health_system
				}
				#Level 1 Trigger
				mr_gui_has_vaccination_tech_trigger = yes
				mr_gui_has_central_archives_tech_trigger = yes
				#Level 2 Trigger
				trigger_if = {
					limit = { has_modifier = panum_medical_techniques_level_1_modifier }
					mr_gui_has_pharmaceuticals_tech_trigger = yes
					mr_gui_has_chemistry_tech_trigger = yes
				}
				#Level 3 Trigger
				trigger_if = {
					limit = { has_modifier = panum_medical_techniques_level_2_modifier }
					mr_gui_has_quinine_tech_trigger = yes
					mr_gui_has_mechanical_tools_tech_trigger = yes
				}
				#Level 4 Trigger
				trigger_if = {
					limit = { has_modifier = panum_medical_techniques_level_3_modifier }
					custom_tooltip = {
						text = MR_PHYSICIAN_HAS_GERM_THEORY_TRIGGER
						panum_has_miasma_theory_trigger = no
					}
					mr_gui_has_microbiology_tech_trigger = yes
				}
				#Level 5 Trigger
				trigger_if = {
					limit = { has_modifier = panum_medical_techniques_level_4_modifier }
					mr_gui_has_malaria_prevention_tech_trigger = yes
					mr_gui_has_central_planning_tech_trigger = yes
				}
				#Level 6 Trigger
				trigger_if = {
					limit = { has_modifier = panum_medical_techniques_level_5_modifier }
					mr_gui_has_antitoxins_tech_trigger = yes
					mr_gui_has_combustion_engine_tech_trigger = yes
				}
				#Level 7 Trigger
				trigger_if = {
					limit = { has_modifier = panum_medical_techniques_level_6_modifier }
					mr_gui_has_antibiotics_tech_trigger = yes
					mr_gui_has_military_aviation_tech_trigger = yes
				}
			}
		}
	}

#	open_panel = tech_tree
	type = important_action
	alert_group = morgenroete
}
