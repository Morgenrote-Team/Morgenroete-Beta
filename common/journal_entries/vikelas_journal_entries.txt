			2 = vikelas.141 #Vikelas 141: No Sailboats!
			2 = vikelas.142 #Vikelas 142: Rigged Football Match?
			2 = vikelas.143 #Vikelas 143: The New Legend of Marathon (organizer)
			2 = vikelas.144 #Vikelas 144: Well organized games (organizer)
			2 = vikelas.145 #Vikelas 145: A celebrated host (organizer)

			#Participant Events again
			
			1 = vikelas.161 #Vikelas 161: Hitchhiker's Guide to the Marathon (participants)
			1 = vikelas.162 #Vikelas 162: A wild Dog appeared (participants)
			1 = vikelas.163 #Vikelas 163: Bull's Eye on the wrong Target (participants)
		}
	}
	
	immediate = {
		random_country = {
			limit = {
				any_scope_state = {
					has_modifier = vikelas_olympics_hosting_modifier
				}
			}
			save_scope_as = vikelas_olympic_country
		}
	}

    fail = {
    }

	on_fail = {
	}

	complete = {
	}

	on_complete = {
	}

	invalid = {
	}

	current_value = {
	}

	goal_add_value = {
	}

	timeout = 30

	on_timeout = {
		if = {
			limit = {
				NOT = {
					has_modifier = vikelas_our_games_our_rules_modifier
				}
			}
			add_modifier = {
				name = vikelas_olympics_hosted_modifier
				years = 20
			}
		}
		random_scope_state = {
			limit = {
				has_modifier = vikelas_olympics_hosting_modifier
			}
			remove_modifier = vikelas_olympics_hosting_modifier
		}
		trigger_event = {
			id = vikelas.108
			popup = yes
		}
		hidden_effect = {
			if = {
				limit = {
					has_modifier = vikelas_our_games_our_rules_modifier
				}
				trigger_event = { #This will ensure that the Games will be held in Greece every four Years
					id = vikelas.109
					days = 1430
				}
			}
			else = {
				random_country = {
					limit = {
						has_variable = vikelas_ioc_founding_country_var
					}
					trigger_event = { #This will ensure that the Board will be re-elected every four years
						id = vikelas.111
						days = 60
					}
				}
			}
		}
	}

	weight = 100
}

je_vikelas_mechanic_engine = {
	icon = "gfx/interface/icons/event_icons/event_autorace.dds"

	group = je_group_vikelas_sports

	scripted_button = vikelas_mechanic_boost_button

	scripted_progress_bar = vikelas_mechanic_progress_bar

	on_monthly_pulse = {
		events = {
			vikelas.584 #Vikelas 584: The V Engine
		}
	}

	immediate = {
		scope:journal_entry = {
			add_modifier = {
				name = vikelas_mechanic_improvement_cost_modifier
				multiplier = money_amount_multiplier_small
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = vikelas_mechanic_progress_bar_at_full_tt
			je:je_vikelas_mechanic_engine = {
				"scripted_bar_progress(vikelas_mechanic_progress_bar)" >= 36
			}
		}
	}

	on_complete = {
		if = {
			limit = {
				has_modifier = vikelas_mechanic_boost_modifier
			}
			remove_modifier = vikelas_mechanic_boost_modifier
		}
		vikelas_mechanic_new_engine_effect = yes
		trigger_event = {
			id = vikelas.581
			popup = yes
		}
	}

	weight = 30

	should_be_pinned_by_default = yes
}

		random_events = {
			18 = vikelas.532 #Vikelas 532: Normal progress
			1 = vikelas.533 #Vikelas 533: Tire damage
			1 = vikelas.534 #Vikelas 534: The Railway Crossing
			1 = vikelas.535 #Vikelas 535: Overtaking Maneuver
			1 = vikelas.538 #Vikelas 538: Wolves & Revolvers
			1 = vikelas.539 #Vikelas 539: Racing car drives into spectators!
			1 = vikelas.540 #Vikelas 540: New Average Speed Record
			1 = vikelas.543 #Vikelas 543: A little help for a friend?
			1 = vikelas.544 #Vikelas 544: Engine problems
			1 = vikelas.545 #Vikelas 545: Fan Gathering
		}
	}

	immediate = {
		set_variable = {
			name = vikelas_endurance_race_progress_var
			value = 0
		}
	}

	complete = {
		scope:journal_entry = { 
			is_goal_complete = yes 
		}
	}

	on_complete = {
		trigger_event = { 
			id = vikelas.541
		}
	}

	current_value = {
		if = {
			limit = {
				scope:journal_entry = {
					goal_value = 0
				}
			}
			value = 0
		}
		else = {
			value = var:vikelas_endurance_race_progress_var #value or variable you use to meassure progress
		}
	}

	goal_add_value = {
		value = 8
	}

	invalid = {
	}

	progressbar = yes

	weight = 100

	should_be_pinned_by_default = yes
}
