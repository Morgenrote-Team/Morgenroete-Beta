#Achromatic Microscope
je_theiler_microscope_achromatic = {
	icon = "gfx/interface/icons/event_icons/event_biology.dds"

	group = je_group_theiler_biology

	is_shown_when_inactive = {
		var:theiler_microscope_var ?= flag:theiler_early_compound_microscope_flag
	}

	possible = {
		has_technology_researched = theiler_microbiology_tech
	}

	immediate = {
		set_variable = {
			name = theiler_build_microscope_achromatic_var
			value = 0
		}
	}

	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					tesla_has_engineer_no_tooltip_trigger = yes
					theiler_has_biologist_no_tooltip_trigger = yes
					var:theiler_build_microscope_achromatic_var < 12
				}
				change_variable = {
					name = theiler_build_microscope_achromatic_var
					add = 1
				}
			}
		}
	}

	complete = {
		any_scope_building = {
			is_building_type = building_glassworks
			level >= 3
		}
		tesla_has_engineer = yes
		theiler_has_biologist_trigger = yes
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		remove_variable = theiler_build_microscope_achromatic_var
		set_variable = {
			name = theiler_microscope_var
			value = flag:theiler_achromatic_microscope_flag
		}
		trigger_event = {
			id = theiler.11
		}
	}

	invalid = {
	}

	goal_add_value = {
		add = 12
	}

	current_value = {
		value = var:theiler_build_microscope_achromatic_var
	}

	progressbar = yes

	weight = 10
}

######Microorganism Research######

je_theiler_microorganism_research = {
	icon = "gfx/interface/icons/event_icons/event_biology.dds"

	group = je_group_theiler_biology

	immediate = {
		set_variable = {
			name = theiler_estimated_microorganisms_on_microorganisms_research_var
			value = theiler_microorganisms_research_monthly_microorganisms_value
		}
	}

	on_monthly_pulse = {
		effect = {
			#Adds microorganismss each month
			if = {
				limit = {
					theiler_has_active_biologist_no_tooltip_trigger = yes
				}

				theiler_add_x_microorganisms_effect = { AMOUNT = theiler_microorganisms_research_monthly_microorganisms_value }
				#Estimated yield
				set_variable = {
					name = theiler_estimated_microorganisms_on_microorganisms_research_var
					value = theiler_microorganisms_research_monthly_microorganisms_value
				}
			}
		}
	}

	invalid = {
		custom_tooltip = {
			text = theiler_biologist_dies_tt
			theiler_has_biologist = no
		}
	}

	on_invalid = {
		remove_variable = theiler_estimated_microorganisms_on_microorganisms_research_var
	}

	weight = 50
}

#Cholera
je_theiler_cholera_pathogen = {
	icon = "gfx/interface/icons/event_icons/event_biology.dds"

	group = je_group_theiler_biology

	scripted_progress_bar = theiler_easy_disease_project_progress_bar

	immediate = {
		scope:journal_entry = {
			add_modifier = {
				name = theiler_pathogen_research_cost_modifier
				multiplier = root.mr_local_project_cost_very_small_value
			}
		}
		if = { #Fallback if the variable does not exist yet
			limit = {
				NOT = {
					has_variable = theiler_cholera_pathogen_clues_var
				}
			}
			set_variable = {
				name = theiler_cholera_pathogen_clues_var
				value = 0
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				#If the Progress reaches the cost, a clue will be added
				limit = {
					je:je_theiler_cholera_pathogen = {
						"scripted_bar_progress(theiler_easy_disease_project_progress_bar)" >= 15
					}
				}
				#The cost will be subtracted from the current progress
				je:je_theiler_cholera_pathogen = {
					add_progress = {
						name = theiler_easy_disease_project_progress_bar
						value = -15
					}
				}
				change_variable = {
					name = theiler_cholera_pathogen_clues_var
					add = 1
				}
			}
		}
		random_events = {
			300 = 0

			2 = theiler.2101 #Theiler 2101: Microbiological Institute +2
			2 = theiler.2102 #Theiler 2102: Improved laboratory method +1
			2 = theiler.2103 #Theiler 2103: International Cooperation +1
			1 = theiler.2104 #Theiler 2104: Laboratory accident -2
			1 = theiler.2105 #Theiler 2105: Contamined Samples -1
			1 = theiler.2106 #Theiler 2106: Researchers' dispute
			5 = theiler.2107 #Theiler 2107: Research assistants +1
		}
	}

	complete = {
		custom_tooltip = {
			text = je_theiler_cholera_pathogen_success
			scope:journal_entry = {
				is_goal_complete = yes
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = theiler.101 #Cholera Pathogen
			popup = yes
		}
		theiler_remove_active_variables_effect = yes
	}

	invalid = {
		custom_tooltip = {
			text = theiler_biologist_dies_tt
			theiler_has_biologist = no
		}
	}

	on_invalid = {
		theiler_remove_active_variables_effect = yes
		var:theiler_biologist_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	current_value = {
		if = {
			limit = {
				scope:journal_entry = {
					goal_value = 0
				}
			}
			value = 0
		}
		else = {
			value = var:theiler_cholera_pathogen_clues_var
		}
	}

	goal_add_value = {
		value = 6
	}

	progressbar = yes

	weight = 30

	should_be_pinned_by_default = yes
}

###Improve Forestry###
je_theiler_improve_agriculture_project = {
	icon = "gfx/interface/icons/event_icons/event_botany.dds"

	scripted_progress_bar = theiler_biologist_improve_agriculture_project_progress_bar

	scripted_button = theiler_stop_biological_project_button

	group = je_group_dubois_nature

	on_monthly_pulse = {
		effect = {
			#Calculates the Progress added (see dubois_scripted_effects for calculation)
			dubois_naturalist_improvement_progress = yes
			#Adds the monthly progress to the variable
			change_variable = {
				name = dubois_naturalist_improvement_progress_var
				add = var:dubois_monthly_naturalist_improvement_progress_var
			}

			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(theiler_biologist_improve_agriculture_project_progress_bar)" >= 24
					}
				}
				scope:journal_entry = {
					add_progress = {
						name = theiler_biologist_improve_agriculture_project_progress_bar
						value = -24
					}
				}
				if = {
					limit = {
						scope:target = {
							NOT = { has_modifier = theiler_building_throughput_add_modifier }
						}
					}
					scope:target = {
						add_modifier = {
							name = theiler_building_throughput_add_modifier
						}
					}
					change_variable = {
						name = dubois_naturalist_improvement_progress_var
						subtract = ito_improvement_progress_cost
					}
				}
				else_if = {
					limit = {
						scope:target = {
							NOT = { has_modifier = theiler_building_production_mult_modifier }
						}
					}
					scope:target = {
						add_modifier = {
							name = theiler_building_production_mult_modifier
						}
					}
					change_variable = {
						name = dubois_naturalist_improvement_progress_var
						subtract = ito_improvement_progress_cost
					}
				}
				else = {
					scope:target = {
						add_modifier = {
							name = theiler_building_goods_input_mult_modifier
						}
					}

					remove_list_variable = {
						name = theiler_improve_agriculture_buildings_chosen_variable_list
						target = scope:target
					}
					if = {
						limit = { has_variable_list = theiler_improve_agriculture_buildings_chosen_variable_list }
						ordered_in_list = {
							variable = theiler_improve_agriculture_buildings_chosen_variable_list
							position = 0

							save_scope_as = target
						}
					}
					change_variable = {
						name = dubois_naturalist_improvement_progress_var
						subtract = ito_improvement_progress_cost
					}
				}
			}
		}
	}

	immediate = {
		set_variable = theiler_can_stop_active_improvement_project_var #This is used to check if the country can stop the project
		scope:journal_entry = {
			add_modifier = {
				name = theiler_improve_agriculture_cost_modifier
				multiplier = root.mr_local_project_cost_small_value
			}
		}
		ordered_in_list = {
			variable = theiler_improve_agriculture_buildings_chosen_variable_list
			position = 0

			save_scope_as = target
		}
	}

	complete = {
		custom_tooltip = {
			text = je_theiler_improve_agriculture_project_complete_tt
			NOT = {
				has_variable_list = theiler_improve_agriculture_buildings_chosen_variable_list
			}
		}
	}

	on_complete = {
		remove_variable = theiler_can_stop_active_improvement_project_var #Needs to be removed, when the JE is removed
		remove_variable = theiler_stop_biological_project_var
		show_as_tooltip = {
			scope:journal_entry = {
				remove_modifier = theiler_improve_agriculture_cost_modifier
			}
		}
		clear_variable_list = theiler_improve_agriculture_buildings_chosen_variable_list

		#Those should be removed in every Naturalist je
		remove_variable = dubois_monthly_naturalist_improvement_progress_var
		remove_variable = naturalist_experience_add_var

		var:dubois_naturalist_var ?= {
			set_character_busy_and_immortal = no
		}
		trigger_event = {
			id = dubois.52
			popup = yes
		}
	}

	invalid = {
		custom_tooltip = {
			text = theiler_stopped_biologist_project_tt
			OR = {
				has_variable = theiler_stop_biological_project_var
				theiler_has_biologist_no_tooltip_trigger = no
			}
		}
	}

	on_invalid = {
		remove_variable = dubois_can_stop_active_improvement_project_var #Needs to be removed, when the JE is removed
		remove_variable = theiler_stop_biological_project_var
		show_as_tooltip = {
			scope:journal_entry = {
				remove_modifier = theiler_improve_agriculture_cost_modifier
			}
		}
		clear_variable_list = theiler_improve_agriculture_buildings_chosen_variable_list

		#Those should be removed in every naturalist je
		remove_variable = dubois_monthly_naturalist_improvement_progress_var
		remove_variable = naturalist_experience_add_var

		var:dubois_naturalist_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	weight = 30

	should_be_pinned_by_default = no
}
