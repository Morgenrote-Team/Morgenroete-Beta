je_gaudi_civil_research_project = {
	icon = "gfx/interface/icons/event_icons/event_architecture.dds"

	group = je_group_tesla_engineers

	scripted_progress_bar = tesla_civil_engineering_research_project_progress_bar

	on_monthly_pulse = {
		effect = {
			#Calculates and executes potential effects
			if = {
				limit = { tesla_has_active_engineer_no_tooltip_trigger = yes }
				mr_journal_entry_project_monthly_effect = {
					VALUE_NAME = tesla_monthly_civil_engineering_progress_for_research_value
					PROGRESS_BAR_NAME = tesla_civil_engineering_research_project_progress_bar
					PROGRESS_COSTS = tesla_research_progress_cost
					PROGRESS_FILLED_EFFECT = tesla_civil_research_project_effect
				}
			}
		}
	}

	immediate = {
		#Sets the Estimated Civil Engineering Progress at the beginning
		scope:journal_entry = {
			set_variable = {
				name = mr_monthly_progress_var
				value = root.tesla_monthly_civil_engineering_progress_for_research_value #Calculated in script value
			}
			set_variable = {
				name = mr_je_progress_var
				value = 0
			}
		}
	}

	invalid = {
		tesla_has_civil_engineer_no_tooltip_trigger = no
	}

	weight = 30

	should_be_pinned_by_default = no
}

je_gaudi_road_project = {
	icon = "gfx/interface/icons/event_icons/paved_roads.dds"

	scripted_button = gaudi_stop_engineering_project_button
	scripted_button = gaudi_additional_support_for_civil_engineer_button
	scripted_button = gaudi_additional_support_for_civil_engineer_stop_button

	group = je_group_tesla_engineers

	on_monthly_pulse = {
		effect = {
			#Calculates the Progress added (see gaudi_scripted_effects for calculation)
			tesla_civil_engineering_progress = yes
			#Adds the monthly progress to the variable
			change_variable = {
				name = tesla_civil_engineering_progress_var
				add = var:gaudi_monthly_civil_engineering_progress_var
			}
			if = {
				#If the Progress reaches the cost, the modifier will be added to a state
				limit = {
					var:tesla_civil_engineering_progress_var > tesla_road_progress_cost
				}
				if = {
					limit = {
						any_scope_state = {
							is_incorporated = yes
							NOT = { has_modifier = gaudi_modern_roads_modifier }
						}
					}
					#The state with the least available infrastructure will be chosen
					ordered_scope_state = {
						position = 0 #We only want the first one here
						limit = {
							is_incorporated = yes
							NOT = { has_modifier = gaudi_modern_roads_modifier }
						}
						order_by = tesla_infrastructure_balance
						add_modifier = {
							name = gaudi_modern_roads_modifier
						}
					}
				}
				else = {
					#The state with the least available infrastructure will be chosen
					ordered_scope_state = {
						position = 0 #We only want the first one here
						limit = { NOT = { has_modifier = gaudi_modern_roads_modifier } }
						order_by = tesla_infrastructure_balance
						add_modifier = {
							name = gaudi_modern_roads_modifier
						}
					}
				}
				#The cost will be subtracted from the current progress
				change_variable = {
					name = tesla_civil_engineering_progress_var
					subtract = tesla_road_progress_cost
				}
			}
			set_variable = {
				name = tesla_num_states_var
				value = root.num_states
			}
			set_variable = {
				name = gaudi_num_states_roads_var
				value = tesla_states_with_roads_value
			}
		}
	}

	immediate = {
		scope:journal_entry = {
			add_modifier = {
				name = tesla_medium_engineering_project_cost_modifier
				multiplier = mr_local_project_cost_medium_value
			}
		}
		set_variable = gaudi_can_stop_active_engineering_project_var #This is used to check if the country has a project already
		#Sets the Civil Engineering Progress at the beginning
		set_variable = {
			name = tesla_civil_engineering_progress_var
			value = 0
		}
		#Sets a variable for the number of states
		set_variable = {
			name = tesla_num_states_var
			value = root.num_states
		}
		#Sets a variable for the number of states with a road modifier
		set_variable = {
			name = gaudi_num_states_roads_var
			value = tesla_states_with_roads_value #See Gaudi Script Values for the calculation
		}
		tesla_civil_engineering_progress = yes #Determines how much Engineering Progress will be added next month
	}

	complete = {
		custom_tooltip = { #All States need to have a roads modifier to end this journal entry
			text = gaudi_states_with_roads_tt
			var:gaudi_num_states_roads_var = var:tesla_num_states_var
		}
	}

	on_complete = {
		remove_variable = gaudi_can_stop_active_engineering_project_var #Needs to be removed, when the JE is removed
		tesla_remove_je_cost_modifiers_effect = yes
		remove_variable = tesla_num_states_var
		remove_variable = gaudi_num_states_roads_var

		#Those should be removed in every civil engineering je
		remove_variable = tesla_civil_engineering_progress_var
		remove_variable = gaudi_monthly_civil_engineering_progress_var

		var:tesla_engineer_var ?= {
			set_character_busy_and_immortal = no
		}
		trigger_event = {
			id = gaudi.1101
			popup = yes
		}

		#Starts new project for AI
		ai_start_engineering_project_for_ai_effect = yes
	}

	current_value = {
		#Keeps track of the Engineering Progress. If the Progress reaches the progress_cost, a modifier will be added.
		value = var:tesla_civil_engineering_progress_var
	}

	goal_add_value = {
		#You can change this in gaudi_script_values
		value = tesla_road_progress_cost
	}

	invalid = {
		custom_tooltip = {
			text = gaudi_stop_engineering_project_tt
			OR = {
				trigger_if = {
					limit = { is_ai = yes }
					tesla_has_engineer = yes
					tesla_has_no_civil_engineer_trigger = yes
				}
				has_variable = gaudi_stop_engineering_project_var
			}
		}
	}

	on_invalid = {
		remove_variable = gaudi_can_stop_active_engineering_project_var #Needs to be removed, when the JE is removed
		tesla_remove_je_cost_modifiers_effect = yes
		remove_variable = tesla_num_states_var
		remove_variable = gaudi_num_states_roads_var
		remove_variable = gaudi_stop_engineering_project_var

		#Those should be removed in every civil engineering je
		remove_variable = tesla_civil_engineering_progress_var
		remove_variable = gaudi_monthly_civil_engineering_progress_var

		var:tesla_engineer_var ?= {
			set_character_busy_and_immortal = no
		}

		#Starts new project for AI
		ai_start_engineering_project_for_ai_effect = yes
	}

	progressbar = yes

	weight = 30

	should_be_pinned_by_default = no
}

je_tesla_irrigation_project = {
	icon = "gfx/interface/icons/event_icons/steamers.dds"

	scripted_button = gaudi_stop_engineering_project_button
	scripted_button = gaudi_additional_support_for_civil_engineer_button
	scripted_button = gaudi_additional_support_for_civil_engineer_stop_button
	scripted_button = tesla_irrigation_project_new_priority_button

	group = je_group_tesla_engineers

	immediate = {
		scope:journal_entry = {
			add_modifier = {
				name = tesla_small_engineering_project_cost_modifier
				multiplier = root.mr_local_project_cost_small_value
			}
		}
		set_variable = gaudi_can_stop_active_engineering_project_var #This is used to check if the country has a project already
		#Sets the Civil Engineering Progress at the beginning
		set_variable = {
			name = tesla_civil_engineering_progress_var
			value = 0
		}
		#Sets a variable for the number of states
		set_variable = {
			name = tesla_num_states_var
			value = root.num_states
		}
		#Sets a variable for the number of states with a road modifier
		set_variable = {
			name = tesla_num_states_irrigation_var
			value = tesla_states_with_irrigation_value #See Gaudi Script Values for the calculation
		}
		tesla_civil_engineering_progress = yes #Determines how much Engineering Progress will be added next month
	}

	on_monthly_pulse = {
		effect = {
			#Calculates the Progress added (see gaudi_scripted_effects for calculation)
			tesla_civil_engineering_progress = yes
			#Adds the monthly progress to the variable
			change_variable = {
				name = tesla_civil_engineering_progress_var
				add = var:gaudi_monthly_civil_engineering_progress_var
			}
			if = {
				#If the Progress reaches the cost, the modifier will be added to a state
				limit = {
					var:tesla_civil_engineering_progress_var > tesla_irrigation_progress_cost
				}
				#Adds Experience to the Engineer
				civil_engineer_experience_gain_effect = { AMOUNT = 3 }
				#Effect if the priority list exists
				if = {
					limit = {
						has_variable_list = prioritized_states_for_irrigation_improvement_variable_list
					}
					#This part removes all states from the priority list if they already have the modifier.
					every_in_list = {
						variable = prioritized_states_for_irrigation_improvement_variable_list
						limit = {
							tesla_state_region_has_irrigation_system = yes
						}

						root = {
							remove_list_variable = {
								name = prioritized_states_for_irrigation_improvement_variable_list
								target = prev
							}
						}
					}
					#Chooses the first entry in the list
					ordered_in_list = {
						variable = prioritized_states_for_irrigation_improvement_variable_list
						limit = {
							tesla_state_region_has_no_irrigation_system = yes
						}
						max = 1 #Only the first one

						#Adds the state trait
						state_region = {
							add_state_trait = tesla_state_trait_irrigation
						}
						#Removes the state from the list
						root = {
							remove_list_variable = {
								name = prioritized_states_for_irrigation_improvement_variable_list
								target = prev
							}
						}
					}
				}
				else_if = {
					limit = {
						any_scope_state = {
							is_incorporated = yes
							tesla_state_region_has_no_irrigation_system = yes
						}
					}
					#The state with the least available infrastructure will be chosen
					ordered_scope_state = {
						position = 0 #We only want the first one here
						limit = {
							is_incorporated = yes
							tesla_state_region_has_no_irrigation_system = yes
						}
						order_by = tesla_state_total_agriculture_level_value
						state_region = {
							add_state_trait = tesla_state_trait_irrigation
						}
					}
				}
				else = {
					#The state with the least available infrastructure will be chosen
					ordered_scope_state = {
						position = 0 #We only want the first one here
						limit = {
							tesla_state_region_has_no_irrigation_system = yes
						}
						order_by = tesla_state_total_agriculture_level_value
						state_region = {
							add_state_trait = tesla_state_trait_irrigation
						}
					}
				}
				#The cost will be subtracted from the current progress
				change_variable = {
					name = tesla_civil_engineering_progress_var
					subtract = tesla_irrigation_progress_cost
				}
			}
			set_variable = {
				name = tesla_num_states_var
				value = root.num_states
			}
			set_variable = {
				name = tesla_num_states_irrigation_var
				value = tesla_states_with_irrigation_value
			}
		}
	}

	complete = {
		custom_tooltip = { #All States need to have a canals modifier to end this journal entry
			text = tesla_states_with_irrigation_tt
			var:tesla_num_states_irrigation_var = var:tesla_num_states_var
		}
	}

	on_complete = {
		remove_variable = gaudi_can_stop_active_engineering_project_var #Needs to be removed, when the JE is removed
		show_as_tooltip = {
			scope:journal_entry = {
				remove_modifier = tesla_medium_engineering_project_cost_modifier
			}
		}
		remove_variable = tesla_num_states_var
		remove_variable = tesla_num_states_irrigation_var

		#Those should be removed in every civil engineering je
		remove_variable = tesla_civil_engineering_progress_var
		remove_variable = gaudi_monthly_civil_engineering_progress_var

		#Should clean the lists
		clear_variable_list = prioritized_states_for_irrigation_improvement_variable_list
		clear_variable_list = states_for_irrigation_improvement_variable_list

		var:tesla_engineer_var ?= {
			set_character_busy_and_immortal = no
		}
		trigger_event = {
			id = gaudi.1111
			popup = yes
		}

		#Starts new project for AI
		ai_start_engineering_project_for_ai_effect = yes
	}

	current_value = {
		#Keeps track of the Engineering Progress. If the Progress reaches the progress_cost, a modifier will be added.
		value = var:tesla_civil_engineering_progress_var
	}

	goal_add_value = {
		#You can change this in gaudi_script_values
		value = tesla_irrigation_progress_cost
	}

	invalid = {
		custom_tooltip = {
			text = gaudi_stop_engineering_project_tt
			OR = {
				trigger_if = {
					limit = { is_ai = yes }
					tesla_has_engineer = yes
					tesla_has_no_civil_engineer_trigger = yes
				}
				has_variable = gaudi_stop_engineering_project_var
			}
		}
	}

	on_invalid = {
		remove_variable = gaudi_can_stop_active_engineering_project_var #Needs to be removed, when the JE is removed
		show_as_tooltip = {
			scope:journal_entry = {
				remove_modifier = tesla_medium_engineering_project_cost_modifier
			}
		}
		remove_variable = tesla_num_states_var
		remove_variable = tesla_num_states_irrigation_var
		remove_variable = gaudi_stop_engineering_project_var

		#Those should be removed in every civil engineering je
		remove_variable = tesla_civil_engineering_progress_var
		remove_variable = gaudi_monthly_civil_engineering_progress_var

		#Should clean the lists
		clear_variable_list = prioritized_states_for_irrigation_improvement_variable_list
		clear_variable_list = states_for_irrigation_improvement_variable_list

		var:tesla_engineer_var ?= {
			set_character_busy_and_immortal = no
		}

		#Starts new project for AI
		ai_start_engineering_project_for_ai_effect = yes
	}

	progressbar = yes

	weight = 30

	should_be_pinned_by_default = no
}
