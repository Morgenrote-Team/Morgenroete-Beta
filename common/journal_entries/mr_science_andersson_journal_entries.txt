je_andersson_speology = {
	icon = "gfx/interface/icons/invention_icons/andersson_anthropology_tech.dds"

	group = je_group_andersson_anthropology

	scripted_progress_bar = je_andersson_speology_research_progress_bar

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(je_andersson_speology_research_progress_bar)" >= 60
					}
				}

				#Adds Minerals and Trait
				anthropologist_experience_gain_effect = { AMOUNT = 10 }
				andersson_add_x_human_fossils_effect = { AMOUNT = 10 }
				if = {
					limit = {
						NOT = {
							var:andersson_anthropologist_var = {
								has_trait = andersson_speologist
							}
						}
					}
					var:andersson_anthropologist_var = {
						add_trait = andersson_speologist
					}
				}

				#Reset Anthropologist Research Bar
				scope:journal_entry = {
					set_bar_progress = {
						name = je_andersson_speology_research_progress_bar
						value = 0
					}
				}

				#Notification for Player
				post_notification = andersson_anthropological_research_notification
			}
		}

		random_events = {
			chance_to_happen = 50
			chance_of_no_event = {
				value = 92 #About every 12 months
			}

			#Anthropological Discoveries
			10 = andersson.101 #Andersson 101: Anthropological Site found (Cave Paintings)
		}
	}

	invalid = {
		custom_tooltip = {
			text = andersson_anthropologist_dies_tt
			andersson_has_anthropologist = no
		}
	}

	on_invalid = {
	}

	weight = 50
}
