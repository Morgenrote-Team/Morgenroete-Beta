######Speology######

je_andersson_speology = {
	icon = "gfx/interface/icons/invention_icons/andersson_anthropology_tech.dds"

	group = je_group_andersson_anthropology

	scripted_progress_bar = je_andersson_speology_research_progress_bar

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(je_andersson_speology_research_progress_bar)" >= 60
					}
				}

				#Adds Minerals and Trait
				anthropologist_experience_gain_effect = { AMOUNT = 10 }
				andersson_add_x_human_fossils_effect = { AMOUNT = 10 }
				#Finds Cave Painting
				if = {
					limit = {
						var:andersson_anthropologist_var = {
							has_trait = andersson_speologist
							OR = {
								has_trait = andersson_anthropologist_experienced
								has_trait = andersson_anthropologist_famous
							}
						}
						any_scope_state = {
							andersson_state_has_cave_paintings_trigger = yes
							andersson_state_has_discovered_cave_paintings_no_tooltip_trigger = no
						}
					}
					trigger_event = {
						id = andersson.101 #Andersson 101: Anthropological Site found (Cave Paintings)
					}
				}
				else_if = {
					limit = {
						andersson_can_find_australopithecus_trigger = yes
						NOT = {
							has_variable = andersson_australopithecus_cave_set_var
						}
						any_scope_state = {
							state_region = {
								is_in_geographic_region = andersson_geographic_region_australopithecus
								andersson_state_has_cave_trigger = no
							}
						}
					}
					trigger_event = {
						id = andersson.105 #Andersson 105: Anthropological Site found (Australopithecus Cave)
					}
				}
				else_if = {
					limit = {
						andersson_can_find_pithecanthropus_trigger = yes
						NOT = {
							has_variable = andersson_pithecanthropus_cave_set_var
						}
						any_scope_state = {
							state_region = {
								is_in_geographic_region = andersson_geographic_region_pithecanthropus
								andersson_state_has_cave_trigger = no
							}
						}
					}
					trigger_event = {
						id = andersson.104 #Andersson 104: Anthropological Site found (Pithecanthropus Cave)
					}
				}
				else_if = {
					limit = {
						andersson_can_find_palaeanthropus_trigger = yes
						NOT = {
							has_variable = andersson_palaeanthropus_cave_set_var
						}
						any_scope_state = {
							state_region = {
								is_in_geographic_region = andersson_geographic_region_palaeanthropus
								andersson_state_has_cave_trigger = no
							}
						}
					}
					trigger_event = {
						id = andersson.103 #Andersson 103: Anthropological Site found (Palaeanthropus Cave)
					}
				}
				else_if = {
					limit = {
						NOT = {
							has_variable = andersson_neoanthropus_cave_set_var
						}
						any_scope_state = {
							is_in_geographic_region = andersson_geographic_region_neoanthropus
							andersson_state_has_cave_trigger = no
						}
					}
					trigger_event = {
						id = andersson.102 #Andersson 102: Anthropological Site found (Neoanthropus Cave)
					}
				}
				#Adds Speologist Trait
				if = {
					limit = {
						NOT = {
							var:andersson_anthropologist_var = {
								has_trait = andersson_speologist
							}
						}
					}
					var:andersson_anthropologist_var = {
						add_trait = andersson_speologist
					}
				}


				#Reset Anthropologist Research Bar
				scope:journal_entry = {
					set_bar_progress = {
						name = je_andersson_speology_research_progress_bar
						value = 0
					}
				}

				#Notification for Player
				post_notification = andersson_anthropological_research_notification
			}
		}
	}

	invalid = {
		custom_tooltip = {
			text = andersson_anthropologist_dies_tt
			andersson_has_anthropologist = no
		}
	}

	on_invalid = {
	}

	weight = 50
}
