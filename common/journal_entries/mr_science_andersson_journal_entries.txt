######Anthropological Research######

je_andersson_anthropological_research = {
	icon = "gfx/interface/icons/invention_icons/andersson_anthropology_tech.dds"

	group = je_group_andersson_anthropology

	scripted_progress_bar = andersson_anthropological_research_progress_bar

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(andersson_anthropological_research_progress_bar)" >= 12
					}
				}

				#Adds Experience and Fossils
				anthropologist_experience_gain_effect = { AMOUNT = 5 }
				andersson_add_x_human_fossils_effect = { AMOUNT = 1 }

				#Paleontologist Trait
				if = {
					limit = {
						var:andersson_anthropologist_var = {
							has_trait = dubois_paleontologist
						}
					}
					andersson_add_x_human_fossils_effect = { AMOUNT = 3 }
				}

				#Archaeologist Trait
				if = {
					limit = {
						var:andersson_anthropologist_var = {
							has_trait = lepsius_archaeologist
						}
					}
					andersson_add_x_prehistoric_artifacts_effect = { AMOUNT = 1 }
				}

				#Archaeologist Trait
				if = {
					limit = {
						var:andersson_anthropologist_var = {
							has_trait = andersson_speleologist
						}
					}
					andersson_add_x_human_fossils_effect = { AMOUNT = 1 }
				}

				#Cave Effect
				if = {
					limit = {
						any_scope_state = {
							OR = {
								has_state_trait = andersson_state_trait_cave
								has_state_trait = andersson_state_trait_cave_paintings
							}
						}
					}
					andersson_add_x_human_fossils_effect = { AMOUNT = 2 }
				}

				#Finds Cave Painting
				if = {
					limit = {
						var:andersson_anthropologist_var = {
							OR = {
								has_trait = andersson_anthropologist_experienced
								has_trait = andersson_anthropologist_famous
							}
						}
						any_scope_state = {
							andersson_state_has_undiscovered_cave_paintings_trigger = yes
						}
					}
					trigger_event = {
						id = andersson.101 #Andersson 101: Anthropological Site found (Cave Paintings)
					}
				}

				#Palentologist & Archaeologist
				if = {
					limit = {
						var:andersson_anthropologist_var = {
							NOR = {
								has_trait = dubois_paleontologist
								has_trait = lepsius_archaeologist
							}
						}
					}
					random_list = {
						1 = {
							var:andersson_anthropologist_var = {
								add_trait = dubois_paleontologist
							}
						}
						1 = {
							var:andersson_anthropologist_var = {
								add_trait = lepsius_archaeologist
							}
						}
						6 = {}
					}
				}

				#Reset Anthropologist Research Bar
				scope:journal_entry = {
					set_bar_progress = {
						name = andersson_anthropological_research_progress_bar
						value = 0
					}
				}

				#Notification for Player
				post_notification = andersson_anthropological_research_notification
			}
		}
	}

	on_yearly_pulse = {
		effect = {
			trigger_event = {
				on_action = andersson_on_yearly_pulse_local_anthropology_events
				days = { 30 365 }
			}
		}
	}

	invalid = {
		custom_tooltip = {
			text = andersson_anthropologist_dies_tt
			andersson_has_anthropologist = no
		}
	}

	on_invalid = {
	}

	weight = 50
}

######Hominid Prospecting###

je_andersson_hominid_prospecting_neoanthropus = {
	icon = "gfx/interface/icons/invention_icons/andersson_anthropology_tech.dds"

	group = je_group_andersson_anthropology

	scripted_progress_bar = andersson_cave_exploration_progress_bar
	scripted_progress_bar = andersson_cave_collection_progress_bar
	scripted_progress_bar = andersson_cave_identification_progress_bar

	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(andersson_cave_exploration_progress_bar)" >= 90
					}
				}
				andersson_hominid_prospecting_exploration_event_effect = yes
			}
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(andersson_cave_collection_progress_bar)" >= 90
					}
				}
				andersson_hominid_prospecting_collecting_event_effect = yes
			}
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(andersson_cave_collection_progress_bar)" >= 50
					}
					andersson_prospecting_neoanthropus_discovery_trigger = yes
				}
				trigger_event = { id = andersson.221 }
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = andersson_cave_identification_progress_bar_at_full_tt
			scope:journal_entry = {
				"scripted_bar_progress(andersson_cave_identification_progress_bar)" >= 90
			}
		}
	}

	on_complete = {
		trigger_event = { id = andersson.211 }
	}

	invalid = {
		custom_tooltip = {
			text = andersson_anthropologist_dies_tt
			andersson_has_anthropologist = no
		}
	}

	on_invalid = {
		andersson_remove_prospecting_effect = yes
		remove_variable = andersson_hominid_prospecting_neoanthropus_var
	}

	weight = 50
}

	icon = "gfx/interface/icons/invention_icons/andersson_anthropology_tech.dds"

	group = je_group_andersson_anthropology

	scripted_progress_bar = andersson_cave_exploration_progress_bar
	scripted_progress_bar = andersson_cave_collection_progress_bar
	scripted_progress_bar = andersson_cave_identification_progress_bar

	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(andersson_cave_exploration_progress_bar)" >= 90
					}
				}
				andersson_hominid_prospecting_exploration_event_effect = yes
			}
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(andersson_cave_collection_progress_bar)" >= 90
					}
				}
				andersson_hominid_prospecting_collecting_event_effect = yes
			}
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(andersson_cave_collection_progress_bar)" >= 50
					}
					andersson_expedition_discovery_trigger = yes
				}
				trigger_event = { id = andersson.225 }
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = andersson_cave_identification_progress_bar_at_full_tt
			scope:journal_entry = {
				"scripted_bar_progress(andersson_cave_identification_progress_bar)" >= 90
			}
		}
	}

	on_complete = {
		trigger_event = { id = andersson.213 }
	}

	invalid = {
		OR = {
			custom_tooltip = {
				text = andersson_anthropologist_dies_tt
				andersson_has_anthropologist = no
			}
			custom_tooltip = {
				text = andersson_je_no_subject_anymore_tt
				NOT = {
					any_direct_subject = {
						any_scope_state = {
							has_variable = andersson_hominid_expedition_var
						}
					}
				}
			}
		}
	}

	on_invalid = {
		andersson_remove_expedition_effect = yes
		remove_variable = andersson_hominid_expedition_neoanthropus_var
		remove_variable = andersson_hominid_expedition_palaeanthropus_var
		remove_variable = andersson_hominid_expedition_pithecanthropus_var
		remove_variable = andersson_hominid_expedition_australopithecus_var
	}

	weight = 50
}
