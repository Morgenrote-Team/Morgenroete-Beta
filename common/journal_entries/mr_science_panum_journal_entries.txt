je_panum_medical_system_project = {
	icon = "gfx/interface/icons/event_icons/event_medicine.dds"

	scripted_button = panum_stop_medical_project_button

	group = je_group_panum_medicine

	scripted_progress_bar = panum_medical_system_project_progress_bar

	on_monthly_pulse = {
		effect = {
			if = {
				#If the Progress reaches the cost, the PM will be unlocked in a state
				limit = {
					je:je_panum_medical_system_project = {
						"scripted_bar_progress(panum_medical_system_project_progress_bar)" >= 30
					}
				}
				#Physician gets experience
				physician_experience_gain_effect = { AMOUNT = 20 }
				#The state with the next available hospital > clinic will be chosen
				if = {
					limit = {
						any_scope_state = {
							has_building = building_government_administration
							any_scope_building = {
								is_building_type = building_government_administration
								level >= 20
								has_active_production_method = pm_panum_clinic
							}
						}
					}
					ordered_scope_state = {
						limit = {
							has_building = building_government_administration
							any_scope_building = {
								is_building_type = building_government_administration
								level >= 20
								has_active_production_method = pm_panum_clinic
							}
						}
						order_by = panum_administration_buildings_value
						max = 1 #Limits to the state with highest urbanization
						check_range_bounds = no #prevents errors

						activate_production_method = {
							building_type = building_government_administration
							production_method = pm_panum_hospital
						}
					}
				}
				else_if = {
					limit = {
						any_scope_state = {
							has_building = building_government_administration
							any_scope_building = {
								is_building_type = building_government_administration
								level >= 20
								has_active_production_method = pm_panum_no_clinic
							}
						}
					}
					ordered_scope_state = {
						limit = {
							has_building = building_government_administration
							any_scope_building = {
								is_building_type = building_government_administration
								level >= 20
								has_active_production_method = pm_panum_no_clinic
							}
						}
						order_by = panum_administration_buildings_value
						max = 1 #Limits to the state with highest urbanization
						check_range_bounds = no #prevents errors

						activate_production_method = {
							building_type = building_government_administration
							production_method = pm_panum_clinic
						}
					}
				}
				else_if = {
					limit = {
						any_scope_state = {
							has_building = building_government_administration
							any_scope_building = {
								is_building_type = building_government_administration
								level >= 5
								level < 20
								has_active_production_method = pm_panum_no_clinic
							}
						}
					}
					ordered_scope_state = {
						limit = {
							has_building = building_government_administration
							any_scope_building = {
								is_building_type = building_government_administration
								level >= 5
								level < 20
								has_active_production_method = pm_panum_no_clinic
							}
						}
						order_by = panum_administration_buildings_value
						max = 1 #Limits to the state with highest urbanization
						check_range_bounds = no #prevents errors

						activate_production_method = {
							building_type = building_government_administration
							production_method = pm_panum_clinic
						}
					}
				}
				#The cost will be subtracted from the current progress
				je:je_panum_medical_system_project = {
					add_progress = {
						name = panum_medical_system_project_progress_bar
						value = -30
					}
				}
			}
		}
		events = {
			panum.2102 #Panum 2102: First Clinic
			panum.2103 #Panum 2103: First Hospital
		}

		random_events = {
			30 = 0

			1 = panum.2111 #Panum 2111: Gray's Anatomy
			1 = panum.2112 #Panum 2112: Diagram of Causes of Mortality
			1 = panum.2113 #Panum 2113: Omnis cellula e cellula
			1 = panum.2114 #Panum 2114: Born criminal

			1 = panum.2131 #Panum 2131: A Plate to win
		}
	}

	immediate = {
		set_variable = panum_can_stop_active_medical_project_var #This is used to check if the country has a project already
		scope:journal_entry = {
			add_modifier = {
				name = panum_medical_system_cost_modifier
				multiplier = root.mr_local_project_cost_very_small_value
			}
		}
	}

	complete = {
		any_scope_state = {
			OR = {
				custom_tooltip = {
					text = je_panum_medical_system_project_success_1
					any_scope_building = {
						is_building_type = building_government_administration
						level >= 20
						has_active_production_method = pm_panum_hospital
					}
				}
				custom_tooltip = {
					text = je_panum_medical_system_project_success_2
					any_scope_building = {
						is_building_type = building_government_administration
						level >= 5
						level < 20
						OR = {
							has_active_production_method = pm_panum_hospital
							has_active_production_method = pm_panum_clinic
						}
					}
				}
				custom_tooltip = {
					text = je_panum_medical_system_project_success_3
					any_scope_building = {
						is_building_type = building_government_administration
						level < 5
					}
				}
				NOT = { has_building = building_government_administration }
			}
			percent >= 1
		}
	}

	on_complete = {
		remove_variable = mr_physician_expand_medical_system_var
		remove_variable = panum_can_stop_active_medical_project_var #Needs to be removed, when the JE is removed
		trigger_event = {
			id = panum.2101
			popup = yes
		}
	}

	invalid = {
		custom_tooltip = {
			text = panum_stop_medical_project_tt
			OR = {
				custom_tooltip = {
					text = je_physician_dies_tt
					NOT = {
						has_variable = panum_physician_var
					}
				}
				has_variable = panum_stop_medical_project_var
			}
		}
	}

	on_invalid = {
		remove_variable = mr_physician_expand_medical_system_var
		remove_variable = panum_stop_medical_project_var
		remove_variable = panum_can_stop_active_medical_project_var #Needs to be removed, when the JE is removed
		var:panum_physician_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	weight = 30
}

	icon = "gfx/interface/icons/event_icons/event_medicine.dds"

	group = je_group_panum_medicine

	complete = {
		NOT = {
			any_scope_state = {
				panum_state_has_spanish_flu_trigger = yes
			}
		}
	}

	on_complete = {
		custom_tooltip = {
			text = je_panum_spanish_flu_research_complete_tt
			if = {
				limit = {
					has_technology_researched = antibiotics
					var:panum_physician_var.var:panum_physician_experience_var >= 250
					theiler_has_microbiologist_trigger = yes
					panum_has_miasma_theory_trigger = no
				}
				trigger_event = {
					id = panum.1111 #Pathogen
					days = { 7 14 }
				}
			}
			else_if = {
				limit = {
					has_technology_researched = panum_antitoxins_tech
					var:panum_physician_var.var:panum_physician_experience_var >= 200
				}
				trigger_event = {
					id = panum.1112 #Origin
					days = { 7 14 }
				}
			}
			else_if = {
				limit = {
					var:panum_physician_var.var:panum_physician_experience_var >= 150
				}
				trigger_event = {
					id = panum.1113 #Treatment
					days = { 7 14 }
				}
			}
			else_if = {
				limit = {
					var:panum_physician_var.var:panum_physician_experience_var >= 100
				}
				trigger_event = {
					id = panum.1114 #Description
					days = { 7 14 }
				}
			}
			else = {
				trigger_event = {
					id = panum.1115 #Nothing
					days = { 7 14 }
				}
			}
		}
	}

	weight = 50

je_panum_cholera_description = {
	icon = "gfx/interface/icons/event_icons/event_medicine.dds"

	group = je_group_panum_medicine

	on_weekly_pulse = {
		random_events = {
			40 = 0
			4 = panum.121 #Panum 121: Cholera Hint (hidden)
			2 = panum.122 #Panum 122: Cholera Incubation
			2 = panum.123 #Panum 123: Cholera Symptoms
			1 = panum.124 #Panum 124: Cholera Morbus and Asiatic Cholera
		}
	}

	immediate = {
		if = { #Fallback if the variable does not exist yet
			limit = {
				NOT = {
					has_variable = panum_cholera_description_clues_var
				}
			}
			set_variable = {
				name = panum_cholera_description_clues_var
				value = 0
			}
		}
		random_scope_state = {
			limit = {
				panum_state_has_cholera_trigger = yes
			}
			save_scope_as = panum_disease_state
		}
	}

    fail = {
		NOT = {
			has_journal_entry = je_panum_cholera
		}
    }

	on_fail = {
		trigger_event = {
			id = panum.131 #Unfinished Description
			popup = yes
		}
		panum_remove_active_variables_effect = yes
	}

	complete = {
		custom_tooltip = {
			text = je_panum_description_success
			scope:journal_entry = {
				is_goal_complete = yes
			}
		}
	}

	on_complete = {
		set_variable = {
			name = panum_disease_title_var
			value = 1
		}
		trigger_event = {
			id = panum.138
			popup = yes
		}
		panum_remove_active_variables_effect = yes
	}

	current_value = {
		if = {
			limit = {
				scope:journal_entry = {
					goal_value = 0
				}
			}
			value = 0
		}
		else = {
			value = var:panum_cholera_description_clues_var
		}
	}

	goal_add_value = {
		value = 6
	}

	invalid = {
		custom_tooltip = {
			text = je_physician_dies_tt
			NOT = {
				has_variable = panum_physician_var
			}
		}
	}

	on_invalid = {
		panum_remove_active_variables_effect = yes
		var:panum_physician_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	progressbar = yes

	weight = 30

	should_be_pinned_by_default_uninvolved_or_context = yes
}
