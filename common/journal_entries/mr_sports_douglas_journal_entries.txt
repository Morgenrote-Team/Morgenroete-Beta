###Mountaineering Training###

je_douglas_mountaineering_training = {
	icon = "gfx/interface/icons/event_icons/event_mountaineering.dds"

	group = je_group_vikelas_sports

	scripted_button = douglas_stop_mountaineering_project_button

	on_yearly_pulse = {
		random_events = {
			1 = 0
			1 = douglas.121 #Douglas 121: Courage Forged at Home
			1 = douglas.122 #Douglas 122: Steps Never Wasted
			1 = douglas.123 #Douglas 123: Beyond the Known Maps
			1 = douglas.124 #Douglas 124: Steel and Stone
			1 = douglas.131 #Douglas 121: Courage Forged at Home (hidden)
			1 = douglas.132 #Douglas 122: Steps Never Wasted (hidden)
			1 = douglas.133 #Douglas 123: Beyond the Known Maps (hidden)
			1 = douglas.134 #Douglas 124: Steel and Stone (hidden)
		}
	}

	immediate = {
		set_variable = douglas_can_stop_active_mountaineering_project #This is used to check if the country has a project already
	}

	timeout = 1095

	on_timeout = {
		trigger_event = {
			id = douglas.104
			popup = yes
		}
	}

	invalid = {
		custom_tooltip = {
			text = douglas_stop_mountaineering_project_tt
			has_variable = douglas_stop_mountaineering_project_var
		}
	}

	on_invalid = {
		remove_variable = douglas_stop_mountaineering_project_var
		remove_variable = douglas_can_stop_active_mountaineering_project #Needs to be removed, when the JE is removed
		var:douglas_mountaineer_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	fail = {
		custom_tooltip = {
			text = je_mountaineer_dies_tt
			NOT = {
				has_variable = douglas_mountaineer_var
			}
		}
	}

	on_fail = {
		remove_variable = douglas_stop_mountaineering_project_var
		remove_variable = douglas_can_stop_active_mountaineering_project #Needs to be removed, when the JE is removed
		var:douglas_mountaineer_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	weight = 25
}

###Mountaineering Equipment###

je_douglas_mountaineering_equipment = {
	icon = "gfx/interface/icons/event_icons/event_mountaineering.dds"

	group = je_group_vikelas_sports

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = douglas_mountaineering_equipment_progress
				add = 1
			}
			if = {
				limit = {
					var:douglas_mountaineering_equipment_progress >= 12
				}
				douglas_mountaineering_equipment_modifier_effect = yes
			}
		}
	}

	immediate = {
		set_variable = {
			name = douglas_mountaineering_equipment_progress
			value = 0
		}
	}

	complete = {
		custom_tooltip = {
			text = je_douglas_mountaineering_equipment_success
			has_variable = douglas_mountaineering_equipment_complete_var
		}
	}

	on_complete = {
		douglas_mountaineering_equipment_event_effect = yes
		var:douglas_mountaineer_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	fail = {
		custom_tooltip = {
			text = je_mountaineer_dies_tt
			NOT = {
				has_variable = douglas_mountaineer_var
			}
		}
	}

	on_fail = {
		var:douglas_mountaineer_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	current_value = {
		value = var:douglas_mountaineering_equipment_progress
	}

	goal_add_value = {
		add = 12
	}

	progressbar = yes

	weight = 25
}

###Mountaineering Expedition######

je_douglas_mountaineering_expedition = {
	icon = "gfx/interface/icons/event_icons/event_mountaineering.dds"

	group = je_group_vikelas_sports

	scripted_progress_bar = douglas_mountain_expedition_progress_bar
	scripted_progress_bar = douglas_mountain_danger_progress_bar

	on_monthly_pulse = {
		random_events = {
			1 = 0
			1 = douglas.221 #Douglas 221: An Easy Climb
			1 = douglas.222 #Douglas 222: A Slope
			1 = douglas.223 #Douglas 223: Steep Wall
			1 = douglas.224 #Douglas 224: Avalanche
			1 = douglas.225 #Douglas 225: Abyss
			1 = douglas.226 #Douglas 226: Death Step
			1 = douglas.227 #Douglas 227: Weather Change
			1 = douglas.228 #Douglas 228: Rockfall
			1 = douglas.229 #Douglas 229: Dizzying Heights
			1 = douglas.230 #Douglas 230: Readings in Stone
			1 = douglas.231 #Douglas 231: Scientist does Science
			1 = douglas.232 #Douglas 232: A Fine Mineral
			1 = douglas.233 #Douglas 233: The Red Panda
		}
	}

	immediate = {
		scope:journal_entry = {
			add_modifier = {
				name = douglas_mountaineering_expedition_cost_modifier
				multiplier = root.mr_local_project_cost_very_small_value
			}
		}
		if = {
			limit = {
				var:douglas_mountaineer_var = {
					OR = {
						has_trait = brave
						has_trait = charismatic
						has_trait = douglas_mountaineer_veteran
					}
				}
			}
			trigger_event = {
				id = douglas.208
				days = { 7 14 }
			}
		}
		if = {
			limit = {
				douglas_has_advanced_equpiment_trigger = yes
			}
			trigger_event = {
				id = douglas.209
				days = { 21 28 }
			}
		}
		if = {
			limit = {
				has_variable = agassiz_geologist_var
			}
			trigger_event = {
				id = douglas.210
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = douglas_mountain_expedition_progress_bar_at_full_tt
			je:je_douglas_mountaineering_expedition = {
				"scripted_bar_progress(douglas_mountain_expedition_progress_bar)" >= 10
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = douglas.211
		}
		var:douglas_mountaineer_var ?= {
			set_character_busy_and_immortal = no
		}
		add_modifier = {
			name = douglas_mountaineering_success_modifier
			years = 3
		}
	}

	invalid = {
		custom_tooltip = {
			text = je_mountaineer_dies_tt
			NOT = {
				has_variable = douglas_mountaineer_var
			}
		}
	}

	on_invalid = {
		remove_variable = douglas_mountain_expedition_var
	}

	timeout = douglas_mountain_motivation_value

	on_timeout = {
		trigger_event = {
			id = douglas.212
		}
		var:douglas_mountaineer_var ?= {
			set_character_busy_and_immortal = no
		}
		add_modifier = {
			name = douglas_mountaineering_disappointment_modifier
			years = 2
		}
	}

	fail = {
		custom_tooltip = {
			text = douglas_mountain_danger_progress_bar_at_full_tt
			je:je_douglas_mountaineering_expedition = {
				"scripted_bar_progress(douglas_mountain_danger_progress_bar)" >= 5
			}
		}
	}

	on_fail = {
		trigger_event = {
			id = douglas.213
		}
		var:douglas_mountaineer_var ?= {
			set_character_busy_and_immortal = no
		}
		add_modifier = {
			name = douglas_mountaineering_disaster_modifier
			years = 3
		}
	}

	should_be_pinned_by_default_uninvolved_or_context = yes

	weight = 25
}
