###PALEONTOLOGICAL EXPEDITION###

je_dubois_paleontological_expedition = {

	icon = "gfx/interface/icons/event_icons/event_fossil.dds"

	group = je_group_dubois_nature

	scripted_progress_bar = dubois_paleontological_expedition_progress_bar
	scripted_progress_bar = dubois_paleontological_expedition_success_bar

	on_monthly_pulse = {
		effect = {
			#Monthly Events
			trigger_event = {
				on_action  = dubois_on_monthly_pulse_paleontological_expedition
				days = { 1 31 }
			}

			#Monthly change of success chance
			set_variable = {
				name = dubois_paleontological_expedition_estimated_progress_var
				value = dubois_paleontology_progress_max_1_value
			}
		}
	}

	immediate = {
		#Set success chance
		set_variable = {
			name = dubois_paleontological_expedition_estimated_progress_var
			value = dubois_paleontology_progress_max_1_value
		}
	}

	complete = {
		custom_tooltip = {
			text = je_dubois_paleontological_expedition_goal_tt
			scope:journal_entry = {
				"scripted_bar_progress(dubois_paleontological_expedition_progress_bar)" >= 12
			}
		}
	}

	on_complete = {
		#This effect is used to make sure the excavation is set up in the right continent.
		if = {
			limit = {
				scope:journal_entry = {
					has_modifier = dubois_paleontological_expedition_africa_modifier
				}
			}
			set_variable = dubois_expedition_in_africa_var
		}
		else_if = {
			limit = {
				scope:journal_entry = {
					has_modifier = dubois_paleontological_expedition_asia_modifier
				}
			}
			set_variable = dubois_expedition_in_asia_var
		}
		else_if = {
			limit = {
				scope:journal_entry = {
					has_modifier = dubois_paleontological_expedition_australia_modifier
				}
			}
			set_variable = dubois_expedition_in_australia_var
		}
		else_if = {
			limit = {
				scope:journal_entry = {
					has_modifier = dubois_paleontological_expedition_europe_modifier
				}
			}
			set_variable = dubois_expedition_in_europe_var
		}
		else_if = {
			limit = {
				scope:journal_entry = {
					has_modifier = dubois_paleontological_expedition_north_america_modifier
				}
			}
			set_variable = dubois_expedition_in_north_america_var
		}
		else_if = {
			limit = {
				scope:journal_entry = {
					has_modifier = dubois_paleontological_expedition_south_america_modifier
				}
			}
			set_variable = dubois_expedition_in_south_america_var
		}
		else_if = {
			limit = {
				scope:journal_entry = {
					has_modifier = dubois_paleontological_expedition_local_modifier
				}
			}
			set_variable = dubois_expedition_local_var
		}

		#Paleontologist Experience gain
		paleontologist_experience_gain_effect = { AMOUNT = 10 }

		#Frees Paleontologist
		var:dubois_paleontologist_var ?= {
			set_character_busy_and_immortal = no
		}

		#Saves the succes value
		dubois_paleontological_expedition_succes_value_effect = yes

		#Event that actually adds the fossils/bones and allows to establish an excavation.
		trigger_event = dubois.241

		#Modifier to make another expedition impossible.
		add_modifier = {
			name = dubois_had_expedition_modifier
			years = 2
		}

		#Starts the next project for AI or automated player
		ai_paleontologist_automation_effect = yes
	}

	should_be_pinned_by_default_uninvolved_or_context = yes

	weight = 25
}

###DINOSAUR EXCAVATION######

je_dubois_dinosaur_excavation_africa = {

	icon = "gfx/interface/icons/event_icons/event_fossil.dds"

	group = je_group_dubois_nature

	scripted_button = dubois_send_paleontologist_to_excavation_button
	scripted_button = dubois_recall_paleontologist_from_excavation_button
	scripted_button = dubois_recall_african_excavation_button

	modifiers_while_active = {
		dubois_paleontological_excavation_africa_modifier
	}

	on_yearly_pulse = {
		effect = {
			#Triggers Special Fossil Events and updates cost
			dubois_paleontological_excavation_yearly_effect = yes

			#Ends Excavation for AI
			if = {
				limit = {
					is_ai = yes
					OR = {
						#Insuficient money
						mr_can_afford_very_small_money_multiplier_trigger = no
						#Should be ended if maximum bones from country were found and not all bones were found yet
						AND = {
							dubois_has_maximum_african_dinosaur_bone = yes
							dubois_has_all_possible_bones_trigger = no
						}
					}
				}

				set_variable = dubois_recall_african_excavation_var
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			1000 = 0
			10 = dubois.213 #Fossil Fragments
			10 = dubois.214 #Dinosaur Bone
			5 = dubois.215 #Dinosaur Skull
			5 = dubois.216 #Dinosaur Skeleton
		}
		effect = {
			#JE uses value to to keep calculations from script value down
			scope:journal_entry = {
				set_variable = {
					name = dubois_yearly_regular_fossils_from_african_excavation_var
					value = root.dubois_yearly_regular_fossils_from_european_excavation_value
				}
			}
			#This is used for multiple JEs, therefore it is on the country scope.
			set_variable = {
				name = dubois_yearly_estimated_regular_fossils_var
				value = dubois_yearly_regular_fossils_value
			}
		}
	}

	immediate = {
		#JE uses value to to keep calculations from script value down
		scope:journal_entry = {
			set_variable = {
				name = dubois_yearly_regular_fossils_from_african_excavation_var
				value = root.dubois_yearly_regular_fossils_from_european_excavation_value
			}
		}
		#This is used for multiple JEs, therefore it is on the country scope.
		set_variable = {
			name = dubois_yearly_estimated_regular_fossils_var
			value = dubois_yearly_regular_fossils_value
		}
	}

	complete = {
		custom_tooltip = {
			text = je_dubois_dinosaur_excavation_goal
			always = no
		}
	}

	invalid = {
		has_variable = dubois_recall_african_excavation_var
	}

	on_invalid = {
		remove_variable = dubois_recall_african_excavation_var
		dubois_unuse_paleontological_team_effect = yes
	}

	weight = 25
}

###Prehistoric Scenery###
je_prehistoric_scenery = {
	icon = "gfx/interface/icons/event_icons/event_fossil.dds"

	scripted_button = dubois_stop_paleontologist_project_button

	group = je_group_dubois_nature

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = dubois_paleontologist_scenery_progress_var
				add = 1
			}
		}
		random_events = {
			50 = 0
			1 = dubois.272 #Dubois 272: Materials
			1 = dubois.273 #Dubois 273: A ridiculous job?
			1 = dubois.274 #Dubois 274: Student interest
		}
	}

	immediate = {
		set_variable = {
			name = dubois_paleontologist_scenery_progress_var
			value = 0
		}
		set_variable = dubois_can_stop_active_paleontologist_project_var
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		trigger_event = {
			id = dubois.271
		}
		if = {
			limit = {
				has_variable = dubois_triassic_scenery_ongoing_var
			}
			set_variable = dubois_triassic_scenery_var
		}
		else_if = {
			limit = {
				has_variable = dubois_jurassic_scenery_ongoing_var
			}
			set_variable = dubois_jurassic_scenery_var
		}
		else_if = {
			limit = {
				has_variable = dubois_cretaceous_scenery_ongoing_var
			}
			set_variable = dubois_cretaceous_scenery_var
		}

		#Starts the next project for AI or automated player
		ai_paleontologist_automation_effect = yes
	}

	current_value = {
		value = var:dubois_paleontologist_scenery_progress_var
	}

	goal_add_value = {
		value = 24
	}

	invalid = {
		OR = {
			custom_tooltip = {
				text = je_paleontologist_dies_tt
				NOT = {
					has_variable = dubois_paleontologist_var
				}
			}
			custom_tooltip = {
				text = dubois_stop_paleontologist_project_tt
				has_variable = dubois_stop_paleontologist_project_var
			}
			custom_tooltip = {
				text = dubois_no_university_for_scenery_tt
				NOT = {
					any_scope_building = {
						is_building_type = building_university
						NOR = {
							has_modifier = dubois_cretaceous_scenery_modifier
							has_modifier = dubois_jurassic_scenery_modifier
							has_modifier = dubois_triassic_scenery_modifier
						}
					}
				}
			}
		}
	}

	on_invalid = {
		remove_variable = dubois_can_stop_active_paleontologist_project_var #Needs to be removed, when the JE is removed
		remove_variable = dubois_stop_paleontologist_project_var
		var:dubois_paleontologist_var ?= {
			set_character_busy_and_immortal = no
		}
		remove_variable = dubois_cretaceous_scenery_ongoing_var
		remove_variable = dubois_jurassic_scenery_ongoing_var
		remove_variable = dubois_triassic_scenery_ongoing_var
	}

	progressbar = yes

	weight = 30
}
