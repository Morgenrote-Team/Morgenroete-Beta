#Word across the Ocean
pbi_je_objective_across_the_sea = {
	should_be_pinned_by_default = yes
	icon = "gfx/interface/icons/invention_icons/screw_frigate.dds"

	group = je_group_pbi

	immediate = {
		
	}

	current_value = {
		value = var:pbi_var_je_objective_across_the_sea
	}
	
	goal_add_value = {
		add = 20
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	on_complete = {
		tesla_electrical_engineering_large_add_technology = yes
		add_modifier = {
			name = pbi_country_modifier_transoceanic_telegraph_cable
			years = 50
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOR = {
						var:pbi_var_je_objective_across_the_sea = 1
						var:pbi_var_je_objective_across_the_sea = 8
						var:pbi_var_je_objective_across_the_sea >= 18
					}
				}
				change_variable = {
					name = pbi_var_je_objective_across_the_sea
					add = 1
				}
			}
		}
		events = {
			pbi_objective_publishing_industry_events.103 #Fund-raising
			pbi_objective_publishing_industry_events.111 #Triumphant return of the submarine cable laying fleet
		}
		random_events = {
			10 = 0
			###Phase 1###
			5 = pbi_objective_publishing_industry_events.113 #Steady fundraising
			5 = pbi_objective_publishing_industry_events.104 #Difficult fundraising
			1 = pbi_objective_publishing_industry_events.105 #[concept_interest_group] Assistance
			1 = pbi_objective_publishing_industry_events.106 #[concept_government] Assistance
			###Phase 2###
			2 = pbi_objective_publishing_industry_events.108 #Cable disconnected
			3 = pbi_objective_publishing_industry_events.109 #Storm disrupts cable installation
			1 = pbi_objective_publishing_industry_events.110 #Giant shadow
			###Science###
			5 = rapanui.511 #Rapanui 511: Whales & Dolphins
			5 = rapanui.512 #Rapanui 512: Fish
			5 = rapanui.513 #Rapanui 513: Experiments
			2 = rapanui.521 #Rapanui 514: Middle-Atlantic Ridge
		}
	}

	progressbar = yes

	invalid = {
		custom_tooltip = {
			text = pbi_je_objective_across_the_sea_flag_not_character_in_laying
			var:pbi_var_je_objective_across_the_sea = 0
		}
	}
	on_invalid = {
		trigger_event = {
			id = pbi_objective_publishing_industry_events.101 #Business Failure
		}
		remove_variable = rapanui_transatlantic_cable_start_var
		set_variable = {
			name = pbi_var_je_objective_across_the_sea
			value = 0
		}
	}

	scripted_button = rapanui_transatlantic_cable_start_button

	weight = 500
	transferable = yes
}

#Newspaper Novel
pbi_je_objective_bestseller = {
	should_be_pinned_by_default = yes
	icon = "gfx/interface/icons/event_icons/tutorial_icon.dds"

	group = je_group_pbi

	is_progressing = {
		manzoni_has_writer = yes
	}

	immediate = {
		set_variable = {
			name = pbi_var_je_objective_bestseller
			value = 0
		}
		var:manzoni_writer_var = {
			set_variable = manzoni_writer_is_writing_newspaper_novel_var
		}
	}

	current_value = {
		value = var:pbi_var_je_objective_bestseller
	}
	
	goal_add_value = {
		add = 18 #1.5 year
	}

	complete = {
		manzoni_has_writer = yes
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	on_complete = {
		trigger_event = {
			id = pbi_objective_publishing_industry_events.200
			popup = yes
		}
		var:manzoni_writer_var = {
			remove_variable = manzoni_writer_is_writing_newspaper_novel_var
		}
	}

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = pbi_var_je_objective_bestseller
				add = 1
			}
		}
		random_events = {
			10 = 0
			3 = pbi_objective_publishing_industry_events.203 #筆が進まない
			3 = pbi_objective_publishing_industry_events.204 #作家が校閲を認めない
			3 = pbi_objective_publishing_industry_events.205 #内容への批判
			3 = pbi_objective_publishing_industry_events.206 #剽窃疑惑
			3 = pbi_objective_publishing_industry_events.207 #予言的事件で作品が注目される
			3 = pbi_objective_publishing_industry_events.208 #著名人の好意的感想
			3 = pbi_objective_publishing_industry_events.209 #著名人の批判的感想
			3 = pbi_objective_publishing_industry_events.210 #徐々に人気が出る
			3 = pbi_objective_publishing_industry_events.211 #特定地域のポップに人気
			3 = pbi_objective_publishing_industry_events.212 #出版停止
		}
	}

	progressbar = yes

	invalid = {
		OR = {
			manzoni_has_writer = no
			var:manzoni_writer_var = {
				NOT = {
					has_variable = manzoni_writer_is_writing_newspaper_novel_var
				}
			}
		}
	}
	on_invalid = {
		trigger_event = {
			id = pbi_objective_publishing_industry_events.201
		}
	}

	weight = 500
	transferable = yes
}

#Journalists on the Battlefield
pbi_je_objective_from_the_battlefield = {
	should_be_pinned_by_default = no
	icon = "gfx/interface/icons/invention_icons/mandatory_service.dds"

	group = je_group_pbi

	is_shown_when_inactive = {
		has_technology_researched = mass_communication
	}

	possible = {
		has_technology_researched = war_propaganda
	}

	is_progressing = {
		any_scope_building = {
			is_building_type = building_manzoni_publishing_industry
			is_under_construction = no
			cash_reserves_ratio > 0.25
			occupancy >= 0.9
			level >= 5
		}
		any_scope_war = {
			any_scope_front = {
				any_scope_state = {
					NOT = { owner = ROOT }
				}
				any_scope_general = {
					owner = ROOT
				}
			}
		}
	}

	immediate = {
		set_variable = {
			name = pbi_var_je_objective_from_the_battlefield
			value = 0
		}
	}

	current_value = {
		value = var:pbi_var_je_objective_from_the_battlefield
	}
	
	goal_add_value = {
		add = 12 #3 months
	}

	complete = {
		any_scope_building = {
			is_building_type = building_manzoni_publishing_industry
			is_under_construction = no
			cash_reserves_ratio > 0.25
			occupancy >= 0.9
			level >= 5
		}
		any_scope_war = {
			any_scope_front = {
				any_scope_state = {
					NOT = { owner = ROOT }
				}
				any_scope_general = {
					owner = ROOT
				}
			}
		}
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		trigger_event = {
			id = pbi_objective_publishing_industry_events.300
			popup = yes
		}
	}

	on_weekly_pulse = {
		effect = {
			if = {
				limit = {
					has_variable = pbi_var_je_objective_from_the_battlefield
					var:pbi_var_je_objective_from_the_battlefield < 1
				}
				set_variable = {
					name = pbi_var_je_objective_from_the_battlefield
					value = 0
				}
			}
			if = {
				limit = {
					any_scope_building = {
						is_building_type = building_manzoni_publishing_industry
						is_under_construction = no
						cash_reserves_ratio > 0.25
						occupancy >= 0.9
						level >= 10
					}
					any_scope_war = {
						any_scope_front = {
							any_scope_state = {
								NOT = { owner = ROOT }
							}
							any_scope_general = {
								owner = ROOT
							}
						}
					}
				}
				change_variable = {
					name = pbi_var_je_objective_from_the_battlefield
					add = 1
				}
			}
		}
	}

	progressbar = yes

	invalid = {
		var:pbi_var_je_objective_from_the_battlefield > 0
		NOT = {
			any_diplomatic_play = {
				is_war = yes
			}
		}
	}
	on_invalid = {
		set_variable = {
			name = pbi_var_je_objective_from_the_battlefield
			value = 0
		}
	}

	weight = 500
}
