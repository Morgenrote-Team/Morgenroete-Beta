###Archaeological Projects in [ROOT]###

je_lepsius_archaeological_default_project = {
	icon = "gfx/interface/icons/event_icons/event_archaeology.dds"

	group = je_group_lepsius_archaeology

	scripted_progress_bar = je_lepsius_archaeological_default_project_progress_bar

	on_monthly_pulse = {
		effect = {
			#Calculates and executes potential effects
			if = {
				limit = {
					scope:journal_entry = {
						"scripted_bar_progress(je_lepsius_archaeological_default_project_progress_bar)" >= 12
					}
				}

				#Resets Progress Bar
				scope:journal_entry = {
					set_bar_progress = {
						name = je_lepsius_archaeological_default_project_progress_bar
						value = 0
					}
				}

				#Experience for the Archaeologist
				archaeologist_experience_gain_effect = { AMOUNT = 5 }

				#Early Archaeology: +1 Local Artifact
				if = {
					limit = {
						has_technology_researched = lepsius_early_archaeology_tech
					}
					lepsius_add_x_local_artifacts_effect = { AMOUNT = 1 }
				}

				#Modern Archaeology: +1 Local Artifact
				if = {
					limit = {
						has_technology_researched = lepsius_modern_archaeology_tech
					}
					lepsius_add_x_local_artifacts_effect = { AMOUNT = 1 }
				}

				#Local Archaeologist: +1 Local Artifact
				if = {
					limit = {
						var:lepsius_archaeologist_var = {
							has_trait = lepsius_local_specialist
						}
					}
					lepsius_add_x_local_artifacts_effect = { AMOUNT = 1 }
				}

				#Minor Excavation Site: +1 Local Artifact
				if = {
					limit = {
						lepsius_has_minor_excavation_site_trigger = yes
					}
					lepsius_add_x_local_artifacts_effect = { AMOUNT = 1 }
				}

				if = {
					limit = {
						NOT = {
							var:lepsius_archaeologist_var = {
								has_trait = lepsius_local_specialist
							}
						}
					}
					random_list = {
						1 = {
							var:lepsius_archaeologist_var = {
								add_trait = lepsius_local_specialist
							}
						}
						3 = {}
					}
				}

				#Adds Artifacts
				random_list = {
					1 = { #Egyptian
						trigger = {
							any_scope_state = {
								OR = {
									has_state_trait = lepsius_state_trait_lower_egypt
									has_state_trait = lepsius_state_trait_upper_egypt
								}
							}
						}

						lepsius_add_x_egyptian_artifacts_effect = { AMOUNT = 1 }
					}
					1 = { #Mesopotamian
						trigger = {
							any_scope_state = {
								OR = {
									has_state_trait = lepsius_state_trait_babylonia
									has_state_trait = lepsius_state_trait_assyria
								}
							}
						}

						lepsius_add_x_mesopotamian_artifacts_effect = { AMOUNT = 1 }
					}
					1 = { #Aegean
						trigger = {
							any_scope_state = {
								OR = {
									has_state_trait = lepsius_state_trait_argolis
									has_state_trait = lepsius_state_trait_crete
								}
							}
						}

						lepsius_add_x_aegean_artifacts_effect = { AMOUNT = 1 }
					}
					1 = { #Precolumbian
						trigger = {
							any_scope_state = {
								OR = {
									has_state_trait = lepsius_state_trait_yucatan
									has_state_trait = lepsius_state_trait_peru
								}
							}
						}

						lepsius_add_x_precolumbian_artifacts_effect = { AMOUNT = 1 }
					}
					1 = { #Local

						lepsius_add_x_local_artifacts_effect = { AMOUNT = 1 }
					}
				}
			}
		}
	}

	on_yearly_pulse = {
		effect = {
			trigger_event = {
				on_action = lepsius_on_local_major_artifact_events
			}
			trigger_event = {
				on_action = lepsius_on_yearly_pulse_local_archaeology_events
				days = { 30 365 }
			}
		}
	}

	invalid = {
		lepsius_has_archaeologist_no_tooltip_trigger = no
	}

	on_invalid = {
		var:lepsius_archaeologist_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	weight = 30
}

#Decipherment of the Akkadian language, access to Babylonia and Assyria (early archaeology is needed)

je_lepsius_decipherment_akkadian = {
	icon = "gfx/interface/icons/event_icons/event_archaeology.dds"

	group = je_group_lepsius_archaeology

	is_shown_when_inactive = {
		has_technology_researched = lepsius_antiquarianism_tech
		game_date >= 1850.1.1
		NOT = {
			has_global_variable = lepsius_decipherment_akkadian_var
		}
	}

	possible = {
		has_technology_researched = lepsius_early_archaeology_tech
	}

	complete = {
		any_scope_building = {
			is_building_type = building_lepsius_archaeological_museum
		}
		country_has_building_type_levels = {
			target = bt:building_university
			value  >= 10
		}
		lepsius_has_archaeologist = yes
		hidden_trigger = {
			NOT = {
				has_variable = lepsius_akkadian_activator_var
			}
		}
	}

	immediate = {
		set_variable = {
			name = lepsius_akkadian_activator_var
			days = 2
		}
	}

	fail = {
		has_global_variable = lepsius_decipherment_akkadian_var
    }

	on_complete = {
		add_technology_researched = lepsius_specialization_mesopotamia_tech
		add_modifier = {
			name = lepsius_decipherment_akkadian_modifier
		}
		trigger_event = { id = lepsius.4 popup = yes }
	}

	weight = 10

	should_be_pinned_by_default_uninvolved_or_context = yes
}

					}
				}
				25 = {
					trigger_event = {
						id = lepsius.813 #3 clues
					}

						if = {
							limit = {
								var:lepsius_archaeologist_var ?= {
									var:lepsius_archaeologist_experience_var >= 150
								}
							}
							add = 15
						}
						if = {
							limit = {
								var:lepsius_archaeologist_var ?= {
									var:lepsius_archaeologist_experience_var >= 250
								}
							}
							add = 15
						}
						if = {
							limit = {
								var:lepsius_archaeologist_var ?= {
									var:lepsius_archaeologist_experience_var >= 350
								}
							}
							add = 15
						}
						if = {
							limit = {
								has_modifier = lepsius_archaeological_society_modifier

							add = 10

						if = {
							limit = {

							add = 10
						}

			}
		}

		remove_variable = lepsius_archaeological_inspection_active_var

je_lepsius_ishtar_gate_transport = {
	icon = "gfx/interface/icons/event_icons/event_archaeology.dds"

	group = je_group_lepsius_archaeology

	scripted_progress_bar = lepsius_monument_ishtar_progress_bar
	scripted_progress_bar = lepsius_monument_restoration_progress_bar
	scripted_progress_bar = lepsius_monument_reconstruction_progress_bar

	immediate = {
		scope:journal_entry = {
			add_modifier = {
				name = lepsius_ishtar_gate_cost_modifier
				multiplier = root.mr_project_cost_very_small_value
			}
		}
	}

	on_yearly_pulse = {
		effect = {
			if = {
				limit = {
					has_modifier = lepsius_babylonia_excavation_modifier
				}
				random_list = {
					#Normal Transport
					50 = {
						je:je_lepsius_ishtar_gate_transport = {
							add_progress = {
								name = lepsius_monument_ishtar_progress_bar
								value = 20000
							}
						}
						post_notification = lepsius_ishtar_normal_transport_notification
					}
					#Double Transport
					10 = {
						je:je_lepsius_ishtar_gate_transport = {
							add_progress = {
								name = lepsius_monument_ishtar_progress_bar
								value = 40000
							}
						}
						post_notification = lepsius_ishtar_double_transport_notification
					}
					#Half Transport
					10 = {
						je:je_lepsius_ishtar_gate_transport = {
							add_progress = {
								name = lepsius_monument_ishtar_progress_bar
								value = 10000
							}
						}
						post_notification = lepsius_ishtar_half_transport_notification
					}
					#Bad Transport
					5 = {
						je:je_lepsius_ishtar_gate_transport = {
							add_progress = {
								name = lepsius_monument_ishtar_progress_bar
								value = 5000
							}
						}
						post_notification = lepsius_ishtar_bad_transport_notification
					}
				}
			}
			if = {
				limit = {
					NOT = {
						has_variable = lepsius_ishtar_gate_minimum_var
					}
					scope:journal_entry = {
						"scripted_bar_progress(lepsius_monument_ishtar_progress_bar)" >= 100000
					}
				}
				set_variable = lepsius_ishtar_gate_minimum_var
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_lepsius_ishtar_gate_transport_success
			scope:journal_entry = {
				"scripted_bar_progress(lepsius_monument_ishtar_progress_bar)" >= 200000
			}
			scope:journal_entry = {
				"scripted_bar_progress(lepsius_monument_restoration_progress_bar)" >= 75
			}
			scope:journal_entry = {
				"scripted_bar_progress(lepsius_monument_reconstruction_progress_bar)" >= 50
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = lepsius.530
		}
	}

    weight = 10
    transferable = no
}
