######Public Construction####

je_gaudi_public_construction = {
	icon = "gfx/interface/icons/event_icons/event_architecture.dds"

	group = je_group_gaudi_architecture

	immediate = {
		scope:journal_entry = {
			set_variable = {
				name = gaudi_architect_month_counter_var
				value = 0
			}
		}
	}

	on_monthly_pulse = {
		effect = {
			#Removes the Modifier
			scope:journal_entry = {
				if = {
					limit = {
						has_modifier = gaudi_architect_public_construction_modifier
					}
					remove_modifier = gaudi_architect_public_construction_modifier
				}
			}
			#Only if the Architect is not busy
			if = {
				limit = { gaudi_has_active_architect_no_tooltip_trigger = yes }

				#Adds the modifier if the JE doesn't have it yet
				scope:journal_entry = {
					if = {
						limit = {
							NOT = {
								has_modifier = gaudi_architect_public_construction_modifier
							}
						}
						add_modifier = {
							name = gaudi_architect_public_construction_modifier
							multiplier = gaudi_architect_public_construction_value #Multiplies the base construction (1) with the public construction value
						}
					}
				}

				#Changes the counter
				scope:journal_entry = {
					if = {
						limit = {
							NOT = {
								has_variable = gaudi_architect_month_counter_var
							}
						}
						set_variable = {
							name = gaudi_architect_month_counter_var
							value = 0
						}
					}
					change_variable = {
						name = gaudi_architect_month_counter_var
						add = 1
					}
				}

				#If the Counter reaches 12, the architect gets some Reputation
				if = {
					limit = {
						scope:journal_entry = {
							var:gaudi_architect_month_counter_var >= 12
						}
					}

					#Adds Reputation to the Architect
					architect_experience_gain_effect = { AMOUNT = 10 }

					#Resets the Counter
					scope:journal_entry = {
						set_variable = {
							name = gaudi_architect_month_counter_var
							value = 0
						}
					}
				}
			}
		}
	}

	fail = {
		gaudi_has_architect = no
	}

	on_fail = {
		var:gaudi_architect_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	weight = 25
}

je_gaudi_urban_planning_project = {
	icon = "gfx/interface/icons/event_icons/event_architecture.dds"

	group = je_group_gaudi_architecture

	scripted_button = gaudi_stop_architectural_project_button

	scripted_progress_bar = gaudi_urban_planning_project_progress_bar

	on_monthly_pulse = {
		effect = {
			if = {
				#If the Progress reaches the cost, the Modifier will be unlocked in a state
				limit = {
					je:je_gaudi_urban_planning_project = {
						"scripted_bar_progress(gaudi_urban_planning_project_progress_bar)" >= 24
					}
				}
				#Architect gets experience
				architect_experience_gain_effect = { AMOUNT = 20 }
				#The state with the next available urban center will be chosen
				if = {
					limit = {
						scope:target = {
							NOT = { has_modifier = gaudi_urban_building_throughput_add_modifier }
						}
					}
					scope:target = {
						add_modifier = {
							name = gaudi_urban_building_throughput_add_modifier
						}
					}
				}
				else_if = {
					limit = {
						scope:target = {
							NOT = { has_modifier = gaudi_urban_building_production_mult_modifier }
						}
					}
					scope:target = {
						add_modifier = {
							name = gaudi_urban_building_production_mult_modifier
						}
					}
				}
				else = {
					scope:target = {
						add_modifier = {
							name = gaudi_urban_building_goods_input_mult_modifier
						}
					}
				}
				#The cost will be subtracted from the current progress
				je:je_gaudi_urban_planning_project = {
					add_progress = {
						name = gaudi_urban_planning_project_progress_bar
						value = -24
					}
				}
			}
		}
	}

	immediate = {
		set_variable = gaudi_can_stop_active_architectural_project_var #This is used to check if the country has a project already
	}

	complete = {
		custom_tooltip = {
			text = gaudi_building_has_throughput_modifier_tt
			scope:target = {
				has_modifier = gaudi_urban_building_throughput_add_modifier
			}
		}
		custom_tooltip = {
			text = gaudi_building_has_production_modifier_tt
			scope:target = {
				has_modifier = gaudi_urban_building_production_mult_modifier
			}
		}
		custom_tooltip = {
			text = gaudi_building_has_input_modifier_tt
			scope:target = {
				has_modifier = gaudi_urban_building_goods_input_mult_modifier
			}
		}
	}

	on_complete = {
		remove_variable = gaudi_can_stop_active_architectural_project_var #Needs to be removed, when the JE is removed

		var:gaudi_architect_var ?= {
			set_character_busy_and_immortal = no
		}
		trigger_event = {
			id = gaudi.9
			popup = yes
		}
	}

	invalid = {
		custom_tooltip = {
			text = gaudi_stop_architectural_project_tt
			has_variable = gaudi_stop_architectural_project_var
		}
	}

	on_invalid = {
		remove_variable = gaudi_stop_architectural_project_var
		remove_variable = gaudi_can_stop_active_architectural_project_var #Needs to be removed, when the JE is removed
		var:gaudi_architect_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	fail = {
		custom_tooltip = {
			text = je_architect_dies_tt
			NOT = {
				has_variable = gaudi_architect_var
			}
		}
	}

	on_fail = {
		remove_variable = gaudi_stop_architectural_project_var
		remove_variable = gaudi_can_stop_active_architectural_project_var #Needs to be removed, when the JE is removed
		var:gaudi_architect_var ?= {
			set_character_busy_and_immortal = no
		}
	}

	weight = 25

	should_be_pinned_by_default = yes
}
