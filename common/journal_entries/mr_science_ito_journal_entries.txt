######NATIVE ANIMAL######

je_dubois_native_animal = {
	icon = "gfx/interface/icons/event_icons/event_animals.dds"

	group = je_group_dubois_nature

	immediate = {
		set_variable = {
			name = naturalist_native_animal_progress_var
			value = 0
		}
	}

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = naturalist_native_animal_progress_var
				add = {
					value = ito_naturalist_native_animal_progress_value
				}
			}
		}
	}

	current_value = {
		value = var:naturalist_native_animal_progress_var
	}

	goal_add_value = {
		value = 50
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		trigger_event = {
			id = dubois.22
		}
	}

	invalid = {
		custom_tooltip = {
			text = je_naturalist_dies_tt
			NOT = {
				has_variable = dubois_naturalist_var
			}
		}
	}

	on_invalid = {
		remove_ongoing_animal_documentation_effect = yes
	}

	progressbar = yes

	weight = 50
}

######ANIMAL BREEDING######

je_animal_breeding = {

	group = je_group_dubois_nature

	on_yearly_pulse = {
		effect = {
			set_variable = {
				name = dubois_last_animal_polpulation
				value = var:dubois_breeding_animal_amount_var
			}
			set_variable = {
				name = dubois_breeding_animal_amount_var
				value = {
					value = var:dubois_last_animal_polpulation
					multiply = ito_breeding_multiplier
					round = yes
				}
			}
			set_variable = {
				name = dubois_animal_breeding_change_var
				value = {
					value = var:dubois_breeding_animal_amount_var
					subtract = var:dubois_last_animal_polpulation
				}
			}

			post_notification = dubois_animal_breeding_notification
		}
	}

	immediate = {
		set_variable = {
			name = dubois_breeding_animal_amount_var
			value = 20
		}
	}

    fail = {
		custom_tooltip = {
			text = je_breeding_programm_failed_tt
			var:dubois_breeding_animal_amount_var <= 1
		}
    }

	on_fail = {
		trigger_event = {
			id = dubois.1322
			popup = yes
		}
	}

	complete = {
		custom_tooltip = {
			text = je_breeding_programm_succesfull_tt
			scope:journal_entry = {
				is_goal_complete = yes
			}
		}
	}

	on_complete = {
			trigger_event = {
				id = dubois.1321
				popup = yes
			}
		remove_variable = dubois_breeding_animal_amount_var
	}

	current_value = {
		if = {
			limit = {
				scope:journal_entry = {
					goal_value = 0
				}
			}
			value = 0
		}
		else = {
			value = var:dubois_breeding_animal_amount_var #value or variable you use to meassure progress
		}
	}

	goal_add_value = {
		value = 4000
	}

	progressbar = yes

    weight = 200
    should_be_pinned_by_default = yes
}

######Specimen Collection######

je_ito_specimen_collection = {
	icon = "gfx/interface/icons/event_icons/event_botany.dds"

	group = je_group_dubois_nature

	immediate = {
		set_variable = {
			name = ito_estimated_specimen_on_specimen_collection_var
			value = ito_specimen_collection_monthly_specimen_rounded_value
		}
	}

	on_monthly_pulse = {
		effect = {
			#Adds Specimens each month
			if = {
				limit = {
					dubois_has_active_naturalist_no_tooltip = yes
				}

				dubois_add_x_specimen_effect = { AMOUNT = ito_specimen_collection_monthly_specimen_value }
				#Estimated yield
				set_variable = {
					name = ito_estimated_specimen_on_specimen_collection_var
					value = ito_specimen_collection_monthly_specimen_rounded_value
				}
			}
		}
	}

	invalid = {
		custom_tooltip = {
			text = ito_naturalist_is_dead_tt
			dubois_has_naturalist = no
		}
	}

	on_invalid = {
		remove_variable = ito_estimated_specimen_on_specimen_collection_var
	}

	weight = 50
}

######BOTANICAL EXPEDITION######

je_ito_botanical_expedition = {
	icon = "gfx/interface/icons/event_icons/event_botany.dds"

	group = je_group_dubois_nature

	scripted_progress_bar = ito_botanical_expedition_progress_bar
	scripted_progress_bar = ito_botanical_expedition_medicinal_plants_progress_bar

	immediate = {
		set_variable = {
			name = ito_total_specimen_current_botanical_expedition_var
			value = 0
		}
		set_variable = {
			name = ito_estimated_specimen_on_botanical_expedition_var
			value = ito_botanical_expedition_monthly_specimen_rounded_value
		}

		scope:journal_entry = {
			add_modifier = {
				name = dubois_botanical_expedition_cost_modifier
				multiplier = root.mr_project_cost_very_small_value
			}
		}
	}

	on_monthly_pulse = {
		#Effect to add Specimen
		effect = {
			#Adds Specimen each month
			dubois_add_x_specimen_effect = { AMOUNT = ito_botanical_expedition_monthly_specimen_value }
			change_variable = {
				name = ito_total_specimen_current_botanical_expedition_var
				add = ito_botanical_expedition_monthly_specimen_value
			}
			#Estimated yield
			set_variable = {
				name = ito_estimated_specimen_on_botanical_expedition_var
				value = ito_botanical_expedition_monthly_specimen_rounded_value
			}
		}

		#Monthly Events
		random_events = {
			300 = 0

			#Random Botanical Events
			1 = dubois.302 #Dubois 302: Victoria amazonica (Amazonia)
		#	1 = dubois.303 #Dubois 303: Attacus atlas (Indochina)
		#	1 = dubois.305 #dubois 305: Goliathus (Congo)
			3 = dubois.306 #Dubois 306: New species of orchids
			3 = dubois.307 #Dubois 307: Medicinal plants
			3 = dubois.308 #Dubois 308: Beautiful, poisonous flowers
		#	3 = dubois.309 #Dubois 309: IG questions botanical expedition #TODO: Figure a good way out to integrate this
			3 = dubois.311 #Dubois 311: Exotic Seeds
		}
	}

	complete = {
		custom_tooltip = {
			text = je_ito_botanical_expedition_goal_tt
			scope:journal_entry = {
				"scripted_bar_progress(ito_botanical_expedition_progress_bar)" >= 12
			}
		}
	}

	on_complete = {
		var:dubois_naturalist_var ?= {
			set_character_busy_and_immortal = no
		}
		show_as_tooltip = {
			scope:journal_entry = {
				remove_modifier = dubois_botanical_expedition_cost_modifier
			}
		}
		save_scope_value_as = {
			name = medicinal_herbs_scope_value
			value = "je:je_ito_botanical_expedition.scripted_bar_progress(ito_botanical_expedition_medicinal_plants_progress_bar)"
		}

		trigger_event = dubois.390

		remove_variable = ito_estimated_specimen_on_botanical_expedition_var
	}

	invalid = {
		custom_tooltip = {
			text = ito_naturalist_is_dead_tt
			NOT = {
				has_variable = dubois_naturalist_var
			}
		}
	}

	on_invalid = {
		show_as_tooltip = {
			scope:journal_entry = {
				remove_modifier = dubois_botanical_expedition_cost_modifier
			}
		}
		remove_variable = ito_total_specimen_current_botanical_expedition_var
		remove_variable = ito_estimated_specimen_on_botanical_expedition_var
	}

	weight = 50
}

###Improve Forestry###
je_ito_improve_forestry_project = {
	icon = "gfx/interface/icons/goods_icons/wood.dds"

	scripted_progress_bar = ito_naturalist_improve_forestry_project_progress_bar

	scripted_button = dubois_stop_naturalist_project_button
	scripted_button = je_ito_improve_forestry_project_change_list_button

	group = je_group_dubois_nature

	immediate = {
		scope:journal_entry = {
			add_modifier = {
				name = dubois_naturalist_forestry_cost_modifier
				multiplier = root.mr_local_project_cost_very_small_value
			}
			set_variable = {
				name = mr_num_improved_buildings_var
				value = 0
			}
		}
		ordered_in_list = {
			variable = ito_improve_forestry_buildings_chosen_variable_list
			position = 0

			scope:journal_entry = {
				set_variable = {
					name = mr_target_building_var
					value = prev
				}
			}
		}
		mr_journal_entry_save_remaining_list_entries_as_variable_effect = { LIST_NAME = ito_improve_forestry_buildings_chosen_variable_list }
	}

	on_monthly_pulse = {
		effect = {
			mr_journal_entry_building_3_modifier_project_monthly_effect = {
				PROGRESS_BAR_NAME = ito_naturalist_improve_forestry_project_progress_bar
				PROGRESS_COST = 24
				MODIFIER_1 = dubois_building_throughput_add_modifier
				MODIFIER_2 = dubois_building_production_mult_modifier
				MODIFIER_3 = dubois_building_goods_input_mult_modifier
				BUILDING_LIST_NAME = ito_improve_forestry_buildings_chosen_variable_list
				ROLE = naturalist
			}
		}
	}

	complete = {
		custom_tooltip = {
			text = je_improve_building_project_complete_tt
			NOT = {
				has_variable_list = ito_improve_forestry_buildings_chosen_variable_list
			}
		}
	}

	on_complete = {
		remove_variable = dubois_stop_naturalist_improvement_project_var
		show_as_tooltip = {
			scope:journal_entry = {
				remove_modifier = dubois_naturalist_forestry_cost_modifier
			}
		}
		clear_variable_list = ito_improve_forestry_buildings_chosen_variable_list

		ito_set_naturalist_unbusy_effect = yes
		if = {
			limit = { ito_is_naturalist_automated_trigger = no }
			trigger_event = {
				id = dubois.52
				popup = yes
			}
		}
		set_variable = {
			name = mr_num_improved_forestry_buildings_var
			value = scope:journal_entry.var:mr_num_improved_buildings_var
		}
	}

	invalid = {
		custom_tooltip = {
			text = dubois_stop_naturalist_improvement_project_tt
			OR = {
				has_variable = dubois_stop_naturalist_improvement_project_var
				ito_has_naturalist_no_tooltip_trigger = no
			}
		}
	}

	on_invalid = {
		remove_variable = dubois_stop_naturalist_improvement_project_var
		show_as_tooltip = {
			je:je_ito_improve_forestry_project = {
				remove_modifier = dubois_naturalist_forestry_cost_modifier
			}
		}
		clear_variable_list = ito_improve_forestry_buildings_chosen_variable_list
		theiler_set_biologist_unbusy_effect = yes
	}

	weight = 30

	should_be_pinned_by_default = no
}

je_dubois_safari_hippopotamus = {
	icon = "gfx/interface/icons/event_icons/event_animals.dds"

	group = je_group_dubois_nature

	scripted_progress_bar = dubois_safari_trap_progress_bar
	scripted_progress_bar = dubois_safari_track_progress_bar

	on_monthly_pulse = {
		events = {
			dubois.38 #Dubois 38: Safari Extras
		}
		random_events = {
			2 = 0
			4 = dubois.32 #Standard event: Trap or Follow
			1 = dubois.33 #Rare Flowers
			1 = dubois.34 #Unknown Species
		}
	}

	immediate = {
		var:dubois_naturalist_var = {
			save_scope_as = dubois_naturalist_scope
		}
	}

	complete = {
		custom_tooltip = {
			text = je_dubois_safari_3_traps_tt
			je:je_dubois_safari_hippopotamus = {
				"scripted_bar_progress(dubois_safari_trap_progress_bar)" >= 3
			}
		}
		custom_tooltip = {
			text = je_dubois_safari_1_track_tt
			je:je_dubois_safari_hippopotamus = {
				"scripted_bar_progress(dubois_safari_track_progress_bar)" >= 1
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = dubois.39
			popup = yes
		}
	}

	fail = {
		custom_tooltip = {
			text = je_dubois_safari_fail_tt
			has_variable = dubois_safari_accident_var
		}
	}

	on_fail = {
		trigger_event = {
			id = dubois.37
		}
	}

	invalid = {
		custom_tooltip = {
			text = je_naturalist_dies_tt
			NOT = {
				has_variable = dubois_naturalist_var
			}
		}
		has_zoo_animal_hippopotamus_trigger = yes
	}

	on_invalid = {
		custom_tooltip = {
			text = je_dubois_safari_invalid_tt
			remove_modifier = dubois_on_safari_modifier
			add_modifier = {
				name = dubois_had_safari_modifier
				years = 3
			}
			dubois_remove_safari_effect = yes
		}
		if = {
			limit = { ito_has_naturalist_no_tooltip_trigger = yes }
			var:dubois_naturalist_var = {
				set_character_busy_and_immortal = no
				remove_variable = dubois_naturalist_on_safari_var
				remove_modifier = dubois_animal_hunter_modifier
			}
		}
	}

	should_be_pinned_by_default = yes

	weight = 35

	timeout = 300

	on_timeout = {
		trigger_event = {
			id = dubois.40
			popup = yes
		}
	}
}
