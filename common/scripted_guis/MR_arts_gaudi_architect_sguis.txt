#MR_ARCHITECT_MODERN_SPLENDOUR_BUTTON
mr_architect_modern_splendour_button_sgui = {
    scope = country
	is_shown = {
		NOR = {
			has_journal_entry = je_gaudi_splendour_bauhaus
			has_journal_entry = je_gaudi_splendour_neoclassical
			has_journal_entry = je_gaudi_splendour_art_deco
		}
		NOR = {
			has_modifier = gaudi_bauhaus_splendour_modifier
			has_modifier = gaudi_neoclassical_splendour_modifier
			has_modifier = gaudi_art_deco_splendour_modifier
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_ARCHITECT_ACTIVE_TRIGGER_TT
			gaudi_has_active_architect = yes
		}
		custom_tooltip = {
			text = MR_MODERN_ARCHITECTURE_SCHOOL_TRIGGER_TT
			OR = {
				has_modifier = gaudi_bauhaus_modifier
				has_modifier = gaudi_neoclassicism_modifier
				has_modifier = gaudi_art_deco_modifier
			}
		}
    }
    effect = {
		if = {
			limit = {
				has_modifier = gaudi_bauhaus_modifier
			}
			add_journal_entry = {
				type = je_gaudi_splendour_bauhaus
			}
		}
		if = {
			limit = {
				has_modifier = gaudi_neoclassicism_modifier
			}
			add_journal_entry = {
				type = je_gaudi_splendour_neoclassical
			}
		}
		if = {
			limit = {
				has_modifier = gaudi_art_deco_modifier
			}
			add_journal_entry = {
				type = je_gaudi_splendour_art_deco
			}
		}
		if = {
			limit = {
				has_variable = gaudi_architect_var
			}
			var:gaudi_architect_var = {
				set_character_busy_and_immortal = yes
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 3 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
mr_architect_modern_splendour_active_sgui = {
	scope = country

	is_shown = {
		OR = {
			has_journal_entry = je_gaudi_splendour_bauhaus
			has_journal_entry = je_gaudi_splendour_neoclassical
			has_journal_entry = je_gaudi_splendour_art_deco
		}
	}
}

#MR_ARCHITECT_TREATISE_BUTTON
mr_architect_treatise_button_sgui = {
    scope = country
	is_shown = {
		AND = {
			has_variable = gaudi_architect_var
			var:gaudi_architect_var = {
				NOT = {
					has_variable = gaudi_architect_achievement_treatise_var
				}
			}
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_ARCHITECT_ACTIVE_TRIGGER_TT
			gaudi_has_active_architect = yes
		}
		custom_tooltip = {
			text = MR_ARCHITECT_NO_TREATISE_TRIGGER_TT
			has_variable = gaudi_architect_var
			var:gaudi_architect_var = {
				NOT = {
					has_variable = gaudi_architect_achievement_treatise_var
				}
			}
		}
		custom_tooltip = {
			text = MR_ARCHITECT_TREATISE_EXPERIENCE_TRIGGER_TT
			has_variable = gaudi_architect_var
			var:gaudi_architect_var = {
				var:gaudi_architect_experience >= 250
			}
		}
		trigger_if = {
			limit = {
				has_variable = gaudi_architecture_treatise_var
			}
			custom_tooltip = {
				text = MR_BETTER_TREATISE_TRIGGER_TT
				has_variable = gaudi_architect_var
				gaudi_architecture_treatise_value > var:gaudi_architecture_treatise_var
			}
		}
    }
    effect = {
		add_journal_entry = {
			type = je_gaudi_architecture_treatise
		}
		if = {
			limit = {
				NOT = {
					has_variable = gaudi_architecture_treatise_var
				}
			}
			set_variable = {
				name = gaudi_architecture_treatise_var
				value = 0
			}
		}
		if = {
			limit = {
				has_variable = gaudi_architect_var
			}
			var:gaudi_architect_var = {
				set_character_busy_and_immortal = yes
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 3 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
mr_architect_treatise_button_active_sgui = {
	scope = country

	is_shown = {
		has_journal_entry = je_gaudi_architecture_treatise
	}
}

#MR_ARCHITECT_CAPITAL_EXPANSION_BUTTON
mr_architect_capital_expansion_button_sgui = {
    scope = country
	is_shown = {
		NOT = {
			capital = {
				has_modifier = gaudi_capital_expansion_level_4_modifier
			}
		}
	}
    is_valid = {
		hidden_trigger = {
			NOR = {
				gaudi_has_active_architect = no
				any_scope_state = {
					is_capital = yes
					has_modifier = gaudi_capital_expansion_level_4_modifier
				}
			}
		}
		custom_tooltip = {
			text = MR_ARCHITECT_ACTIVE_TRIGGER_TT
			gaudi_has_active_architect = yes
		}
		#Level 1 Trigger: Traditional Architect
		mr_gui_has_urban_planning_tech_trigger = yes
		mr_gui_has_classicism_tech_trigger = yes
		#Level 2 Trigger: Historicist Architect
		trigger_if = {
			limit = {
				capital = {
					has_modifier = gaudi_capital_expansion_level_1_modifier
				}
			}
			custom_tooltip = {
				text = MR_HISTORICIST_ARCHITECT_TRIGGER_TT
				has_variable = gaudi_architect_var
				var:gaudi_architect_var = {
					OR = {
						has_trait = gaudi_historicism_style
						has_trait = gaudi_bauhaus_style
						has_trait = gaudi_neoclassical_style
						has_trait = gaudi_art_deco_style
					}
				}
			}
			mr_gui_has_modern_sewerage_tech_trigger = yes
			mr_gui_has_romanticism_tech_trigger = yes
		}
		#Level 3 Trigger: Historicist Architect
		trigger_if = {
			limit = {
				capital = {
					has_modifier = gaudi_capital_expansion_level_2_modifier
				}
			}
			custom_tooltip = {
				text = MR_HISTORICIST_ARCHITECT_TRIGGER_TT
				has_variable = gaudi_architect_var
				var:gaudi_architect_var = {
					OR = {
						has_trait = gaudi_historicism_style
						has_trait = gaudi_bauhaus_style
						has_trait = gaudi_neoclassical_style
						has_trait = gaudi_art_deco_style
					}
				}
			}
			mr_gui_has_steel_frame_buildings_tech_trigger = yes
			mr_gui_has_realism_tech_trigger = yes
		}
		#Level 4 Trigger: Modern Architect
		trigger_if = {
			limit = {
				capital = {
					has_modifier = gaudi_capital_expansion_level_3_modifier
				}
			}
			custom_tooltip = {
				text = MR_MODERN_ARCHITECT_TRIGGER_TT
				has_variable = gaudi_architect_var
				var:gaudi_architect_var = {
					OR = {
						has_trait = gaudi_bauhaus_style
						has_trait = gaudi_neoclassical_style
						has_trait = gaudi_art_deco_style
					}
				}
			}
			mr_gui_has_elevator_tech_trigger = yes
			OR = {
				mr_gui_has_art_deco_tech_trigger = yes
				mr_gui_has_neoclassicism_tech_trigger = yes
				mr_gui_has_new_objectivity_tech_trigger = yes
			}
		}
    }
    effect = {
		show_as_tooltip = {
			add_modifier = {
				name = gaudi_capital_expansion_cost_modifier
				multiplier = mr_local_project_cost_very_small_value
			}
		}
		add_journal_entry = { type = je_gaudi_capital_expansion_project }
		if = {
			limit = {
				has_variable = gaudi_architect_var
			}
			var:gaudi_architect_var = {
				set_character_busy_and_immortal = yes
			}
		}
    }
}

    scope = country
	is_shown = { #Should be shown if the project is active
		has_journal_entry = je_gaudi_capital_expansion_project

#MR_ARCHITECT_INVITE_INDEPENDENT_ARCHITECT
mr_invite_architect_button_sgui = {
	#!!!If you change any Trigger or effects here, change them in the AI version too (gaudi_ai_hire_architect_effect)!!!
    scope = country
	saved_scopes = {
        chosen_architect
    }
    is_valid = {
		#Has Urban Planning Technology researched
		mr_gui_has_urban_planning_tech_trigger = yes

		#not hired an independent architect recently
		custom_tooltip = {
			text = MR_ARCHITECT_NOT_HIRED_INDEPENDENT_ARCHITECT_RECENTLY_TRIGGER_TT
			NOT = {
				has_modifier = gaudi_recently_hired_independent_architect_modifier
			}
		}

		#This Architect was not recently sent away
		custom_tooltip = {
			text = MR_ARCHITECT_WAS_NOT_RECENTLY_SENT_AWAY_TT
			NOT = {
				scope:chosen_architect = {
					AND = {
						has_variable = gaudi_architect_unavailable_for_this_var
						var:gaudi_architect_unavailable_for_this_var = root
					}
				}
			}
		}
    }
    effect = {
		custom_tooltip = {
			text = MR_ARCHITECT_INVITE_INDEPENDENT_ARCHITECT_EFFECT_TT

			set_architect_independent_architect_effect = yes
			set_variable = {
				name = gaudi_architect_var
				value = scope:chosen_architect
			}

			if = { #Used for error prevention
				limit = { has_variable = gaudi_architect_var }
				var:gaudi_architect_var = {
					transfer_character = root
					add_modifier = artists_character_is_official_architect_modifier
					remove_list_global_variable = {
						name = mr_independent_architect_global_list
						target = this
					}
				}
			}

			if = {
				limit = {
					NOT = { has_journal_entry = je_gaudi_public_construction }
				}
				add_journal_entry = {
					type = je_gaudi_public_construction
				}
			}
		}

		add_modifier = {
			name = gaudi_recently_hired_independent_architect_modifier
			years = 3
		}
		add_treasury = scope:chosen_architect.gaudi_independent_architect_cost_value
    }
}
