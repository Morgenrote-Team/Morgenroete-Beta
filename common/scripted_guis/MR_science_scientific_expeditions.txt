######ARCTIC EXPEDITION######
mr_arctica_expedition_button_sgui = {
    scope = country

    is_valid = {
		mr_gui_has_zeppelins_tech_trigger = yes
		custom_tooltip = {
			text = MR_ZEPPELIN_OBSERVATORY_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_verrier_observatory
				OR = {
					has_active_production_method = pm_curtiss_weather_station
					has_active_production_method = pm_curtiss_modern_weather_station
				}
			}
		}
		custom_tooltip = {
			text = MR_ZEPPELIN_PILOT_TRIGGER_TT
			has_variable = curtiss_pilot_var
			var:curtiss_pilot_var ?= {
				is_busy = no
			}
		}
		custom_tooltip = {
			text = MR_ZEPPELIN_RECENT_EXPEDITION_TRIGGER_TT
			NOR = {
				has_modifier = curtiss_arctica_expedition_modifier
				has_modifier = curtiss_had_arctica_expedition_modifier
			}
		}
    }
    effect = {
		set_variable = {
			name = curtiss_zeppelin_progress
			value = 0
		}
		set_variable = {
			name = curtiss_zeppelin_science
			value = 0
		}
		set_variable = {
			name = curtiss_zeppelin_danger
			value = 0
		}
		add_modifier = {
			name = curtiss_arctica_expedition_modifier
		}
		add_modifier = {
			name = curtiss_zeppelin_expedition_cost_modifier
			multiplier = mr_project_cost_very_small_value
		}
		trigger_event = {
			id = curtiss.520
			popup = yes
		}
    }
}
mr_arctica_expedition_active_sgui = {
    scope = country

    is_shown = {
		has_modifier = curtiss_arctica_expedition_modifier
    }
}

#MR_TRANSATLANTIC_CABLE_EXPEDITION_BUTTON_TT
mr_transatlantic_cable_expedition_button_sgui = {
	scope = country
	is_shown = {
		NOT = {
			has_global_variable = tesla_transatlantic_cable_country_global_var
		}
	}
	is_valid = {
		mr_gui_has_screw_frigate_tech_trigger = yes
		trigger_if = {
			limit = {
				is_player = yes
			}
			tesla_has_active_engineer_trigger = yes
		}
		#AI normally isn't smart enough to "unbusy" their engineer for this project, so it may start it even with a busy engineer
		trigger_if = {
			limit = {
				is_ai = yes
			}
			has_variable = tesla_engineer_var
		}
		custom_tooltip = {
			text = MR_TRANSATLANTIC_CABLE_HAS_TELEGRAPH_COMPANY_TRIGGER_TT
			any_company = {
				is_company_type = company_type:mr_basic_telegraphy
			}
		}
		custom_tooltip = {
			text = MR_TRANSATLANTIC_CABLE_INVESTMENT_POOL_TRIGGER_TT
			investment_pool >= 5000000
		}
		custom_tooltip = {
			text = MR_TRANSATLANTIC_CABLE_EXPEDITION_BUTTON_NO_TRANSATLANTIC_RADIO_TRIGGER_TT
			NOT = {
				has_global_variable = gaudi_transatlantic_radio_var
			}
		}
		custom_tooltip = {
			text = MR_TRANSATLANTIC_CABLE_EXPEDITION_BUTTON_NO_RECENT_FAIL_TRIGGER_TT
			NOT = {
				has_modifier = tesla_transatlantic_cable_expedition_failed_modifier
			}
		}
		custom_tooltip = {
			text = MR_TRANSATLANTIC_CABLE_EXPEDITION_BUTTON_NOT_ONGOING_TRIGGER_TT
			NOT = { has_journal_entry = je_tesla_transatlantic_cable }
		}
	}
	effect = {
		if = {
			limit = {
				is_player = yes
				tesla_has_active_engineer_trigger = yes
			}
			var:tesla_engineer_var = {
				set_character_busy_and_immortal = yes
			}
		}
		add_journal_entry = {
			type = je_tesla_transatlantic_cable
		}
		trigger_event = {
			id = tesla.3001
			popup = yes
		}
	}
}
mr_transatlantic_cable_expedition_active_sgui = {
    scope = country

    is_shown = {
		has_journal_entry = je_tesla_transatlantic_cable
    }
}

#MR_VENUS_TRANSIT_1874_EXPEDITION_BUTTON_TT
mr_venus_transit_1874_expedition_button_sgui = {
	scope = country
	is_shown = {
		NOT = {
			has_global_variable = verrier_venus_transit_1874_happened_global_var
		}
	}
	is_valid = {
		custom_tooltip = {
			text = MR_VENUS_YEAR_1872_TRIGGER_TT
			game_date > 1872.1.1
			NOT = {
				game_date >= 1874.1.1
			}
		}
		mr_gui_has_modern_astronomy_tech_trigger = yes
		mr_gui_has_camera_tech_trigger = yes
		custom_tooltip = {
			text = MR_ASTEROID_CONGRESS_OBSERVATORY_BUILDING_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_verrier_observatory
				level > 0
			}
		}
		custom_tooltip = {
			text = MR_ASTEROID_CONGRESS_ASTRONOMER_TRIGGER_TT
			verrier_has_astronomer = yes
		}
		custom_tooltip = {
			text = MR_VENUS_TRANSIT_POSSIBLE_1874_TRIGGER_TT
			OR = {
				any_scope_state = {
					is_incorporated = yes
					verrier_venus_transit_1874_region = yes
				}
				any_country = {
					is_subject_of = ROOT
					any_scope_state = {
						verrier_venus_transit_1874_region = yes
					}
				}
				any_country = {
					has_diplomatic_relevance = ROOT
					NOT = {
						has_modifier = verrier_foreign_venus_expedition_modifier
					}
					any_scope_state = {
						is_incorporated = yes
						verrier_venus_transit_1874_region = yes
					}
					relations:root >= relations_threshold:cordial
					is_subject = no
					is_revolutionary = no
					is_at_war = no
				}
			}
		}
		hidden_trigger = {
			NOR = {
				has_variable = verrier_venus_transit_select_var
				has_modifier = verrier_venus_transit_preparation_modifier
			}
		}
	}
	effect = {
		set_variable = verrier_venus_transit_select_var
		trigger_event = {
			id = verrier.311
			popup = yes
		}
	}
}
