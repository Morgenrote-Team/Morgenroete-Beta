#MR_PADDOCK_IMPROVE_ENGINE_BUTTON
mr_paddock_improve_engine_button_sgui = {
    scope = country
    is_valid = {
		hidden_trigger = {
			has_variable = vikelas_engine_level_var
			NOR = {
				var:vikelas_engine_level_var = 4
				vikelas_mechanic_improves_race_car_trigger = yes
			}
		}
		custom_tooltip = {
			text = MR_MECHANIC_TRIGGER
			has_variable = vikelas_mechanic_var
		}
		#Level 2 Trigger
		trigger_if = {
			limit = { var:vikelas_engine_level_var = 1 }
			custom_tooltip = {
				text = MR_PADDOCK_50_MECHANIC_INNOVATION_TRIGGER_TT
				var:vikelas_mechanical_innovation_var >= 50
			}
		}
		#Level 3 Trigger
		trigger_if = {
			limit = { var:vikelas_engine_level_var = 2 }
			custom_tooltip = {
				text = MR_PADDOCK_MOTOR_INDUSTRY_LEVEL_5_TRIGGER_TT
				country_has_building_type_levels = {  
					target = bt:building_motor_industry  
					value  >= 5
				}
			}
			custom_tooltip = {
				text = MR_PADDOCK_100_MECHANIC_INNOVATION_TRIGGER_TT
				var:vikelas_mechanical_innovation_var >= 100
			}
		}
		#Level 4 Trigger
		trigger_if = {
			limit = { var:vikelas_engine_level_var = 3 }
			mr_gui_has_compression_ignition_tech_trigger = yes
			custom_tooltip = {
				text = MR_PADDOCK_MOTOR_INDUSTRY_LEVEL_10_TRIGGER_TT
				country_has_building_type_levels = {  
					target = bt:building_motor_industry  
					value  >= 10
				}
			}
			custom_tooltip = {
				text = MR_PADDOCK_150_MECHANIC_INNOVATION_TRIGGER_TT
				var:vikelas_mechanical_innovation_var >= 150
			}
		}
    }
    effect = {
		add_journal_entry = { type = je_vikelas_mechanic_engine }
		if = {
			limit = {
				var:vikelas_engine_level_var = 3
			}
			vikelas_add_x_mechanical_innovation_effect = { 
				AMOUNT = -150
			}
		}
		else_if = {
			limit = {
				var:vikelas_engine_level_var = 2
			}
			vikelas_add_x_mechanical_innovation_effect = { 
				AMOUNT = -100
			}
		}
		else_if = {
			limit = {
				var:vikelas_engine_level_var = 1
			}
			vikelas_add_x_mechanical_innovation_effect = { 
				AMOUNT = -50
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
		modifier = {
			trigger = {
				vikelas_autorace_racer_speed_value < vikelas_autorace_racer_reliability_value
			}
			add = -100 #The AI will try to update the military medicine first, because the traits are rare
		}
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}

#MR_PADDOCK_FIRE_RACER_BUTTON
mr_paddock_fire_racer_button_sgui = {
    scope = country
	is_shown = {
		always = yes
	}
    is_valid = {
		custom_tooltip = {
			text = MR_HAS_RACER_TT
			has_variable = vikelas_racer_var
		}
		custom_tooltip = {
			text = MR_NOT_HAS_NEW_RACER_TT
			has_modifier = vikelas_new_racer_modifier
		}
    }
    effect = {
		kill_racer_effect = yes
		add_modifier = {
			name = vikelas_new_racer_modifier
			years = 2
		}
		trigger_event = {
			id = vikelas.595
		}
    }
}

#MR_PADDOCK_INNOVATION_TO_RESEARCH_BUTTON
mr_paddock_innovation_to_research_button_sgui = {
    scope = country
	is_shown = {
		always = yes
	}
    is_valid = {
		custom_tooltip = {
			text = MR_PADDOCK_150_MECHANIC_INNOVATION_TRIGGER_TT
			trigger_if = {
				limit = { has_variable = vikelas_mechanical_innovation_var }
				var:vikelas_mechanical_innovation_var >= 150
			}
			trigger_else = { always = no }
		}
    }
    effect = {
		vikelas_add_x_mechanical_innovation_effect = { 
			AMOUNT = -150
		}
		vikelas_convert_mechanical_innovation_to_research_effect = yes
    }
}
