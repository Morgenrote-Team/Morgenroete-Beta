#Transfer from general list to prioritized list
mr_transfer_from_general_list_to_prioritized_list_topographic_project_button_sgui = {
	scope = country
	effect = {
		remove_list_variable = {
			name = states_for_topographic_improvement_variable_list
			target = scope:chosen_state
		}
		add_to_variable_list = {
			name = prioritized_states_for_topographic_improvement_variable_list
			target = scope:chosen_state
		}
	}
}

#Sort by Incorporation Status
mr_sort_topographic_state_by_status_button_sgui = {
	scope = country
	effect = {
		ordered_in_list = {
			variable = states_for_topographic_improvement_variable_list
			limit = {
				is_incorporated = yes
			}
			order_by = gaudi_infrastructure_balance
			max = 1000
			check_range_bounds = no
	
			root = {
				add_to_variable_list = {
					name = placeholder_list
					target = prev
				}
			}
		}
		ordered_in_list = {
			variable = states_for_topographic_improvement_variable_list
			limit = {
				is_incorporated = no
			}
			order_by = gaudi_infrastructure_balance
			max = 1000
			check_range_bounds = no
	
			root = {
				add_to_variable_list = {
					name = placeholder_list
					target = prev
				}
			}
		}
		clear_variable_list = states_for_topographic_improvement_variable_list
		every_in_list = {
			variable = placeholder_list
	
			root = {
				add_to_variable_list = {
					name = states_for_topographic_improvement_variable_list
					target = prev
				}
			}
		}
		clear_variable_list = placeholder_list
	}
}
