#General Transfer from general list to prioritized list
mr_transfer_from_general_list_to_prioritized_list_button_sgui = {
	scope = country
	effect = {
		chosen_state = {
			prev = {
				remove_list_variable = {
					name = states_for_improvement_variable_list
					target = prev
				}
				add_to_variable_list = {
					name = prioritized_states_for_improvement_variable_list
					target = prev
				}
			}
		}
	}
}

#General Transfer from prioritized list to general list
mr_transfer_from_prioritized_list_to_general_list_button_sgui = {

}
