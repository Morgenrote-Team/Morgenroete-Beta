mr_engineer_improve_construction_window_start_button_sgui = {
	scope = country
	is_shown = {
		NOT = { has_journal_entry = je_gaudi_construction_sector_project }
	}
	is_valid = {
		custom_tooltip = {
			text = MR_HAS_ANY_CHOSEN_BUILDINGS_TRIGGER_TT
			has_variable_list = tesla_improve_construction_buildings_chosen_variable_list
		}
	}
	effect = {
		remove_variable = gaudi_improve_construction_sector_building_choser_var

		add_journal_entry = { type = je_gaudi_construction_sector_project }
		show_as_tooltip = {
			add_modifier = {
				name = tesla_small_engineering_project_cost_modifier
				multiplier = mr_local_project_cost_small_value
			}
		}
		var:tesla_engineer_var ?= {
			set_character_busy_and_immortal = yes
		}
		clear_variable_list  = tesla_improve_construction_sector_building_buildings_variable_list
	}
}
