		clear_variable_list  = theiler_improve_agriculture_buildings_chosen_variable_list
	}
}

mr_biologist_improve_agriculture_window_start_button_sgui = {
	scope = country
	is_shown = {
		NOT = { has_journal_entry = je_theiler_improve_agriculture_project }
	}
	is_valid = {
		custom_tooltip = {
			text = MR_HAS_ANY_CHOSEN_BUILDINGS_TRIGGER_TT
			has_variable_list = theiler_improve_agriculture_buildings_chosen_variable_list
		}
	}
	effect = {
		remove_variable = theiler_improve_agriculture_project_building_chooser_var

		add_journal_entry = { type = je_theiler_improve_agriculture_project }
		show_as_tooltip = {
			add_modifier = {
				name = theiler_improve_agriculture_cost_modifier
				multiplier = mr_local_project_cost_small_value
			}
		}
		var:theiler_biologist_var ?= {
			set_character_busy_and_immortal = yes
		}
		clear_variable_list  = theiler_improve_agriculture_buildings_variable_list
	}
}

mr_biologist_improve_agriculture_window_change_chosen_buildings_button_sgui = {
	scope = country

			has_variable_list = theiler_improve_agriculture_buildings_chosen_variable_list

		remove_variable = theiler_improve_agriculture_project_building_chooser_var
		clear_variable_list  = theiler_improve_agriculture_buildings_variable_list

	scope = country
	effect = {
		clear_variable_list = mendelejew_improve_chemical_plants_buildings_variable_list
		ordered_scope_building = {
			limit = {
				level > 0
				is_building_type = building_chemical_plant
				NAND = {
					has_modifier = mendelejew_building_throughput_add_modifier
					has_modifier = mendelejew_building_production_mult_modifier
					has_modifier = mendelejew_building_goods_input_mult_modifier

				add_to_variable_list = {
