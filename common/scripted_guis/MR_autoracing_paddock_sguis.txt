#MR_PADDOCK_IMPROVE_ENGINE_BUTTON
mr_paddock_improve_engine_button = {
    scope = country
    is_valid = {
		hidden_trigger = {
			NOT = {
				has_variable = vikelas_engine_level_4
			}
		}
		custom_tooltip = {
			text = MR_MECHANIC_TRIGGER
			has_variable = vikelas_mechanic_var
		}
		#Level 2 Trigger
		trigger_if = {
			limit = { has_variable = vikelas_engine_level_1 }
			custom_tooltip = {
				text = MR_PADDOCK_50_MECHANIC_INNOVATION_TRIGGER_TT
				var:vikelas_mechanical_innovation_var >= 50
			}
		}
		#Level 3 Trigger
		trigger_if = {
			limit = { has_variable = vikelas_engine_level_2 }
			custom_tooltip = {
				text = MR_PADDOCK_MOTOR_INDUSTRY_LEVEL_5_TRIGGER_TT
				country_has_building_type_levels = {  
					target = bt:building_motor_industry  
					value  >= 5
				}
			}
			custom_tooltip = {
				text = MR_PADDOCK_100_MECHANIC_INNOVATION_TRIGGER_TT
				var:vikelas_mechanical_innovation_var >= 100
			}
		}
		#Level 4 Trigger
		trigger_if = {
			limit = { has_variable = vikelas_engine_level_3 }
			mr_gui_has_compression_ignition_tech_trigger = yes
			custom_tooltip = {
				text = MR_PADDOCK_MOTOR_INDUSTRY_LEVEL_10_TRIGGER_TT
				country_has_building_type_levels = {  
					target = bt:building_motor_industry  
					value  >= 10
				}
			}
			custom_tooltip = {
				text = MR_PADDOCK_150_MECHANIC_INNOVATION_TRIGGER_TT
				var:vikelas_mechanical_innovation_var >= 150
			}
		}
    }
    effect = {
		#add_journal_entry = { type = je_vikelas_mechanic_engine }
		if = {
			limit = {
				has_variable = vikelas_engine_level_3
			}
			vikelas_add_x_mechanical_innovation_effect = { 
				AMOUNT = -150
			}
		}
		else_if = {
			limit = {
				has_variable = vikelas_engine_level_2
			}
			vikelas_add_x_mechanical_innovation_effect = { 
				AMOUNT = -100
			}
		}
		else_if = {
			limit = {
				has_variable = vikelas_engine_level_1
			}
			vikelas_add_x_mechanical_innovation_effect = { 
				AMOUNT = -50
			}
		}
    }
}
