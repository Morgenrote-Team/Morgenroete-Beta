#MR_CHANGE_TO_CIVIL_EGINEER_BUTTON_TT
gaudi_change_to_civil_engineer_button_sgui = {
	scope = country
	is_valid = {
		custom_tooltip = {
			text = MR_HAS_ENGINEER_TT
			tesla_has_engineer = yes
		}
		custom_tooltip = {
			text = MR_CHANGE_TO_CIVIL_EGINEER_BUTTON_HAS_NO_CIVIL_ENGINEER_TT
			tesla_has_no_civil_engineer_trigger = yes
		}
		
		custom_tooltip = {
			text = MR_CHANGE_TO_CIVIL_EGINEER_BUTTON_NO_RECENT_SPECIALIZATION_SWITCH_TT
			var:tesla_engineer_var ?= {
				NOT = { has_modifier = academics_engineer_recent_specialization_switch_modifier }
			}
		}
	}
	effect = {
		gaudi_set_civil_engineer_effect = yes
		if = {
			limit = {
				has_variable = tesla_engineer_var
			}
			var:tesla_engineer_var = {
				add_modifier = {
					name = academics_engineer_recent_specialization_switch_modifier
					years = 5
				}
			}
		}
	}
}

#Civil Engineer Railway project
mr_civil_engineer_railway_button = {
	scope = country
	is_valid = {
		tesla_has_civil_engineer_trigger = yes
		tesla_has_active_engineer_trigger = yes
		custom_tooltip = {
			text = MR_CIVIL_ENGINEER_RAILWAY_TRIGGER_TT
			any_scope_building = {
				tesla_railway_valid_for_improvement_trigger = yes
			}
		}
	}
	effect = {
		clear_variable_list = states_for_railway_improvement_variable_list
		ordered_scope_state = {
			limit = {
				any_scope_building = {
					tesla_railway_valid_for_improvement_trigger = yes
				}
			}
			order_by = tesla_infrastructure_balance
			max = 1000
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = states_for_railway_improvement_variable_list
					target = prev
				}
			}
		}
		set_variable = mr_railway_state_prioritizing_window_active_var
		show_as_tooltip = {
			add_modifier = {
				name = gaudi_engineer_project_medium_cost_modifier
				multiplier = money_amount_multiplier_small
			}
		}
	}
}

	scope = country
	is_valid = {
		tesla_has_aeronautical_engineer_trigger = yes
		custom_tooltip = {
			text = MR_NO_TESTPLANE_MODIFIER_TT
			NOT = {
				has_modifier = curtiss_testplane_cost_modifier
			}
		}
		trigger_if = {
			limit = {
				is_ai = yes
			}

			curtiss_has_pilot = yes
			NOT = {
				has_modifier = curtiss_airplane_speed_record_modifier
			}
			mr_gui_has_combustion_engine_tech_trigger = yes
			mr_gui_has_military_aviation_tech_trigger = yes
			curtiss_has_aeroplane_production_size_x_tooltip_trigger = { LEVEL = 1 }
		}
		trigger_else = {
			tesla_has_active_engineer_trigger = yes
			curtiss_has_pilot = yes
			custom_tooltip = {
				text = MR_AERONAUTICAL_ENGINEER_NO_RECORD_TECH_TRIGGER_TT
				NOT = {
					has_modifier = curtiss_airplane_speed_record_modifier
				}
			}
			mr_gui_has_combustion_engine_tech_trigger = yes
			mr_gui_has_military_aviation_tech_trigger = yes
			curtiss_has_aeroplane_production_size_x_tooltip_trigger = { LEVEL = 1 }
		}
	}
	effect = {
		add_journal_entry = {
        	type = je_curtiss_testplane_build
        }
		add_modifier = {
			name = curtiss_testplane_cost_modifier
			multiplier = money_amount_multiplier_very_small
		}
		#Makes the Engineer busy
		if = {
			limit = {
				has_variable = tesla_engineer_var
			}
			var:tesla_engineer_var = {
				set_character_busy_and_immortal = yes
			}
		}
	}

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 4 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
#Active
mr_aeronautical_engineer_speed_airplane_active = {
	scope = country
	is_shown = {
		has_journal_entry = je_curtiss_testplane_build
	}
}
