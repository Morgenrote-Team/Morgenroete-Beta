#MR_PHYSICIAN_EXPAND_MEDICAL_SYSTEM_BUTTON
mr_physician_expand_medical_system_button_sgui = {
    scope = country
    is_valid = {
		hidden_trigger = {
			NOT = {
				has_variable = panum_physician_action_active_var
			}
		}
		custom_tooltip = {
			text = MR_ACTIVE_PHYSICIAN_TRIGGER
			has_variable = panum_physician_var
			var:panum_physician_var = {
				is_busy = no
			}
		}
		custom_tooltip = {
			text = MR_PHYSICIAN_HAS_HEALTH_SYSTEM_TRIGGER_TT
			NOT = {
				has_law = law_type:law_no_health_system
			}
		}
		custom_tooltip = {
			text = MR_PHYSICIAN_HAS_HEALTH_SYSTEM_INSTITUTION_TRIGGER_TT
			institution_investment_level = {
				institution	= institution_health_system
				value >= 1
			}
		}
		custom_tooltip = {
			text = MR_PHYSICIAN_HAS_POSSIBLE_CLINIC_OR_POSSIBLE_HOSPITAL_TRIGGER_TT
			OR = {
				any_scope_state = {
					any_scope_building = {
						is_building_type = building_government_administration
						level >= 5
						has_active_production_method = pm_panum_no_clinic
					}
				}
				any_scope_state = {
					any_scope_building = {
						is_building_type = building_government_administration
						level >= 20
						has_active_production_method = pm_panum_clinic
					}
				}
			}
		}
    }
    effect = {
		set_variable = mr_physician_expand_medical_system_var
		show_as_tooltip = {
			add_modifier = {
				name = panum_medical_system_cost_modifier
				multiplier = money_amount_multiplier_small
			}
		}
		add_journal_entry = { type = je_panum_medical_system_project }
		if = {
			limit = {
				has_variable = panum_physician_var
			}
			var:panum_physician_var = {
				set_character_busy_and_immortal = yes
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
		modifier = {
			trigger = {
				var:panum_physician_var ?= {
					OR = {
						has_trait = panum_military_physician
						has_trait = panum_red_cross
					}
				}
				has_variable = panum_physician_advanced_medical_support_var
				var:panum_physician_advanced_medical_support_var > 2
			}
			add = -100 #The AI will try to update the military medicine first, because the traits are rare
		}
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
#Checks if this is button is active
mr_physician_expand_medical_system_active_sgui = {
	is_shown = {
		has_variable = mr_physician_expand_medical_system_var
	}
}

#MR_PHYSICIAN_INTRODUCE_MEDICAL_TECHNIQUES_LEVEL_BUTTON
mr_physician_introduce_medical_techniques_level_button_sgui = {
    scope = country
	is_shown = {
		NOT = { has_modifier = panum_medical_techniques_level_7_modifier }
	}
    is_valid = {
		hidden_trigger = {
			NOR = {
				has_variable = panum_physician_action_active_var
				has_modifier = panum_medical_techniques_level_7_modifier
			}
		}
		custom_tooltip = {
			text = MR_ACTIVE_PHYSICIAN_TRIGGER
			has_variable = panum_physician_var
			var:panum_physician_var = {
				is_busy = no
			}
		}
		custom_tooltip = {
			text = MR_PHYSICIAN_HAS_HEALTH_SYSTEM_TRIGGER_TT
			NOT = {
				has_law = law_type:law_no_health_system
			}
		}
		#Level 1 Trigger
		mr_gui_has_vaccination_tech_trigger = yes
		mr_gui_has_central_archives_tech_trigger = yes
		#Level 2 Trigger
		trigger_if = {
			limit = { has_modifier = panum_medical_techniques_level_1_modifier }
			mr_gui_has_pharmaceuticals_tech_trigger = yes
			mr_gui_has_chemistry_tech_trigger = yes
		}
		#Level 3 Trigger
		trigger_if = {
			limit = { has_modifier = panum_medical_techniques_level_2_modifier }
			mr_gui_has_quinine_tech_trigger = yes
			mr_gui_has_mechanical_tools_tech_trigger = yes
		}
		#Level 4 Trigger
		trigger_if = {
			limit = { has_modifier = panum_medical_techniques_level_3_modifier }
			custom_tooltip = {
				text = MR_PHYSICIAN_HAS_GERM_THEORY_TRIGGER
				panum_has_miasma_theory_trigger = no
			}
			mr_gui_has_microbiology_tech_trigger = yes
		}
		#Level 5 Trigger
		trigger_if = {
			limit = { has_modifier = panum_medical_techniques_level_4_modifier }
			mr_gui_has_malaria_prevention_tech_trigger = yes
			mr_gui_has_central_planning_tech_trigger = yes
		}
		#Level 6 Trigger
		trigger_if = {
			limit = { has_modifier = panum_medical_techniques_level_5_modifier }
			mr_gui_has_antitoxins_tech_trigger = yes
			mr_gui_has_combustion_engine_tech_trigger = yes
		}
		#Level 7 Trigger
		trigger_if = {
			limit = { has_modifier = panum_medical_techniques_level_6_modifier }
			mr_gui_has_antibiotics_tech_trigger = yes
			mr_gui_has_military_aviation_tech_trigger = yes
		}
    }
    effect = {
		set_variable = mr_physician_introduce_medical_techniques_var
		show_as_tooltip = {
			add_modifier = {
				name = panum_medical_techniques_cost_modifier
				multiplier = money_amount_multiplier_small
			}
		}
		add_journal_entry = { type = je_panum_medical_techniques_project }
		if = {
			limit = {
				has_variable = panum_physician_var
			}
			var:panum_physician_var = {
				set_character_busy_and_immortal = yes
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
		modifier = {
			trigger = {
				var:panum_physician_var ?= {
					OR = {
						has_trait = panum_military_physician
						has_trait = panum_red_cross
					}
				}
				has_variable = panum_physician_advanced_medical_support_var
				var:panum_physician_advanced_medical_support_var > 2
			}
			add = -100 #The AI will try to update the military medicine first, because the traits are rare
		}
		modifier = {
			trigger = {
				institution_investment_level = {
					institution	= institution_health_system
					value >= 1
				}
				OR = {
					any_scope_state = {
						any_scope_building = {
							is_building_type = building_government_administration
							level >= 5
							has_active_production_method = pm_panum_no_clinic
						}
					}
					any_scope_state = {
						any_scope_building = {
							is_building_type = building_government_administration
							level >= 20
							has_active_production_method = pm_panum_clinic
						}
					}
				}
			}
			add = -100 #The AI will try to expand the medical System second, because the traits are rare
		}
	}
	ai_frequency = 5 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
