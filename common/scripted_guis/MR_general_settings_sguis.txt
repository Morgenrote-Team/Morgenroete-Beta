# GUI Settings

mr_setting_academic_compact_sgui = {
    scope = country

    is_shown = {
        has_variable = mr_setting_academic_compact_var
    }

    effect = {
        if = {
            limit = {
                has_variable = mr_setting_academic_compact_var
            }
            remove_variable = mr_setting_academic_compact_var
        }
        else = {
            set_variable = mr_setting_academic_compact_var
        }
    }
}
