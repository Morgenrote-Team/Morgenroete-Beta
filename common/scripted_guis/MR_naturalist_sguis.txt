#MR_SAFARI_EXPEDITION_BUTTON_TT
mr_safari_expedition_button = {
    scope = country
    is_valid = {
		mr_gui_has_theory_of_evolution_tech_trigger = yes
		custom_tooltip = {
			text = MR_SAFARI_EXPEDITION_CIVILIZING_MISSION_TRIGGER_TT
			has_technology_researched = civilizing_mission
		}
		custom_tooltip = {
			text = MR_VISIT_ZOO_BUILDING_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_dubois_zoo
				level > 0
			}
		}
		custom_tooltip = {
			text = MR_SAFARI_EXPEDITION_NATURALIST_TRIGGER_TT
			dubois_has_active_naturalist = yes
		}
		custom_tooltip = {
			text = MR_SAFARI_EXPEDITION_HABITAT_TRIGGER_TT
			dubois_safari_possible = yes
		}
		custom_tooltip = {
			text = MR_SAFARI_EXPEDITION_RECENT_TRIGGER_TT
			NOR = {
				has_modifier = dubois_on_safari_modifier
				has_modifier = dubois_had_safari_modifier
			}
		}
    }
    effect = {
        add_modifier = {
			name = dubois_on_safari_modifier
		}
		trigger_event = {
			id = dubois.31
			popup = yes
		}
		hidden_effect = {
			if = {
				limit = { has_variable = dubois_naturalist_var }
				var:dubois_naturalist_var = {
					add_modifier = {
						name = dubois_animal_hunter_modifier
					}
					set_character_busy_and_immortal = yes
					set_variable = dubois_naturalist_on_safari_var
				}
			}
		}
    }
}

mr_safari_expedition_active_sgui = {
    scope = country

		has_modifier = dubois_on_safari_modifier
    }
}

#MR_ANIMAL_DOCUMENTATION_BUTTON_TT
mr_animal_documentation_button = {
	scope = country
	is_valid = {

	ai_frequency = 4 # Named value determining how frequently the AI will evaluate this SGUI (in months)

#MR_BOTANICAL_EXPEDITION_RECALL_BUTTON_TT
mr_botanical_expedition_recall_button = {
    scope = country
	is_shown = {
		has_modifier = dubois_botanical_expedition_modifier
	}
    is_valid = {
		custom_tooltip = {
			text = MR_BOTANICAL_EXPEDITION_RECALL_TRIGGER_TT
			NOR = {
				has_modifier = dubois_naturalist_travell_to_location_modifier
				has_modifier = dubois_naturalist_travell_from_location_modifier
			}
			dubois_naturalist_sick = no
		}
    }
    effect = {
		custom_tooltip = {
			text = MR_BOTANICAL_EXPEDITION_RECALL_NATURALIST_EFFECT_TT
			if = {
				limit = { has_variable = dubois_naturalist_var }
				var:dubois_naturalist_var = {
					set_character_busy_and_immortal = no
					remove_variable = dubois_naturalist_on_botanical_expedition_var
				}
			}
		}
		dubois_set_naturalist_to_default = yes
		remove_modifier = dubois_botanical_expedition_modifier
		remove_modifier = dubois_botanical_expedition_cost_modifier
		add_modifier = {
			name = dubois_naturalist_travell_from_location_modifier
			days = 90
		}
    }
}

#MR_NATURALIST_IMPROVE_FORESTRY_BUTTON
mr_naturalist_improve_forestry_button = {
	scope = country
	is_valid = {
		custom_tooltip = {
			text = MR_ALREADY_CHOOSING_BUILDING_TRIGGER_TT
			NOT = {
				has_variable = dubois_improve_forestry_project_building_choser_var
			}
		}
		custom_tooltip = {
			text = MR_SAFARI_EXPEDITION_NATURALIST_TRIGGER_TT
			dubois_has_active_naturalist = yes
		}
		custom_tooltip = {
			text = MR_NATURALIST_IMPROVE_FORESTRY_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_logging_camp
				level > 0
				NAND = {
					has_modifier = dubois_building_throughput_add_modifier
					has_modifier = dubois_building_production_mult_modifier
					has_modifier = dubois_building_goods_input_mult_modifier
				}
			}
		}
	}
	effect = {
		if = {
			limit = {
				is_ai = yes
			}
			ordered_scope_building = {
				limit = {
					level > 0
					is_building_type = building_logging_camp
					NAND = {
						has_modifier = dubois_building_throughput_add_modifier
						has_modifier = dubois_building_production_mult_modifier
						has_modifier = dubois_building_goods_input_mult_modifier
					}
				}
				order_by = earnings
				save_scope_as = chosen_building
			}
			add_journal_entry = {
				type = je_dubois_improve_forestry_project
				target = scope:chosen_building
			}
			je:je_dubois_improve_forestry_project = {
				add_modifier = {
					name = dubois_naturalist_forestry_cost_modifier
					multiplier = money_amount_multiplier_small
				}
			}
			if = {
				limit = {
					has_variable = dubois_naturalist_var
				}
				var:dubois_naturalist_var = {
					set_character_busy_and_immortal = yes
				}
			}
		}
		else = {
			clear_variable_list = dubois_improve_forestry_buildings_variable_list
			ordered_scope_building = {
				limit = {
					level > 0
					is_building_type = building_logging_camp
					NAND = {
						has_modifier = dubois_building_throughput_add_modifier
						has_modifier = dubois_building_production_mult_modifier
						has_modifier = dubois_building_goods_input_mult_modifier
					}
				}
				order_by = earnings
				max = 1000
				check_range_bounds = no

				root = {
					add_to_variable_list = {
						name = dubois_improve_forestry_buildings_variable_list
						target = prev
					}
				}
			}
			set_variable = dubois_improve_forestry_project_building_choser_var
			show_as_tooltip = {
				add_modifier = {
					name = dubois_naturalist_forestry_cost_modifier
					multiplier = money_amount_multiplier_small
				}
			}
		}
	}

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
		modifier = {
			trigger = {
				market = {
					OR = {
						mg:wood ?= { market_goods_has_goods_shortage = yes }
						mg:hardwood ?= { market_goods_has_goods_shortage = yes }
					}
				}
			}
			add = 500
		}
	}
	ai_frequency = 4 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}

	}
}

#MR_HIPPOCAMPUS_CONGRESS_BUTTON
mr_hippocampus_congress_button = {
    scope = country
	is_shown = {
		NOT = {
			has_global_variable = dubois_hippocampus_congress_hosted_var
		}
		NOT = {
			any_country = {
				has_modifier = dubois_congress_host_modifier
			}
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_MAJOR_POWER_HIGHER_TRIGGER_TT
			country_rank >= rank_value:major_power
		}
		mr_gui_has_theory_of_evolution_tech_trigger = yes
		custom_tooltip = {
			text = MR_UNIVERSITY_BUILDING_LEVEL_10_TRIGGER_TT
			country_has_building_type_levels = {  
				target = bt:building_university  
				value  >= 10
			}
		}
		custom_tooltip = {
			text = MR_VISIT_ZOO_BUILDING_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_dubois_zoo
				level > 0
			}
		}
		custom_tooltip = {
			text = MR_NATURALIST_TRIGGER_TT
			dubois_has_naturalist = yes
		}
		custom_tooltip = {
			text = MR_HIPPOCAMPUS_CONGRESS_5_NATURALISTS_TRIGGER_TT
			any_country = {
				dubois_has_naturalist = yes
				count >= 5
			}
		}
		custom_tooltip = {
			text = MR_HIPPOCAMPUS_CONGRESS_SPECIMEN_TRIGGER_TT
			var:dubois_specimen >= 20000
		}
		custom_tooltip = {
			text = MR_HIPPOCAMPUS_CONGRESS_GLOBAL_SPECIMEN_TRIGGER_TT
			global_var:dubois_global_specimen >= 100000
		}
    }
    effect = {
        trigger_event = { 
			id = dubois.901
			popup = yes 
		}
		set_global_variable = dubois_hippocampus_congress_hosted_var
		add_modifier = {
			name = dubois_congress_host_modifier
			years = 5
		}
    }
}

#MR_EXTEND_ZOO_SMALL_BUTTON
mr_extend_zoo_small_button = {
    scope = country
	is_shown = {
		any_scope_state = {
			is_capital = yes
			is_production_method_active = {
				building_type = building_dubois_zoo
				production_method = pm_zoo_level_1
			}
			any_scope_building = {
				is_building_type = building_dubois_zoo
				level > 0
			}
		}
		NOT = {
			has_variable = dubois_zoo_first_extension
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_EXTEND_ZOO_SMALL_10_TOTAL_ANIMALS_TRIGGER_TT
			var:dubois_zoo_animal_total >= 10
		}
		custom_tooltip = {
			text = MR_EXTEND_ZOO_SMALL_2_SPECIAL_ANIMALS_TRIGGER_TT
			OR = {
				var:dubois_zoo_animal_africa >= 2
				var:dubois_zoo_animal_asia >= 2
				var:dubois_zoo_animal_australia >= 2
			}
		}
		custom_tooltip = {
			text = MR_EXTEND_ZOO_SMALL_CURRENT_EXPANSION_TRIGGER_TT
			NOT = {
				has_modifier = dubois_zoo_extension_modifier
			}
		}
    }
    effect = {
        set_variable = dubois_zoo_first_extension
		random_scope_state = {
			limit = {
				has_building = building_dubois_zoo
				is_capital = yes
			}
			activate_production_method = {
				building_type = building_dubois_zoo
				production_method = pm_zoo_level_2
			}
		}
		add_modifier = {
			name = dubois_zoo_extension_modifier
			days = 300
		}
    }
}
