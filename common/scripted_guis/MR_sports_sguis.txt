#MR_SCHOOL_SPORTS_BUTTON
mr_school_sport_button = {
    scope = country
	is_shown = {
		NOT = {
			has_variable = vikelas_school_sport_var
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_SCHOOL_SPORTS_PUBLIC_SCHOOLS_TRIGGER_TT
			has_law = law_type:law_public_schools
		}
		custom_tooltip = {
			text = MR_SCHOOL_SPORTS_IG_ATHLETE_LEADER_TRIGGER_TT
			any_interest_group = {
				is_in_government = yes
				leader = {
					has_trait = vikelas_athlete
				}
			}
		}
		custom_tooltip = {
			text = MR_SCHOOL_SPORTS_SPORTS_FACILITIES_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_government_administration
				has_active_production_method = pm_vikelas_sports_facilities
			}
		}
		mr_gui_has_theory_of_evolution_tech_trigger = yes
		mr_gui_has_organized_sports_tech_trigger = yes
    }
    effect = {
		set_variable = vikelas_school_sport_var
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}

#MR_FIFA_BUTTON
vikelas_fifa_button = {
    scope = country
	is_shown = {
		NOT = {
			has_global_variable = vikelas_fifa_founded_global_var
		}
		has_variable = vikelas_football_country_var
	}
    is_valid = {
		custom_tooltip = {
			text = MR_FIFA_TRIGGER_TT
			any_country = {
				has_variable = vikelas_football_association_var
				count >= 4
			}
		}
		mr_gui_has_organized_sports_tech_trigger = yes
    }
    effect = {
		custom_tooltip = vikelas_become_fifa_president_tt
		set_global_variable = vikelas_fifa_founded_global_var
		set_variable = vikelas_fifa_founding_country_var
		capital = {
			create_building = {
				building = building_vikelas_fifa
				level = 1
			}
			state_region = {
				set_global_variable = { #This is used in the GUI to display the state
            		name = vikelas_fifa_seat_global_var
           			value = this
        		}
			}
		}
		country_definition = {
			set_global_variable = {
				name = vikelas_fifa_president_global_var
				value = ROOT
			}
		}
		every_country = {
			limit = {
				has_variable = vikelas_football_association_var
			}
			trigger_event = {
				id = vikelas.401
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}

#MR_WORLDCUP_BUTTON
vikelas_football_worldcup_button = {
    scope = country
	is_shown = {
		NOT = {
			has_global_variable = vikelas_football_worldcup_global_var
		}
		has_global_variable = vikelas_fifa_founded_global_var
		has_variable = vikelas_football_country_var
	}
    is_valid = {
		custom_tooltip = {
			text = MR_FOOTBALL_WORLDCUP_FOOTBALL_ASSOCIATION_TRIGGER_TT
			any_country = {
				has_variable = vikelas_football_association_var
				count >= 8
			}
		}
		mr_gui_has_mass_culture_tech_trigger = yes
		mr_gui_has_organized_sports_tech_trigger = yes
    }
    effect = {
		set_global_variable = vikelas_football_worldcup_global_var
		add_modifier = {
			name = vikelas_fifa_worldcup_host_modifier
			days = 120
		}
		every_country = {
			limit = {
				has_variable = vikelas_football_association_var
			}
			trigger_event = {
				id = vikelas.411
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
