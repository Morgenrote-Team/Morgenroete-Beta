mr_mountain_expedition_button = {
    scope = country
	
    is_valid = {
		custom_tooltip = {
			text = MR_MOUNTAIN_UNCONQUERED_SUMMIT_TRIGGER_TT
			OR = {
				dubois_has_unconquered_summit_no_tt_trigger = yes
				any_country = {
					dubois_has_unconquered_summit_no_tt_trigger = yes
					has_diplomatic_pact = {
						who = ROOT
						type = mountaineering_license
					}
				}
			}
		}
		mr_gui_has_prospecting_tech_trigger = yes
		mr_gui_has_nationalism_tech_trigger = yes
		custom_tooltip = {
			text = MR_MOUNTAIN_GENERAL_TRIGGER_TT
			any_scope_character = {
				has_role = general
				is_busy = no
				OR = {
					has_trait = dubois_mountaineer
					has_trait = explorer
					has_trait = mountain_commander
					has_trait = surveyor
				}
			}
		}
		custom_tooltip = {
			text = MR_MOUNTAIN_RECENT_EXPEDITION_TRIGGER_TT
			NOR = {
				has_modifier = dubois_summit_level_1_modifier
				has_modifier = dubois_summit_level_2_modifier
				has_modifier = dubois_summit_level_3_modifier
				has_modifier = dubois_summit_level_4_modifier
				has_modifier = dubois_summit_level_5_modifier
				has_modifier = dubois_summit_level_6_modifier
				has_modifier = dubois_mountaineering_disaster_modifier
				has_modifier = dubois_mountaineering_disappointment_modifier
				has_journal_entry = je_dubois_mountain_expedition
				has_variable = dubois_current_mountain_expedition_var
			}
		}
    }
    effect = {
		add_treasury = -25000
		set_variable = {
			name = dubois_current_mountain_expedition_var
			days = 180
		}
		trigger_event = { 
			id = dubois.411
			popup = yes 
		}
    }
}

#MR_MOUNTAINEERING_CLUB_BUTTON
mr_mountaineering_club_button = {
    scope = country
	is_shown = {
		NOT = {
			has_variable = douglas_mountaineering_club_var
		}
	}
    is_valid = {
		mr_gui_has_geology_tech_trigger = yes
		mr_gui_has_nationalism_tech_trigger = yes
		custom_tooltip = {
			text = MR_MOUNTAIN_GENERAL_TRIGGER_TT
			any_scope_character = {
				has_role = general
				is_busy = no
				OR = {
					has_trait = dubois_mountaineer
					has_trait = explorer
					has_trait = mountain_commander
					has_trait = surveyor
				}
			}
		}
    }
    effect = {
		set_variable = douglas_mountaineering_club_var
		trigger_event = {
			id = douglas.101
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}

#MR_ALPINE_CLUB_BUTTON
mr_alpine_club_button = {
    scope = country
	is_shown = {
		NOT = {
			has_global_variable = douglas_alpine_club_founded_global_var
		}
		has_variable = douglas_mountaineering_club_var
	}
    is_valid = {
		custom_tooltip = {
			text = MR_ALPINE_CLUB_TRIGGER_TT
			any_country = {
				has_variable = douglas_mountaineering_club_var
				count >= 6
			}
		}
		
    }
    effect = {
		set_global_variable = douglas_alpine_club_founded_global_var
		custom_tooltip = douglas_become_alpine_club_president_tt
		set_variable = douglas_alpine_club_founding_country_var
		capital = {
			create_building = {
				building = building_douglas_alpine_club
				level = 1
			}
			state_region = {
				set_global_variable = { #This is used in the GUI to display the state
            		name = douglas_alpine_club_seat_global_var
           			value = this
        		}
			}
		}
		country_definition = {
			set_global_variable = {
				name = douglas_alpine_club_president_global_var
				value = ROOT
			}
		}
		every_country = {
			limit = {
				has_variable = douglas_mountaineering_club_var
			}
			trigger_event = {
				id = douglas.102
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
