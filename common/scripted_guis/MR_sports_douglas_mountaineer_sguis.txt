mr_mountain_expedition_level_1_button_sgui = {
    scope = country
	is_shown = {
		any_country = {
			douglas_has_unconquered_level_1_summit_trigger = yes
		}
	}
    is_valid = {
		douglas_has_active_mountaineer = yes
		custom_tooltip = {
			text = MR_MOUNTAIN_EQUIPMENT_1_TRIGGER_TT
			OR = {
				has_modifier = douglas_mountaineering_equipment_level_1_modifier
				has_modifier = douglas_mountaineering_equipment_level_2_modifier
				has_modifier = douglas_mountaineering_equipment_level_3_modifier
				has_modifier = douglas_mountaineering_equipment_level_4_modifier
				has_modifier = douglas_mountaineering_equipment_level_5_modifier
			}
		}
		custom_tooltip = {
			text = MR_MOUNTAIN_UNCONQUERED_LEVEL_1_SUMMIT_TRIGGER_TT
			OR = {
				douglas_has_unconquered_level_1_summit_trigger = yes
				any_country = {
					douglas_has_unconquered_level_1_summit_trigger = yes
					has_diplomatic_pact = {
						who = ROOT
						type = mountaineering_license
					}
				}
			}
		}
		custom_tooltip = {
			text = MR_MOUNTAIN_RECENT_EXPEDITION_TRIGGER_TT
			NOR = {
				has_modifier = douglas_mountaineering_success_modifier
				has_modifier = douglas_mountaineering_disaster_modifier
				has_modifier = douglas_mountaineering_disappointment_modifier
			}
		}
    }
    effect = {
		add_treasury = -10000
		if = {
			limit = {
				has_variable = douglas_mountaineer_var
			}
			var:douglas_mountaineer_var = {
				set_character_busy_and_immortal = yes
			}
		}
		trigger_event = {
			id = douglas.201
			popup = yes
		}
    }
}

#MR_MOUNTAINEERING_EQUIPMENT_BUTTON
mr_mountaineering_equipment_button_sgui = {
    scope = country
	is_shown = {
		NOT = {
			has_modifier = douglas_mountaineering_equipment_level_5_modifier
		}
	}
    is_valid = {
		hidden_trigger = {
			NOR = {
				douglas_has_active_mountaineer = no
				has_modifier = douglas_mountaineering_equipment_level_5_modifier
			}
		}
		douglas_has_active_mountaineer = yes
		custom_tooltip = {
			text = MR_ALPINE_HERITAGE_25_TRIGGER_TT
			var:douglas_alpine_heritage_var >= 25
		}
		#Level 1 Trigger: Geology
		mr_gui_has_geology_tech_trigger = yes
		#Level 2 Trigger: Logistics (Rope Group)
		trigger_if = {
			limit = {
				has_modifier = douglas_mountaineering_equipment_level_1_modifier
			}
			mr_gui_has_logistics_tech_trigger = yes
		}
		#Level 3 Trigger: Modern Nursing (Severe Injuries)
		trigger_if = {
			limit = {
				has_modifier = douglas_mountaineering_equipment_level_2_modifier
			}
			mr_gui_has_modern_nursing_tech_trigger = yes
		}
		#Level 4 Trigger: Seismology
		trigger_if = {
			limit = {
				has_modifier = douglas_mountaineering_equipment_level_3_modifier
			}
			mr_gui_has_seismology_tech_trigger = yes
		}
		#Level 5 Trigger: Chemical Warfare (Oxygen)
		trigger_if = {
			limit = {
				has_modifier = douglas_mountaineering_equipment_level_4_modifier
			}
			mr_gui_has_chemical_warfare_tech_trigger = yes
		}

    }
    effect = {
		douglas_add_x_alpine_heritage_effect = {
			AMOUNT = -25
		}
		add_journal_entry = { type = je_douglas_mountaineering_equipment }
		if = {
			limit = {
				has_variable = douglas_mountaineer_var
			}
			var:douglas_mountaineer_var = {
				set_character_busy_and_immortal = yes
			}
		}
    }
}

			id = douglas.102
		}

				id = douglas.301
			}
		}
