# GUI Settings

mr_setting_academic_compact = {
    scope = country

    is_valid = {
        has_variable = mr_setting_academic_compact
    }

    effect = {
        if = {
            limit = {
                has_variable = mr_setting_academic_compact
            }
            remove_variable = mr_setting_academic_compact
        }
        else = {
            set_variable = mr_setting_academic_compact
        }
    }
}
