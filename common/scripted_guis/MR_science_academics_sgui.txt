### ACADEMICS BUTTONS ###

# Chemist

mr_academics_hire_chemist_button_sgui = {
	scope = country
	is_shown = {
		mendelejew_has_chemist = yes
	}
	is_valid = {
		NOT = { has_variable = mr_academics_hire_chemist_window_active_var }
	}
	effect = {
		academics_hire_academic_open_window_effect = { ROLE = chemist PROJECT = mendelejew }
	}
}

mr_academics_hire_chemist_active_sgui = {
	scope = country
	is_shown = {
		has_variable = mr_academics_hire_chemist_window_active_var
	}
}

mr_academics_hire_chemist_close_button_sgui = {
	scope = country
	effect = {
		remove_variable = mr_academics_hire_chemist_window_active_var
	}
}

mr_academics_hire_this_chemist_button_sgui = {
	scope = country
	saved_scopes = {
        chosen_character
    }
	effect = {
		academics_hire_academic_hire_button_effect = { ROLE = chemist PROJECT = mendelejew }
	}
}

mr_academics_fire_this_chemist_button_sgui = {
	scope = country
	effect = {
		academics_hire_academic_fire_button_effect = { ROLE = chemist PROJECT = mendelejew }
	}
}
