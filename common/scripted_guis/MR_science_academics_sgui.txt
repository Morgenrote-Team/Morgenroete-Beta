### ACADEMICS BUTTONS ###

mr_academics_hire_chemist_button_sgui = {
	scope = country
	is_shown = {
		mendelejew_has_chemist = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		custom_tooltip = {
			text = mr_academics_hire_chemist_button_tt

			clear_variable_list = mr_academics_hire_chemist_variable_list
			ordered_character = {
				order_by = var:mendelejew_chemist_experience_var

				limit = {
					has_role = character_role_mendelejew_chemist
					has_variable = mendelejew_chemist_experience_var
					NOT = {
						owner.var:mendelejew_chemist_var ?= this
					}
				}

				root = {
					add_to_variable_list = {
						name = mr_academics_hire_chemist_variable_list
						target = prev
					}
				}
			}
			set_variable = mr_academics_hire_chemist_window_active_var
		}
	}
}
mr_academics_hire_chemist_active_sgui = {
	scope = country
	is_shown = {
		has_variable = mr_academics_hire_chemist_window_active_var
	}
}
mr_academics_hire_chemist_close_button_sgui = {
	scope = country
	effect = {
		remove_variable = mr_academics_hire_chemist_window_active_var
	}
}
mr_academics_hire_this_chemist_button_sgui = {
	scope = country
	saved_scopes = {
        chosen_character
    }
	effect = {
		clear_variable_list = mr_academics_hire_chemist_variable_list
		remove_variable = mr_academics_hire_chemist_window_active_var

		scope:chosen_character = {
			save_scope_as = mendelejew_new_chemist_scope
			transfer_character = root
		}

		mendelejew_set_new_chemist_as_chemist_effect = yes
	}
}
