#MR_ARCHITECT_MODERN_SPLENDOUR_BUTTON
mr_architect_modern_splendour_button = {
    scope = country
	is_shown = {
		NOR = {
			has_journal_entry = je_gaudi_splendour_bauhaus
			has_journal_entry = je_gaudi_splendour_neoclassical
			has_journal_entry = je_gaudi_splendour_art_deco
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_ARCHITECT_ACTIVE_TRIGGER_TT
			gaudi_has_active_architect = yes
		}
		custom_tooltip = {
			text = MR_MODERN_ARCHITECTURE_SCHOOL_TRIGGER_TT
			OR = {
				has_modifier = gaudi_bauhaus_modifier
				has_modifier = gaudi_neoclassicism_modifier
				has_modifier = gaudi_art_deco_modifier
			}
		}
    }
    effect = {
		if = {
			limit = {
				has_modifier = gaudi_bauhaus_modifier
			}
			add_journal_entry = {
				type = je_gaudi_splendour_bauhaus
			}
		}
		if = {
			limit = {
				has_modifier = gaudi_neoclassicism_modifier
			}
			add_journal_entry = {
				type = je_gaudi_splendour_neoclassical
			}
		}
		if = {
			limit = {
				has_modifier = gaudi_art_deco_modifier
			}
			add_journal_entry = {
				type = je_gaudi_splendour_art_deco
			}
		}
		if = {
			limit = {
				has_variable = gaudi_architect_var
			}
			var:gaudi_architect_var = {
				set_character_busy_and_immortal = yes
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 3 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}

mr_architect_modern_splendour_active_sgui = {
	scope = country
	
	is_shown = {
		OR = {
			has_journal_entry = je_gaudi_splendour_bauhaus
			has_journal_entry = je_gaudi_splendour_neoclassical
			has_journal_entry = je_gaudi_splendour_art_deco
		}
	}
}
