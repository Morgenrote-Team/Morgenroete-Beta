#MR_ANTHROPOLOGIST_CAVE_EXPLORATION_BUTTON
mr_anthropologist_cave_exploration_button_sgui = {
    scope = country
    is_valid = {
		custom_tooltip = {
			text = MR_ACTIVE_ANTHROPOLOGIST_TRIGGER_TT
			andersson_has_active_anthropologist = yes
		}
		custom_tooltip = {
			text = MR_ANTHROPOLOGIST_HAS_CAVE_TRIGGER_TT
			any_scope_state = {
				has_state_trait = andersson_state_trait_cave
			}
		}
		custom_tooltip = {
			text = MR_ANTHROPOLOGIST_IS_SPELEOLOGIST_TRIGGER_TT
			var:andersson_anthropologist_var ?= {
				has_trait = andersson_speleologist
			}
		}
		custom_tooltip = {
			text = MR_ANTHROPOLOGIST_NOT_HAS_CAVE_MODIFIER_TRIGGER_TT
			NOT = {
				has_modifier = andersson_cave_exploration_modifier
			}
		}
	}
	effect = {
		add_modifier = {
			name = andersson_cave_exploration_modifier
		}
		add_journal_entry = {
			type = je_andersson_cave_exploration
		}
		if = {
			limit = {
				has_variable = andersson_anthropologist_var
			}
			var:andersson_anthropologist_var = {
				set_character_busy_and_immortal = yes
			}
		}
		random_scope_state = {
			limit = {
				has_state_trait = andersson_state_trait_cave
			}
			set_variable = andersson_hominid_prospecting_var
		}
	}
}

mr_anthropologist_cave_exploration_active_sgui = {
    scope = country
	is_shown = {
		has_journal_entry = je_andersson_cave_exploration
	}
}

mr_anthropologist_neolithic_button_sgui = {
    scope = country
	is_shown = {
		NOT = {
			has_variable = andersson_neolithic_scenery_var
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_ACTIVE_ANTHROPOLOGIST_TRIGGER_TT
			andersson_has_active_anthropologist = yes
		}
		custom_tooltip = {
			text = MR_ANDERSSON_20_BONES_TRIGGER_TT
			has_variable = andersson_total_human_fossil
			var:andersson_total_human_fossil >= 20
		}
		custom_tooltip = {
			text = MR_NO_OUR_ANCESTORS_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_university
				NOR = {
					has_modifier = andersson_neolithic_scenery_modifier
					has_modifier = andersson_mesolithic_scenery_modifier
					has_modifier = andersson_upper_paleolithic_scenery_modifier
				}
			}
		}
    }
    effect = {
		set_variable = andersson_neolithic_scenery_ongoing_var
		add_journal_entry = {
			type = je_andersson_our_ancestors
		}
		if = {
			limit = {
				has_variable = andersson_anthropologist_var
			}
			var:andersson_anthropologist_var = {
				set_character_busy_and_immortal = yes
			}
		}
    }
}

#MR_ANTHROPOLOGIST_EXPOSE_PILTDOWN_FRAUD_BUTTON
mr_anthropologist_expose_piltdown_fraud_button_sgui = {
    scope = country
	is_shown = {
		has_global_variable = andersson_piltdown_country_global_var
		NOT = {
			has_global_variable = andersson_piltdown_exposed_country_global_var
		}
	}
    is_valid = {
		custom_tooltip = {
            text = MR_ANTHROPOLOGY_INSTITUTE_TRIGGER_TT
            andersson_has_institute_trigger = yes
        }
		custom_tooltip = {
            text = MR_HAS_ANTHROPOLOGIST_TRIGGER_TT
            andersson_has_active_anthropologist = yes
        }
		custom_tooltip = {
			text = MR_ANTHROPOLOGY_RATING_PILTDOWN_FRAUD_TT
			var:andersson_anthropological_rating >= 100
		}
		custom_tooltip = {
            text = MR_ANTHROPOLOGY_PILTDOWN_NO_RECENT_TRY_TRIGGER_TT
            NOT = {
				has_modifier = andersson_piltdown_debate_lost_modifier
			}
        }
		custom_tooltip = {
            text = MR_ANTHROPOLOGY_PILTDOWN_NO_CURRENT_DEBATE_TRIGGER_TT
            NOT = {
				any_country = {
					has_variable = andersson_piltdown_ongoing_debate_var
				}
			}
        }
		custom_tooltip = {
            text = MR_ANTHROPOLOGY_PILTDOWN_CAN_BE_EXPOSED_TRIGGER_TT
            any_country = {
				andersson_has_piltdown_fossils_trigger = yes
				NOT = {
					has_modifier = andersson_piltdown_debate_won_modifier
				}
			}
        }
    }
    effect = {
		set_variable = andersson_piltdown_ongoing_debate_var
		trigger_event = {
			id = andersson.511
			popup = yes
		}
    }
}

#MR_ORIGIN_CONGRESS_BUTTON
mr_origin_congress_button_sgui = {
    scope = country
	is_shown = {
		has_variable = andersson_anthropology_update_var
		NOT = {
			has_global_variable = andersson_origin_congress_hosted_var
		}
		NOT = {
			any_country = {
				has_modifier = andersson_congress_host_modifier
			}
		}
	}
    is_valid = {
		mr_gui_has_modern_archaeology_tech_trigger = yes
		mr_gui_has_genetics_tech_trigger = yes
		custom_tooltip = {
			text = MR_UNIVERSITY_BUILDING_LEVEL_25_TRIGGER_TT
			country_has_building_type_levels = {
				target = bt:building_university
				value  >= 25
			}
		}
		custom_tooltip = {
            text = MR_ANTHROPOLOGY_INSTITUTE_TRIGGER_TT
            andersson_has_institute_trigger = yes
        }
		custom_tooltip = {
            text = MR_HAS_ANTHROPOLOGIST_TRIGGER_TT
            andersson_has_anthropologist = yes
        }
		custom_tooltip = {
            text = MR_ANTHROPOLOGY_10_COUNTRIES_WITH_ANTHROPOLOGIST_TRIGGER_TT
            any_country = {
				andersson_has_anthropologist = yes
				count >= 10
			}
        }
		custom_tooltip = {
			text = MR_ORIGIN_CONGRESS_HUMAN_FOSSIL_TT
			var:andersson_total_human_fossil >= 30
		}
		custom_tooltip = {
			text = MR_ORIGIN_CONGRESS_SPECIAL_HUMAN_FOSSIL_TT
			OR = {
				andersson_has_maximum_pithecanthropus_fossils_trigger = yes
				andersson_has_maximum_australopithecus_fossils_trigger = yes
			}
		}
		custom_tooltip = {
			text = MR_ORIGIN_CONGRESS_GLOBAL_HUMAN_FOSSIL_TT
			global_var:andersson_global_human_fossil_global_var >= 250
		}
    }
    effect = {
		trigger_event = {
			id = andersson.911
			popup = yes
		}
		set_global_variable = andersson_origin_congress_hosted_var
		add_modifier = {
			name = andersson_congress_host_modifier
			years = 5
		}
    }
}
