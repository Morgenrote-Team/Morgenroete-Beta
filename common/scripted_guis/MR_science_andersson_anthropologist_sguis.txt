#MR_ANTHROPOLOGIST_SPELEOLOGY_BUTTON
mr_anthropologist_speleology_button_sgui = {
    scope = country
    is_valid = {
		andersson_has_anthropologist = yes
		NOT = {
			has_journal_entry = je_andersson_speleology
		}
	}
	effect = {
		add_journal_entry = { type = je_andersson_speleology }
	}
}

#MR_ANTHROPOLOGIST_CAVE_EXPLORATION_BUTTON
mr_anthropologist_cave_exploration_button_sgui = {
    scope = country
    is_valid = {
		custom_tooltip = {
			text = MR_ACTIVE_ANTHROPOLOGIST_TRIGGER_TT
			andersson_has_active_anthropologist = yes
		}
		custom_tooltip = {
			text = MR_ANTHROPOLOGIST_HAS_CAVE_TRIGGER_TT
			OR = {
				andersson_state_has_unexplored_cave_paintings_trigger = yes
				andersson_country_has_cave_trigger = yes
			}
		}
		custom_tooltip = {
			text = MR_ANTHROPOLOGIST_NOT_HAS_CAVE_MODIFIER_TRIGGER_TT
			NOT = {
				any_scope_state = {
					has_modifier = andersson_cave_exploration_modifier
				}
			}
		}
		
	}
	effect = {
		trigger_event = {
			id = andersson.201
			popup = yes
		}
		hidden_effect = {
			set_variable = andersson_cave_exploration_destination_selection_var
		}
	}
}

mr_cave_exploration_active_sgui = {

#MR_ORIGIN_CONGRESS_BUTTON
mr_origin_congress_button_sgui = {
    scope = country
	is_shown = {
		NOT = {
			has_global_variable = andersson_origin_congress_hosted_var
		}
		NOT = {
			any_country = {
				has_modifier = andersson_congress_host_modifier
			}
		}
	}
    is_valid = {
		mr_gui_has_modern_archaeology_tech_trigger = yes
		mr_gui_has_genetics_tech_trigger = yes
		custom_tooltip = {
			text = MR_UNIVERSITY_BUILDING_LEVEL_25_TRIGGER_TT
			country_has_building_type_levels = {
				target = bt:building_university
				value  >= 25
			}
		}
		custom_tooltip = {
            text = MR_ANTHROPOLOGY_INSTITUTE_TRIGGER_TT
            andersson_has_institute_trigger = yes
        }
		custom_tooltip = {
            text = MR_HAS_ANTHROPOLOGIST_TRIGGER_TT
            andersson_has_anthropologist = yes
        }
		custom_tooltip = {
            text = MR_ANTHROPOLOGY_10_COUNTRIES_WITH_ANTHROPOLOGIST_TRIGGER_TT
            any_country = {
				andersson_has_anthropologist = yes
				count >= 10
			}
        }
		custom_tooltip = {
			text = MR_ORIGIN_CONGRESS_HUMAN_FOSSIL_TT
			var:andersson_total_human_fossil >= 30
		}
		custom_tooltip = {
			text = MR_ORIGIN_CONGRESS_SPECIAL_HUMAN_FOSSIL_TT
			OR = {
				andersson_has_maximum_pithecanthropus_fossils_trigger = yes
				andersson_has_maximum_australopithecus_fossils_trigger = yes
			}
		}
		custom_tooltip = {
			text = MR_ORIGIN_CONGRESS_GLOBAL_HUMAN_FOSSIL_TT
			global_var:andersson_global_human_fossil_global_var >= 250
		}
    }
    effect = {
		trigger_event = {
			id = andersson.911
			popup = yes
		}
		set_global_variable = andersson_origin_congress_hosted_var
		add_modifier = {
			name = andersson_congress_host_modifier
			years = 5
		}
    }
}
