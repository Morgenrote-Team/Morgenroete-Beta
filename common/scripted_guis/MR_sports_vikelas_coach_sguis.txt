#MR_INTERNATIONAL_SPORTS_EVENT
mr_international_sports_event_sgui = {
    scope = country
	is_shown = {
		vikelas_has_national_sport_trigger = yes
		OR = {
			AND = {
				is_power_bloc_leader = yes
				root.power_bloc = {
					num_power_bloc_members = 4
				}
			}
			num_subjects = 3
		}
	}
    is_valid = {
		mr_gui_has_vikelas_international_sports_tech_trigger = yes
		custom_tooltip = {
			text = MR_INTERNATIONAL_SPORTS_MEMBER_TRIGGER_TT
			OR = {
				root.power_bloc = {
					any_power_bloc_member = {
						count >= 4
						vikelas_has_sports_association_trigger = yes
						var:vikelas_national_sport_var ?= root.var:vikelas_national_sport_var
					}
				}
				any_direct_subject = {
					count >= 3
					vikelas_has_sports_association_trigger = yes
					var:vikelas_national_sport_var ?= root.var:vikelas_national_sport_var
				}
			}
		}
		custom_tooltip = {
			text = MR_INTERNATIONAL_SPORTS_NO_FIFA_TRIGGER_TT
			NAND = {
				has_variable = vikelas_fifa_member_var
				vikelas_has_national_sport_football_trigger = yes
			}
		}
		custom_tooltip = {
			text = MR_INTERNATIONAL_SPORTS_MODIFIER_TRIGGER_TT
			NOR = {
				has_modifier = vikelas_international_sports_event_modifier
				has_modifier = vikelas_recent_international_sports_event_modifier
			}
		}
    }
    effect = {
		trigger_event = {
			id = vikelas.395
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}

#MR_FIFA_BUTTON
mr_fifa_button_sgui = {
    scope = country
	is_shown = {
		NOT = {
			has_global_variable = vikelas_fifa_founded_global_var
		}
		has_variable = vikelas_football_country_var
	}
    is_valid = {
		custom_tooltip = {
			text = MR_FIFA_TRIGGER_TT
			any_country = {
				vikelas_has_football_association_trigger = yes
				count >= 6
			}
		}
		mr_gui_has_vikelas_international_sports_tech_trigger = yes
    }
    effect = {
		custom_tooltip = vikelas_become_fifa_president_tt
		set_global_variable = vikelas_fifa_founded_global_var
		set_variable = vikelas_fifa_founding_country_var
		capital = {
			create_building = {
				building = building_vikelas_fifa
				level = 1
			}
			state_region = {
				set_global_variable = { #This is used in the GUI to display the state
            		name = vikelas_fifa_seat_global_var
           			value = this
        		}
			}
		}
		country_definition = {
			set_global_variable = {
				name = vikelas_fifa_president_global_var
				value = ROOT
			}
		}
		every_country = {
			limit = {
				vikelas_has_football_association_trigger = yes
			}
			trigger_event = {
				id = vikelas.401
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}

#MR_WORLDCUP_BUTTON
vikelas_football_worldcup_button_sgui = {
    scope = country
	is_shown = {
		NOT = {
			has_global_variable = vikelas_football_worldcup_global_var
		}
		has_global_variable = vikelas_fifa_founded_global_var
		has_variable = vikelas_football_country_var
	}
    is_valid = {
		custom_tooltip = {
			text = MR_FOOTBALL_WORLDCUP_FOOTBALL_ASSOCIATION_TRIGGER_TT
			any_country = {
				vikelas_has_football_association_trigger = yes
				count >= 10
			}
		}
		mr_gui_has_mass_culture_tech_trigger = yes
		mr_gui_has_vikelas_international_sports_tech_trigger = yes
    }
    effect = {
		set_global_variable = vikelas_football_worldcup_global_var
		set_global_variable = {
			name = vikelas_num_football_worldcup_global_var
			value = 0
		}
		set_variable = {
			name = vikelas_recent_fifa_host_var
			years = 20
		}
		add_modifier = {
			name = vikelas_fifa_worldcup_host_modifier
			days = 365
		}
		hidden_effect = {
			trigger_event = {
				id = vikelas.410
				days = 1
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)

		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
