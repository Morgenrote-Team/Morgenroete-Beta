#MR_AUTORACING_FOUND_AUTOMOBILE_CLUB_BUTTON
mr_autoracing_found_automobile_club_button = {
    scope = country
	is_shown = {
		NOT = {
			has_variable = vikelas_automobile_club_var
		}
	}
    is_valid = {
		mr_gui_has_combustion_engine_tech_trigger = yes
    }
    effect = {
		set_variable = vikelas_automobile_club_var
		hidden_effect = {
			if = {
				limit = { has_variable = vikelas_automobile_club_var }
				create_new_racer_effect = yes
				vikelas_set_new_racer_as_racer_effect = yes
			}
		}
		trigger_event = {
			id = vikelas.501
		}
    }
}

#MR_AUTORACING_LONG_DISTANCE_BUTTON
mr_autoracing_long_distance_button = {
    scope = country
	is_shown = {
		has_variable = vikelas_automobile_club_var
	}
    is_valid = {
		mr_gui_has_combustion_engine_tech_trigger = yes
		custom_tooltip = {
			text = MR_AUTORACING_LONG_DISTANCE_MODIFIER_TRIGGER_TT
			NOR = {
				has_modifier = vikelas_recent_autorace_modifier
				has_modifier = vikelas_ongoing_autorace_modifier
			}
		}
		custom_tooltip = {
			text = MR_AUTORACING_LONG_DISTANCE_STATE_TRIGGER_TT
			any_scope_state = {
				count >= 2
			}
		}
    }
    effect = {
		add_modifier = {
			name = vikelas_ongoing_autorace_modifier
		}
		trigger_event = {
			id = vikelas.511
		}
    }
}

#MR_AUTORACING_FOUND_GRAND_PRIX_BUTTON
mr_autoracing_found_grand_prix_button = {
	scope = country
	is_shown = {
		NOT = {
			has_global_variable = vikelas_autoracing_grand_prix_founded_var
		}
		has_variable = vikelas_automobile_club_var
	}
	is_valid = {
		mr_gui_has_combustion_engine_tech_trigger = yes
		mr_gui_has_organized_sports_tech_trigger = yes
		custom_tooltip = {
			text = MR_AUTORACING_FOUND_GRAND_PRIX_BUTTON_TRIGGER_TT
			any_country = {
				has_variable = vikelas_automobile_club_var
				has_technology_researched = organized_sports
				count >= 5
			}
		}
	}
	effect = {
		set_global_variable = vikelas_autoracing_grand_prix_founded_var
		capital = {
			create_building = {
				building = building_vikelas_aiacr
				level = 1
			}
		}
		trigger_event = {
			id = vikelas.507
			popup = yes
		}
	}
}

#MR_AUTORACING_FOUND_TEAM_BUTTON
mr_autoracing_found_team_button = {
	scope = country
	is_shown = {
		NOT = {
			has_variable = vikelas_autoracing_team_var
		}
		has_variable = vikelas_automobile_club_var
	}
	is_valid = {
		custom_tooltip = {
			text = MR_HAS_ENGINEER_TT
			gaudi_has_engineer = yes
		}
		custom_tooltip = {
			text = MR_AUTORACING_FOUND_TEAM_BUTTON_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_motor_industry
				has_active_production_method = pm_automobile_production
				level >= 2
			}
		}
	}
	effect = {
		set_variable = vikelas_autoracing_team_var
		trigger_event = {
			id = vikelas.505
		}
	}
}

#MR_AUTORACING_RACE_TRACK_BUTTON
mr_autoracing_race_track_button = {
	scope = country
	is_shown = {
		NOT = {
			any_scope_state = {
				any_scope_building = {
					is_building_type = building_vikelas_circuit
				}
			}
		}
		has_variable = vikelas_automobile_club_var
		has_global_variable = vikelas_autoracing_grand_prix_founded_var
	}
	is_valid = {
		mr_gui_has_combustion_engine_tech_trigger = yes
		mr_gui_has_organized_sports_tech_trigger = yes
		mr_gui_has_paved_roads_tech_trigger = yes
		custom_tooltip = {
			text = MR_HAS_RACER_TT
			has_variable = vikelas_racer_var
		}
		custom_tooltip = {
			text = MR_HAS_ENGINEER_TT
			has_variable = gaudi_engineer_var
		}
	}
	effect = {
		vikelas_add_circuit_effect = yes
	}
}
