#MR_CHEMIST_SYNTHETIC_OIL_BUTTON
mr_chemist_synthetic_oil_button_sgui = {
    scope = country
	is_shown = {
		NOT = {
			has_technology_researched = mendelejew_synthetic_fuel_tech
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_CHEMIST_ACTIVE_TRIGGER_TT
			mendelejew_has_active_chemist = yes
		}
		mr_gui_has_modern_physics_tech_trigger = yes
		mr_gui_has_plastics_tech_trigger = yes
		mr_gui_has_combustion_engine_tech_trigger = yes
		custom_tooltip = {
			text = MR_UNIVERSITY_BUILDING_LEVEL_10_TRIGGER_TT
			country_has_building_type_levels = {
				target = bt:building_university
				value  >= 10
			}
		}
		custom_tooltip = {
			text = MR_ASTEROID_CONGRESS_OBSERVATORY_BUILDING_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_verrier_observatory
				level > 0
			}
		}
		custom_tooltip = {
			text = MR_CHEMIST_HAS_IMPROVED_SYNTHETICS_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_synthetics_plant
				level > 0
				has_modifier = mendelejew_building_throughput_add_modifier
				has_modifier = mendelejew_building_production_mult_modifier
				has_modifier = mendelejew_building_goods_input_mult_modifier
			}
		}
    }
    effect = {
        add_journal_entry = { type = je_mendelejew_synthetic_oil }
		show_as_tooltip = {
			add_modifier = {
				name = mendelejew_chemist_synthetic_oil_cost_modifier
				multiplier = mr_local_project_cost_medium_value
			}
		}
		if = {
			limit = {
				has_variable = mendelejew_chemist_var
			}
			var:mendelejew_chemist_var = {
				set_character_busy_and_immortal = yes
			}
		}
    }

	#See ai_chemistry_automation_effect for AI
}

#MR_CHEMIST_FERTILIZER_BUTTON
mr_chemist_fertilizer_button_sgui = {
	scope = country
	saved_scopes = {
        chosen_building
    }
	is_valid = {
		custom_tooltip = {
			text = MR_ALREADY_CHOOSING_BUILDING_TRIGGER_TT
			NOT = {
				has_variable = mendelejew_improve_chemical_plants_building_choser_var
			}
		}
		custom_tooltip = {
			text = MR_CHEMIST_ACTIVE_TRIGGER_TT
			mendelejew_has_active_chemist = yes
		}
		custom_tooltip = {
			text = MR_CHEMIST_IMPROVE_FERTILIZER_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_chemical_plant
				level > 0
				NAND = {
					has_modifier = mendelejew_building_throughput_add_modifier
					has_modifier = mendelejew_building_production_mult_modifier
					has_modifier = mendelejew_building_goods_input_mult_modifier
				}
			}
		}
	}
	effect = {
		clear_variable_list = mendelejew_improve_chemical_plants_buildings_variable_list
		ordered_scope_building = {
			limit = {
				level > 0
				is_building_type = building_chemical_plant
				NAND = {
					has_modifier = mendelejew_building_throughput_add_modifier
					has_modifier = mendelejew_building_production_mult_modifier
					has_modifier = mendelejew_building_goods_input_mult_modifier
				}
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = mendelejew_improve_chemical_plants_buildings_variable_list
					target = prev
				}
			}
		}
		set_variable = mendelejew_improve_chemical_plants_building_choser_var
		show_as_tooltip = {
			add_modifier = {
				name = mendelejew_chemist_fertilizer_cost_modifier
				multiplier = mr_local_project_cost_small_value
			}
		}
	}

	#See ai_chemistry_automation_effect for AI
}

#MR_CHEMIST_CONGRESS_BUTTON
mr_chemist_congress_button_sgui = {
    scope = country
	is_shown = {
		NOT = {
			has_global_variable = mendelejew_congress_hosted_var
		}
		NOT = {
			any_country = {
				has_modifier = mendelejew_congress_host_modifier
			}
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_MAJOR_POWER_HIGHER_TRIGGER_TT
			country_rank >= rank_value:major_power
		}
		mr_gui_has_electromagnetism_tech_trigger = yes
		custom_tooltip = {
			text = MR_UNIVERSITY_BUILDING_LEVEL_10_TRIGGER_TT
			country_has_building_type_levels = {
				target = bt:building_university
				value  >= 10
			}
		}
		custom_tooltip = {
			text = MR_ASTEROID_CONGRESS_OBSERVATORY_BUILDING_TRIGGER_TT
			any_scope_building = {
				is_building_type = building_verrier_observatory
				level > 0
			}
		}
		custom_tooltip = {
			text = MR_CHEMIST_TRIGGER_TT
			mendelejew_has_chemist = yes
		}
		custom_tooltip = {
			text = MR_CHEMIST_CONGRESS_5_CHEMISTS_TRIGGER_TT
			any_country = {
				mendelejew_has_chemist = yes
				count >= 5
			}
		}
		custom_tooltip = {
			text = MR_CHEMIST_CONGRESS_ELEMENTS_TRIGGER_TT
			global_var:mendelejew_isolated_elements_global_number_var >= 60
		}
    }
