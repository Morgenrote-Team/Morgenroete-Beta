#Transfer from general list to prioritized list
mr_transfer_from_general_list_to_prioritized_list_railway_project_button_sgui = {
	scope = country
	saved_scopes = {
        chosen_state
    }
	effect = {
		remove_list_variable = {
			name = states_for_railway_improvement_variable_list
			target = scope:chosen_state
		}
		add_to_variable_list = {
			name = prioritized_states_for_railway_improvement_variable_list
			target = scope:chosen_state
		}
	}
}

#Sort by Incorporation Status
mr_sort_railway_state_by_status_button_sgui = {
	scope = country
	effect = {
		ordered_in_list = {
			variable = states_for_railway_improvement_variable_list
			limit = {
				is_incorporated = yes
			}
			order_by = tesla_infrastructure_balance
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = placeholder_list
					target = prev
				}
			}
		}
		ordered_in_list = {
			variable = states_for_railway_improvement_variable_list
			limit = {
				is_incorporated = no
			}
			order_by = tesla_infrastructure_balance
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = placeholder_list
					target = prev
				}
			}
		}
		clear_variable_list = states_for_railway_improvement_variable_list
		every_in_list = {
			variable = placeholder_list

			root = {
				add_to_variable_list = {
					name = states_for_railway_improvement_variable_list
					target = prev
				}
			}
		}
		clear_variable_list = placeholder_list
	}
}

#Visibility
mr_state_prioritizing_railway_project_window_active_sgui = {
	scope = country
	is_shown = {
		has_variable = mr_railway_state_prioritizing_window_active_var
		OR = {
			#Active Engineer
			AND = {
				has_variable = tesla_engineer_var
				var:tesla_engineer_var = {
					is_busy = no
				}
			}
			has_journal_entry = je_gaudi_railway_project
		}
	}
}
#Closing
mr_state_prioritizing_railway_project_window_close_button_sgui = {
	scope = country
	effect = {
		remove_variable = mr_railway_state_prioritizing_window_active_var
	}
}
#Start Project
mr_state_prioritizing_railway_project_window_start_button_sgui = {
	scope = country
	is_shown = {
		NOT = { has_journal_entry = je_gaudi_railway_project }
	}
	effect = {
		remove_variable = mr_railway_state_prioritizing_window_active_var

		add_journal_entry = { type = je_gaudi_railway_project }
		je:je_gaudi_railway_project = {
			add_modifier = {
				name = tesla_medium_engineering_project_cost_modifier
				multiplier = mr_local_project_cost_medium_value
			}
		}
		show_as_tooltip = {
			add_modifier = {
				name = tesla_medium_engineering_project_cost_modifier
				multiplier = mr_local_project_cost_medium_value
			}
		}
		if = {
			limit = {
				has_variable = tesla_engineer_var
			}
			var:tesla_engineer_var = {
				set_character_busy_and_immortal = yes
			}
		}
		clear_variable_list  = states_for_railway_improvement_variable_list #Removes the list
	}
}
#Change Priority
mr_state_prioritizing_railway_project_window_change_priority_button_sgui = {
	scope = country
	is_shown = {
		has_journal_entry = je_gaudi_railway_project
	}
	effect = {
		remove_variable = mr_railway_state_prioritizing_window_active_var
		clear_variable_list  = states_for_railway_improvement_variable_list #Removes the list
	}
}
