# Science Panel
mr_set_science_panel_open_sgui = {
	scope = country
	effect = {
		set_variable = mr_science_panel_open
	}
}
mr_set_science_panel_closed_sgui = {
	scope = country
	effect = {
		remove_variable = mr_science_panel_open
	}
}
