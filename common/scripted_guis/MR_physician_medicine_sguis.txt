#MR_SIGN_FIRST_CONVENTION_BUTTON
mr_sign_first_convention_button = {
    scope = country
	is_shown = {
		NOT = { has_modifier = panum_first_geneva_convention_modifier }
	}
    is_valid = {
		custom_tooltip = {
			text = MR_FIRST_CONVENTION_PROPOSED_TRIGGER
			has_global_variable = panum_first_convention_proposed_global_var
		}
		custom_tooltip = {
			text = MR_RED_CROSS_MEMBER_TRIGGER
			has_variable = panum_red_cross_member_var
		}
		custom_tooltip = {
			text = MR_FIRST_CONVENTION_NOT_SIGNED_TRIGGER_TT
			NOT = {
				has_modifier = panum_first_geneva_convention_modifier
			}
		}
		custom_tooltip = {
			text = MR_FIRST_CONVENTION_NOT_ABOLISHED_TRIGGER_TT
			NOT = {
				has_modifier = panum_abolished_first_geneva_convention_modifier
			}
		}
    }
    effect = {
		change_infamy = -2
		add_modifier = {
			name = panum_first_geneva_convention_modifier
		}
		custom_tooltip = {
			text = panum_sign_convention_relation_effect
