	scope = country

#MR_BIOLOGIST_FIND_PATHOGEN_BUTTON
mr_biologist_find_pathogen_button_sgui = {
    scope = country
    is_valid = {
		hidden_trigger = {
			NOT = {
				has_variable = theiler_biologist_action_active_var
			}
		}
		mr_gui_has_microbiology_tech_trigger = yes
		mr_gui_has_aniline_tech_trigger = yes
		custom_tooltip = {
			text = MR_PHYSICIAN_HAS_GERM_THEORY_TRIGGER
			panum_has_miasma_theory_trigger = no
		}
		custom_tooltip = {
			text = MR_ACTIVE_BIOLOGIST_TRIGGER
			has_variable = theiler_biologist_var
			var:theiler_biologist_var = {
				is_busy = no
			}
		}
		custom_tooltip = {
			text = MR_ACHROMATIC_MICROSCOPE_TRIGGER
			OR = {
				theiler_has_achromatic_microscope_trigger = yes
				theiler_has_electron_microscope_trigger = yes
			}
		}
		custom_tooltip = {
			text = MR_BIOLOGIST_CAN_FIND_PATHOGEN_TRIGGER_TT
			theiler_can_find_pathogen_trigger = yes
		}
    }
    effect = {
		mr_biologist_find_pathogen_effect = yes
    }
}
#Checks if this is button is active
mr_biologist_find_pathogen_active_sgui = {
	is_shown = {
		has_variable = theiler_biologist_find_pathogen_active_var
	}
}

### Biologist Automation ###

mr_biologist_automation_gui = {
	scope = country
	is_shown = {
		NOT = { has_variable = theiler_biology_automation_var }
	}
	is_valid = {
		always = yes
	}
	effect = {
		custom_tooltip = mr_biologist_automation_gui_tt
		set_variable = theiler_biology_automation_var
		ai_biology_automation_effect = yes
	}
}
mr_stop_biologist_automation_gui = {
	scope = country
	is_shown = {
		has_variable = theiler_biology_automation_var
	}
	is_valid = {
		always = yes
	}
	effect = {
		custom_tooltip = mr_stop_biologist_automation_gui_tt
		remove_variable = theiler_biology_automation_var
	}
}
