    scope = country
    is_valid = {
		hidden_trigger = {
			NOT = {
				has_variable = theiler_biologist_action_active_var
			}
		}
		mr_gui_has_microbiology_tech_trigger = yes
		mr_gui_has_aniline_tech_trigger = yes
		custom_tooltip = {
			text = MR_PHYSICIAN_HAS_GERM_THEORY_TRIGGER
			panum_has_miasma_theory_trigger = no
		}
		custom_tooltip = {
			text = MR_ACTIVE_BIOLOGIST_TRIGGER
			has_variable = theiler_biologist_var
			var:theiler_biologist_var = {
				is_busy = no
			}
		}
		custom_tooltip = {
			text = MR_ACHROMATIC_MICROSCOPE_TRIGGER
			OR = {
				theiler_has_achromatic_microscope_trigger = yes
				theiler_has_electron_microscope_trigger = yes
			}
		}
		custom_tooltip = {
			text = MR_BIOLOGIST_CAN_FIND_PATHOGEN_TRIGGER_TT
			theiler_can_find_pathogen_trigger = yes
		}
    }
    effect = {
		set_variable = theiler_biologist_action_active_var
		set_variable = theiler_biologist_find_pathogen_active_var
		trigger_event = {
			id = theiler.51
			popup = yes
		}
    }
}

#MR_BIOLOGIST_IMPROVE_AGRICULTURE_BUTTON
mr_biologist_improve_agriculture_button_sgui = {
	scope = country
	is_valid = {
		custom_tooltip = {
			text = MR_ALREADY_CHOOSING_BUILDING_TRIGGER_TT
			NOT = {
				has_variable = theiler_improve_agriculture_project_building_choser_var
			}
		}
		theiler_has_active_biologist_trigger = yes
		theiler_country_has_building_valid_for_agriculture_project_trigger = yes
	}
	effect = {
		if = {
			limit = {
				is_ai = yes
			}
			clear_variable_list = theiler_improve_agriculture_buildings_chosen_variable_list
			ordered_scope_building = {
				position = 0 #We only want the first one here
				limit = {
					theiler_building_valid_for_agriculture_project_no_tooltip_trigger = yes
				}
				order_by = earnings
				root = {
					add_to_variable_list = {
						name = theiler_improve_agriculture_buildings_chosen_variable_list
						target = prev
					}
				}
			}
			add_journal_entry = {
				type = je_theiler_improve_agriculture_project
			}
			var:theiler_biologist_var ?= {
				set_character_busy_and_immortal = yes
			}
		}
		else = {
			clear_variable_list = theiler_improve_agriculture_buildings_variable_list
			ordered_scope_building = {
				limit = {
					theiler_building_valid_for_agriculture_project_no_tooltip_trigger = yes
				}
				order_by = earnings
				check_range_bounds = no

				root = {
					add_to_variable_list = {
						name = theiler_improve_agriculture_buildings_variable_list
						target = prev
					}
				}
			}
			set_variable = theiler_improve_agriculture_project_building_choser_var
			show_as_tooltip = {
				add_modifier = {
					name = theiler_improve_agriculture_cost_modifier
					multiplier = mr_local_project_cost_small_value
				}
			}
		}
	}

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 0 # The chance that the AI will activate this SGUI (script value between 1 and 100)
		modifier = { #BIOLOGY TODO: Add this
			trigger = {
				market = {
					OR = {
						mg:wood ?= { market_goods_has_goods_shortage = yes }
						mg:hardwood ?= { market_goods_has_goods_shortage = yes }
					}
				}
			}
			add = 500
		}
	}
	ai_frequency = 4 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
mr_biologist_improve_agriculture_active_sgui = {
	scope = country
	is_shown = {
		has_journal_entry = je_theiler_improve_agriculture_project
	}
}
