#MR_BIOLOGIST_MICROORGANISM_RESEARCH_BUTTON
mr_biologist_microorganism_research_button_sgui = {
	scope = country
	is_valid = {
		theiler_has_biologist = yes
		NOT = {
			has_journal_entry = je_theiler_microorganism_research
		}
	}
	effect = {
		add_journal_entry = { type = je_theiler_microorganism_research }
	}
}
#Active
mr_biologist_microorganism_research_active_sgui = {
	scope = country
	is_shown = {
		has_journal_entry = je_theiler_microorganism_research
		theiler_has_active_biologist_no_tooltip_trigger = yes
	}
}

#MR_BIOLOGIST_FIND_PATHOGEN_BUTTON
mr_biologist_find_pathogen_button_sgui = {
    scope = country
    is_valid = {
		hidden_trigger = {
			NOT = {
				has_variable = theiler_biologist_action_active_var
			}
		}
		mr_gui_has_microbiology_tech_trigger = yes
		mr_gui_has_aniline_tech_trigger = yes
		custom_tooltip = {
			text = MR_ACTIVE_BIOLOGIST_TRIGGER
			has_variable = theiler_biologist_var
			var:theiler_biologist_var = {
				is_busy = no
			}
		}
		custom_tooltip = {
			text = MR_ACHROMATIC_MICROSCOPE_TRIGGER
			OR = {
				theiler_has_achromatic_microscope_trigger = yes
				theiler_has_electron_microscope_trigger = yes
			}
		}
		custom_tooltip = {
			text = MR_BIOLOGIST_CAN_FIND_PATHOGEN_TRIGGER_TT
			theiler_can_find_pathogen_trigger = yes
		}
    }
    effect = {
		set_variable = theiler_biologist_action_active_var
		set_variable = theiler_biologist_find_pathogen_active_var
		trigger_event = { 
			id = theiler.51
			popup = yes 
		}
    }
}
#Checks if this is button is active
mr_biologist_find_pathogen_active_sgui = {
	is_shown = {
		has_variable = theiler_biologist_find_pathogen_active_var
	}
}
