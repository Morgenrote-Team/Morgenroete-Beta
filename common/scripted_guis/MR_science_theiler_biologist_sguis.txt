#MR_BIOLOGIST_FIND_PATHOGEN_BUTTON
mr_biologist_find_pathogen_button_sgui = {
    scope = country
	is_shown = {
		OR = {
			theiler_has_cholera_pathogen_trigger = no
			theiler_has_tuberculosis_pathogen_trigger = no
			theiler_has_smallpox_pathogen_trigger = no
			theiler_has_typhus_pathogen_trigger = no
			theiler_has_yellow_fever_pathogen_trigger = no
			theiler_has_measles_pathogen_trigger = no
			theiler_has_scarlet_pathogen_trigger = no
			theiler_has_plague_pathogen_trigger = no
			theiler_has_diphteria_pathogen_trigger = no
			theiler_has_polio_pathogen_trigger = no
		}
	}
    is_valid = {
		hidden_trigger = {
			NOT = {
				has_variable = theiler_biologist_action_active_var
			}
		}
		mr_gui_has_microbiology_tech_trigger = yes
		mr_gui_has_aniline_tech_trigger = yes
		custom_tooltip = {
			text = MR_PHYSICIAN_HAS_GERM_THEORY_TRIGGER
			panum_has_miasma_theory_trigger = no
		}
		custom_tooltip = {
			text = MR_ACTIVE_BIOLOGIST_TRIGGER
			has_variable = theiler_biologist_var
			var:theiler_biologist_var = {
				is_busy = no
			}
		}
		custom_tooltip = {
			text = MR_ACHROMATIC_MICROSCOPE_TRIGGER_TT
			OR = {
				theiler_has_achromatic_microscope_trigger = yes
				theiler_has_electron_microscope_trigger = yes
			}
		}
		trigger_if = {

		}
		custom_tooltip = {
			text = MR_BIOLOGIST_CAN_FIND_PATHOGEN_TRIGGER_TT
			theiler_can_find_pathogen_trigger = yes
		}
    }
    effect = {
		mr_biologist_find_pathogen_effect = yes
    }
}

	}

#MR_BIOLOGIST_LEAD_ARSENATE_BUTTON
MR_BIOLOGIST_LEAD_ARSENATE_BUTTON_sgui = {
    scope = country
	is_shown = {
		NOR = {
			has_variable = theiler_codling_moth_prevention_var
			has_variable = theiler_potato_beetle_prevention_var
		}
		OR = {
			any_scope_state = {
				is_in_geographic_region = theiler_geographic_region_codling_moth
			}
			AND = {
				has_global_variable = theiler_potato_beetle_active_global_var
				any_scope_state = {
					is_in_geographic_region = theiler_geographic_region_potato_beetle
				}
			}
		}
	}
    is_valid = {
		mr_gui_has_early_pesticides_tech_trigger = yes
		mendelejew_has_chemist = yes
		custom_tooltip = {
			text = MR_MOTH_OR_BEETLE_STATE_TRIGGER
			any_scope_state = {
				OR = {
					is_in_geographic_region = theiler_geographic_region_potato_beetle
					is_in_geographic_region = theiler_geographic_region_codling_moth
				}
			}
		}
		custom_tooltip = {
			text = MR_ACTIVE_BIOLOGIST_TRIGGER
			has_variable = theiler_biologist_var
			var:theiler_biologist_var = {
				is_busy = no
			}
		}
    }
    effect = {
		MR_BIOLOGIST_LEAD_ARSENATE_BUTTON_effect = yes
    }
}
