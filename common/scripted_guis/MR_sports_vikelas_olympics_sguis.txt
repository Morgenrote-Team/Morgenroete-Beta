#MR_IOC_FOUNDATION_BUTTON
mr_ioc_foundation_button_sgui = {
    scope = country
	is_shown = {
		NOT = {
			has_global_variable = vikelas_ioc_founded_var
		}
	}
    is_valid = {
		mr_gui_has_vikelas_international_sports_tech_trigger = yes
		custom_tooltip = {
			text = MR_IOC_FOUNDATION_HAS_ACADEMIC_TRIGGER_TT
			OR = {
				tesla_has_engineer = yes
				verrier_has_astronomer = yes
				dubois_has_naturalist = yes
				dubois_has_paleontologist = yes
				lepsius_has_archaeologist = yes
				andersson_has_anthropologist = yes
			}
		}
		custom_tooltip = {
			text = MR_IOC_FOUNDATION_PEACE_TRIGGER_TT
			is_at_war = no
		}
    }
    effect = {
		vikelas_set_basic_olympic_disciplines_effect = yes
		set_global_variable = vikelas_ioc_founded_var
		set_variable = vikelas_ioc_founding_country_var
		set_variable = vikelas_ioc_member_var
		trigger_event = { 
			id = vikelas.100
			popup = yes 
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}

mr_olympic_flame_button_sgui = {
    scope = country
	is_shown = {
		has_variable = vikelas_ioc_member_var
		NOT = {
			has_global_variable = vikelas_olympic_flame_var
		}
	}
    is_valid = {
		custom_tooltip = {
			text = MR_IOC_OLYMPIC_FLAME_IS_HOST_TRIGGER_TT
			has_modifier = vikelas_olympics_preparation_modifier
		}
		mr_gui_has_mass_culture_tech_trigger = yes
		mr_gui_has_vikelas_international_sports_tech_trigger = yes
		custom_tooltip = {
			text = MR_ARCHAEOLOGY_HAS_ARCHAEOLOGIST_TRIGGER_TT
			lepsius_has_archaeologist = yes
		}
    }
    effect = {
		set_global_variable = vikelas_olympic_flame_var
		change_infamy = -10
		hidden_effect = {
			random_country = {
				limit = {
					any_scope_state = {
						state_region = s:STATE_PELOPONNESE
					}
				}
				trigger_event = {
					id = vikelas.181
				}
			}
		}
		every_country = {
			limit = {
				has_variable = vikelas_ioc_board_member_var
				NOT = { root = this }
			}
			change_relations = {
				country = ROOT
				value = 5
			}
		}
    }

	ai_is_valid = { always = yes }	# Whether this SGUI is available to the AI or not (default false)
	ai_chance = {
		base = 100 # The chance that the AI will activate this SGUI (script value between 1 and 100)
	}
	ai_frequency = 6 # Named value determining how frequently the AI will evaluate this SGUI (in months)
}
