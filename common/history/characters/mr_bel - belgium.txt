CHARACTERS = {
	c:BEL ?= {

		### Writers ###
		if = { #Hendrik Conscience, Writer
			limit = {
				has_game_rule = artists_starting_historical_artists_rule
				year >= 1835 #Not sure when he started writing but two years later "In 't Wonderjaer" was published
				year < 1883 #Death
			}
			create_character = {
				template = mr_bel_conscience_hendrik_character_template
				save_scope_as = manzoni_writer_scope
			}
			scope:manzoni_writer_scope = {
				set_variable = mr_is_writer_var
				set_variable = {
					name = manzoni_writer_experience
					value = 102
				}
				set_variable = {
					name = manzoni_num_publications_var
					value = 3
				}
				place_character_in_void = 1200
			}

			set_variable = {
				name = manzoni_writer_var
				value = scope:manzoni_writer_scope
			}
		}

		### Painters ###
		if = { #François-Joseph Navez, Painter
			limit = {
				has_game_rule = artists_starting_historical_artists_rule
				year >= 1810 #helped create the Société des amateurs d'arts
				year < 1851 #Resigned
			}
			create_character = {
				template = mr_bel_navez_francois_joseph_character_template
				save_scope_as = klimt_painter_scope
			}
			scope:klimt_painter_scope = {
				set_variable = mr_is_painter_var
				set_variable = mr_no_headgear_var
				set_variable = {
					name = klimt_painter_experience
					value = 100
				}
				set_variable = {
					name = klimt_painter_num_paintings_var
					value = 8
				}
				set_variable = {
					name = klimt_painter_num_masterworks_var
					value = 0
				}
				place_character_in_void = 1200
			}

			set_variable = {
				name = klimt_classic_masterwork_var
				value = 0
			}

			set_variable = {
				name = klimt_painter_var
				value = scope:klimt_painter_scope
			}
		}
	}
}
