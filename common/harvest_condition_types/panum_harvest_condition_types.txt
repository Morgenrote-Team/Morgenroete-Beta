panum_cholera_condition = {
	trigger = {
		panum_state_can_have_cholera_trigger = yes
	}

	incompatible_with = panum_tuberculosis_condition
	incompatible_with = panum_smallpox_condition
	incompatible_with = panum_typhus_condition
	incompatible_with = panum_yellow_fever_condition
	incompatible_with = panum_measles_condition
	incompatible_with = panum_scarlet_condition
	incompatible_with = panum_plague_condition
	incompatible_with = panum_diphteria_condition
	incompatible_with = panum_polio_condition
	incompatible_with = panum_spanish_flu_condition
	# Incompatible with any other disease

	range = {
		integer_range = {
			min = 1
			max = 4
		}
	}

	duration = {
		fixed_range = {
			min = 40
			max = 80
		}
	}

	modifier = {
		#Basic Cholera Effects
		state_mortality_mult = 0.05
		state_tax_collection_mult = -0.05
		state_migration_pull_mult = -0.05
		building_throughput_add = -0.05
		state_infrastructure_mult = -0.05
	}

	intensity = {
		fixed_range = {
			min = 1
			max = 4
		}
	}

	chance = {
		value = 0.06
		if = {
			limit = {
				panum_has_cholera_treatment_trigger = yes
			}
			divide = 10
		}
		if = {
			limit = {
				any_neighbouring_state = {
					panum_state_has_cholera_trigger = yes
				}
			}
			multiply = 5
		}
		if = {
			limit = {
				any_scope_state = {
					total_urbanization > 2500
				}
			}
			multiply = 2
		}
		if = {
			limit = {
				any_scope_state = {
					total_urbanization > 2000
				}
			}
			multiply = 1.5
		}
		if = {
			limit = {
				any_scope_state = {
					total_urbanization > 1500
				}
			}
			multiply = 1.5
		}
		if = {
			limit = {
				any_scope_state = {
					total_urbanization > 1000
				}
			}
			multiply = 1.1
		}
		if = {
			limit = {
				any_scope_state = {
					total_urbanization > 900
				}
			}
			multiply = 1.1
		}
		if = {
			limit = {
				any_scope_state = {
					total_urbanization > 800
				}
			}
			multiply = 1.1
		}
		if = {
			limit = {
				any_scope_state = {
					total_urbanization > 700
				}
			}
			multiply = 1.1
		}
		if = {
			limit = {
				any_scope_state = {
					total_urbanization > 600
				}
			}
			multiply = 1.1
		}
	}

	color = { 0 150 250 }
	icon = "gfx/interface/icons/harvest_condition_icons/panum_condition_cholera.dds"
	#graphics = drought
	incompatible_terrain = { }
	map_texture = "gfx/map/map_painting/harvest_conditions_map_patterns/harvest_conditions_general_disease.dds"
}
