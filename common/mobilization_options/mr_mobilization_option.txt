mobilization_option_bicycle_recon = {
	texture = "gfx/interface/icons/mobilization_options/mr_bicycle_recon.dds"

	possible = {
		market ?= {
			mg:rubber ?= {
				market_goods_sell_orders > 0
			}
		}
	}
	
	on_activate_while_mobilized = {
		custom_tooltip = {
			text = mobilization_option_it_hurts_organization_when_you_adjust_equipment_tt
			add_organization = {
				value = organization
				multiply = -0.5
			}
		}
	}
	
	on_deactivate = {
		custom_tooltip = {
			text = mobilization_option_it_hurts_organization_when_you_adjust_equipment_tt
			add_organization = {
				value = organization
				multiply = -0.5
			}
		}
	}
	
	unlocking_technologies = {
		vikelas_bicycles_tech
	}
	upkeep_modifier = {
		goods_input_rubber_add = 0.35
		goods_input_steel_add = 0.2
	}
	unit_modifier = {
		unit_occupation_mult = 0.15
		military_formation_army_movement_speed_mult = 0.1
		character_battle_condition_surprise_maneuver_mult = 0.15
	}
	ai_weight = {
		value = 1
	}

    group = reconnaissance
}
