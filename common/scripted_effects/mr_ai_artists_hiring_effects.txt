### Composer ###

#Effect used by the AI to hire a Travelling Composer
#!!! If you change this Effect, make sure to change it for the "mr_invite_composer_button_sgui" Button too !!!
elgar_ai_hire_composer_effect = {
	if = {
		limit = {
			NOR = {
				has_modifier = elgar_recently_hired_travelling_composer_modifier
				has_modifier = elgar_music_repression_modifier
			#	has_variable = elgar_composer_var
			}
			has_technology_researched = elgar_classicism_tech
			elgar_has_opera_size_x_trigger = { LEVEL = 1 }
		}

		#This if construct chooses the experience level of possible Composers
		if = { #All Composers possible
			limit = {
				elgar_tier_5_composer_trigger = yes
			}
			elgar_hire_travelling_composer_effect = { EXPERIENCE = 10000 }
		}
		else_if = { #Up to 500 Reputation
			limit = {
				elgar_tier_4_composer_trigger = yes
			}

			elgar_hire_travelling_composer_effect = { EXPERIENCE = 400 }
		}
		else_if = { #Up to 450 Reputation
			limit = {
				elgar_tier_3_composer_trigger = yes
			}

			elgar_hire_travelling_composer_effect = { EXPERIENCE = 300 }
		}
		else_if = { #Up to 350 Reputation
			limit = {
				elgar_tier_2_composer_trigger = yes
			}

			elgar_hire_travelling_composer_effect = { EXPERIENCE = 200 }
		}
		else_if = { #Up to 250 Reputation
			limit = {
				elgar_tier_1_composer_trigger = yes
			}

			elgar_hire_travelling_composer_effect = { EXPERIENCE = 100 }
		}
	}
}

#This Effect is used to either directly choose a Travelling Composer or replace an existing one.
elgar_hire_travelling_composer_effect = {
	# EXPERIENCE = maximum experience of the composer
	if = {
		limit = { has_variable = elgar_composer_var }
		if = {
			limit = {
				any_in_global_list = {
					variable = mr_travelling_composer_global_list

					var:elgar_composer_experience < $EXPERIENCE$
					var:elgar_composer_experience >= elgar_compser_experience_plus_50_value
					NOT = {
						var:elgar_composer_unavailable_for_this_var ?= root
					}
				}
			}
			ordered_in_global_list = {
				variable = mr_travelling_composer_global_list

				position = 0 #We only want the first one here
				order_by = var:elgar_composer_experience
				check_range_bounds = no

				limit = {
					var:elgar_composer_experience < $EXPERIENCE$
					elgar_national_composer_trigger = yes
					NOT = {
						var:elgar_composer_unavailable_for_this_var ?= root
					}
				}

				save_scope_as = chosen_composer

				root = {
					set_composer_travelling_composer_effect = yes
					remove_modifier = elgar_conductor_exhausted_modifier
					set_variable = {
						name = elgar_composer_var
						value = scope:chosen_composer
					}
				}

				if = {
					limit = {
						NOT = {
							owner ?= root
						}
					}
					transfer_character = root
				}
				add_modifier = {
					name = artists_character_is_official_composer_modifier
				}
				remove_list_global_variable = {
					name = mr_travelling_composer_global_list
					target = this
				}
				set_character_busy_and_immortal = no

				root = {
					add_modifier = {
						name = elgar_recently_hired_travelling_composer_modifier
						years = 3
					}
				}

				elgar_travelling_composer_character_set_up_cost_effect = yes
			}
		}
	}
	else = {
		if = {
			limit = {
				any_in_global_list = {
					variable = mr_travelling_composer_global_list

					var:elgar_composer_experience < $EXPERIENCE$
					NOT = {
						var:elgar_composer_unavailable_for_this_var ?= root
					}
				}
			}
			ordered_in_global_list = {
				variable = mr_travelling_composer_global_list

				position = 0 #We only want the first one here
				order_by = var:elgar_composer_experience
				check_range_bounds = no

				limit = {
					var:elgar_composer_experience < $EXPERIENCE$
					elgar_national_composer_trigger = yes
					NOT = {
						var:elgar_composer_unavailable_for_this_var ?= root
					}
				}

				save_scope_as = chosen_composer

				root = {
					set_variable = {
						name = elgar_composer_var
						value = scope:chosen_composer
					}
				}

				if = {
					limit = {
						NOT = {
							owner ?= root
						}
					}
					transfer_character = root
				}
				add_modifier = {
					name = artists_character_is_official_composer_modifier
				}
				remove_list_global_variable = {
					name = mr_travelling_composer_global_list
					target = this
				}

				root = {
					add_modifier = {
						name = elgar_recently_hired_travelling_composer_modifier
						years = 3
					}
				}
				scope:chosen_composer = {
					elgar_travelling_composer_character_set_up_cost_effect = yes
					set_character_busy_and_immortal = no
				}
			}
		}
	}

	#Used to count the number of Great composers
	if = {
		limit = {
			var:elgar_composer_var ?= {
				has_trait = elgar_great_composer
			}
		}

		elgar_add_great_composer_count_effect = yes
	}
}

### Architect ###

#Effect used by the AI to hire an Independent Architect
#!!! If you change this Effect, make sure to change it for the "mr_invite_architect_button_sgui" Button too !!!
gaudi_ai_hire_architect_effect = {
	#This part is used if there is an architect already
	if = {
		limit = { has_variable = gaudi_architect_var }
		if = {
			limit = {
				any_in_global_list = {
					variable = mr_independent_architect_global_list

					root.gold_reserves > gaudi_independent_architect_cost_value
					var:gaudi_architect_experience >= gaudi_architect_experience_plus_50_value
					NOT = {
						var:gaudi_architect_unavailable_for_this_var ?= root
					}
				}
			}
			ordered_in_global_list = {
				position = 0 #We only want the first one here
				variable = mr_independent_architect_global_list

				order_by = var:gaudi_architect_experience
				check_range_bounds = no

				limit = {
					has_variable = gaudi_architect_experience
					root.gold_reserves > gaudi_independent_architect_cost_value
					var:gaudi_architect_experience >= gaudi_architect_experience_plus_50_value
					NOT = {
						var:gaudi_architect_unavailable_for_this_var ?= root
					}
				}

					set_architect_independent_architect_effect = yes

						name = gaudi_architect_var

						name = gaudi_recently_hired_independent_architect_modifier
						years = 3
					}

					#Cost of the Architect
					add_treasury = prev.gaudi_independent_architect_cost_value
				}
			}
		}
	}
	else = {
		if = {
			limit = {
				any_in_global_list = {

					root.gold_reserves > gaudi_independent_architect_cost_value

				}
			}

				position = 0 #We only want the first one here
				variable = mr_independent_architect_global_list

				order_by = var:gaudi_architect_experience
				check_range_bounds = no

				limit = {
					root.gold_reserves > gaudi_independent_architect_cost_value

						name = gaudi_architect_var

					}
				}

						}
