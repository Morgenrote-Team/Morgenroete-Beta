###Determine the artifacts at the end of an Expedition###

lepsius_archaeological_expedition_artifacts_effect = {
	if = {
		limit = {
			var:lepsius_expedition_success < 3
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 1 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 3
			var:lepsius_expedition_success < 5
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 2 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 5
			var:lepsius_expedition_success < 8
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 4 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 8
			var:lepsius_expedition_success < 12
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 6 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 12
			var:lepsius_expedition_success < 15
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 8 }
	}
	if = {
		limit = {
			var:lepsius_expedition_success >= 15
		}
		lepsius_add_x_minor_artifacts = { AMOUNT = 10 }
	}
}

###Check Archaeologist Rank###

check_lepsius_archaeologist_rank = {
	if = {
		limit = {
			var:lepsius_archaeologist_var = {
				var:lepsius_archaeologist_experience_var >= 3
				NOT = {
					var:lepsius_archaeologist_experience_var >= 5
				}
				NOR = {
					has_trait = lepsius_archaeologist_experienced
					has_trait = lepsius_archaeologist_famous
				}
			}
		}
		remove_trait = lepsius_archaeologist
		add_trait = lepsius_archaeologist_experienced
	}
	if = {
		limit = {
			var:lepsius_archaeologist_var = {
				var:lepsius_archaeologist_experience_var >= 5
				NOT = {
					has_trait = lepsius_archaeologist_famous
				}
			}
		}
		remove_trait = lepsius_archaeologist_experienced
		add_trait = lepsius_archaeologist_famous
	}
}

	}

		limit = { $AMOUNT$ = -1 }

		limit = { $AMOUNT$ = 0 }
		custom_tooltip = lepsius_add_x_fareastern_artifacts_0_tt

		limit = { $AMOUNT$ = 1 }
		custom_tooltip = lepsius_add_x_fareastern_artifacts_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = lepsius_add_x_fareastern_artifacts_plus_x_tt

# Creates a list of the ten countries with the most artifacts
#As this effect is called in an effect, it creates the list with normal variables
#Look at the customizable localizations to see how the countries are displayed.
#Basically the custom loc uses the variable to display both the countries in their order as well as the variables of those countries
lepsius_create_total_artifacts_ranking_effect = {
	ordered_country = {
		limit = {
			NOT = { #Hopefully keeps revolutions to appear in the ranking
				is_revolutionary = yes
			}
		}
		order_by = var:lepsius_total_artifacts
		max = 10 #Limits the list to the ten top countires
		check_range_bounds = no #prevents errors

		trigger_event = {

		}
	}
}

#Adds the Archaeologists as global variables
lepsius_create_archaeologist_ranking_effect = {
	#Removes the global variable list
	clear_global_variable_list = mr_archaeologist_ranking_global_list

	ordered_character = {
		limit = {
			has_role = character_role_lepsius_archaeologist
		}
		order_by = var:lepsius_archaeologist_experience_var
		check_range_bounds = no

		#This part adds all the countries to the list
		add_to_global_variable_list = {
			name = mr_archaeologist_ranking_global_list
			target = this
		}
	}
}
