#Effect used to add or remove progress from the scripted progress bar.
# AMOUNT = amount of progress to add/subtract
ito_add_x_medicinal_plants_progress_effect = {
	# Amount = amount of Progress to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if local_var is negative
		limit = { $AMOUNT$ < 0 }
		custom_tooltip = ito_add_x_medicinal_plants_progress_minus_x_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = ito_add_x_medicinal_plants_progress_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 0 }
		custom_tooltip = ito_add_x_medicinal_plants_progress_plus_x_tt
	}

	hidden_effect = {
		je:je_ito_botanical_expedition = {
			add_progress = {
				name = ito_botanical_expedition_medicinal_plants_progress_bar
				value = $AMOUNT$
			}
		}
	}
}

ito_add_x_aloe_vera_effect = { # AMOUNT = amount of Medicinal Plants to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = amount
		value = {
			value = $AMOUNT$
			round = yes
		}
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = ito_add_x_aloe_vera_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = ito_add_x_aloe_vera_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = ito_add_x_aloe_vera_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = ito_add_x_aloe_vera_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = ito_add_x_aloe_vera_plus_x_tt
	}

	#This part adds the AMOUNT of Medicinal Plants
	if = { #Fallback if the variable does not exist yet

	#Removes the global variable list
	clear_global_variable_list = mr_naturalist_ranking_global_list

		limit = {
			has_role = character_role_ito_naturalist
		}
		order_by = var:dubois_naturalist_experience_var
		check_range_bounds = no

		#This part adds all the countries to the list
		add_to_global_variable_list = {
			name = mr_naturalist_ranking_global_list
			target = this
		}
	}
}
