###### Chemist Project effects ######

### Element Isolation ###
#Button Effect
mendelejew_element_isolation_button_effect = {
	mendelejew_add_examined_element_from_list_effect = yes
	mendelejew_add_isolation_je_duration_effect = yes
	if = {
		limit = {
			has_variable = mendelejew_chemist_var
		}
		var:mendelejew_chemist_var = {
			set_character_busy_and_immortal = yes
		}
	}
	agassiz_add_x_generic_minerals_effect = {
		AMOUNT = -10000
	}
	hidden_effect = {
		add_journal_entry = {
			type = je_mendelejew_element_isolation
		}
	}
	if = {
		limit = { mendelejew_is_chemistry_automated_trigger = no }
		trigger_event = {
			id = mendelejew.302
			popup = yes
		}
	}
}
