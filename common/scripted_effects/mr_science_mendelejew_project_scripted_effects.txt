### Element Isolation ###
#Button Effect
mendelejew_element_isolation_button_effect = {
	hidden_effect = {
		mendelejew_add_examined_element_from_list_effect = yes
		mendelejew_add_isolation_je_duration_effect = yes
	}
	var:mendelejew_chemist_var ?= {
		set_character_busy_and_immortal = yes
	}
	agassiz_add_x_generic_minerals_effect = {
		AMOUNT = -10000
	}
	hidden_effect = {
		add_journal_entry = {
			type = je_mendelejew_element_isolation
		}
	}
	if = {
		limit = { mendelejew_is_chemistry_automated_trigger = no }
		trigger_event = {
			id = mendelejew.302
			popup = yes
		}
	}
}

### Improve Chemical Plant Project ###
#Button Effect
mr_chemist_improve_chemical_plant_button_effect = {
	if = {
		limit = {
			mendelejew_is_chemistry_automated_trigger = yes
		}
		clear_variable_list = mendelejew_improve_chemical_plants_buildings_chosen_variable_list
		ordered_scope_building = {
			position = 0 #We only want the first one here
			limit = {
				mendelejew_building_valid_for_chemical_plant_project_no_tooltip_trigger = yes
			}
			order_by = earnings
			root = {
				add_to_variable_list = {
					name = mendelejew_improve_chemical_plants_buildings_chosen_variable_list
					target = prev
				}
			}
		}
		add_journal_entry = {
			type = je_mendelejew_improve_fertilizer_project
		}
		var:mendelejew_chemist_var ?= {
			set_character_busy_and_immortal = yes
		}
	}
	else = {
		clear_variable_list = mendelejew_improve_chemical_plants_buildings_variable_list
		ordered_scope_building = {
			limit = {
				mendelejew_building_valid_for_chemical_plant_project_no_tooltip_trigger = yes
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = mendelejew_improve_chemical_plants_buildings_variable_list
					target = prev
				}
			}
		}
		set_variable = mendelejew_improve_chemical_plants_building_choser_var
		show_as_tooltip = {
			add_modifier = {
				name = mendelejew_chemist_fertilizer_cost_modifier
				multiplier = mr_local_project_cost_small_value
			}
			add_journal_entry = {
				type = je_mendelejew_improve_fertilizer_project
			}
		}
	}
}

### Improve Synthetics Factory Project ###
#Button Effect
mr_chemist_improve_synthetics_factory_button_effect = {
	if = {
		limit = {
			mendelejew_is_chemistry_automated_trigger = yes
		}
		clear_variable_list = mendelejew_improve_synthetics_factory_buildings_chosen_variable_list
		#Chooses a building depending on market supply and price of goods
		random_list = {
			1 = { #Synthetics Factory
				trigger = {
					any_scope_building = {
						mendelejew_building_valid_for_synthetics_factory_synthetics_project_no_tooltip_trigger = yes
					}
				}

				ordered_scope_building = {
					position = 0 #We only want the first one here
					limit = {
						mendelejew_building_valid_for_synthetics_factory_synthetics_project_no_tooltip_trigger = yes
					}
					order_by = earnings
					root = {
						add_to_variable_list = {
							name = mendelejew_improve_synthetics_factory_buildings_chosen_variable_list
							target = prev
						}
					}
				}

				modifier = {
					if = {
						limit = {
							market ?= {
								OR = {
									mg:silk ?= { market_goods_has_goods_shortage = yes }
									mg:dye ?= { market_goods_has_goods_shortage = yes }
									mg:silk = {
										market_goods_pricier > 0.25
									}
									mg:dye = {
										market_goods_pricier > 0.25
									}
								}
							}
						}
						add = 99
					}
				}
			}
			1 = { #Hydrogenation Plants
				trigger = {
					any_scope_building = {
						mendelejew_building_valid_for_synthetics_factory_hydrogenation_project_no_tooltip_trigger = yes
					}
				}

				ordered_scope_building = {
					position = 0 #We only want the first one here
					limit = {
						mendelejew_building_valid_for_synthetics_factory_hydrogenation_project_no_tooltip_trigger = yes
					}
					order_by = earnings
					root = {
						add_to_variable_list = {
							name = mendelejew_improve_synthetics_factory_buildings_chosen_variable_list
							target = prev
						}
					}
				}

				modifier = {
					if = {
						limit = {
							market ?= {
								OR = {
									mg:oil ?= { market_goods_has_goods_shortage = yes }
									mg:oil = {
										market_goods_pricier > 0.25
									}
								}
							}
						}
						add = 99
					}
				}
			}
			1 = { #Synthetic Rubber Factory
				trigger = {
					any_scope_building = {
						mendelejew_building_valid_for_synthetics_factory_synthetic_rubber_project_no_tooltip_trigger = yes
					}
				}

				ordered_scope_building = {
					position = 0 #We only want the first one here
					limit = {
						mendelejew_building_valid_for_synthetics_factory_synthetic_rubber_project_no_tooltip_trigger = yes
					}
					order_by = earnings
					root = {
						add_to_variable_list = {
							name = mendelejew_improve_synthetics_factory_buildings_chosen_variable_list
							target = prev
						}
					}
				}

				modifier = {
					if = {
						limit = {
							market ?= {
								OR = {
									mg:rubber ?= { market_goods_has_goods_shortage = yes }
									mg:rubber = {
										market_goods_pricier > 0.25
									}
								}
							}
						}
						add = 99
					}
				}
			}
		}
		add_journal_entry = {
			type = je_mendelejew_improve_synthetics_project
		}
		var:mendelejew_chemist_var ?= {
			set_character_busy_and_immortal = yes
		}
	}
	else = {
		clear_variable_list = mendelejew_improve_synthetics_factory_buildings_variable_list
		ordered_scope_building = {
			limit = {
				mendelejew_building_valid_for_synthetics_factory_project_no_tooltip_trigger = yes
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = mendelejew_improve_synthetics_factory_buildings_variable_list
					target = prev
				}
			}
		}
		set_variable = mendelejew_improve_synthetics_factory_building_choser_var
		show_as_tooltip = {
			add_modifier = {
				name = mendelejew_chemist_synthetics_cost_modifier
				multiplier = mr_local_project_cost_small_value
			}
			add_journal_entry = {
				type = je_mendelejew_improve_synthetics_project
			}
		}
	}
}
