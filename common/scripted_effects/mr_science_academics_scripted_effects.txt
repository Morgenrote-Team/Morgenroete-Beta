#All Academics (and Special Characters) should have a variable assigned to the character:

######GENERAL######

set_retirement_variable_effect = {
	if = {
		limit = {
			NOT = { has_variable = random_retirement_chooser_var }
		}
		set_variable = {
			name = random_retirement_chooser_var
			value = { 0 30 }
		}
	}
}

######ENGINEER######

#This effect kills the current engineer
# !!! Change stuff in both of the following effects !!!
kill_engineer_effect = {
	var:tesla_engineer_var = {
		kill_character = yes
	}

	remove_variable = tesla_engineer_var
	remove_modifier = gaudi_engineer_technology_spread_modifier
}
#This Effect retires the current Engineer
# Use "set_retirement_variable_effect" in the immediate effect when using this effect.
retire_engineer_effect = {
	custom_tooltip = {
		text = retire_engineer_effect_tt

		var:tesla_engineer_var = {
			kill_character = yes
		}
	}

	remove_variable = tesla_engineer_var
	remove_modifier = gaudi_engineer_technology_spread_modifier
}

#Removes the official Engineer without killing or retiring
tesla_remove_official_engineer_effect = {
	var:tesla_engineer_var ?= {
		set_character_busy_and_immortal = no
	}
	remove_variable = tesla_engineer_var
}

# Set new Engineer as official Engineer
tesla_set_new_engineer_as_engineer_effect = {
	custom_tooltip = {
		text = gaudi_replace_engineer_with_new_engineer_tt

		if = {
			limit = {
				has_variable = tesla_engineer_var
			}
			var:tesla_engineer_var = {
				set_character_busy_and_immortal = no
			}
		}

		set_variable = {
			name = tesla_engineer_var
			value = scope:tesla_new_engineer_scope
		}
		remove_modifier = gaudi_engineer_technology_spread_modifier
		add_modifier = {
			name = gaudi_engineer_technology_spread_modifier
			multiplier = tesla_percentage_more_advanced_countries_value
		}

		remove_variable = tesla_new_engineer_var

		if = {
			limit = { tesla_has_active_engineering_project_with_busy_engineer_trigger = yes }
			var:tesla_engineer_var = {
				set_character_busy_and_immortal = yes
			}
		}

		#Sets the Default Project up
		tesla_default_project_set_up_effect = yes

		#Starts projects as the AI
		ai_start_engineering_project_for_ai_effect = yes
	}
}

#Character Civil Engineer Experience Gain Effect
# scope = character
tesla_character_civil_engineer_experience_gain_effect = { #adds Chemist XP to the Character
	# AMOUNT = amount of XP to add
	custom_description = {
		text = "tesla_character_civil_engineer_experience_gain_effect_loc"
		subject = this
		value = $AMOUNT$

		#Fallback if there is no Experience yet
		if = {
			limit = {
				NOT = {
					has_variable = tesla_civil_engineer_experience_var
				}
			}
			engineer_experience_set_up_effect = yes
		}
		#Actual change of Experience
		change_variable = {
			name = tesla_civil_engineer_experience_var
			add = $AMOUNT$
		}
		#Clamping to prevent overflow
		clamp_variable = {
			name = tesla_civil_engineer_experience_var
			min = 0
			max = 300
		}
	}

	#Add Traits if the character does not have one yet
	if = {
		limit = {
			var:tesla_civil_engineer_experience_var >= 100
			NOT = {
				has_trait = tesla_civil_engineer
			}
		}
		if = {
			limit = {
				exists = owner
				owner.var:tesla_engineer_var ?= this
			}
			owner ?= {
				trigger_event = {
					id = academics.140
				}
			}
		}
		else = {
			post_notification = academics_character_is_civil_engineer_notification
		}
		hidden_effect = {
			add_trait = tesla_civil_engineer
		}
	}
	#Add Academic Title if Character does not have one yet
	academics_civil_engineer_add_academic_title_effect = yes
}

#Sets the Engineer to be a Civil Engineer
gaudi_set_civil_engineer_effect = {
	custom_tooltip = {
		text = gaudi_set_civil_engineer_tt

		if = {
			limit = {
				has_variable = tesla_engineer_var
			}
			var:tesla_engineer_var = {
				remove_modifier = academics_is_mechanical_engineer_modifier
				remove_modifier = academics_is_military_engineer_modifier
				remove_modifier = academics_is_electrical_engineer_modifier
				remove_modifier = academics_is_aeronautical_engineer_modifier

				add_modifier = {
					name = academics_is_civil_engineer_modifier
				}
			}

			#Adds the Default Project for the Engineer
			add_journal_entry = {
				type = je_gaudi_civil_research_project
			}
		}
	}
}

#Character Astronomer Experience Gain Effect
# scope = character
verrier_character_astronomer_experience_gain_effect = { #adds Astronomer XP to the Character
	# AMOUNT = amount of XP to add
	custom_description = {
		text = "verrier_character_astronomer_experience_gain_effect_loc"
		subject = this
		value = $AMOUNT$

		#Fallback if there is no Experience yet
		if = {
			limit = {
				NOT = {
					has_variable = verrier_astronomer_experience_var
				}
			}
			astronomer_experience_set_up_effect = yes
		}
		#Actual change of Experience
		change_variable = {
			name = verrier_astronomer_experience_var
			add = $AMOUNT$
		}
		#Clamping to prevent overflow
		clamp_variable = {
			name = verrier_astronomer_experience_var
			min = 0
			max = 300
		}
	}

	#Add Traits if the character does not have one yet
	if = {
		limit = {
			var:verrier_astronomer_experience_var >= 100
			var:verrier_astronomer_experience_var < 300
			NOR = {
				has_trait = verrier_astronomer_experienced
				has_trait = verrier_astronomer_famous
			}
		}
		if = {
			limit = {
				exists = owner
				owner.var:verrier_astronomer_var ?= this
			}
			owner ?= {
				trigger_event = {
					id = academics.102
				}
			}
		}
		else = {
			post_notification = academics_character_is_experienced_notification
		}
		hidden_effect = {
			remove_trait = verrier_astronomer
			add_trait = verrier_astronomer_experienced
		}
	}
	if = {
		limit = {
			var:verrier_astronomer_experience_var >= 300
			NOT = {
				has_trait = verrier_astronomer_famous
			}
		}
		if = {
			limit = {
				exists = owner
				owner.var:verrier_astronomer_var ?= this
			}
			owner ?= {
				trigger_event = {
					id = academics.103
				}
			}
		}
		else = {
			post_notification = academics_character_is_famous_notification
		}
		hidden_effect = {
			if = {
				limit = { has_trait = verrier_astronomer }
				remove_trait = verrier_astronomer
			}
			remove_trait = verrier_astronomer_experienced
			add_trait = verrier_astronomer_famous
		}
	}

	#Add Academic Title if Character does not have one yet
	academics_astronomer_add_academic_title_effect = yes
}

					id = academics.116

#Astronomical missions
academics_add_astronomical_mission = {
	if = {
		limit = { #Fallback if the variable doesn't exist
			NOT = { has_variable = academics_mission_counter_var }
		}
		set_variable = { #used to count the academic missions
			name = academics_mission_counter_var
			value = 0
		}
	}
	change_variable = { #The mission counter is used to not flood the player with missions. It is raised by 1 for each mission and gets checked before adding a new mission.
		name = academics_mission_counter_var
		add = 1
	}
	set_variable = academics_astronomical_mission_var #A country should have maximum one astronomical mission. This variable is used to check that
	academics_set_mission_timer_effect = yes #Sets a timer for New missions to be unavailable
}

academics_remove_astronomical_mission = {
	change_variable = { #Removes the mission from the mission counter
		name = academics_mission_counter_var
		add = -1
	}
	remove_variable = academics_astronomical_mission_var #removes the astronomical mission and allows new astronomical missions
}

#Effect to add Title randomly depending on various factors
# EXPERIENCE_VARIABLE = experience_variable of the character
# BASE_CHANCE_DOCTOR = Base chance for the specific role
# BASE_CHANCE_PROFESSOR = Base chance for Professor
academics_character_add_title_from_experience_gain_effect = {
	if = {
		limit = {
			owner = {
				has_technology_researched = academia
			}
		}

		#Doctor's degree
		if = {
			limit = {
				NOT = { has_modifier = academics_character_is_doctor_modifier }
			}

			if = {
				limit = {
					OR = {
						$EXPERIENCE_VARIABLE$ >= 150
						has_role = character_role_panum_physician
					}
				}

				academics_add_doctor_title_effect = yes
			}
			else = {
				random_list = {
					#Nothing
					0 = {
						modifier = {
							add = {
								value = 100
								subtract = {
									value = {
										value = $EXPERIENCE_VARIABLE$
										add = $BASE_CHANCE_DOCTOR$
										multiply = 0.3
									}
									multiply = academics_traits_for_degree_gain_chance_value
									min = 1
									max = 100
								}
							}
						}
					}
					#Gain Degree
					0 = {
						academics_add_doctor_title_effect = yes

						modifier = {
							add = {
								value = {
									value = $EXPERIENCE_VARIABLE$
									add = $BASE_CHANCE_DOCTOR$
									multiply = 0.3
								}
								multiply = academics_traits_for_degree_gain_chance_value
								min = 1
								max = 100
							}
						}
					}
				}
			}
		}
		#Professor
		else_if = {
			limit = {
				NOT = { has_modifier = academics_character_is_professor_modifier }
			}

			if = {
				limit = { $EXPERIENCE_VARIABLE$ >= 300 }

				academics_add_professor_title_effect = yes
			}
			else = {
				random_list = {
					#Nothing
					0 = {
						modifier = {
							add = {
								value = 100
								subtract = {
									value = {
										value = $EXPERIENCE_VARIABLE$
										add = $BASE_CHANCE_PROFESSOR$
										multiply = 0.3
									}
									multiply = academics_traits_for_degree_gain_chance_value
									min = 1
									max = 100
								}
							}
						}
					}
					#Gain Degree
					0 = {
						academics_add_professor_title_effect = yes

						modifier = {
							add = {
								value = {
									value = $EXPERIENCE_VARIABLE$
									add = $BASE_CHANCE_PROFESSOR$
									multiply = 0.3
								}
								multiply = academics_traits_for_degree_gain_chance_value
								min = 1
								max = 100
							}
						}
					}
				}
			}
		}
	}
