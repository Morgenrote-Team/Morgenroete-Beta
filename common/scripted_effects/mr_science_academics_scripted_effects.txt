######GENERAL######

set_retirement_variable_effect = {
	if = {
		limit = {
			NOT = { has_variable = random_retirement_chooser_var }
		}
		set_variable = {
			name = random_retirement_chooser_var
			value = { 0 30 }
		}
	}
}

#This effect kills the current engineer
# !!! Change stuff in both of the following effects !!!
kill_engineer_effect = {
	var:tesla_engineer_var = {
		free_character_from_void = no
	}

	remove_variable = tesla_engineer_var
	remove_modifier = gaudi_engineer_technology_spread_modifier

	if = {
		limit = { has_modifier = academics_engineer_emeritus_delayed_modifier }
		remove_modifier = academics_engineer_emeritus_delayed_modifier
	}
}
#This Effect retires the current Engineer
# Use "set_retirement_variable_effect" in the immediate effect when using this effect.
retire_engineer_effect = {
	custom_tooltip = {
		text = retire_engineer_effect_tt

		var:tesla_engineer_var = {
			free_character_from_void = no
		}
	}

	remove_variable = tesla_engineer_var
	remove_modifier = gaudi_engineer_technology_spread_modifier

	if = {
		limit = { has_modifier = academics_engineer_emeritus_delayed_modifier }
		remove_modifier = academics_engineer_emeritus_delayed_modifier
	}
}

# Set new Engineer as official Engineer
tesla_set_new_engineer_as_engineer_effect = {
	custom_tooltip = {
		text = gaudi_replace_engineer_with_new_engineer_tt

		if = {
			limit = {
				has_variable = tesla_engineer_var
			}
			var:tesla_engineer_var = {
				free_character_from_void = no
			}
		}

		set_variable = {
			name = tesla_engineer_var
			value = scope:tesla_new_engineer_scope
		}
		remove_modifier = gaudi_engineer_technology_spread_modifier
		add_modifier = {
			name = gaudi_engineer_technology_spread_modifier
			multiplier = tesla_percentage_more_advanced_countries_value
		}

		remove_variable = tesla_new_engineer_var

		if = {
			limit = { tesla_has_active_engineering_project_with_busy_engineer_trigger = yes }
			var:tesla_engineer_var = {
				set_character_busy_and_immortal = yes
			}
		}

		#Sets the Default Project up
		tesla_default_project_set_up_effect = yes

		#Starts projects as the AI
		ai_start_engineering_project_for_ai_effect = yes
	}
}

# Remove new Engineer
gaudi_remove_new_engineer_effect = {
	scope:tesla_new_engineer_scope = {
		free_character_from_void = no
	}

	remove_variable = tesla_new_engineer_var
}

#Set Engineer unbusy
# Made this effect to have a central effect for it and to fire the ai_automation from the same effect
tesla_set_engineer_unbusy_effect = {
	var:tesla_engineer_var ?= {
		set_character_busy_and_immortal = no
	}
	ai_start_engineering_project_for_ai_effect = yes
}

# Used to set up the Experience of an engineer
engineer_experience_set_up_effect = {
	if = {
		limit = {
			OR = {
				has_trait = tesla_civil_engineer
				has_trait = tesla_mechanical_engineer
				has_trait = tesla_military_engineer
				has_trait = tesla_electrical_engineer
				has_trait = tesla_aeronautical_engineer
			}
		}
		random_list = {
			1 = {
				trigger = { has_trait = tesla_civil_engineer }

				#Effect
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_civil_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_civil_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
			}
			1 = {
				trigger = { has_trait = tesla_mechanical_engineer }

				#Effect
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_mechanical_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_mechanical_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
			}
			1 = {
				trigger = { has_trait = tesla_military_engineer }

				#Effect
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_military_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_military_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
			}
			1 = {
				trigger = { has_trait = tesla_electrical_engineer }

				#Effect
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_electrical_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_electrical_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
			}
			1 = {
				trigger = { has_trait = tesla_aeronautical_engineer }

				#Effect
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_aeronautical_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_aeronautical_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
			}
		}
	}
	else = {
		random_list = { #Here, a random focus is chosen for the engineer. Then the Experience and trait are set up for them.
			1 = {
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_civil_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_civil_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
				add_trait = tesla_civil_engineer
			}
			1 = {
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_mechanical_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_civil_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
				add_trait = tesla_mechanical_engineer
			}
			1 = {
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_military_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_civil_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
				add_trait = tesla_military_engineer
				modifier = {
					if = {
						limit = {
							interest_group = {
								is_interest_group_type = ig_armed_forces
							}
						}
						add = 5
					}
				}
			}
			1 = {
				trigger = {
					owner = {
						has_technology_researched = verrier_physics_tech
					}
				}
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_electrical_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_electrical_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
				add_trait = tesla_electrical_engineer
			}
			1 = {
				trigger = {
					owner = {
						has_technology_researched = curtiss_reconnaissance_balloon_tech
					}
				}
				add_modifier = { #This variable specialises the engineer. It can be changed by the player during the game
					name = academics_is_aeronautical_engineer_modifier
				}
				general_experience_set_up_effect = {
					TYPE = tesla_aeronautical_engineer_experience_var
					UNI_PM = pm_dubois_university_mathematics
				}
				add_trait = tesla_aeronautical_engineer
			}
		}
	}
	if = { #Sets all not yet existing experience types
		limit = {
			NOT = {
				has_variable = tesla_civil_engineer_experience_var
			}
		}
		set_variable = {
			name = tesla_civil_engineer_experience_var
			value = 0
		}
	}
	if = {
		limit = {
			NOT = {
				has_variable = tesla_mechanical_engineer_experience_var
			}
		}
		set_variable = {
			name = tesla_mechanical_engineer_experience_var
			value = 0
		}
	}
	if = {
		limit = {
			NOT = {
				has_variable = tesla_military_engineer_experience_var
			}
		}
		set_variable = {
			name = tesla_military_engineer_experience_var
			value = 0
		}
	}
	if = {
		limit = {
			NOT = {
				has_variable = tesla_electrical_engineer_experience_var
			}
		}
		set_variable = {
			name = tesla_electrical_engineer_experience_var
			value = 0
		}
	}
	if = {
		limit = {
			NOT = {
				has_variable = tesla_aeronautical_engineer_experience_var
			}
		}
		set_variable = {
			name = tesla_aeronautical_engineer_experience_var
			value = 0
		}
	}
}

#Engineer Experience gain
engineer_experience_gain_effect = { #adds XP to the Engineer
	# Amount = amount of XP to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = xp_amount
		value = $AMOUNT$
	}

	if = {
		limit = {
			has_variable = tesla_engineer_var #should only add experience if there actually is an Engineer
		}
		if = {
			limit = {
				var:tesla_engineer_var = {
					has_modifier = academics_is_civil_engineer_modifier
				}
			}
			custom_tooltip = {
				text = gaudi_civil_engineer_experience_gain_tt
				if = {
					limit = {
						NOT = {
							var:tesla_engineer_var = { #Fallback to make sure there are no errors if the Engineer does not have the variable yet
								has_variable = tesla_civil_engineer_experience_var
							}
						}
					}
					var:tesla_engineer_var = { #Sets the experience in case it was not set yet
						engineer_experience_set_up_effect = yes
					}
				}
				var:tesla_engineer_var = {
					change_variable = {
						name = tesla_civil_engineer_experience_var
						add = $AMOUNT$
					}
					clamp_variable = {
						name = tesla_civil_engineer_experience_var
						min = 0
						max = 300
					}
				}
			}
		}
		else_if = {

# Set new Astronomer as official Astronomer
verrier_set_new_astronomer_as_astronomer_effect = {
	custom_tooltip = {
		text = verrier_replace_astronomer_with_new_astronomer_tt

		if = {
			limit = {
				has_variable = verrier_astronomer_var
			}
			var:verrier_astronomer_var = {
				free_character_from_void = no
			}
		}

		set_variable = {
			name = verrier_astronomer_var
			value = scope:verrier_new_astronomer_scope
		}

		#Sets the astronomer busy if there is an active project
		if = {
			limit = { verrier_astronomer_project_active_trigger = yes }
			var:verrier_astronomer_var = {
				set_character_busy_and_immortal = yes
			}
		}

		#Extracts the name as a variable
		add_to_global_variable_list = {
			name = com_character_name_extract
			target = var:verrier_new_astronomer_var
		}

		remove_variable = verrier_new_astronomer_var

		#Starts Project for new Astronomer if automation is active
		ai_astronomy_automation_effect = yes
	}
}

# Used to set up the Experience of a character
astronomer_experience_set_up_effect = {
	#Adds astronomer experience (see effect for further information)
	general_experience_set_up_effect = {
		TYPE = verrier_astronomer_experience_var
		UNI_PM = pm_dubois_university_mathematics
	}

	if = { #Adds the Experienced Astronomer Trait if Experience is above 100
		limit = {
			var:verrier_astronomer_experience_var >= 100
			NOT = {
				var:verrier_astronomer_experience_var >= 300
			}
			NOR = {
				has_trait = verrier_astronomer_experienced
				has_trait = verrier_astronomer_famous
			}
		}
		remove_trait = verrier_astronomer
		add_trait = verrier_astronomer_experienced
	}
	if = { #Adds the Famous Astronomer Trait if Experience is above 300
		limit = {
			var:verrier_astronomer_experience_var >= 300
			NOT = {
				has_trait = verrier_astronomer_famous
			}
		}
		remove_trait = verrier_astronomer
		remove_trait = verrier_astronomer_experienced
		add_trait = verrier_astronomer_famous
	}
}

#Astronomical missions
academics_add_astronomical_mission = {
	if = {
		limit = { #Fallback if the variable doesn't exist
			NOT = { has_variable = academics_mission_counter_var }
		}
		set_variable = { #used to count the academic missions
			name = academics_mission_counter_var
			value = 0
		}
	}
	change_variable = { #The mission counter is used to not flood the player with missions. It is raised by 1 for each mission and gets checked before adding a new mission.
		name = academics_mission_counter_var
		add = 1
	}
	set_variable = academics_astronomical_mission_var #A country should have maximum one astronomical mission. This variable is used to check that
	academics_set_mission_timer_effect = yes #Sets a timer for New missions to be unavailable
}

academics_remove_astronomical_mission = {
	change_variable = { #Removes the mission from the mission counter
		name = academics_mission_counter_var
		add = -1
	}
	remove_variable = academics_astronomical_mission_var #removes the astronomical mission and allows new astronomical missions
}

	}

#This Effect sets up a specific kind of experience, depending on the input
### TYPE = experience type
### UNI_PM = Production Method of the University
general_experience_set_up_effect = {
	if = {
		limit = {
			NOT = {
				has_variable = $TYPE$
			}
		}
		if = { #Each year over 30 gives 9 Experience
			limit = {
				age > 30
			}
			set_variable = {
				name = $TYPE$
				value = age
			}
			change_variable = {
				name = $TYPE$
				subtract = 30
			}
			change_variable = {
				name = $TYPE$
				multiply = 9
			}
		}
		else = {
			set_variable = {
				name = $TYPE$
				value = 0
			}
		}
		# 5 Experience per 10% of Technical-Mathematical Universities
		if = { #Added a first check to reduce number of checks (divide and conquer)
			limit = {
				owner = {
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.5
						has_active_production_method = $UNI_PM$
					}
				}
			}
			if = {
				limit = {
					owner = {
						any_scope_building = {
							filter = {
								is_building_type = building_university
							}
							percent >= 1
							has_active_production_method = $UNI_PM$
						}
					}
				}
				change_variable = {
					name = $TYPE$
					add = 50
				}
			}
			else_if = {

							filter = {
								is_building_type = building_university

							percent >= 0.9

					}
				}
				change_variable = {
					name = $TYPE$
					add = 45

						}
					}

				}

							percent >= 0.2

					}

				value = 0
			}
		}
	}

#Engineer
academics_engineer_cleanup_effect = {
	ordered_character_in_void = {
		limit = {
			has_variable = mr_is_engineer_var
		}
		check_range_bounds = no

		if = {
			limit = {
				OR = {
					owner ?= {
						NOT = {
							var:tesla_engineer_var ?= prev
						}
					}
					NOT = {
						exists = owner
					}
				}
			}

			kill_character = yes
			free_character_from_void = no
		}
	}
}
