#Handles Chemistry for the AI and player automation

# This effect will choose projects depending on various circumstances
ai_chemistry_automation_effect = {
	if = {
		limit = {
			mendelejew_is_chemistry_automated_trigger = yes
			mendelejew_has_active_chemist_no_tooltip_trigger = yes
		}

		#Uses a random list to decide which project to do.
		# This allows for weighting using various factors.
		random_list = {
			### Element Isolation ###
			100 = {
				#Triggers
				trigger = {
					any_scope_state = {
						is_production_method_active = {
							building_type = building_university
							production_method = pm_dubois_university_mathematics
						}
					}
					any_scope_building = {
						is_building_type = building_verrier_observatory
						level > 0
					}
					has_variable = agassiz_generic_minerals_var
					var:agassiz_generic_minerals_var >= 10000
					mendelejew_element_can_isolate_trigger = yes
				}

				#Modifiers / Weighting
				modifier = {
					if = {
						limit = {
							var:mendelejew_chemist_var ?= {
								has_trait = mendelejew_chemist_inspector
							}
						}
						add = 1000
					}
				}

				#Effect
				mendelejew_element_isolation_button_effect = yes
			}
			### Element Discovery ###
			100 = {
				#Triggers
				trigger = {
					any_scope_building = {
						is_building_type = building_university
						level > 0
					}
					any_scope_building = {
						is_building_type = building_verrier_observatory
						level > 0
					}
					has_variable = agassiz_generic_minerals_var
					var:agassiz_generic_minerals_var >= 5000
					mendelejew_element_can_discover_trigger = yes
				}

				#Modifiers / Weighting
				modifier = {
					if = {
						limit = {
							var:mendelejew_chemist_var ?= {
								has_trait = mendelejew_chemist_analyst
							}
						}
						add = 100
					}
				}

				#Effect
				mendelejew_element_discovery_button_effect = yes
			}
			### Element Theorization ###
			100 = {
				#Triggers
				trigger = {
					any_scope_building = {
						is_building_type = building_university
						level > 0
					}
					any_scope_building = {
						is_building_type = building_verrier_observatory
						level > 0
					}
					mendelejew_element_can_theorize_trigger = yes
				}

				#Modifiers / Weighting
				modifier = {
					if = {
						limit = {
							var:mendelejew_chemist_var ?= {
								has_trait = mendelejew_chemist_theorist
							}
						}
						add = 100
					}
					if = {
						limit = {
							var:mendelejew_chemist_var ?= {
								has_trait = mendelejew_chemist_periodic_table_inventor
							}
						}
						add = 100
					}
				}

				#Effect
				mendelejew_element_theorization_button_effect = yes
			}
			### Improve Fertilizer Plant ###
			10 = {
				#Triggers
				trigger = {
					NOT = {
						has_variable = mendelejew_improve_chemical_plants_building_choser_var
					}
					any_scope_building = {
						is_building_type = building_chemical_plant
						level > 0
						NAND = {
							has_modifier = mendelejew_building_throughput_add_modifier
							has_modifier = mendelejew_building_production_mult_modifier
							has_modifier = mendelejew_building_goods_input_mult_modifier
						}
					}
				}

				#Modifiers / Weighting
				modifier = {
					#Shortage --> Very important
					if = {
						limit = {
							market = {
								mg:fertilizer ?= { market_goods_has_goods_shortage = yes }
							}
						}
						add = 1000
					}
					#High Prices --> important
					if = {
						limit = {
							market = {
								mg:fertilizer = {
									market_goods_pricier > 0.25
								}
							}
						}
						add = 190
					}
					#Ai Focus on Economy --> kind of important
					if = {
						limit = {
							OR = {
								has_strategy = ai_strategy_agricultural_expansion
								has_strategy = ai_strategy_industrial_expansion
							}
						}
						add = 90
					}
				}

				#Effect
				mr_chemist_improve_chemical_plant_button_effect = yes
			}
			### Improve Explosives Factories ###
			10 = {
				#Triggers
				trigger = {
					NOT = {
						has_variable = mendelejew_improve_explosives_factory_building_choser_var
					}
					any_scope_building = {
						is_building_type = building_explosives_factory
						level > 0
						NAND = {
							has_modifier = mendelejew_building_throughput_add_modifier
							has_modifier = mendelejew_building_production_mult_modifier
							has_modifier = mendelejew_building_goods_input_mult_modifier
						}
					}
				}

				#Modifiers / Weighting
				modifier = {
					#Shortage --> Very important
					if = {
						limit = {
							market = {
								mg:explosives ?= { market_goods_has_goods_shortage = yes }
							}
						}
						add = 1000
					}
					#High Prices --> important
					if = {
						limit = {
							market = {
								mg:explosives = {
									market_goods_pricier > 0.25
								}
							}
						}
						add = 190
					}
					#Ai Focus on Economy --> kind of important
					if = {
						limit = {
							OR = {
								has_strategy = ai_strategy_agricultural_expansion
								has_strategy = ai_strategy_industrial_expansion
							}
						}
						add = 90
					}
				}

				#Effect
				ordered_scope_building = {
					position = 0 #We only want the first one here
					limit = {
						level > 0
						is_building_type = building_explosives_factory
						NAND = {
							has_modifier = mendelejew_building_throughput_add_modifier
							has_modifier = mendelejew_building_production_mult_modifier
							has_modifier = mendelejew_building_goods_input_mult_modifier
						}
					}
					order_by = earnings
					save_scope_as = chosen_building
				}
				add_journal_entry = {
					type = je_mendelejew_improve_explosives_project
					target = scope:chosen_building
				}
				je:je_mendelejew_improve_explosives_project = {
					add_modifier = {
						name = mendelejew_chemist_explosives_cost_modifier
						multiplier = root.mr_local_project_cost_small_value
					}
				}
				if = {
					limit = {
						has_variable = mendelejew_chemist_var
					}
					var:mendelejew_chemist_var = {
						set_character_busy_and_immortal = yes
					}
				}
			}
			### Invent Synthetic Oil ###
			100 = {
				#Triggers
				trigger = {
					NOT = {
						has_technology_researched = mendelejew_synthetic_fuel_tech
					}
					has_technology_researched = verrier_modern_physics_tech
					has_technology_researched = plastics
					has_technology_researched = combustion_engine
					any_scope_building = {
						is_building_type = building_verrier_observatory
						level > 0
					}
					country_has_building_type_levels = {
						target = bt:building_university
						value  >= 10
					}
					any_scope_building = {
						is_building_type = building_synthetics_plant
						level > 0
						has_modifier = mendelejew_building_throughput_add_modifier
						has_modifier = mendelejew_building_production_mult_modifier
						has_modifier = mendelejew_building_goods_input_mult_modifier
					}
				}

				#Modifiers / Weighting
				modifier = {
					#Germany --> Higher Chance
					if = {
						limit = {
							OR = {
								c:GER ?= this
								was_formed_from = GER
							}
						}
						add = 1000
					}
					#Shortage --> Very important
					if = {
						limit = {
							market = {
								mg:oil ?= { market_goods_has_goods_shortage = yes }
							}
						}
						add = 5000
					}
					#High Prices --> important
					if = {
						limit = {
							market = {
								mg:oil = {
									market_goods_pricier > 0.25
								}
							}
						}
						add = 2500
					}
					#Ai Focus on Economy --> kind of important
					if = {
						limit = {
							OR = {
								has_strategy = ai_strategy_agricultural_expansion
								has_strategy = ai_strategy_industrial_expansion
							}
						}
						add = 100
					}
				}

				#Effect
				add_journal_entry = { type = je_mendelejew_synthetic_oil }
				if = {
					limit = {
						has_variable = mendelejew_chemist_var
					}
					var:mendelejew_chemist_var = {
						set_character_busy_and_immortal = yes
					}
				}
			}
			### Invent Synthetic Rubber ###
			100 = {
				#Triggers
				trigger = {
					NOT = {
						has_technology_researched = mendelejew_synthetic_rubber_tech
					}
					has_technology_researched = verrier_modern_physics_tech
					has_technology_researched = plastics
					has_technology_researched = combustion_engine
					any_scope_building = {
						is_building_type = building_verrier_observatory
						level > 0
					}
					country_has_building_type_levels = {
						target = bt:building_university
						value  >= 10
					}
					any_scope_building = {
						is_building_type = building_synthetics_plant
						level > 0
						has_modifier = mendelejew_building_throughput_add_modifier
						has_modifier = mendelejew_building_production_mult_modifier
						has_modifier = mendelejew_building_goods_input_mult_modifier
					}
				}

				#Modifiers / Weighting
				modifier = {
					#Germany --> Higher Chance
					if = {
						limit = {
							OR = {
								c:GER ?= this
								was_formed_from = GER
							}
						}
						add = 1000
					}
					#Shortage --> Very important
					if = {
						limit = {
							market = {
								mg:rubber ?= { market_goods_has_goods_shortage = yes }
							}
						}
						add = 5000
					}
					#High Prices --> important
					if = {
						limit = {
							market = {
								mg:rubber = {
									market_goods_pricier > 0.25
								}
							}
						}
						add = 2500
					}
					#Ai Focus on Economy --> kind of important
					if = {
						limit = {
							OR = {
								has_strategy = ai_strategy_agricultural_expansion
								has_strategy = ai_strategy_industrial_expansion
							}
						}
						add = 100
					}
				}

				#Effect
				add_journal_entry = { type = je_mendelejew_synthetic_rubber }
				if = {
					limit = {
						has_variable = mendelejew_chemist_var
					}
					var:mendelejew_chemist_var = {
						set_character_busy_and_immortal = yes
					}
				}
			}
		}
	}
}
