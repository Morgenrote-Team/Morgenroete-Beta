mr_create_random_physicist_for_character_creation_effect = {
	random_list = {
		1 = {
			create_character = {
				female = yes
				trait_generation = {}
				save_scope_as = curie_new_physicist_scope
				role = character_role_curie_physicist
			}
			scope:curie_new_physicist_scope = {
				add_trait = curie_physicist
				mr_trait_generation_physicist_effect = yes
				mr_make_character_physicist_effect = yes
			}

			set_variable = {
				name = curie_new_physicist_var
				value = scope:curie_new_physicist_scope
			}

			modifier = {
				if = {
					limit = { has_law = law_type:law_women_own_property }
					add = 4
				}
				if = {
					limit = { has_law = law_type:law_women_in_the_workplace }
					add = 9
				}
				if = {
					limit = { has_law = law_type:law_womens_suffrage }
					add = 19
				}
			}
		}
		99 = {
			create_character = {
				trait_generation = {}
				save_scope_as = curie_new_physicist_scope
				role = character_role_curie_physicist
			}
			scope:curie_new_physicist_scope = {
				add_trait = curie_physicist
				mr_trait_generation_physicist_effect = yes
				mr_make_character_physicist_effect = yes
			}

			set_variable = {
				name = curie_new_physicist_var
				value = scope:curie_new_physicist_scope
			}

			modifier = {
				if = {
					limit = { has_law = law_type:law_women_own_property }
					add = -4
				}
				if = {
					limit = { has_law = law_type:law_women_in_the_workplace }
					add = -9
				}
				if = {
					limit = { has_law = law_type:law_womens_suffrage }
					add = -19
				}
			}
		}
	}
}

mr_make_character_physicist_effect = {
	physicist_experience_set_up_effect = yes
	set_variable = mr_is_mr_character
	academics_character_set_role_retirement_effect = {
		ROLE = character_role_curie_physicist
		AGE_LEVEL = 2
	}
