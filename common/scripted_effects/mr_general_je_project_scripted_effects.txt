###### General Project effect ######

#### General effect to use in JEs that calculates progress, etc. ###
#Takes these Parameters:
# VALUE_NAME = name of the value for the monthly progress
# PROGRESS_BAR_NAME = Name of the Progress Bar used in the JE
# PROGRESS_COSTS = name of script_value that controls the cost of the next step
# PROGRESS_FILLED_EFFECT = effect that gets executed when the progress bar is filled
mr_journal_entry_project_monthly_effect = {
	#Sets Variables if they are not yet set
	scope:journal_entry = {
		if = {
			limit = {
				NOT = { has_variable = mr_monthly_progress_var }
			}

			set_variable = {
				name = mr_monthly_progress_var
				value = root.$VALUE_NAME$ #Calculated in script value
			}
		}
		if = {
			limit = {
				NOT = { has_variable = mr_je_progress_var }
			}

			set_variable = {
				name = mr_je_progress_var
				value = 0
			}
		}
	}

	#Adds monthly change
	scope:journal_entry = {
		set_variable = {
			name = mr_monthly_progress_var
			value = root.$VALUE_NAME$
		}
		change_variable = {
			name = mr_je_progress_var
			add = root.$VALUE_NAME$
		}
	}

	#If the Progress reaches the cost, the modifier will be added to a state
	if = {
		limit = {
			scope:journal_entry = {
				var:mr_je_progress_var >= $PROGRESS_COSTS$
			}
		}

		#The cost will be subtracted from the current progress
		scope:journal_entry = {
			change_variable = {
				name = mr_je_progress_var
				subtract = $PROGRESS_COSTS$
			}
			if = {
				limit = { var:mr_je_progress_var >= $PROGRESS_COSTS$ }
				set_bar_progress = {
					name = $PROGRESS_BAR_NAME$
					value = $PROGRESS_COSTS$
				}
			}
			else = {
				set_bar_progress = {
					name = $PROGRESS_BAR_NAME$
					value = var:mr_je_progress_var
				}
			}
		}

		#Effect when stage is completed
		$PROGRESS_FILLED_EFFECT$ = yes
	}
}

#### General effect to use in JEs that handles the monthly progress (1 Modifier) ###
#Takes these Parameters:
# PROGRESS_BAR_NAME = Name of the Progress Bar used in the JE
# PROGRESS_COST = name of script_value or number that controls the cost of the next step
# MODIFIER_1 = name of the first modifier to be added
# BUILDING_LIST_NAME = name of the variable list in which the buildingsa are saved
# ROLE = Role name for experience gain effect
mr_journal_entry_building_1_modifier_project_monthly_effect = {
	scope:journal_entry = {
		#Executed if Progress is high enough
		if = {
			limit = {
				scripted_bar_progress = {
					name = $PROGRESS_BAR_NAME$
					value >= $PROGRESS_COST$
				}
			}
			#Removes Progress
			set_bar_progress = {
				name = $PROGRESS_BAR_NAME$
				value = 0
			}
			#Adds third Modifier, Experience and chooses new building if possible
			var:mr_target_building_var ?= {
				add_modifier = {
					name = $MODIFIER_1$
				}
			}

			root = {
				#Removes old building from List
				ordered_in_list = {
					variable = $BUILDING_LIST_NAME$
					position = 0

					root = {
						remove_list_variable = {
							name = $BUILDING_LIST_NAME$
							target = prev
						}
					}
				}
				#Adds new building if possible
				if = {
					limit = { has_variable_list = $BUILDING_LIST_NAME$ }
					ordered_in_list = {
						variable = $BUILDING_LIST_NAME$
						position = 0

						scope:journal_entry = {
							set_variable = {
								name = mr_target_building_var
								value = prev
							}
						}
					}
				}
				mr_academics_small_experience_gain_effect = { ROLE = $ROLE$ }
			}

			change_variable = {
				name = mr_num_improved_buildings_var
				add = 1
			}
		}
	}
	mr_journal_entry_save_remaining_list_entries_as_variable_effect = { LIST_NAME = $BUILDING_LIST_NAME$ }
}

#### General effect to use in JEs that handles the monthly progress ###
#Takes these Parameters:
# PROGRESS_BAR_NAME = Name of the Progress Bar used in the JE
# PROGRESS_COST = name of script_value or number that controls the cost of the next step
# MODIFIER_1 = name of the first modifier to be added
# MODIFIER_2 = name of the second modifier to be added
# MODIFIER_3 = name of the third modifier to be added
# BUILDING_LIST_NAME = name of the variable list in which the buildingsa are saved
# ROLE = Role name for experience gain effect
mr_journal_entry_building_3_modifier_project_monthly_effect = {
	scope:journal_entry = {
		#Executed if Progress is high enough
		if = {
			limit = {
				scripted_bar_progress = {
					name = $PROGRESS_BAR_NAME$
					value >= $PROGRESS_COST$
				}
			}
			#Removes Progress
			set_bar_progress = {
				name = $PROGRESS_BAR_NAME$
				value = 0
			}
			#Adds first Modifier
			if = {
				limit = {
					var:mr_target_building_var ?= {
						NOT = { has_modifier = $MODIFIER_1$ }
					}
				}
				var:mr_target_building_var ?= {
					add_modifier = {
						name = $MODIFIER_1$
					}
				}
			}
			#Adds second Modifier
			else_if = {
				limit = {
					var:mr_target_building_var ?= {
						NOT = { has_modifier = $MODIFIER_2$ }
					}
				}
				var:mr_target_building_var ?= {
					add_modifier = {
						name = $MODIFIER_2$
					}
				}
			}
			#Adds third Modifier, Experience and chooses new building if possible
			else = {
				var:mr_target_building_var ?= {
					add_modifier = {
						name = $MODIFIER_3$
					}
				}

				root = {
					#Removes old building from List
					ordered_in_list = {
						variable = $BUILDING_LIST_NAME$
						position = 0

						root = {
							remove_list_variable = {
								name = $BUILDING_LIST_NAME$
								target = prev
							}
						}
					}
					#Adds new building if possible
					if = {
						limit = { has_variable_list = $BUILDING_LIST_NAME$ }
						ordered_in_list = {
							variable = $BUILDING_LIST_NAME$
							position = 0

							scope:journal_entry = {
								set_variable = {
									name = mr_target_building_var
									value = prev
								}
							}
						}
					}
					mr_academics_small_experience_gain_effect = { ROLE = $ROLE$ }
				}

				change_variable = {
					name = mr_num_improved_buildings_var
					add = 1
				}
			}
		}
	}
	mr_journal_entry_save_remaining_list_entries_as_variable_effect = { LIST_NAME = $BUILDING_LIST_NAME$ }
}

mr_journal_entry_save_remaining_list_entries_as_variable_effect = {
	scope:journal_entry = {
		set_variable = {
			name  = mr_number_of_remaining_list_entries_var
			value = {
				root = {
					every_in_list = {
						variable = $LIST_NAME$
						add = 1
					}
				}
				subtract = 1
			}
		}
	}
}
