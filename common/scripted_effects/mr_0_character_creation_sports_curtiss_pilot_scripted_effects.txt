# Creates a new Pilot without losing the old one
#This one was a little harder as the scope was only valid in the random_list
#Therefore a Variable is set to save the scope again. then gets deleted
mr_create_historical_pilot_for_character_creation_effect = {
	random_list = {
		25 = { #Antoine de Saint-Exupéry, France
			trigger = {
				NOT = { has_global_variable = mr_de_saint_exupery_created_global_var }
				country_has_primary_culture = cu:french
				has_state_in_state_region = STATE_RHONE
				year >= 1922 #Sometimes around then was military pilot
				year <= 1940 #Age 40
			}
			create_character = {
				template = mr_fra_de_saint_exupery_antoine_character_template
				save_scope_as = curtiss_new_pilot_scope
			}
			scope:curtiss_new_pilot_scope = {
				add_trait = curtiss_pilot
				set_variable = mr_is_pilot_var
				pilot_experience_set_up_effect = yes
				place_character_in_void = 1200
			}

			set_variable = {
				name = curtiss_new_pilot_var
				value = scope:curtiss_new_pilot_scope
			}

			set_global_variable = mr_de_saint_exupery_created_global_var

			modifier = {
				add = {
					value = year
					subtract = 1900
					subtract = 25
					min = 0
					multiply = 2
					multiply = -0.25
				}
			}
		}
		200 = { #Manfred von Richthofen, Germany
			trigger = {
				NOT = { has_global_variable = curtiss_richthofen_created_global_var }
				OR = {
					country_has_primary_culture = cu:north_german
					country_has_primary_culture = cu:south_german
				}
				has_state_in_state_region = STATE_LOWER_SILESIA #Born in Kleinburg/Borek
				year >= 1915
				year <= 1932 #Age 40
			}
			create_character = {
				template = mr_ger_richthofen_manfred_character_template
				save_scope_as = curtiss_new_pilot_scope
			}
			scope:curtiss_new_pilot_scope = {
				set_variable = mr_is_pilot_var
				pilot_experience_set_up_effect = yes
				place_character_in_void = 1200
			}

			set_variable = {
				name = curtiss_new_pilot_var
				value = scope:curtiss_new_pilot_scope
			}

			set_global_variable = curtiss_richthofen_created_global_var

			modifier = {
				add = {
					value = year
					subtract = 1892
					subtract = 25
					min = 0
					multiply = 2
					multiply = -2
				}
			}
		}
		200 = { #Anthony Fokker, Netherlands - By Cipherparadox
			trigger = {
				NOT = { has_global_variable = curtiss_fokker_anthony_created_global_var }
				OR = {
					has_state_in_state_region = STATE_EAST_JAVA #Born in Blitar
					has_state_in_state_region = STATE_HOLLAND #Family returned to Haarlem
				}
				country_average_cultural_acceptance = {
					target = cu:dutch
					value >= acceptance_status_4
				}
				year >= 1910 #Age 20
				year <= 1930 #Age 40
			}
			create_character = {
				template = mr_net_fokker_anthony_character_template
				save_scope_as = curtiss_new_pilot_scope
			}
			scope:curtiss_new_pilot_scope = {
				set_variable = mr_is_pilot_var
				pilot_experience_set_up_effect = yes
				place_character_in_void = 1200
			}

			set_variable = {
				name = curtiss_new_pilot_var
				value = scope:curtiss_new_pilot_scope
			}

			set_global_variable = curtiss_fokker_anthony_created_global_var

			modifier = {
				add = {
					value = year
					subtract = 1890
					subtract = 25
					min = 0
					multiply = 2
					multiply = -2
				}
			}
		}
		10 = { #Nimitmongkol Nawarat, Siam
			trigger = {
				NOT = { has_global_variable = mr_nimitmongkol_nawarat_created_global_var }
				has_state_in_state_region = STATE_BANGKOK
				country_average_cultural_acceptance = {
					target = cu:thai
					value >= acceptance_status_4
				}
				year >= 1924 #Graduated Flight School around this time
				year <= 1948 #Age 40
			}
			create_character = {
				template = mr_sia_nawarat_nimitmongkol_character_template
				save_scope_as = curtiss_new_pilot_scope
			}
			scope:curtiss_new_pilot_scope = {
				set_variable = mr_is_pilot_var
				pilot_experience_set_up_effect = yes
				place_character_in_void = 1200
			}

			set_variable = {
				name = curtiss_new_pilot_var
				value = scope:curtiss_new_pilot_scope
			}

			set_global_variable = mr_nimitmongkol_nawarat_created_global_var

			modifier = {
				add = {
					value = year
					subtract = 1908
					subtract = 25
					min = 0
					multiply = 2
					multiply = -0.1
				}
			}
		}
	}
}

mr_create_random_pilot_for_character_creation_effect = {
	random_list = {
		1 = {
			create_character = {
				female = yes
				trait_generation = {}
				save_scope_as = curtiss_new_pilot_scope
			}
			scope:curtiss_new_pilot_scope = {
				add_trait = curtiss_pilot
				set_variable = mr_is_pilot_var
				pilot_experience_set_up_effect = yes
				mr_trait_generation_pilot_effect = yes
				place_character_in_void = 1200
			}
			set_variable = {
				name = curtiss_new_pilot_var
				value = scope:curtiss_new_pilot_scope
			}

			modifier = {
				if = {
					limit = { has_law = law_type:law_women_own_property }
					add = 4
				}
				if = {
					limit = { has_law = law_type:law_women_in_the_workplace }
					add = 9
				}
				if = {
					limit = { has_law = law_type:law_womens_suffrage }
					add = 19
				}
			}
		}
		99 = {
			create_character = {
				age = { 20 25 }
				trait_generation = {}
				save_scope_as = curtiss_new_pilot_scope
			}
			scope:curtiss_new_pilot_scope = {
				add_trait = curtiss_pilot
				set_variable = mr_is_pilot_var
				pilot_experience_set_up_effect = yes
				mr_trait_generation_pilot_effect = yes
				place_character_in_void = 1200
			}
			set_variable = {
				name = curtiss_new_pilot_var
				value = scope:curtiss_new_pilot_scope
			}

			modifier = {
				if = {
					limit = { has_law = law_type:law_women_own_property }
					add = -4
				}
				if = {
					limit = { has_law = law_type:law_women_in_the_workplace }
					add = -9
				}
				if = {
					limit = { has_law = law_type:law_womens_suffrage }
					add = -19
				}
			}
		}
	}
}
