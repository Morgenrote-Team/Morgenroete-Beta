## Generic Effects
mr_update_conference_participant_list_effect = {
	# Call this from scope:journal_entry
	clear_variable_list = mr_conference_participants_list
	if = {
		limit = {
			has_global_variable = $type$
			exists = global_var:$type$
		}
		every_country = {
			limit = {
				is_involved_in_journal_entry = $type$
			}
			every_scope_character = {
				limit = {
					has_role = $role$
				}
				global_var:$type$ = {
					add_to_variable_list = {
						name = mr_conference_participants_list
						target = prev
					}
				}
			}
		}
		global_var:$type$ = {
			every_in_list = {
				variable = mr_conference_participants_list
				set_variable = {
					name = mr_scientist_votes_var
					value = mr_scientist_votes_value
				}
			}
		}
	}
	else = {
		error_log = "mr_update_conference_participant_list_effect was called, but the specified global variable is either not set or its scope does not exist."
	}
}

mr_update_leading_scientist_list_effect = {
	# Call this from scope:journal_entry
	# This is a truncated variant of mr_update_conference_participant_list_effect
	# Use this for non-voting conferences, when the leading scientist is only needed for cosmetic purposes
	clear_variable_list = mr_conference_participants_list
	if = {
		limit = {
			has_global_variable = $type$
			exists = global_var:$type$
		}
		every_country = {
			limit = {
				is_involved_in_journal_entry = $type$
			}
			var:$scientist$ = {
				global_var:$type$ = {
					add_to_variable_list = {
						name = mr_conference_participants_list
						target = prev
					}
				}
			}
		}
	}
	else = {
		error_log = "mr_update_leading_scientist_list_effect was called, but the specified global variable is either not set or its scope does not exist."
	}
}

mr_update_conference_position_values_effect = {
	# Call this from scope:journal_entry
	set_variable = {
		name = mr_$position$_votes_var
		value = {
			every_country = {
				limit = {
					is_involved_in_journal_entry = $type$
				}
				every_scope_character = {
					limit = {
						has_role = $role$
						var:mr_conference_stance_var ?= flag:mr_stance_$position$
					}
					add = mr_scientist_votes_value
				}
			}
		}
	}
}

mr_check_conference_chair_effect = {
	# This is a safety effect in case a conference chair dies/retires mid-conference
	# Call this from scope:journal_entry
	if = {
		limit = {
			NOT = {
				exists = var:com_interorg_chair_var
			}
		}
		ordered_in_list = {
			variable = mr_conference_participants_list
			order_by = owner.prestige
			position = 0
			save_scope_as = mr_conference_chair_scope
		}
		set_variable = {
			name = com_interorg_chair_var
			value = scope:mr_conference_chair_scope
		}
	}
}
