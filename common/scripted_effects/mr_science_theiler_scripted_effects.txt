theiler_set_physician_demands_medicinal_plant_effect = {
	set_variable = theiler_medicinal_plant_demand_active_var
	random_list = {
		1 = { trigger = { theiler_has_pharmaceutical_strychnine_trigger = no } set_variable = { name = theiler_physician_demand_var value = flag:theiler_pharmaceutical_strychnine_flag } }
		1 = { trigger = { theiler_has_pharmaceutical_emetine_trigger = no } set_variable = { name = theiler_physician_demand_var value = flag:theiler_pharmaceutical_emetine_flag } }
		1 = { trigger = { theiler_has_pharmaceutical_atropine_trigger = no panum_has_medcial_technique_level_1_or_higher_no_tooltip_trigger = yes } set_variable = { name = theiler_physician_demand_var value = flag:theiler_pharmaceutical_atropine_flag } }

# This effect adds 1 to the total amount of pharmaceuticals.
theiler_add_1_total_pharmaceutical_effect = {
	if = {
		limit = {
			NOT = { has_variable = theiler_num_pharmaceuticals_var }
		}
		set_variable = {
			name = theiler_num_pharmaceuticals_var
			value = 1
		}
	}
	else = {
		change_variable = {
			name = theiler_num_pharmaceuticals_var
			add = 1
		}
	}
}

# This effect adds the Strychnine Pharmaceutical
theiler_add_pharmaceutical_strychnine_effect = {
	custom_tooltip = {
		text = theiler_add_pharmaceutical_strychnine_effect_tt
		set_variable = theiler_pharmaceutical_strychnine_var
		theiler_add_1_total_pharmaceutical_effect = yes
	}
	add_modifier = {
		name = theiler_biological_breakthrough_modifier
		years = 10
	}
	if = {
		limit = {
			NOT = {
				has_global_variable = theiler_pharmaceutical_strychnine_invented_global_var
			}
		}
		set_global_variable = {
			name = theiler_pharmaceutical_strychnine_year_global_var
			value = year
		}
		country_definition = {
			set_global_variable = {
				name = theiler_pharmaceutical_strychnine_invented_global_var
				value = this
			}
		}
		var:theiler_biologist_var = {
			set_variable = {
				name = theiler_pharmaceutical_strychnine_achievement_var
				value = flag:theiler_pharmaceutical_strychnine_achievement_loc
			}
		}
		biologist_experience_gain_effect = { AMOUNT = 20 }
		panum_small_add_technology = yes
		hidden_effect = {
			every_country = {
				limit = {
					is_player = yes
				}
				post_notification = theiler_pharmaceutical_strychnine_notification
			}
		}
		#For each First Pathogen, add 6 to the medical score
		panum_add_x_medical_score_points_effect = { AMOUNT = 4 }
	}
	else = {
		biologist_experience_gain_effect = { AMOUNT = 20 }
		#For each Pathogen, add 3 to the medical score
		panum_add_x_medical_score_points_effect = { AMOUNT = 2 }
	}
}

#Effect to add $AMOUNT$ Microorganisms
theiler_add_x_microorganisms_effect = {
	# Amount = amount of microorganisms to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = microorganisms_amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = theiler_add_x_microorganisms_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = theiler_add_x_microorganisms_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = theiler_add_x_microorganisms_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = theiler_add_x_microorganisms_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = theiler_add_x_microorganisms_plus_x_tt
	}

	#Yearly microorganisms are used for the botanical report, they are added with each normal microorganisms
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = theiler_yearly_microorganisms_var
			}
		}
		set_variable = {
			name = theiler_yearly_microorganisms_var
			value = 0
		}
	}
	change_variable = {
		name = theiler_yearly_microorganisms_var
		add = $AMOUNT$
	}
	change_global_variable = {
		name = theiler_global_microorganisms_global_var
		add = $AMOUNT$
	}

	#This part adds the AMOUNT of microorganisms
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = theiler_microorganisms_var
			}
		}
		set_variable = {
			name = theiler_microorganisms_var
			value = 0
		}
	}
	change_variable = {
		name = theiler_microorganisms_var
		add = scope:microorganisms_amount
	}

	#theiler_biologist_total_microorganisms_var is used to keep track of all the microorganisms that a specific biologist found.
	#It is used to show how successful the biologist is.
	#Every time a normal biologist object is added, the same number gets added here.
	if = {
		limit = { #Only works if there actually is a biologist
			has_variable = theiler_biologist_var
		}
		var:theiler_biologist_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = theiler_biologist_total_microorganisms_var
					}
				}
				set_variable = {
					name = theiler_biologist_total_microorganisms_var
					value = 0
				}
			}
			change_variable = { #The amount of microorganisms that this biologist discovered
				name = theiler_biologist_total_microorganisms_var
				add = scope:microorganisms_amount
			}
		}
	}
}
