###Adding artifacts###

#Effect to add $AMOUNT$ Minor Egyptian Artifacts
lepsius_add_x_egyptian_artifacts_effect = {
	# Amount = amount of artifacts to add
	custom_description = {
		text = "lepsius_add_x_minor_egyptian_artifacts_effect_loc"
		subject = this
		value = $AMOUNT$

		#Total Minor Artifacts of country
		mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_total_artifacts }

		#Total Minor Artifacts in the world
		change_global_variable = {
			name = lepsius_global_artifacts
			add = $AMOUNT$
		}

		#Minor Egyptian Artifacts of the country
		mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_egyptian_artifact_minor }

		#Adds Artifacts to the Archaeologist
		var:lepsius_archaeologist_var ?= {
			#Counts how many Minor Egyptian Artifacts the character found
			mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_egyptian_artifact_minor_archaeologist_var }
			#Counts how many Minor Artifacts the character found
			mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_total_artifact_minor_archaeologist_var }
		}
	}
}
