###Adding minor artifacts###

#Effect to add $AMOUNT$ Minor Egyptian Artifacts
lepsius_add_x_egyptian_artifacts_effect = {
	# Amount = amount of artifacts to add
	custom_description = {
		text = "lepsius_add_x_minor_egyptian_artifacts_effect_loc"
		subject = this
		value = $AMOUNT$

		#Total Minor Artifacts of country
		mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_total_artifacts }

		#Total Minor Artifacts in the world
		mr_change_global_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_global_artifacts }

		#Minor Egyptian Artifacts of the country
		mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_egyptian_artifact_minor }

		#Adds Artifacts to the Archaeologist
		var:lepsius_archaeologist_var ?= {
			#Counts how many Minor Egyptian Artifacts the character found
			mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_egyptian_artifact_minor_archaeologist_var }
			#Counts how many Minor Artifacts the character found
			mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_total_artifact_minor_archaeologist_var }
		}
	}
}

#Adds "AMOUNT" minor artifacts
lepsius_add_x_minor_artifacts = {
	# Amount = amount of artifacts to add
	if = {
		limit = {
			lepsius_expedition_in_egypt = yes
		}
		lepsius_add_x_egyptian_artifacts_effect = { AMOUNT = $AMOUNT$ }
	}
	if = {
		limit = {
			lepsius_expedition_in_mesopotamia = yes
		}
		lepsius_add_x_mesopotamian_artifacts_effect = { AMOUNT = $AMOUNT$ }
	}
	if = {
		limit = {
			lepsius_expedition_in_greece = yes
		}
		lepsius_add_x_aegean_artifacts_effect = { AMOUNT = $AMOUNT$ }
	}
	if = {
		limit = {
			lepsius_expedition_in_south_america = yes
		}
		lepsius_add_x_precolumbian_artifacts_effect = { AMOUNT = $AMOUNT$ }
	}
}

lepsius_add_x_major_egyptian_artifact = {

		text = "lepsius_add_x_major_egyptian_artifact_loc"
		subject = this
		value = $AMOUNT$

		mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_total_major_artifacts_var }

		mr_change_global_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_global_artifacts }
		#Total Egyptian Major Artifacts of Country
		mr_change_variable_min_0_effect = { AMOUNT = $AMOUNT$ NAME = lepsius_egyptian_artifact_major }
	}
}
