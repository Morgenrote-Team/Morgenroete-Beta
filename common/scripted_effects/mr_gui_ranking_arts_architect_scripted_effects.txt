### Architectural Splendour Ranking List ###
#Creates a list of the countries ranked by their architectural splendour.
#As this effect is called in an effect, it creates a global variable list.
gaudi_create_architectural_splendour_score_ranking_list_effect = {
	#Removes the global variable list
	clear_global_variable_list = mr_architect_architectural_splendour_ranking_global_list

	set_local_variable = {
		name = mr_current_rank_local_var
		value = 0
	}

	ordered_country = {
		limit = {
			has_variable = gaudi_architectural_splendour_var
			is_revolutionary = no #Hopefully keeps revolutions from appearing in the ranking
			NOR = {
				c:PRC ?= this
				is_country_type = decentralized
			}
		}
		order_by = var:gaudi_architectural_splendour_var
		check_range_bounds = no #prevents errors

		#Keeps Track of the rank so it can be saved as a variable
		change_local_variable = {
			name = mr_current_rank_local_var
			add = 1
		}
		if = {
			limit = {
				this = {
					is_player = yes
				}
			}
			set_variable = {
				name = gaudi_architectural_splendour_player_rank_var
				value = local_var:mr_current_rank_local_var
			}
		}

		#This part adds all the countries to the list
		add_to_global_variable_list = {
			name = mr_architect_architectural_splendour_ranking_global_list
			target = this
		}
	}
}
