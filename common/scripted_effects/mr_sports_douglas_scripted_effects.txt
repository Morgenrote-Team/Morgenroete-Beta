#Effect to add $AMOUNT$ Alpine Heritage
douglas_add_x_alpine_heritage_effect = {
	# Amount = amount of Alpine Heritage to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = alpine_heritage_amount
		value = $AMOUNT$
	}
	
	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if local_var is negative
		limit = { $AMOUNT$ < 0 }
		custom_tooltip = douglas_add_x_alpine_heritage_minus_x_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = douglas_add_x_alpine_heritage_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 0 }
		custom_tooltip = douglas_add_x_alpine_heritage_plus_x_tt
	}

	#This part adds the AMOUNT of Alpine Heritage
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = douglas_alpine_heritage_var
			}
		}
		set_variable = {
			name = douglas_alpine_heritage_var
			value = 0
		}
	}
	change_variable = {
		name = douglas_alpine_heritage_var
		add = scope:alpine_heritage_amount
	}
}

#This effect is used to add the yearly Alpine Heritage
douglas_yearly_alpine_heritage_effect = {
	if = { #Effect only gets executed if the technology Prospecting is researched
		limit = {
			has_technology_researched = prospecting
		}

		#Set up
		if = {
			limit = { NOT = { has_variable = douglas_alpine_heritage_var } }
			set_variable = {
				name = douglas_alpine_heritage_var
				value = 0
			}
		}

		### Actual calculation done in script_value ###
		if = {
			limit = { douglas_yearly_alpine_heritage_value > 0 }
			set_variable = {
				name = douglas_estimated_yearly_alpine_heritage_var
				value = douglas_yearly_alpine_heritage_value
			}
		}
		else = {
			set_variable = {
				name = douglas_estimated_yearly_alpine_heritage_var
				value = 0
			}
		}
		
		#Add yearly variable to the total variable
		if = {
			limit = { var:douglas_estimated_yearly_alpine_heritage_var > 0 }
			change_variable = {
				name = douglas_alpine_heritage_var
				add = var:douglas_estimated_yearly_alpine_heritage_var
			}
		}
	}
	else = {
		set_variable = {
			name = douglas_estimated_yearly_alpine_heritage_var
			value = 0
		}
	}
}

#This effect adds historical mountain conquests at game start
douglas_set_start_mountaineering_effect = {
	if = {
		limit = {
			game_date >= 1823.1.1 #Mauna Kea conquered
			exists = c:USA
		}
		c:USA = {
			set_variable = douglas_mauna_kea_conqueror_var
			country_definition = {
				set_global_variable = {
					name = douglas_mauna_kea_conqueror_global_var
					value = this
				}
			}
		}
		s:STATE_HAWAIIAN_ISLANDS = {
			add_state_trait = douglas_state_trait_mauna_kea_conquered
			remove_state_trait = douglas_state_trait_mauna_kea
		}
		set_global_variable = {
			name = douglas_mauna_kea_year_global_var
			value = 1823
		}
	}
	if = {
		limit = {
			game_date >= 1786.1.1 #Mont Blanc conquered
			exists = c:SAR
		}
		c:SAR = {
			set_variable = douglas_mont_blanc_conqueror_var
			country_definition = {
				set_global_variable = {
					name = douglas_mont_blanc_conqueror_global_var
					value = this
				}
			}
		}
		s:STATE_PIEDMONT = {
			add_state_trait = douglas_state_trait_mont_blanc_conquered
			remove_state_trait = douglas_state_trait_mont_blanc
		}
		set_global_variable = {
			name = douglas_mont_blanc_year_global_var
			value = 1786
		}
	}
	if = {
		limit = {
			game_date >= 1519.1.1 #Popocatepetl conquered
			exists = c:SPA
		}
		c:SPA = {
			set_variable = douglas_popocatepetl_conqueror_var
			country_definition = {
				set_global_variable = {
					name = douglas_popocatepetl_conqueror_global_var
					value = this
				}
			}
		}
		s:STATE_MEXICO = {
			add_state_trait = douglas_state_trait_popocatepetl_conquered
			remove_state_trait = douglas_state_trait_popocatepetl
		}
		set_global_variable = {
			name = douglas_popocatepetl_year_global_var
			value = 1519
		}
	}
	######HIGHEST MOUNTAIN######
	if = {
		limit = {
			game_date >= 1808.1.1 #Dhalaugiri is considered the highest mountain
		}
		set_global_variable = {
			name = douglas_highest_mountain_global_var
			value = flag:douglas_dhalaugiri_highest_mountain_flag
		}
	}
	else_if = {
		limit = {
			game_date >= 1735.1.1 #Chimborazo is considered the highest mountain
		}
		set_global_variable = {
			name = douglas_highest_mountain_global_var
			value = flag:douglas_chimborazo_highest_mountain_flag
		}
	}
}

#Effect to add Board Members which are considered leading mountaineering countries
douglas_add_alpine_board_effect = {
	custom_tooltip = douglas_alpine_club_new_board_tt
	#Removes the NATIONAL variables
	every_country = {
		limit = {
			has_variable = douglas_alpine_club_board_member_var
		}
		remove_variable = douglas_alpine_club_board_member_var
	}
	#Removes the GLOBAL ranking variables
	remove_global_variable = douglas_alpine_club_board_member_1_global_var
	remove_global_variable = douglas_alpine_club_board_member_2_global_var
	remove_global_variable = douglas_alpine_club_board_member_3_global_var
	remove_global_variable = douglas_alpine_club_board_member_4_global_var
	remove_global_variable = douglas_alpine_club_board_member_5_global_var

	ordered_country = {
		limit = {
			has_variable = douglas_mountaineering_club_var
			NOT = {
				has_variable = douglas_alpine_club_founding_country_var
			}
			is_revolutionary = no
		}
		order_by = douglas_alpine_board_member_quality_value
		max = 5 #Limits the list to the five top countries
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = { 
					has_global_variable = douglas_alpine_club_board_member_1_global_var
				}
			}
			set_variable = douglas_alpine_club_board_member_var
			set_global_variable = {
				name =  douglas_alpine_club_board_member_1_global_var
				value = THIS
			}
		}
		else_if = {
			limit = {
				NOT = { 
					has_global_variable = douglas_alpine_club_board_member_2_global_var
				}
			}
			set_variable = douglas_alpine_club_board_member_var
			set_global_variable = {
				name =  douglas_alpine_club_board_member_2_global_var
				value = THIS
			}
		}
		else_if = {
			limit = {
				NOT = { 
					has_global_variable = douglas_alpine_club_board_member_3_global_var
				}
			}
			set_variable = douglas_alpine_club_board_member_var
			set_global_variable = {
				name =  douglas_alpine_club_board_member_3_global_var
				value = THIS
			}
		}
		else_if = {
			limit = {
				NOT = { 
					has_global_variable = douglas_alpine_club_board_member_4_global_var
				}
			}
			set_variable = douglas_alpine_club_board_member_var
			set_global_variable = {
				name =  douglas_alpine_club_board_member_4_global_var
				value = THIS
			}
		}
		else_if = {
			limit = {
				NOT = { 
					has_global_variable = douglas_alpine_club_board_member_5_global_var
				}
			}
			set_variable = douglas_alpine_club_board_member_var
			set_global_variable = {
				name =  douglas_alpine_club_board_member_5_global_var
				value = THIS
			}
		}
	}
}

douglas_mountaineering_equipment_modifier_effect = {
	if = {
		limit = {
			has_modifier = douglas_mountaineering_equipment_level_4_modifier
		}
		remove_modifier = douglas_mountaineering_equipment_level_4_modifier
		add_modifier = {
			name = douglas_mountaineering_equipment_level_5_modifier
		}
	}
	else_if = {
		limit = {
			has_modifier = douglas_mountaineering_equipment_level_3_modifier
		}
		remove_modifier = douglas_mountaineering_equipment_level_3_modifier
		add_modifier = {
			name = douglas_mountaineering_equipment_level_4_modifier
		}
	}
	else_if = {
		limit = {
			has_modifier = douglas_mountaineering_equipment_level_2_modifier
		}
		remove_modifier = douglas_mountaineering_equipment_level_2_modifier
		add_modifier = {
			name = douglas_mountaineering_equipment_level_3_modifier
		}
	}
	else_if = {
		limit = {
			has_modifier = douglas_mountaineering_equipment_level_1_modifier
		}
		remove_modifier = douglas_mountaineering_equipment_level_1_modifier
		add_modifier = {
			name = douglas_mountaineering_equipment_level_2_modifier
		}
	}
	else_if = {
		limit = {
			NOR = {
				has_modifier = douglas_mountaineering_equipment_level_1_modifier
				has_modifier = douglas_mountaineering_equipment_level_2_modifier
				has_modifier = douglas_mountaineering_equipment_level_3_modifier
				has_modifier = douglas_mountaineering_equipment_level_4_modifier
				has_modifier = douglas_mountaineering_equipment_level_5_modifier
			}
		}
		add_modifier = {
			name = douglas_mountaineering_equipment_level_1_modifier
		}
	}
	set_variable = douglas_mountaineering_equipment_complete_var
}

#Effect to update Mountaineering Prestige
change_douglas_summit_prestige_effect = {
	custom_tooltip = {
		text = change_douglas_summit_prestige_effect_tt

		while = {
			limit = {
				has_modifier = douglas_summit_prestige_modifier
			}
			remove_modifier = douglas_summit_prestige_modifier
		}
		if = {
			limit = { douglas_summit_prestige_value > 0 }
			add_modifier = {
				name = douglas_summit_prestige_modifier
				multiplier = douglas_summit_prestige_value
			}
		}
	}
}

	}

douglas_add_variables = {
	every_country = {
		if = {
			limit = { NOT = { has_variable = douglas_alpine_heritage_var } }
			set_variable = {
				name = douglas_alpine_heritage_var
				value = 0
			}
		}
		if = {
			limit = { NOT = { has_variable = douglas_num_mountain_ascents_var } }
			set_variable = {
				name = douglas_num_mountain_ascents_var
				value = 0
			}
		}
		if = {
			limit = { NOT = { has_variable = douglas_summits_conquered_var } }
			set_variable = {
				name = douglas_summits_conquered_var
				value = 0
			}
		}
		if = {
			limit = { NOT = { has_variable = douglas_num_mountain_disasters_var } }
			set_variable = {
				name = douglas_num_mountain_disasters_var
				value = 0
			}
		}
		if = {
			limit = { NOT = { has_variable = douglas_num_climbs_aborted_var } }
			set_variable = {
				name = douglas_num_climbs_aborted_var
				value = 0
			}
		}
		set_variable = douglas_variables_var
	}
}
