### Urban Planning Project ###
#Button Effect
mr_architect_urban_planning_button_effect = {
	if = {
		limit = {
			gaudi_is_architecture_automated_trigger = yes
		}
		clear_variable_list = gaudi_architect_urban_planning_buildings_chosen_variable_list
		ordered_scope_building = {
			position = 0 #We only want the first one here
			limit = {
				gaudi_building_valid_for_urban_planning_project_no_tooltip_trigger = yes
			}
			order_by = earnings
			root = {
				add_to_variable_list = {
					name = gaudi_architect_urban_planning_buildings_chosen_variable_list
					target = prev
				}
			}
		}
		add_journal_entry = {
			type = je_gaudi_urban_planning_project
		}
		var:gaudi_architect_var ?= {
			set_character_busy_and_immortal = yes
		}
	}
	else = {
		clear_variable_list = gaudi_architect_urban_planning_buildings_variable_list
		ordered_scope_building = {
			limit = {
				gaudi_building_valid_for_urban_planning_project_no_tooltip_trigger = yes
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = gaudi_architect_urban_planning_buildings_variable_list
					target = prev
				}
			}
		}
		set_variable = gaudi_architect_urban_planning_building_choser_var
		show_as_tooltip = {
			add_modifier = {
				name = gaudi_urban_planning_cost_modifier
				multiplier = mr_local_project_cost_very_small_value
			}
		}
	}
}
