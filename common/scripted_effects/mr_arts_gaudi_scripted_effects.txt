#Effect to add $AMOUNT$ Architectural Splendour
gaudi_add_x_architectural_splendour_effect = {
	# Amount = amount of Architectural Splendour to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = architectural_splendour_amount
		value = $AMOUNT$
	}
	
	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if local_var is negative
		limit = { $AMOUNT$ < 0 }
		custom_tooltip = gaudi_add_x_architectural_splendour_minus_x_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = gaudi_add_x_architectural_splendour_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 0 }
		custom_tooltip = gaudi_add_x_architectural_splendour_plus_x_tt
	}

	#This part adds the AMOUNT of Architectural Splendour
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = gaudi_architectural_splendour_var
			}
		}
		set_variable = {
			name = gaudi_architectural_splendour_var
			value = 0
		}
	}
	change_variable = {
		name = gaudi_architectural_splendour_var
		add = scope:architectural_splendour_amount
	}

	#gaudi_architect_total_architectural_splendour_var is used to keep track of all the Architectural Splendour that a specific architect created.
	#It is used to show how successful the architect is.
	#Every time a normal Architectural Splendour Point is added, the same number gets added here.
	if = {
		limit = { #Only works if there actually is a architect
			has_variable = gaudi_architect_var
		}
		var:gaudi_architect_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = gaudi_architect_total_architectural_splendour_var
					}
				}
				set_variable = {
					name = gaudi_architect_total_architectural_splendour_var
					value = 0
				}
			}
			change_variable = { #The amount of paintingal Tradition that this architect discovered
				name = gaudi_architect_total_architectural_splendour_var
				add = scope:architectural_splendour_amount
			}
		}
	}
}

######CAPITAL EXPANSION######

gaudi_capital_expansion_modifier_effect = {
	if = {
		limit = {
			any_scope_state = {
				is_capital = yes
				has_modifier = gaudi_capital_expansion_level_3_modifier
			}
		}
		random_scope_state = {
			limit = {
				is_capital = yes
				has_modifier = gaudi_capital_expansion_level_3_modifier
			}
			remove_modifier = gaudi_capital_expansion_level_3_modifier
			add_modifier = {
				name = gaudi_capital_expansion_level_4_modifier
			}
		}
	}
	else_if = {
		limit = {
			any_scope_state = {
				is_capital = yes
				has_modifier = gaudi_capital_expansion_level_2_modifier
			}
		}
		random_scope_state = {
			limit = {
				is_capital = yes
				has_modifier = gaudi_capital_expansion_level_2_modifier
			}
			remove_modifier = gaudi_capital_expansion_level_2_modifier
			add_modifier = {
				name = gaudi_capital_expansion_level_3_modifier
			}
		}
	}
	else_if = {
		limit = {
			any_scope_state = {
				is_capital = yes
				has_modifier = gaudi_capital_expansion_level_1_modifier
			}
		}
		random_scope_state = {
			limit = {
				is_capital = yes
				has_modifier = gaudi_capital_expansion_level_1_modifier
			}
			remove_modifier = gaudi_capital_expansion_level_1_modifier
			add_modifier = {
				name = gaudi_capital_expansion_level_2_modifier
			}
		}
	}
	else_if = {
		limit = {
			any_scope_state = {
				is_capital = yes
				NOR = {
					has_modifier = gaudi_capital_expansion_level_1_modifier
					has_modifier = gaudi_capital_expansion_level_2_modifier
					has_modifier = gaudi_capital_expansion_level_3_modifier
					has_modifier = gaudi_capital_expansion_level_4_modifier
				}
			}
		}
		random_scope_state = {
			limit = {
				is_capital = yes
			}
			add_modifier = {
				name = gaudi_capital_expansion_level_1_modifier
			}
		}
	}
	set_variable = gaudi_capital_expansion_complete_var
}

gaudi_capital_expansion_event_effect = {
	custom_tooltip = {
		text = gaudi_capital_expansion_advance_tt
		if = {
			limit = {
				any_scope_state = {
					is_capital = yes
					has_modifier = gaudi_capital_expansion_level_1_modifier
				}
			}
			trigger_event = {
				id = gaudi.41
			}
		}
		else_if = {
			limit = {

					is_capital = yes

				}
			}
			trigger_event = {

				}
			}
			trigger_event = {

		else_if = {

	}

###Ranking for Architects###

#Adds the Architects as global variables
gaudi_create_architect_ranking_effect = {
	#Removes the global variable list
	clear_global_variable_list = mr_architect_ranking_global_list

	ordered_character_in_void = {
		limit = {
			OR = {
				has_variable = mr_is_architect_var
				has_variable = mr_is_independent_architect_var
			}
		}
		order_by = var:gaudi_architect_experience
		check_range_bounds = no

		#This part adds all the countries to the list
		add_to_global_variable_list = {
			name = mr_architect_ranking_global_list
			target = this
		}
	}
}
