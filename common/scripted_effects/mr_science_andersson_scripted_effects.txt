#Add Caves

lepsius_add_caves = {
	#EUROPE
	while = {
		count = 3

		random_state = {
			limit = {
				is_in_geographic_region = geographic_region_europe
				lepsius_cromagnon_region = yes
				lepsius_has_cave = no
			}
			state_region = {
				set_variable = lepsius_cromagnon_cave_var
			}
		}
	}
	while = {
		count = 3

		random_state = {
			limit = {
				is_in_geographic_region = geographic_region_europe
				lepsius_neanderthalensis_region = yes

lepsius_set_cave_site = {
	random_scope_state = {
		limit = {
			has_modifier = lepsius_cave_examination_state_modifier
		}
		save_scope_as = lepsius_cave_site
	}
}

lepsius_set_cave_leader = {
	var:andersson_anthropologist_var = {
		save_scope_as = andersson_anthropologist_scope
	}
}

#Add Unspecific Human Fossils

lepsius_add_x_human_fossils_effect = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = fossil_amount
		value = $AMOUNT$
	}

	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = lepsius_add_x_human_fossils_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = lepsius_add_x_human_fossils_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = lepsius_add_x_human_fossils_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = lepsius_add_x_human_fossils_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = lepsius_add_x_human_fossils_plus_x_tt
	}

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_total_human_fossil
			}
		}
		set_variable = {
			name = lepsius_total_human_fossil
			value = 0
		}
	}
	change_variable = {
		name = lepsius_total_human_fossil
		add = $AMOUNT$
	}
	change_global_variable = {
		name = lepsius_global_human_fossil
		add = $AMOUNT$
	}
	if = {
		limit = { #Only works if there actually is an anthropologist
			has_variable = andersson_anthropologist_var
		}
		var:andersson_anthropologist_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = lepsius_human_fossil_anthropologist_var
					}
				}
				set_variable = {
					name = lepsius_human_fossil_anthropologist_var
					value = 0
				}
			}
			change_variable = { #The amount of Artifacts that this archaeologist discovered
				name = lepsius_human_fossil_anthropologist_var
				add = $AMOUNT$
			}
		}
	}
}

#Add Specific Human Fossils

lepsius_add_cromagnon_fossil_effect = {
	# Amount not needed

	custom_tooltip = lepsius_add_cromagnon_fossil_tt

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_total_human_fossil
			}
		}
		set_variable = {
			name = lepsius_total_human_fossil
			value = 0
		}
	}
	change_variable = { #Total human fossils the TAG found
		name = lepsius_total_human_fossil
		add = 1
	}
	change_global_variable = { #Total human fossil found worldwide
		name = lepsius_global_human_fossil
		add = 1
	}

	#Total Human Fossils the Anthropologist found
	if = {
		limit = { has_variable = andersson_anthropologist_var }
		var:andersson_anthropologist_var = {
			if = {
				limit = { NOT = { has_variable = lepsius_total_human_fossils_anthropologist_var } }
				set_variable = {
					name = lepsius_total_human_fossils_anthropologist_var
					value = 0
				}
			}
			change_variable = {
				name = lepsius_total_human_fossils_anthropologist_var
				add = 1
			}
		}
	}

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = lepsius_cromagnon_human_fossil
			}
		}
		set_variable = {
			name = lepsius_cromagnon_human_fossil
			value = 0
		}
	}
	change_variable = { #Add unknown Cro-Magnon
		name = lepsius_cromagnon_human_fossil
		add = 1
	}
	if = {
		limit = {
			var:lepsius_cromagnon_human_fossil >= 3
			NOT = {
				has_variable = lepsius_cromagnon_complete
			}
		}
		trigger_event = {
			id = lepsius.1321
			popup = yes
		}
	}

	#Total Cro-Magnon Fossils the Anthropologist found
	if = {
		limit = { has_variable = andersson_anthropologist_var }
		var:andersson_anthropologist_var = {
			if = {
				limit = { NOT = { has_variable = lepsius_total_cromagnon_fossils_anthropologist_var } }
				set_variable = {
					name = lepsius_total_cromagnon_fossils_anthropologist_var
					value = 0
				}
			}
			change_variable = {
				name = lepsius_total_cromagnon_fossils_anthropologist_var
				add = 1
			}
		}
	}
}

#Adds the Anthropologists as global variables
andersson_create_anthropologist_ranking_effect = {
	#Removes the global variable list
	clear_global_variable_list = mr_anthropologist_ranking_global_list

	ordered_character = {
		limit = {
			has_role = character_role_andersson_anthropologist
			has_variable = andersson_anthropologist_experience_var
		}
		order_by = var:andersson_anthropologist_experience_var
		check_range_bounds = no

		#This part adds all the countries to the list
		add_to_global_variable_list = {
			name = mr_anthropologist_ranking_global_list
			target = this
		}
	}
}
