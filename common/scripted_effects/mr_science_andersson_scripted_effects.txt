### Anthropologist ###

#Set Anthropologist busy
andersson_set_anthropologist_busy_effect = {
	var:andersson_anthropologist_var ?= {
		set_character_busy_and_immortal = yes
	}
}

andersson_set_cave_leader = {
	var:andersson_anthropologist_var ?= {
		save_scope_as = andersson_anthropologist_scope
	}
}

andersson_set_cave_site = {
	if = {
		limit = {
			has_modifier = andersson_cave_exploration_modifier
		}
		random_scope_state = {
			limit = {
				has_variable = andersson_hominid_prospecting_var
			}
			save_scope_as = andersson_cave_site
		}
	}
	if = {
		limit = {
			has_modifier = andersson_hominid_expedition_modifier
		}
		random_direct_subject = {
			limit = {
				any_scope_state = {
					has_variable = andersson_hominid_expedition_var
				}
			}
			random_scope_state = {
				limit = {
					has_variable = andersson_hominid_expedition_var
				}
				save_scope_as = andersson_cave_site
			}
		}
	}
}

#Add Cave Paintings at Game Start
andersson_add_cave_paintings_effect = {
	#HISTORICAL SITES
	every_state_region = {
		limit = {
			OR = {
				this = s:STATE_UTAH #Horseshoe Canyon
				this = s:STATE_PATAGONIA #Cueva de las Manos
				this = s:STATE_PIAUI #Toca do Boqueirão da Pedra Furada
				this = s:STATE_ASTURIAS #Altamira
				this = s:STATE_GUYENNE #Vézère Valley
				this = s:STATE_EAST_SAHARA #Tassili n'Ajjer
				this = s:STATE_ZULULAND #San Rock Art
				this = s:STATE_CENTRAL_PROVINCES #Bhimbetka rock shelters
				this = s:STATE_GUANGXI #Zuojiang Huashan Rock Art
				this = s:STATE_CELEBES #Maros-Pangkep
				this = s:STATE_WESTERN_AUSTRALIA #Gwion Gwion rock art
			}
		}
		set_variable = andersson_cave_var
	}

	#NORTH AMERICA
	while = {
		count = 2

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_vero
				NOT = {
					has_variable = andersson_cave_var
				}
			}
			state_region = {
				set_variable = andersson_cave_var
			}
		}
	}

	#SOUTH AMERICA
	while = {
		count = 2

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_lagoa_santa
				NOT = {
					has_variable = andersson_cave_var
				}
			}
			state_region = {
				set_variable = andersson_cave_var
			}
		}
	}
	#EUROPE
	while = {
		count = 3

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_cro_magnon
				NOT = {
					has_variable = andersson_cave_var
				}
			}
			state_region = {
				set_variable = andersson_cave_var
			}
		}
	}
	#AFRICA
	while = {
		count = 2

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_boskop
				NOT = {
					has_variable = andersson_cave_var
				}
			}
			state_region = {
				set_variable = andersson_cave_var
			}
		}
	}
	#ASIA
	while = {
		count = 3

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_tianyuan
				NOT = {
					has_variable = andersson_cave_var

andersson_remove_prospecting_effect = {
	remove_modifier = andersson_cave_exploration_modifier
	remove_variable = andersson_hominid_prospecting_exploration_completed_var
	remove_variable = andersson_hominid_prospecting_collection_completed_var
	random_scope_state = {
		limit = {
			has_variable = andersson_hominid_prospecting_var
		}
		remove_variable = andersson_hominid_prospecting_var
	}
}

			any_scope_state = {

				has_variable = andersson_hominid_expedition_var

}

#Add Neoanthropus Skeleton Effect
andersson_add_x_neoanthropus_skeleton_effect = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = artifact_amount
		value = $AMOUNT$
	}

	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = andersson_add_x_neoanthropus_skeleton_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = andersson_add_x_neoanthropus_skeleton_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = andersson_add_x_neoanthropus_skeleton_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = andersson_add_x_neoanthropus_skeleton_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = andersson_add_x_neoanthropus_skeleton_plus_x_tt
	}

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = andersson_neoanthropus_skeleton_var
			}
		}
		set_variable = {
			name = andersson_neoanthropus_skeleton_var
			value = 0
		}
	}
	change_variable = {
		name = andersson_neoanthropus_skeleton_var
		add = $AMOUNT$
	}
	if = {
		limit = { #Only works if there actually is an anthropologist
			has_variable = andersson_anthropologist_var
		}
		var:andersson_anthropologist_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = andersson_neoanthropus_skeleton_anthropologist_var
					}
				}
				set_variable = {
					name = andersson_neoanthropus_skeleton_anthropologist_var
					value = 0
				}
			}
			change_variable = { #The amount of human skeletons that this anthropologist discovered
				name = andersson_neoanthropus_skeleton_anthropologist_var
				add = $AMOUNT$
			}
		}
	}

	andersson_change_anthropological_prestige_by_amount_effect = { AMOUNT = 25 }
}

	clear_global_variable_list = mr_anthropologist_ranking_global_list

	ordered_character = {
		limit = {
			has_role = character_role_andersson_anthropologist

		add_to_global_variable_list = {
			name = mr_anthropologist_ranking_global_list
			target = this
		}
	}
}
