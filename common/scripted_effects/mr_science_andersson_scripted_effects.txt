#Set Anthropologist busy
andersson_set_anthropologist_busy_effect = {
	custom_tooltip = {
		text = andersson_set_anthropologist_busy_tt
		if = {
			limit = { has_variable = andersson_anthropologist_var }
			var:andersson_anthropologist_var = {
				set_character_busy_and_immortal = yes
			}
		}
	}
}

#Add Unspecific Human Fossils
andersson_add_x_prehistoric_artifacts_effect = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = artifact_amount
		value = $AMOUNT$
	}

	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = andersson_add_x_prehistoric_artifacts_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = andersson_add_x_prehistoric_artifacts_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = andersson_add_x_prehistoric_artifacts_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = andersson_add_x_prehistoric_artifacts_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = andersson_add_x_prehistoric_artifacts_plus_x_tt
	}

	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = andersson_prehistoric_artifact_var
			}
		}
		set_variable = {
			name = andersson_prehistoric_artifact_var
			value = 0
		}
	}
	change_variable = {
		name = andersson_prehistoric_artifact_var
		add = $AMOUNT$
	}
	if = {
		limit = { #Only works if there actually is an anthropologist
			has_variable = andersson_anthropologist_var
		}
		var:andersson_anthropologist_var = {
			if = { #Fallback if the variable does not exist yet
				limit = {
					NOT = {
						has_variable = andersson_prehistoric_artifact_anthropologist_var
					}
				}
				set_variable = {
					name = andersson_prehistoric_artifact_anthropologist_var
					value = 0
				}
			}
			change_variable = { #The amount of human fossils that this anthropologist discovered
				name = andersson_prehistoric_artifact_anthropologist_var
				add = $AMOUNT$
			}
		}
	}
}

#Add Cave Paintings at Game Start
andersson_add_cave_paintings_effect = {
	#HISTORICAL SITES
	every_state_region = {
		limit = {
			OR = {
				this = s:STATE_UTAH #Horseshoe Canyon
				this = s:STATE_PATAGONIA #Cueva de las Manos
				this = s:STATE_PIAUI #Toca do Boqueirão da Pedra Furada
				this = s:STATE_ASTURIAS #Altamira
				this = s:STATE_GUYENNE #Vézère Valley
				this = s:STATE_EAST_SAHARA #Tassili n'Ajjer
				this = s:STATE_ZULULAND #San Rock Art
				this = s:STATE_CENTRAL_PROVINCES #Bhimbetka rock shelters
				this = s:STATE_GUANGXI #Zuojiang Huashan Rock Art
				this = s:STATE_CELEBES #Maros-Pangkep
				this = s:STATE_WESTERN_AUSTRALIA #Gwion Gwion rock art
			}
		}
		set_variable = andersson_cave_paintings_var
	}

	#NORTH AMERICA
	while = {
		count = 2

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_vero
				andersson_state_has_cave_paintings_trigger = no
			}
			state_region = {
				set_variable = andersson_cave_paintings_var
			}
		}
	}

	#SOUTH AMERICA
	while = {
		count = 2

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_lagoa_santa
				andersson_state_has_cave_paintings_trigger = no
			}
			state_region = {
				set_variable = andersson_cave_paintings_var
			}
		}
	}
	#EUROPE
	while = {
		count = 3

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_cro_magnon
				andersson_state_has_cave_paintings_trigger = no
			}
			state_region = {
				set_variable = andersson_cave_paintings_var
			}
		}
	}
	#AFRICA
	while = {
		count = 2

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_boskop
				andersson_state_has_cave_paintings_trigger = no
			}
			state_region = {
				set_variable = andersson_cave_paintings_var
			}
		}
	}
	#ASIA
	while = {
		count = 3

		random_state = {
			limit = {
				is_in_geographic_region = andersson_geographic_region_tianyuan
				andersson_state_has_cave_paintings_trigger = no
			}
			state_region = {
				set_variable = andersson_cave_paintings_var
			}
		}
	}
}

###ANTHROPOLOGICAL RATING######

#Specific Human Fossil: 5 Prestige

#Cave Paintings Discovery: 10 Prestige
#Neoanthropus Discovery: 20 Prestige, Skeleton: 25 Prestige
#Palaeanthropus Discovery: 40 Prestige, Skeleton: 50 Prestige
#Pithecanthropus Discovery: 60 Prestige, Skeleton: 75 Prestige
#Australopithecus Discovery: 80 Prestige, Skeleton: 100 Prestige

andersson_change_anthropological_rating = {
	# Amount = amount of artifacts to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = add_anthropological_prestige
		value = $AMOUNT$
	}

	#Used to display the right tooltip
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = andersson_change_anthropological_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = andersson_change_anthropological_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = andersson_change_anthropological_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = andersson_change_anthropological_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = andersson_change_anthropological_plus_x_tt
	}

	#Fallback in case the variable does not exist yet
	if = {
		limit = {
			NOT = {
				has_variable = andersson_anthropological_rating
			}
		}
		set_variable = {
			name = andersson_anthropological_rating
			value = 0
		}
	}

	#Actual Variable Change
	change_variable = {
		name = andersson_anthropological_rating
		add = $AMOUNT$
	}

	#Adds the correct modifier
	hidden_effect = {
		andersson_update_anthropological_prestige_modifiers_effect = yes
	}
}

andersson_update_anthropological_prestige_modifiers_effect = {
	if = {
		limit = {
			has_modifier = andersson_anthropological_rating_modifier
		}
		remove_modifier = andersson_anthropological_rating_modifier
	}
	add_modifier = {
		name = andersson_anthropological_rating_modifier
		multiplier = root.var:andersson_anthropological_rating #Anthropological Rating = Anthropological Prestige
	}
}
