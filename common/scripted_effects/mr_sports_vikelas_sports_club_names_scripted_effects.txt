#Effect to add a sports club to the list of Sports Club
# SPORT = name of the sport e.g. football
# FLAG = loc string with the name of the club
vikelas_add_sports_club_to_list_effect = {
	add_to_variable_list = {
		name = vikelas_$SPORT$_clubs_variable_list
		target = flag:$FLAG$
	}
}

vikelas_football_club_test_effect = {
	random_scope_state = {
		save_scope_as = vikelas_football_state
		random_in_list = {
			variable = vikelas_football_clubs_variable_list
			scope:vikelas_football_state = {
				set_variable = {
					name = vikelas_football_club_var
					value = prev
				}
			}
		}
	}
	custom_tooltip = vikelas_football_club_test_effect_tt
}
