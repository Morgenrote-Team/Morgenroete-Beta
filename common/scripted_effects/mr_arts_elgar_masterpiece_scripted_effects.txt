### Actual Effect to set the Masterpiece ###
# NUMBER = Number of the Masterpiece
# FLAG = flag that defines the name of the piece
elgar_create_masterpiece_effect = {
	set_variable = {
		name = elgar_masterpiece_$NUMBER$_var
		value = flag:$FLAG$
	}
}

elgar_character_create_masterpiece_number_choice_effect = {
	if = {
		limit = { var:elgar_composer_num_masterworks_var = 1 }
		set_variable = {
			name = elgar_masterpiece_1_var
			value = flag:$FLAG$
		}
	}
	else_if = {
		limit = { var:elgar_composer_num_masterworks_var = 2 }
		set_variable = {
			name = elgar_masterpiece_2_var
			value = flag:$FLAG$
		}
	}
	else = {
		set_variable = {
			name = elgar_masterpiece_3_var
			value = flag:$FLAG$
		}
	}
}
