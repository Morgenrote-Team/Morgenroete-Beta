#Includes the other seperate effects to hand out the medals and create the ranking.
vikelas_medals_of_this_olympia_effect = {
	if = {
		limit = { NOT = { has_global_variable = vikelas_olympics_effect_timer } } #Only triggers once
		vikelas_remove_old_medals_effect = yes
		vikelas_add_medals_effect = yes
		vikelas_create_medal_ranking_effect = yes
	}
	set_global_variable = { #This makes sure the effect is only run once per Olympic Games, necessary as the event is fired for every participant
		name = vikelas_olympics_effect_timer
		days = 30
	}
}

#Removes the medals of the last olympia
vikelas_remove_old_medals_effect = {
	every_country = {
		limit = {
			has_variable = vikelas_gold_medals_this_year_var
		}
		remove_variable = vikelas_gold_medals_this_year_var
	}
}

#Creates the Medal table
vikelas_create_medal_ranking_effect = {
	#Removes the GLOBAL ranking variables
	remove_global_variable = vikelas_medal_rank_1_this_year_global_var
	remove_global_variable = vikelas_medal_rank_2_this_year_global_var
	remove_global_variable = vikelas_medal_rank_3_this_year_global_var
	remove_global_variable = vikelas_medal_rank_4_this_year_global_var
	remove_global_variable = vikelas_medal_rank_5_this_year_global_var

	remove_global_variable = vikelas_medal_rank_7_this_year_global_var
	remove_global_variable = vikelas_medal_rank_8_this_year_global_var
	remove_global_variable = vikelas_medal_rank_9_this_year_global_var
	remove_global_variable = vikelas_medal_rank_10_this_year_global_var

	ordered_country = {
		limit = {
			has_variable = vikelas_gold_medals_this_year_var
		}
		order_by = var:vikelas_gold_medals_this_year_var
		max = 10 #Limits the list to the ten top countries
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = { has_global_variable = vikelas_medal_rank_1_this_year_global_var }
			}
			set_global_variable = {
				name = vikelas_medal_rank_1_this_year_global_var

			}
			set_global_variable = {

#Calculates the total number of Participants at the olympics
vikelas_calculate_number_participants_effect = {
	set_global_variable = {
		name = vikelas_olympic_participants_global_var
		value = 0
	}
	every_country = {
		limit = {
			has_variable = vikelas_ioc_member_var
			is_revolutionary = no
		}
		change_global_variable = {
			name = vikelas_olympic_participants_global_var
			add = 1
		}
	}
	vikelas_calculate_number_fifa_members_effect = yes
	vikelas_calculate_number_aiacr_members_effect = yes
	douglas_calculate_number_alpine_club_members_effect = yes
}

#As this effect is called in an effect, it creates the list with GLOBAL variables.
#Look at the customizable localizations to see how the countries are displayed.
#Basically the custom loc uses the variable to display both the countries in their order as well as the variables of those countries
vikelas_gold_medal_ranking_global_effect = {
	#Removes the GLOBAL ranking variables
	remove_global_variable = vikelas_gold_medal_rank_1_global_var

			is_revolutionary = no #Hopefully keeps revolutions from appearing in the ranking

#This effect creates a a variable to display the rank of each country for gold Medals
vikelas_olympic_rank_effect = {
	set_local_variable = { #This is used to keep track of the current rank
		name = current_rank
		value = 0
	}
	ordered_country = {
		limit = {
			has_variable = vikelas_gold_medals_var
			is_revolutionary = no #Hopefully keeps revolutions from appearing in the ranking
			NOR = {
				c:PRC ?= this
				is_country_type = decentralized
			}
		}
		order_by = var:vikelas_gold_medals_var
		check_range_bounds = no #prevents errors

		remove_variable = vikelas_olympic_gold_medal_rank_country_var
		change_local_variable = {
			name = current_rank
			add = 1
		}
		set_variable = { #uses the current rank to set the rank variable for the country.
			name = vikelas_olympic_gold_medal_rank_country_var
			value = local_var:current_rank
		}
	}
}

#This effect replaces the current Prestige Modifier
vikelas_replace_olympic_prestige_modifier_effect = {
	if = { #The if keepes countries that do not have medals from getting the modifier.

			multiplier = root.var:vikelas_gold_medals_var #Multiplies the base prestige per medal (5) with the number of medals
		}
	}
}

#This effect determines which racers may get a position in a City-to-City Race (or get a crash/breakdown)
vikelas_city_race_qualification_effect = {
	#Car Crash Chance: Racer risks crash if speed is higher than safety
	if = {
		limit = {
			vikelas_autorace_racer_speed_value > vikelas_autorace_racer_safety_value
		}
		random_list = {
			65 = {
				set_variable = vikelas_city_race_qualified_var
			}
			35 = {
				trigger_event = {
					id = vikelas.593
					days = 80
				}
			}
		}
	}
	#Arrival Chance: Racer arrives depending on reliability
	else_if = {
		limit = {
			vikelas_autorace_racer_speed_value > vikelas_autorace_racer_reliability_value
		}
		random_list = {
			50 = {
				set_variable = vikelas_city_race_qualified_var
			}
			50 = {
				trigger_event = {
					id = vikelas.594
					days = 80
				}
			}
		}
	}
	else = {
		set_variable = vikelas_city_race_qualified_var
		random_list = {
			2 = { trigger_event = { id = vikelas.513 days = 80 } }
			1 = { trigger_event = { id = vikelas.514 days = 80 } }
			1 = { trigger_event = { id = vikelas.515 days = 80 } }
			1 = { trigger_event = { id = vikelas.516 days = 80 } }
			1 = { trigger_event = { id = vikelas.517 days = 80 } }
			1 = { trigger_event = { id = vikelas.518 days = 80 } }
		}
	}
}

#This effect determines the three fastest racers in a City-to-City Race
vikelas_city_race_winner_effect = {
	#Removes the GLOBAL ranking variables
	remove_global_variable = vikelas_city_race_place_1_global_var
	remove_global_variable = vikelas_city_race_place_2_global_var
	remove_global_variable = vikelas_city_race_place_3_global_var

	ordered_country = {
		limit = {
			has_variable = vikelas_city_race_qualified_var
		}
		order_by = vikelas_autorace_racer_speed_value
		max = 3 #Limits the list to the three top countries
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = {
					has_global_variable = vikelas_city_race_place_1_global_var
				}
			}
			set_global_variable = {
				name = vikelas_city_race_place_1_global_var
				value = this
			}
			save_scope_as = vikelas_first_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = vikelas_city_race_place_2_global_var  }
			}
			set_global_variable = {
				name = vikelas_city_race_place_2_global_var
				value = this
			}
			save_scope_as = vikelas_second_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = vikelas_city_race_place_3_global_var  }
			}
			set_global_variable = {
				name = vikelas_city_race_place_3_global_var
				value = this
			}
			save_scope_as = vikelas_third_winner_scope
		}
	}
}

#Monthly Engine Mechanic Progress
vikelas_monthly_engine_mechanic_progress_effect = {
	if = { #Effect only gets executed if there is a mechanist
		limit = {
			has_variable = vikelas_mechanic_var
		}
		#Set up
		if = {
			limit = { NOT = { has_variable = vikelas_engine_mechanic_progress_var } }
			set_variable = {
				name = vikelas_engine_mechanic_progress_var
				value = 0
			}
		}

		### Actual calculation done in script_value ###
		if = {
			limit = { vikelas_monthly_engine_mechanic_progress_value > 0 }
			set_variable = {
				name = vikelas_estimated_monthly_engine_mechanic_progress_var
				value = vikelas_monthly_engine_mechanic_progress_value
			}
		}
		else = {
			set_variable = {
				name = vikelas_estimated_monthly_engine_mechanic_progress_var
				value = 0
			}
		}

		#Add monthly variable to the total variable
		if = {
			limit = { var:vikelas_estimated_monthly_engine_mechanic_progress_var > 0 }
			change_variable = {
				name = vikelas_engine_mechanic_progress_var
				add = var:vikelas_estimated_monthly_engine_mechanic_progress_var
			}
		}
	}
}

vikelas_grand_prix_championship_effect = {
	#Removes the GLOBAL ranking variables
	remove_global_variable = vikelas_grand_prix_championship_place_1_global_var
	remove_global_variable = vikelas_grand_prix_championship_place_2_global_var
	remove_global_variable = vikelas_grand_prix_championship_place_3_global_var
	remove_global_variable = vikelas_grand_prix_championship_place_4_global_var
	remove_global_variable = vikelas_grand_prix_championship_place_5_global_var

	ordered_country = {
		limit = {
			has_variable = vikelas_race_points_var
			var:vikelas_race_points_var >= 1
		}
		order_by = var:vikelas_race_points_var
		max = 5 #Limits the list to the three top countries
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = {
					has_global_variable = vikelas_grand_prix_championship_place_1_global_var
				}
			}
			set_global_variable = {
				name = vikelas_grand_prix_championship_place_1_global_var
				value = this
			}
			save_scope_as = vikelas_first_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = vikelas_grand_prix_championship_place_2_global_var  }
			}
			set_global_variable = {
				name = vikelas_grand_prix_championship_place_2_global_var
				value = this
			}
			save_scope_as = vikelas_second_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = vikelas_grand_prix_championship_place_3_global_var  }
			}
			set_global_variable = {
				name = vikelas_grand_prix_championship_place_3_global_var
				value = this
			}
			save_scope_as = vikelas_third_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = vikelas_grand_prix_championship_place_4_global_var  }
			}
			set_global_variable = {
				name = vikelas_grand_prix_championship_place_4_global_var

			multiplier = global_var:vikelas_num_basketball_nations_global_var #Multiplies the base prestige (1) and influence (5) with the number of countries
		}
	}

#Determine Cricket Winner
vikelas_create_cricket_winner_teams_effect = {

	ordered_country = {
		limit = {
			OR = {
				is_direct_subject_of = ROOT
				is_in_same_power_bloc = ROOT
			}
			vikelas_has_sports_association_trigger = yes
			var:vikelas_national_sport_var ?= root.var:vikelas_national_sport_var
		}
		order_by = vikelas_national_cricket_team_game_value
		max = 2 #Limits the list to the ten top countries
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = {
					has_global_variable = vikelas_first_cricket_winner_team_global_var
				}
			}
			set_global_variable = {
				name = vikelas_first_cricket_winner_team_global_var
				value = this
			}
			save_scope_as = vikelas_first_winner_team_scope
		}
		else_if = {
			limit = {
				NOT = {
					has_global_variable = vikelas_second_cricket_winner_team_global_var
				}
			}
			set_global_variable = {
				name = vikelas_second_cricket_winner_team_global_var
				value = this
			}
			save_scope_as = vikelas_second_winner_team_scope
		}

		remove_global_variable = vikelas_first_cricket_winner_team_global_var
		remove_global_variable = vikelas_second_cricket_winner_team_global_var
	}
}
