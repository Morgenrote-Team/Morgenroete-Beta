#Includes the other seperate effects to hand out the medals and create the ranking.
vikelas_medals_of_this_olympia_effect = {
	if = {
		limit = { NOT = { has_global_variable = vikelas_olympics_effect_timer } } #Only triggers once
		vikelas_remove_old_medals_effect = yes
		vikelas_add_medals_effect = yes
		vikelas_create_medal_ranking_effect = yes
	}
	set_global_variable = { #This makes sure the effect is only run once per Olympic Games, necessary as the event is fired for every participant
		name = vikelas_olympics_effect_timer
		days = 30
	}
}

#Removes the medals of the last olympia
vikelas_remove_old_medals_effect = {
	every_country = {
		limit = {
			has_variable = vikelas_gold_medals_this_year_var
		}
		remove_variable = vikelas_gold_medals_this_year_var
	}
}

		limit = {
			has_variable = vikelas_gold_medals_this_year_var

#Calculates the total number of Participants at the olypics
vikelas_calculate_number_participants_effect = {
	set_global_variable = {
		name = vikelas_olympic_participants_global_var
		value = 0
	}
	every_country = {
		limit = { has_variable = vikelas_ioc_member_var }
		change_global_variable = {
			name = vikelas_olympic_participants_global_var
			add = 1
		}
	}
}

vikelas_gold_medal_ranking_global_effect = {

	remove_global_variable = vikelas_gold_medal_rank_1_global_var

	# Amount = amount of Mechanical Innovation to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = mechanical_innovation_amount
		value = $AMOUNT$
	}
	
	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if local_var is negative
		limit = { $AMOUNT$ < 0 }
		custom_tooltip = vikelas_add_x_mechanical_innovation_minus_x_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = vikelas_add_x_mechanical_innovation_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 0 }
		custom_tooltip = vikelas_add_x_mechanical_innovation_plus_x_tt
	}

	#This part adds the AMOUNT of Mechanical Innovation
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = vikelas_mechanical_innovation_var
			}
		}
		set_variable = {
			name = vikelas_mechanical_innovation_var
			value = 0
		}
	}
	change_variable = {
		name = vikelas_mechanical_innovation_var
		add = scope:mechanical_innovation_amount
	}
}

#This effect determines which racers may get a position in a City-to-City Race (or get a crash/breakdown)
vikelas_city_race_qualification_effect = {
	#Car Crash Chance: Racer risks crash if speed is higher than safety
	if = {
		limit = {
			vikelas_autorace_racer_speed_value > vikelas_autorace_racer_safety_value
		}
		random_list = {
			65 = {
				set_variable = vikelas_city_race_qualified_var
			}
			35 = {
				trigger_event = {
					id = vikelas.593
					days = 80
				}
			}
		}
	}
	#Arrival Chance: Racer arrives depending on reliability
	else_if = {
		limit = {
			vikelas_autorace_racer_speed_value > vikelas_autorace_racer_reliability_value
		}
		random_list = {
			50 = {
				set_variable = vikelas_city_race_qualified_var
			}
			50 = {
				trigger_event = {
					id = vikelas.594
					days = 80
				}
			}
		}
	}
	else = {
		set_variable = vikelas_city_race_qualified_var
		random_list = {
			2 = { trigger_event = { id = vikelas.513 days = 80 } }
			1 = { trigger_event = { id = vikelas.514 days = 80 } }
			1 = { trigger_event = { id = vikelas.515 days = 80 } }
			1 = { trigger_event = { id = vikelas.516 days = 80 } }
			1 = { trigger_event = { id = vikelas.517 days = 80 } }
			1 = { trigger_event = { id = vikelas.518 days = 80 } }
		}
	}
}

	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = endurance_race_progress_amount
		value = $AMOUNT$
	}
	
	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if local_var is negative
		limit = { $AMOUNT$ < 0 }
		custom_tooltip = vikelas_add_x_endurance_race_progress_minus_x_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = vikelas_add_x_endurance_race_progress_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 0 }
		custom_tooltip = vikelas_add_x_endurance_race_progress_plus_x_tt
	}

	#This part adds the AMOUNT of Endurance Race Progress
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = vikelas_endurance_race_progress_var
			}
		}
		set_variable = {
			name = vikelas_endurance_race_progress_var
			value = 0
		}
	}
	change_variable = {
		name = vikelas_endurance_race_progress_var
		add = scope:endurance_race_progress_amount
	}
}

#Monthly Mechanical Innovation
vikelas_monthly_mechanical_innovation_effect = {
	if = { #Effect only gets executed if there is a mechanist
		limit = {
			has_variable = vikelas_mechanic_var
		}
		#Set up
		if = {
			limit = { NOT = { has_variable = vikelas_mechanical_innovation_var } }
			set_variable = {
				name = vikelas_mechanical_innovation_var
				value = 0
			}
		}

		### Actual calculation done in script_value ###
		if = {
			limit = { vikelas_monthly_mechanical_innovation_value > 0 }
			set_variable = {
				name = vikelas_estimated_monthly_mechanical_innovation_var
				value = vikelas_monthly_mechanical_innovation_value
			}
		}
		else = {
			set_variable = {
				name = vikelas_estimated_monthly_mechanical_innovation_var
				value = 0
			}
		}

		#Add monthly variable to the total variable
		if = {
			limit = { var:vikelas_estimated_monthly_mechanical_innovation_var > 0 }
			change_variable = {
				name = vikelas_mechanical_innovation_var
				add = var:vikelas_estimated_monthly_mechanical_innovation_var
			}
		}
	}
}

vikelas_grand_prix_championship_effect = {
	#Removes the GLOBAL ranking variables
	remove_global_variable = vikelas_grand_prix_championship_place_1_global_var 
	remove_global_variable = vikelas_grand_prix_championship_place_2_global_var 
	remove_global_variable = vikelas_grand_prix_championship_place_3_global_var
	remove_global_variable = vikelas_grand_prix_championship_place_4_global_var 
	remove_global_variable = vikelas_grand_prix_championship_place_5_global_var 

	ordered_country = {
		limit = {
			has_variable = vikelas_race_points_var
			var:vikelas_race_points_var >= 1
		}
		order_by = var:vikelas_race_points_var
		max = 5 #Limits the list to the three top countries
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = { 
					has_global_variable = vikelas_grand_prix_championship_place_1_global_var 
				}
			}
			set_global_variable = {
				name = vikelas_grand_prix_championship_place_1_global_var 
				value = this
			}
			save_scope_as = vikelas_first_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = vikelas_grand_prix_championship_place_2_global_var  }
			}
			set_global_variable = {
				name = vikelas_grand_prix_championship_place_2_global_var
				value = this
			}
			save_scope_as = vikelas_second_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = vikelas_grand_prix_championship_place_3_global_var  }
			}
			set_global_variable = {
				name = vikelas_grand_prix_championship_place_3_global_var
				value = this
			}
			save_scope_as = vikelas_third_winner_scope
		}
		else_if = {
			limit = {
				NOT = { has_global_variable = vikelas_grand_prix_championship_place_4_global_var  }

#Determine Cricket Winner
vikelas_create_cricket_winner_teams_effect = {

	ordered_country = {
		limit = {
			OR = {
				is_direct_subject_of = ROOT
				is_in_same_power_bloc = ROOT
			}
			vikelas_has_sports_association_trigger = yes
			var:vikelas_national_sport_var ?= root.var:vikelas_national_sport_var
		}
		order_by = vikelas_national_cricket_team_game_value
		max = 2 #Limits the list to the ten top countries
		check_range_bounds = no #prevents errors

		#This part checks one after another whether the first, second, etc. ranked countries are saved already.
		#Luckily the ordered_country effect iterates through the countries one after another.
		#So the first country gets saved to the first variable but as that variable now is set,
		# the second country gets saved to the second variable and so on.
		if = {
			limit = {
				NOT = {
					has_global_variable = vikelas_first_cricket_winner_team_global_var
				}
			}
			set_global_variable = {
				name = vikelas_first_cricket_winner_team_global_var 
				value = this
			}
			save_scope_as = vikelas_first_winner_team_scope
		}
		else_if = {
			limit = {
				NOT = {
					has_global_variable = vikelas_second_cricket_winner_team_global_var
				}
			}
			set_global_variable = {
				name = vikelas_second_cricket_winner_team_global_var 
				value = this
			}
			save_scope_as = vikelas_second_winner_team_scope
		}

		remove_global_variable = vikelas_first_cricket_winner_team_global_var
		remove_global_variable = vikelas_second_cricket_winner_team_global_var
	}
}
