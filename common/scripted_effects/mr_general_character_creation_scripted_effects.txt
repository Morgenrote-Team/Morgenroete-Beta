# Gustave Eiffel
#Historical Eiffel
create_eiffel_gustave_effect = {
	create_character = {
		template = mr_fra_eiffel_gustave_character_template

		save_scope_as = gaudi_engineer_scope
	}

	scope:gaudi_engineer_scope = {
		set_variable = mr_no_headgear_var
		mr_make_character_engineer_effect = yes
	}

	set_variable = {
		name = tesla_engineer_var
		value = scope:gaudi_engineer_scope
	}

	set_global_variable = mr_eiffel_created_global_var
}
