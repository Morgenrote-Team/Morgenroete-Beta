###### MAGNETISM ######

#Effect to add $AMOUNT$ Magnetic Data
tesla_add_x_magnetic_data = {
	# Amount = amount of Magnetic Data to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = magnetic_data_amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ < -1 }
		custom_tooltip = tesla_add_x_magnetic_data_minus_x_tt
	}
	if = { #Negative Singular Tooltip if $AMOUNT$ is negative
		limit = { $AMOUNT$ = -1 }
		custom_tooltip = tesla_add_x_magnetic_data_minus_1_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = tesla_add_x_magnetic_data_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ = 1 }
		custom_tooltip = tesla_add_x_magnetic_data_plus_1_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 1 }
		custom_tooltip = tesla_add_x_magnetic_data_plus_x_tt
	}

	#This part adds the AMOUNT of National Magnetic Data
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_variable = gaudi_magnetic_data
			}
		}
		set_variable = {
			name = gaudi_magnetic_data
			value = 0
		}
	}
	change_variable = {
		name = gaudi_magnetic_data
		add = $AMOUNT$
	}

	#This part adds the AMOUNT of Global Magnetic Data
	if = { #Fallback if the variable does not exist yet
		limit = {
			NOT = {
				has_global_variable = gaudi_total_magnetic_data
			}
		}
		set_global_variable = {
			name = gaudi_total_magnetic_data
			value = 0
		}
	}
	change_global_variable = {
		name = gaudi_total_magnetic_data
		add = $AMOUNT$
	}
}

tesla_set_magnetic_association = {
	set_global_variable = {
		name = gaudi_magnetic_association_members
		value = 1
	}
	set_variable = {
		name = gaudi_magnetic_data
		value = 0
	}
	set_global_variable = {
		name = gaudi_total_magnetic_data
		value = 0
	}
	set_variable = gaudi_magnetic_association_member_var

	custom_tooltip = tesla_set_magnetic_association_tt
}

tesla_add_magnetic_association_member = {
	change_global_variable = {
		name = gaudi_magnetic_association_members
		add = 1
	}
	set_variable = gaudi_magnetic_association_member_var

	custom_tooltip = tesla_add_magnetic_association_member_tt
}

	if = {
		limit = {
			has_variable = tesla_engineer_var

	set_variable = {

	}
}

#Effect used to add or remove progress from the scripted progress bar.
# AMOUNT = amount of progress to add/subtract
tesla_add_x_mechanical_computer_progress_effect = {
	# Amount = amount of Progress to add
	save_temporary_scope_value_as = { #This is used to display the amount in the tooltip
		name = amount
		value = $AMOUNT$
	}

	#This part is used to display an accurate tooltip (Gain, loss, etc.)
	if = { #Negative Plural Tooltip if local_var is negative
		limit = { $AMOUNT$ < 0 }
		custom_tooltip = tesla_add_x_mechanical_computer_progress_minus_x_tt
	}
	if = { #Neutral Tooltip if $AMOUNT$ is 0
		limit = { $AMOUNT$ = 0 }
		custom_tooltip = tesla_add_x_mechanical_computer_progress_0_tt
	}
	if = { #Positive Plural Tooltip if $AMOUNT$ is positive
		limit = { $AMOUNT$ > 0 }
		custom_tooltip = tesla_add_x_mechanical_computer_progress_plus_x_tt
	}

	hidden_effect = {
		je:je_gaudi_mechanical_computer_project = {
			add_progress = {
				name = tesla_mechanical_computer_project_progress_bar
				value = $AMOUNT$
			}
		}
	}
}

	#Removes the global variable list
	clear_global_variable_list = mr_engineer_ranking_global_list

	ordered_character_in_void = {
		limit = {
			has_variable = mr_is_engineer_var
		}
		order_by = tesla_engineer_experience_current_specialization_character_scope_value
		check_range_bounds = no

		#This part adds all the countries to the list
		add_to_global_variable_list = {
			name = mr_engineer_ranking_global_list
			target = this
		}
	}
}
