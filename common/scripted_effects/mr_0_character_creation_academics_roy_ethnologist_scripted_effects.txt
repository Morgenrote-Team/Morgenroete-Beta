# Creates a new Ethnologist without losing the old one
#This one was a little harder as the scope was only valid in the random_list
#Therefore a Variable is set to save the scope again. then gets deleted
mr_create_historical_ethnologist_for_character_creation_effect = {
	random_list = {
		50 = { #Andreas Reischek, Austria
			trigger = {
				NOT = { has_global_variable = mr_reischek_andreas_created_global_var }
				has_state_in_state_region = STATE_AUSTRIA #Born in Linz
				country_average_cultural_acceptance = {
					target = cu:south_german
					value >= acceptance_status_4
				}
				game_date < 1900.11.26 #Age 55
				year >= 1870 #age 25
			}
			create_character = {
				template = mr_aus_reischek_andreas_character_template

				save_scope_as = roy_new_ethnologist_scope
			}
			scope:roy_new_ethnologist_scope = {
				set_variable = mr_is_ethnologist_var
				set_variable = mr_no_headgear_var
				ethnologist_experience_set_up_effect = yes
				place_character_in_void = 1200	
			}

			set_variable = {
				name = roy_new_ethnologist_var
				value = scope:roy_new_ethnologist_scope
			}

			set_global_variable = mr_reischek_andreas_created_global_var
		}
		100 = { #Frederik Adolph de Roepstorff, Denmark

				NOT = { has_global_variable = mr_de_roepstorff_frederik_adolph_created_global_var }

					target = cu:danish

				template = mr_den_de_roepstorff_frederik_adolph_character_template

			}

mr_create_random_ethnologist_for_character_creation_effect = {
	random_list = {
		1 = {
			create_character = {
				female = yes
				trait_generation = {}
				save_scope_as = roy_new_ethnologist_scope
			}
			scope:roy_new_ethnologist_scope = {
				add_trait = roy_ethnologist
				set_variable = mr_is_ethnologist_var
				ethnologist_experience_set_up_effect = yes
				mr_trait_generation_ethnologist_effect = yes
				place_character_in_void = 1200
			}
			set_variable = {
				name = roy_new_ethnologist_var
				value = scope:roy_new_ethnologist_scope
			}

			modifier = {
				if = {
					limit = { has_law = law_type:law_women_own_property }
					add = 4
				}
				if = {
					limit = { has_law = law_type:law_women_in_the_workplace }
					add = 9
				}
				if = {
					limit = { has_law = law_type:law_womens_suffrage }
					add = 19
				}
			}
		}
		99 = {
			create_character = {
				trait_generation = {}
				save_scope_as = roy_new_ethnologist_scope
			}
			scope:roy_new_ethnologist_scope = {
				add_trait = roy_ethnologist
				set_variable = mr_is_ethnologist_var
				ethnologist_experience_set_up_effect = yes
				mr_trait_generation_ethnologist_effect = yes
				place_character_in_void = 1200
			}
			set_variable = {
				name = roy_new_ethnologist_var
				value = scope:roy_new_ethnologist_scope
			}

			modifier = {
				if = {
					limit = { has_law = law_type:law_women_own_property }
					add = -4
				}
				if = {
					limit = { has_law = law_type:law_women_in_the_workplace }
					add = -9
				}
				if = {
					limit = { has_law = law_type:law_womens_suffrage }
					add = -19
				}
			}
		}
	}
}
