#Checks if the scope is an AI and then starts Engineering Projects
ai_start_engineering_project_for_ai_effect = {
	if = {
		limit = {
			tesla_has_active_engineer_no_tooltip_trigger = yes
			tesla_is_engineering_automated_trigger = yes
		}
		ai_engineering_effect = yes
	}
}
