#Handles the Engineering projects for the AI
ai_engineering_effect = {
	#Checks if there is an engineer
	if = {
		limit = { tesla_has_active_engineer_no_tooltip_trigger = yes }

		### Civil Engineer ###
		if = {
			limit = {
				tesla_has_civil_engineer_no_tooltip_trigger = yes
			}

			random_list = {
				### Infrastructure Project ###
				100 = {
					#Trigger
					trigger = {
						#Can the AI afford the project?
						OR = {
							weekly_net_fixed_income > tesla_medium_project_cost_weekly_value
							gold_reserves > tesla_medium_project_cost_yearly_value
						}

						#Is there any state where a project is possible?
						any_scope_state = {
							OR = {
								NOT = { has_modifier = gaudi_modern_roads_modifier }
								NOT = { has_modifier = gaudi_new_bridge_modifier }
								NOT = { has_modifier = gaudi_new_tunnel_modifier }
							}
						}
					}

					#Effect
					tesla_civilian_engineer_infrastructure_button_effect = yes

					#Modifier
					modifier = {
						# If a state has a negative infrastructure balance, this project will be more likely
						if = {
							limit = {
								any_scope_state = {
									state_infrastructure_balance < 0
								}
							}

							add = 200
						}
					}
				}
				### Railway Project ###
				100 = {
					#Trigger
					trigger = {
						#Can the AI afford the project?
						OR = {
							weekly_net_fixed_income > tesla_medium_project_cost_weekly_value
							gold_reserves > tesla_medium_project_cost_yearly_value
						}

						#Is there any state where a project is possible?
						any_scope_building = {
							tesla_railway_valid_for_improvement_trigger = yes
						}
					}

					#Effect
					add_journal_entry = { type = je_gaudi_railway_project }
					je:je_gaudi_railway_project = {
						add_modifier = {
							name = gaudi_engineer_project_medium_cost_modifier
							multiplier = money_amount_multiplier_small
						}
					}
					if = {
						limit = {
							has_variable = tesla_engineer_var
						}
						var:tesla_engineer_var = {
							set_character_busy_and_immortal = yes
						}
					}

					#Modifier
					modifier = {
						# If a state has a negative infrastructure balance, this project will be more likely
						if = {
							limit = {
								any_scope_state = {
									state_infrastructure_balance < 0
								}
							}

							add = 200
						}
					}
				}
				### Irrigation Project ###
				200 = {
					#Trigger
					trigger = {
						#Can the AI afford the project?
						OR = {
							weekly_net_fixed_income > tesla_medium_project_cost_weekly_value
							gold_reserves > tesla_medium_project_cost_yearly_value
						}

						#Is there any state where a project is possible?
						any_scope_state = {
							tesla_state_region_has_no_irrigation_system = yes
						}
					}

					#Effect
					add_journal_entry = { type = je_tesla_irrigation_project }
					je:je_tesla_irrigation_project = {
						add_modifier = {
							name = gaudi_engineer_project_low_cost_modifier
							multiplier = money_amount_multiplier_small
						}
					}
					if = {
						limit = {
							has_variable = tesla_engineer_var
						}
						var:tesla_engineer_var = {
							set_character_busy_and_immortal = yes
						}
					}

					#Modifier
					modifier = {
						# Chance is higher for countries focussing on agriculture
						if = {
							limit = {
								has_strategy = ai_strategy_plantation_economy
								has_strategy = ai_strategy_agricultural_expansion
							}

							add = 100
						}
					}
				}
				### Construction Project ###
				200 = {
					#Trigger
					trigger = {
						#Can the AI afford the project?
						OR = {
							weekly_net_fixed_income > tesla_low_project_cost_weekly_value
							gold_reserves > tesla_low_project_cost_yearly_value
						}

						#Is there any building where a project is possible?
						any_scope_building = {
							tesla_construction_sector_valid_for_improvement_trigger = yes
						}
					}

					#Effect
					ordered_scope_building = {
						limit = {
							tesla_construction_sector_valid_for_improvement_trigger = yes
						}
						order_by = level
						max = 1000
						check_range_bounds = no

						save_scope_as = chosen_building
					}
					add_journal_entry = {
						type = je_gaudi_construction_sector_project
						target = scope:chosen_building
					}
					je:je_gaudi_construction_sector_project = {
						add_modifier = {
							name = gaudi_engineer_project_low_cost_modifier
							multiplier = money_amount_multiplier_small
						}
					}
					var:gaudi_engineer_var = {
						set_character_busy_and_immortal = yes
					}

					#Modifier
					modifier = {
						# Chance is higher for countries focussing on industry
						if = {
							limit = {
								has_strategy = ai_strategy_resource_expansion
								has_strategy = ai_strategy_industrial_expansion
								has_strategy = ai_strategy_colonial_extraction
							}

							add = 100
						}
					}
				}
				### Topographic Map Project ###
				300 = {
					#Trigger
					trigger = {
						#Can the AI afford the project?
						OR = {
							weekly_net_fixed_income > tesla_low_project_cost_weekly_value
							gold_reserves > tesla_low_project_cost_yearly_value
						}

						#Is there any building where a project is possible?
						any_scope_state = {
							NOT = { has_modifier = gaudi_topographic_state_map_modifier }
						}
						NOT = { has_journal_entry = je_dufour_carte_dufour }
					}

					#Effect
					add_journal_entry = { type = je_gaudi_topographic_map_project }
					je:je_gaudi_topographic_map_project = {
						add_modifier = {
							name = gaudi_engineer_project_low_cost_modifier
							multiplier = money_amount_multiplier_small
						}
					}
					if = {
						limit = {
							has_variable = tesla_engineer_var
						}
						var:tesla_engineer_var = {
							set_character_busy_and_immortal = yes
						}
					}
				}
				### Civil Engineering Research Project ###
				1 = {
					#Trigger
					trigger = {
						#Can the AI afford the project?
						OR = {
							weekly_net_fixed_income > tesla_low_project_cost_weekly_value
							gold_reserves > tesla_low_project_cost_yearly_value
						}
					}

					#Effect
					add_modifier = {
						name = gaudi_engineer_project_low_cost_modifier
						multiplier = money_amount_multiplier_small
					}
					add_journal_entry = { type = je_gaudi_civil_research_project }
					var:tesla_engineer_var = {
						set_character_busy_and_immortal = yes
					}
				}
			}

			### Mechanical Engineer ###
			if = {
				limit = {
					tesla_has_mechanical_engineer_no_tooltip_trigger = yes
				}

				random_list = {
					### Locomotives Project ###
					100 = {
						#Trigger
						trigger = {
							#Can the AI afford the project?
							OR = {
								weekly_net_fixed_income > tesla_medium_project_cost_weekly_value
								gold_reserves > tesla_medium_project_cost_yearly_value
							}

							#Is there any state where a project is possible?
							any_scope_building = {
								is_building_type = building_railway
								level > 0
								NOR = {
									has_modifier = gaudi_locomotive_tier_1_modifier
									has_modifier = gaudi_locomotive_tier_2_modifier
									has_modifier = gaudi_locomotive_tier_3_modifier
									has_modifier = gaudi_locomotive_tier_4_modifier
									has_modifier = gaudi_locomotive_tier_5_modifier
								}
							}
						}

						#Effect
						add_journal_entry = { type = je_gaudi_new_locomotives_project }
						add_modifier = {
							name = gaudi_engineer_project_medium_cost_modifier
							multiplier = money_amount_multiplier_small
						}
						if = {
							limit = {
								has_variable = tesla_engineer_var
							}
							var:tesla_engineer_var = {
								set_character_busy_and_immortal = yes
							}
						}

						#Modifier
						modifier = {
							# Chance is higher for countries focussing on industry
							if = {
								limit = {
									has_strategy = ai_strategy_resource_expansion
									has_strategy = ai_strategy_industrial_expansion
									has_strategy = ai_strategy_colonial_extraction
								}

								add = 200
							}
						}
					}
					### Improve Industires Project ###
					300 = {
						#Trigger
						trigger = {
							#Can the AI afford the project?
							OR = {
								weekly_net_fixed_income > tesla_medium_project_cost_weekly_value
								gold_reserves > tesla_medium_project_cost_yearly_value
							}

							#Is there any state where a project is possible?
							any_scope_building = {
								tesla_building_valid_for_mechanical_improvement_trigger = yes
							}
						}

						#Effect
						ordered_scope_building = {
							limit = {
								tesla_building_valid_for_mechanical_improvement_trigger = yes
							}
							order_by = earnings
							save_scope_as = chosen_building
						}
						add_journal_entry = {
							type = je_gaudi_modernizing_industries_project
							target = scope:chosen_building
						}
						je:je_gaudi_modernizing_industries_project = {
							add_modifier = {
								name = gaudi_engineer_project_medium_cost_modifier
								multiplier = money_amount_multiplier_small
							}
						}
						if = {
							limit = {
								has_variable = tesla_engineer_var
							}
							var:tesla_engineer_var = {
								set_character_busy_and_immortal = yes
							}
						}

						#Modifier
						modifier = {
							# Chance is higher for countries focussing on industry
							if = {
								limit = {
									has_strategy = ai_strategy_resource_expansion
									has_strategy = ai_strategy_industrial_expansion
									has_strategy = ai_strategy_colonial_extraction
								}

								add = 200
							}
						}
					}


					### Mechanical Engineering Research Project ###
					1 = {
						#Trigger
						trigger = {
							#Can the AI afford the project?
							OR = {
								weekly_net_fixed_income > tesla_low_project_cost_weekly_value
								gold_reserves > tesla_low_project_cost_yearly_value
							}
						}

						#Effect
						add_journal_entry = { type = je_gaudi_mechanical_research_project }
						add_modifier = {
							name = gaudi_engineer_project_low_cost_modifier
							multiplier = money_amount_multiplier_small
						}
						var:tesla_engineer_var = {
							set_character_busy_and_immortal = yes
						}
					}
				}
			}
		}
	}
}
