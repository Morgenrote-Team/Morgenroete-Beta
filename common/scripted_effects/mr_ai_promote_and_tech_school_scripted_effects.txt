#Handles the Tech Schools for the AI
ai_choose_tech_school_effect = {
	if = {
		#Only change the tech school if there is no active tech school yet.
		limit = {
			academics_has_tech_school = no
			NOT = {
				has_modifier = academics_national_academic_confusion_modifier
			}
		}

		#Chooses a Tech School depending on randomness
		random_list = {
			### Avantgarde Intelligentsia ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_avantgarde_intelligentsia_modifier
				}

				#Trigger
				trigger = {
					ig:ig_intelligentsia = {
						is_in_government = yes
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.50
						has_active_production_method = pm_dubois_university_humanities
					}
				}
			}
			### Practical Academia ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_practical_academia_modifier
				}

				#Trigger
				trigger = {
					OR = {
						ig:ig_devout = {
							is_in_government = yes
						}
						ig:ig_trade_unions = {
							is_in_government = yes
						}
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.50
						has_active_production_method = pm_dubois_university_natural_history
					}
				}
			}
			### Military-Industrial Complex ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_military_industrial_complex_modifier
				}

				#Trigger
				trigger = {
					OR = {
						ig:ig_armed_forces = {
							is_in_government = yes
						}
						ig:ig_industrialists = {
							is_in_government = yes
						}
					}
					army_size >= 50
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.50
						has_active_production_method = pm_dubois_university_mathematics
					}
					any_scope_building = {
						OR = {
							AND = {
								is_building_type = building_automotive_industry
								level >= 5
							}
							AND = {
								is_building_type = building_arms_industry
								level >= 5
							}
							AND = {
								is_building_type = building_munition_plant
								level >= 5
							}
							AND = {
								is_building_type = building_artillery_foundry
								level >= 5
							}
						}
					}
				}
			}
			### Military-Industrial Complex ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_business_schools_tycoon_industrialism_modifier
				}

				#Trigger
				trigger = {
					OR = {
						ig:ig_petty_bourgeoisie = {
							is_in_government = yes
						}
						ig:ig_industrialists = {
							is_in_government = yes
						}
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_mathematics
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_humanities
					}
					any_scope_building = {
						OR = {
							AND = {
								is_building_type = building_textile_mill
								level >= 15
							}
							AND = {
								is_building_type = building_furniture_manufactory
								level >= 15
							}
							AND = {
								is_building_type = building_tooling_workshop
								level >= 10
							}
							AND = {
								is_building_type = building_steel_mill
								level >= 10
							}
							AND = {
								is_building_type = building_motor_industry
								level >= 5
							}
							AND = {
								is_building_type = building_electrics_industry
								level >= 5
							}
						}
					}
				}
			}
			### Sea Power & Merchant Marine ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_sea_power_merchant_marine_modifier
				}

				#Trigger
				trigger = {
					OR = {
						ig:ig_armed_forces = {
							is_in_government = yes
						}
						ig:ig_petty_bourgeoisie = {
							is_in_government = yes
						}
					}
					navy_size >= 50
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_mathematics
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_natural_history
					}
				}
			}
			### Useful Ivory Tower ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_useful_ivory_tower_modifier
				}

				#Trigger
				trigger = {
					OR = {
						ig:ig_rural_folk = {
							is_in_government = yes
						}
						ig:ig_landowners = {
							is_in_government = yes
						}
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_humanities
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_natural_history
					}
					OR = {
						any_scope_building = {
							is_building_type = building_lepsius_archaeological_museum
						}
						any_scope_building = {
							is_building_type = building_verrier_observatory
						}
						any_scope_building = {
							is_building_type = building_dubois_zoo
						}
					}
				}
			}
		}
	}
}

		}

			0 = { #Higher base chance

#Handles the promotion of Artists
ai_promote_artists_effect = {
	if = {
		limit = {
			NOT = {
				has_variable = artists_promotion_active
			}
			any_scope_building = {
				is_building_type = building_art_academy
				level > 0
			}
		}
		random_list = {
			### Architect ###
			0 = {
				#Effect
				set_variable = artists_promote_architect_active_var
				set_variable = artists_promotion_active

				#Trigger
				trigger = {
					NOT = {
						has_variable = artists_promote_architect_active_var
					}
					any_scope_building = {
						is_building_type = building_art_academy
						has_active_production_method = pm_artists_arts_academy_school
					}
				}

				#Modifier
				modifier = {
					add = ai_promote_architect_probability_value
				}
			}
			### Composer ###
			0 = {
				#Effect
				set_variable = artists_promote_composer_active_var
				set_variable = artists_promotion_active
	
				#Trigger
				trigger = {
					NOR = {
						has_variable = artists_promote_composer_active_var
						has_variable = artists_promotion_active
					}
					any_scope_building = {
						is_building_type = building_art_academy
						has_active_production_method = pm_artists_arts_academy_conservatory
					}
				}
	
				#Modifier
				modifier = {
					add = ai_promote_composer_probability_value
				}
			}
			### Writer ###
			0 = {
				#Effect
				set_variable = artists_promote_writer_active_var
				set_variable = artists_promotion_active
	
				#Trigger
				trigger = {
					AND = { #Can also promote writers if already has the other artists
						elgar_has_composer_no_tooltip_trigger = yes
						klimt_has_painter_no_tooltip = yes
					}
					NOR = {
						has_variable = artists_promote_writer_active_var
						has_variable = artists_promotion_active
					}
					any_scope_building = {
						is_building_type = building_art_academy
						has_active_production_method = pm_artists_arts_academy_salon
					}
				}
	
				#Modifier
				modifier = {
					add = ai_promote_writer_probability_value
				}
			}
			### Painter ###
			0 = {
				#Effect
				set_variable = artists_promote_painter_active_var
				set_variable = artists_promotion_active
	
				#Trigger
				trigger = {
					NOR = {
						has_variable = artists_promote_painter_active_var
						has_variable = artists_promotion_active
					}
					any_scope_building = {
						is_building_type = building_art_academy
						has_active_production_method = pm_artists_arts_academy_studio
					}
				}
	
				#Modifier
				modifier = {
					add = ai_promote_painter_probability_value
				}
			}
		}
	}
}
