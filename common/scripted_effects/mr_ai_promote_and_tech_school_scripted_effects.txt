#Handles the Tech Schools for the AI
ai_choose_tech_school_effect = {
	if = {
		#Only change the tech school if there is no active tech school yet.
		limit = {
			academics_has_tech_school = no
			NOT = {
				has_modifier = academics_national_academic_confusion_modifier
			}
		}

		#Chooses a Tech School depending on randomness
		random_list = {
			### Avantgarde Intelligentsia ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_avantgarde_intelligentsia_modifier
				}

				#Trigger
				trigger = {
					ig:ig_intelligentsia = {
						is_in_government = yes
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.50
						has_active_production_method = pm_dubois_university_humanities
					}
				}
			}
			### Practical Academia ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_practical_academia_modifier
				}

				#Trigger
				trigger = {
					OR = {
						ig:ig_devout = {
							is_in_government = yes
						}
						ig:ig_trade_unions = {
							is_in_government = yes
						}
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.50
						has_active_production_method = pm_dubois_university_natural_history
					}
				}
			}
			### Military-Industrial Complex ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_military_industrial_complex_modifier
				}

				#Trigger
				trigger = {
					OR = {
						ig:ig_armed_forces = {
							is_in_government = yes
						}
						ig:ig_industrialists = {
							is_in_government = yes
						}
					}
					army_size >= 50
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.50
						has_active_production_method = pm_dubois_university_mathematics
					}
					any_scope_building = {
						OR = {
							AND = {
								is_building_type = building_war_machine_industry
								level >= 5
							}
							AND = {
								is_building_type = building_arms_industry
								level >= 5
							}
							AND = {
								is_building_type = building_munition_plants
								level >= 5
							}
							AND = {
								is_building_type = building_artillery_foundries
								level >= 5
							}
						}
					}
				}
			}
			### Military-Industrial Complex ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_business_schools_tycoon_industrialism_modifier

				trigger = {
					OR = {
						ig:ig_petty_bourgeoisie = {

					}

						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_humanities
					}

						OR = {
							AND = {
								is_building_type = building_textile_mills
								level >= 15
							}
							AND = {
								is_building_type = building_furniture_manufacturies
								level >= 15
							}
							AND = {
								is_building_type = building_tooling_workshops
								level >= 10
							}
							AND = {
								is_building_type = building_steel_mills
								level >= 10
							}
							AND = {
								is_building_type = building_motor_industry
								level >= 5
							}
							AND = {
								is_building_type = building_electrics_industry
								level >= 5
							}
						}
					}
				}
			}
			### Sea Power & Merchant Marine ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier

				}

						percent >= 0.35

					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_natural_history
					}
				}
			}
			### Useful Ivory Tower ###
			10 = {
				#Effect
				academics_remove_focus = yes
				add_modifier = {
					name = academics_national_academic_confusion_modifier
					years = 5
				}
				add_modifier = {
					name = academics_useful_ivory_tower_modifier
				}

				#Trigger
				trigger = {
					OR = {
						ig:ig_rural_folk = {
							is_in_government = yes
						}
						ig:ig_landowners = {
							is_in_government = yes
						}
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_humanities
					}
					any_scope_building = {
						filter = {
							is_building_type = building_university
						}
						percent >= 0.35
						has_active_production_method = pm_dubois_university_natural_history
					}
					OR = {
						any_scope_building = {
							is_building_type = building_lepsius_archaeological_museum
						}
						any_scope_building = {
							is_building_type = building_verrier_observatory
						}
						any_scope_building = {
							is_building_type = building_dubois_zoo
						}
					}
				}
			}
		}
	}
}

#Handles the promotion of Artists

				has_variable = academics_academy_promotion_active
			}
			any_scope_building = {
				is_building_type = building_university
				level > 0
			}
		}

					}
				}

				#Modifier
				modifier = {
					#Nations historically focused on engineering
					if = {
						limit = {
							OR = {
								c:PRU ?= ROOT
								c:ITA ?= ROOT
								c:SWI ?= ROOT
								c:GER ?= ROOT
								c:AUS ?= ROOT
								c:SPA ?= ROOT
								c:BRZ ?= ROOT
							}
						}
						add = 25
					}
					#Smaller countries should favor engineers
					if = {
						limit = {
							country_rank < rank_value:major_power
						}
						add = 50
					}
					#Nations without access to formations, rainforests or excavations sites have a higher chance to promote Engineers

						limit = {
							NOR = {
								dubois_has_formation = yes
								dubois_can_discover_formation = yes
								any_country = {
									has_diplomatic_pact = {
										who = ROOT
										type = paleontological_license
									}

				}

				#Modifier
				modifier = {
					#Major and Great Powers have a higher chance for an Anthropologist
					if = {
						limit = {
							country_rank >= rank_value:major_power
						}
						add = 50
					}
					#Countries with a Archaeological Museum have a higher chance to promote an Anthropologist
					if = {
						limit = {
							lepsius_has_archaeological_museum = yes
						}
						add = 50
					}
					#If the AI has a Physician already, it will not try to get another one except if the existing one is old.
					if = {
						limit = {
							lepsius_has_anthropologist_no_tooltip_trigger = yes
							var:lepsius_anthropologist_var ?= {
								age < 60
							}
						}
						add = -1000
					}
				}
			}
		}
	}
}

#Handles the promotion of Artists
ai_promote_artists_effect = {
	if = {
		limit = {
			NOT = {
				has_variable = artists_promotion_active
			}
			any_scope_building = {
				is_building_type = building_arts_academy
				level > 0
			}
		}
		random_list = {
			### Architect ###
			10 = {
				#Effect
				set_variable = artists_promote_architect_active_var
				set_variable = artists_promotion_active

				#Trigger
				trigger = {
					NOT = {
						has_variable = artists_promote_architect_active_var
					}
					any_scope_building = {
						is_building_type = building_arts_academy
						has_active_production_method = pm_artists_arts_academy_school
					}
				}

				#Modifier
				modifier = {
					if = {
						limit = {
							gaudi_has_architect_no_tooltip_trigger = yes
							var:gaudi_architect_var ?= {
								age < 70
							}
						}
						add = -9 #If the AI has an Architect already, it will rarely try to get another one except if the existing one is old.
					}
				}
			}
			### Composer ###
			10 = {
				#Effect
				set_variable = artists_promote_composer_active_var
				set_variable = artists_promotion_active
	
				#Trigger
				trigger = {
					NOR = {
						has_variable = artists_promote_composer_active_var
						has_variable = artists_promotion_active
					}
					any_scope_building = {
						is_building_type = building_arts_academy
						has_active_production_method = pm_artists_arts_academy_conservatory
					}
				}
	
				#Modifier
				modifier = {
					if = {
						limit = {
							elgar_has_composer_no_tooltip_trigger = yes
							var:elgar_composer_var ?= {
								age < 70
							}
						}
						add = -9 #If the AI has a Composer already, it will rarely try to get another one except if the existing one is old.
					}
					else_if = {
						limit = {
							OR = { #Cultures that are more probable to promote Composers
								country_has_primary_culture = cu:north_italian
								country_has_primary_culture = cu:south_italian
								country_has_primary_culture = cu:russian
								country_has_primary_culture = cu:south_german
							}
						}
						add = 5 #If the AI has a Composer already, it will not try to get another one.
					}
				}
			}
			### Writer ###
			10 = {
				#Effect
				set_variable = artists_promote_writer_active_var
				set_variable = artists_promotion_active
	
				#Trigger
				trigger = {
					OR = {
						AND = { #Can also promote writers if already has the other artists
							elgar_has_composer_no_tooltip_trigger = yes
							klimt_has_painter_no_tooltip = yes
						}
					}
					NOR = {
						has_variable = artists_promote_writer_active_var
						has_variable = artists_promotion_active
					}
					any_scope_building = {
						is_building_type = building_arts_academy
						has_active_production_method = pm_artists_arts_academy_salon
					}
				}
	
				#Modifier
				modifier = {
					if = {
						limit = {
							manzoni_has_writer_no_tooltip = yes
							var:manzoni_writer_var ?= {
								age < 70
							}
						}
						add = -9 #If the AI has a Painter already, it will rarely try to get another one except if the existing one is old.
					}
					if = {
						limit = {
							manzoni_x_or_more_local_writers_trigger = { AMOUNT = 3 }
						}
						add = -5 #If the AI has a 3 or more local writers already, it will have a lowered chance to promote another one.
					}
				}
			}
			### Painter ###
			10 = {
				#Effect
				set_variable = artists_promote_painter_active_var
				set_variable = artists_promotion_active
	
				#Trigger
				trigger = {
					NOR = {
						has_variable = artists_promote_painter_active_var
						has_variable = artists_promotion_active
					}
					any_scope_building = {
						is_building_type = building_arts_academy
						has_active_production_method = pm_artists_arts_academy_studio
					}
				}
	
				#Modifier
				modifier = {
					if = {
						limit = {
							klimt_has_painter_no_tooltip = yes
							var:klimt_painter_var ?= {
								age < 70
							}
						}
						add = -14 #If the AI has a Painter already, it will rarely try to get another one except if the existing one is old.
					}
					else_if = {
						limit = {
							OR = { #Cultures that are more probable to promote Painters
								country_has_primary_culture = cu:french
								country_has_primary_culture = cu:spanish
							}
						}
						add = 5
					}
				}
			}
		}
	}
