#Handles Astronomy for the AI and player automation

# This effect will choose projects in this order:
# - Discover Satellite
# - Lunar Map
ai_astronomy_automation_effect = {
	if = {
		limit = {
			verrier_is_astronomy_automated_trigger = yes
			verrier_has_active_astronomer_no_tooltip_trigger = yes
		}

		#Satellite Discovery
		if = {
			limit = { verrier_astronomy_automation_discover_satellites_possible_trigger = yes }

			#Removes variable in case it somehow is still there
			remove_variable = verrier_stop_astronomer_project_var
			#Makes the Astronomer busy
			if = {
				limit = {
					has_variable = verrier_astronomer_var
				}
				var:verrier_astronomer_var = {
					set_character_busy_and_immortal = yes
				}
			}
			#Actual Choice of the Planet
			random_list = {
				#Jupiter
				1 = {
					trigger = { verrier_can_discover_jupiter_satellite_no_je_trigger = yes }

					add_journal_entry = {
						type = je_verrier_jupiter_satellites
					}
				}
				#Saturn
				1 = {
					trigger = { verrier_can_discover_saturn_satellite_no_je_trigger = yes }

					add_journal_entry = {
						type = je_verrier_saturn_satellites
					}
				}
				#Uranus
				1 = {
					trigger = { verrier_can_discover_uranus_satellite_no_je_trigger = yes }

					add_journal_entry = {
						type = je_verrier_uranus_satellites
					}
				}
				#Neptune
				1 = {
					trigger = { verrier_can_discover_neptune_satellite_no_je_trigger = yes }

					add_journal_entry = {
						type = je_verrier_neptune_satellites
					}
				}
				#Mars
				1 = {
					trigger = { verrier_can_discover_mars_satellite_no_je_trigger = yes }

					add_journal_entry = {
						type = je_verrier_mars_satellites
					}
				}
			}
		}
		#Lunar Map
		if = {
			limit = {
				verrier_astronomy_automation_moon_map_possible_trigger = yes
			}
			#Adds the Journal Entry
			add_journal_entry = { type = je_verrier_charter_moon }
			#Makes the Astronomer busy
			if = {
				limit = {
					has_variable = verrier_astronomer_var
				}
				var:verrier_astronomer_var = {
					set_character_busy_and_immortal = yes
				}
			}
		}
	}
}
