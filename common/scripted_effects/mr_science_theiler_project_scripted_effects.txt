### Improve Agriculture Project ###
#Button Effect
mr_biologist_improve_agriculture_button_effect = {
	if = {
		limit = {
			theiler_is_biology_automated_trigger = yes
		}
		clear_variable_list = theiler_improve_agriculture_buildings_chosen_variable_list
		ordered_scope_building = {
			position = 0 #We only want the first one here
			limit = {
				theiler_building_valid_for_agriculture_project_no_tooltip_trigger = yes
			}
			order_by = earnings
			root = {
				add_to_variable_list = {
					name = theiler_improve_agriculture_buildings_chosen_variable_list
					target = prev
				}
			}
		}
		add_journal_entry = {
			type = je_theiler_improve_agriculture_project
		}
		var:theiler_biologist_var ?= {
			set_character_busy_and_immortal = yes
		}
	}
	else = {
		clear_variable_list = theiler_improve_agriculture_buildings_variable_list
		ordered_scope_building = {
			limit = {
				theiler_building_valid_for_agriculture_project_no_tooltip_trigger = yes
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = theiler_improve_agriculture_buildings_variable_list
					target = prev
				}
			}
		}
		set_variable = theiler_improve_agriculture_project_building_chooser_var
		show_as_tooltip = {
			add_modifier = {
				name = theiler_improve_agriculture_cost_modifier
				multiplier = mr_local_project_cost_small_value
			}
		}
	}
}

theiler_improve_agriculture_additional_effects = {
	theiler_add_x_microorganisms_effect = { AMOUNT = 50 }
	if = {
		limit = {
			var:mr_target_building_var ?= {
				is_building_group = bg_ranching
			}
		}

		theiler_add_x_improved_ranching_building_effect = { AMOUNT = 1 }
		var:theiler_biologist_var ?= {
			if = {
				limit = {
					NOT = { has_trait = ito_zoologist }
				}
				random_list = {
					3 = {
						trigger = { var:theiler_biologist_num_improved_ranching_var < 5 }
					}
					1 = {
						add_trait = ito_zoologist
						modifier = {
							add = var:theiler_biologist_num_improved_ranching_var
						}
					}
				}
			}
		}
	}
	else = {
		theiler_add_x_improved_agricultural_building_effect = { AMOUNT = 1 }

		var:theiler_biologist_var ?= {
			if = {
				limit = {
					NOT = { has_trait = ito_botanist }
				}
				random_list = {
					3 = {
						trigger = { var:theiler_biologist_num_improved_agricultural_var < 5 }
					}
					1 = {
						add_trait = ito_botanist
						modifier = {
							add = var:theiler_biologist_num_improved_agricultural_var
						}
					}
				}
			}
		}
	}
}

# AMOUNT = amount to change variable by
theiler_add_x_improved_ranching_building_effect = {
	var:theiler_biologist_var ?= {
		if = {
			limit = {
				NOT = { has_variable = theiler_biologist_num_improved_ranching_var }
			}
			set_variable = {
				name = theiler_biologist_num_improved_ranching_var
				value = 0
			}
		}
		change_variable = {
			name = theiler_biologist_num_improved_ranching_var
			add = $AMOUNT$
		}
	}
}
