mr_create_random_racer_for_character_creation_effect = {
	random_list = {
		1 = {
			create_character = {
				female = yes
				trait_generation = {
					add_random_trait = personality
					random_list = {
						1 = { add_trait = vikelas_speedster_racer }
						1 = { add_trait = vikelas_enduring_racer }
						1 = { add_trait = vikelas_prudent_racer }
					}
					random_list = {
						10 = {}
						15 = { add_trait = gaudi_mechanical_engineer }
						10 = { add_trait = curtiss_pilot }
						10 = { add_trait = brave }
						10 = { add_trait = ambitious }
						10 = { add_trait = persistent }
						10 = { add_trait = cautious }
						5 = { add_trait = reckless }
						5 = { add_trait = tactful }
						5 = { add_trait = meticulous }
						5 = { add_trait = ruthless }
						5 = { add_trait = wrathful }
						5 = { add_trait = reserved }
						5 = { add_trait = pious }
					}
					random_list = {
						80 = {}
						5 = { add_trait = alcoholic }
						5 = { add_trait = opium_addiction }
						5 = { add_trait = cocaine_addiction }
						5 = { add_trait = expensive_tastes }
					}
				}
				save_scope_as = vikelas_new_racer_scope
			}
			scope:vikelas_new_racer_scope = {
				add_trait = vikelas_racer
				set_variable = mr_is_racer_var
				racer_experience_set_up_effect = yes
				place_character_in_void = 1200	
			}
			set_variable = {
				name = vikelas_new_racer_var
				value = scope:vikelas_new_racer_scope
			}

			modifier = {
				if = {
					limit = { has_law = law_type:law_women_own_property }
					add = 4
				}
				if = {
					limit = { has_law = law_type:law_women_in_the_workplace }
					add = 9
				}
				if = {
					limit = { has_law = law_type:law_womens_suffrage }
					add = 19
				}
			}
		}
		99 = {
