### Medical Score Ranking List ###
#Creates a list of the countries ranked by their medical score.
#As this effect is called in an effect, it creates a global variable list.
panum_create_medical_score_ranking_list_effect = {
	#Removes the global variable list
	clear_global_variable_list = mr_medical_score_ranking_global_list

	set_local_variable = {
		name = mr_current_rank_local_var
		value = 0
	}

	ordered_country = {
		limit = {
			is_revolutionary = no #Hopefully keeps revolutions from appearing in the ranking
			NOR = {
				c:PRC ?= this
				is_country_type = decentralized
			}
		}
		order_by = var:panum_medical_score_points_var
		check_range_bounds = no #prevents errors

		#Keeps Track of the rank so it can be saved as a variable
		change_local_variable = {
			name = mr_current_rank_local_var
			add = 1
		}
		if = {
			limit = {
				this = {
					is_player = yes
				}
			}
			set_variable = {
				name = panum_medical_score_player_rank_var
				value = local_var:mr_current_rank_local_var
			}
		}

		#This part adds all the countries to the list
		add_to_global_variable_list = {
			name = mr_medical_score_ranking_global_list
			target = this
		}
	}
}
