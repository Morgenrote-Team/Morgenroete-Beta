###### General Project effect ######

### Chooses the correct Research Project for the Engineer ###
tesla_default_project_set_up_effect = {
	if = {
		limit = {
			tesla_has_civil_engineer_no_tooltip_trigger = yes
			NOT = { has_journal_entry = je_gaudi_civil_research_project }
		}
		add_journal_entry = {
			type = je_gaudi_civil_research_project
		}
	}
	if = {
		limit = {
			tesla_has_mechanical_engineer_no_tooltip_trigger = yes
			NOT = { has_journal_entry = je_gaudi_mechanical_research_project }
		}
		add_journal_entry = {
			type = je_gaudi_mechanical_research_project
		}
	}
	if = {
		limit = {
			tesla_has_military_engineer_no_tooltip_trigger = yes
			NOT = { has_journal_entry = je_gaudi_military_research_project }
		}
		add_journal_entry = {
			type = je_gaudi_military_research_project
		}
	}
	if = {
		limit = {
			tesla_has_electrical_engineer_no_tooltip_trigger = yes
			NOT = { has_journal_entry = je_gaudi_electrical_research_project }
		}
		add_journal_entry = {
			type = je_gaudi_electrical_research_project
		}
	}
	if = {
		limit = {
			tesla_has_aeronautical_engineer_no_tooltip_trigger = yes
			NOT = { has_journal_entry = je_gaudi_aviation_research_project }
		}
		add_journal_entry = {
			type = je_gaudi_aviation_research_project
		}
	}
}

###### Civil Engineering Projects effects ######

#Civil Engineering Research Effect
tesla_civil_research_project_effect = {
	tesla_civil_engineering_small_add_technology = yes
	civil_engineer_experience_gain_effect = { AMOUNT = 5 }
	post_notification = tesla_civil_engineering_technology_boost_from_engineer_notification
}

###### Mechanical Engineering Projects effects ######

#Mechanical Engineering Research Effect

	tesla_mechanical_engineering_small_add_technology = yes
	mechanical_engineer_experience_gain_effect = { AMOUNT = 5 }
	post_notification = tesla_mechanical_engineering_technology_boost_from_engineer_notification
}

###### Military Engineering Projects effects ######

#Military Engineering Research Effect
tesla_military_research_project_effect = {
	#Chooses the Bonus depending on the focus
	if = {
		limit = {
			scope:journal_entry = {
				has_variable = tesla_naval_research_focus_var
			}
		}
		tesla_naval_engineering_small_add_technology = yes
		military_engineer_experience_gain_effect = { AMOUNT = 5 }
		post_notification = tesla_naval_engineering_technology_boost_from_engineer_notification
	}
	else = {
		tesla_military_engineering_small_add_technology = yes
		military_engineer_experience_gain_effect = { AMOUNT = 5 }
		post_notification = tesla_military_engineering_technology_boost_from_engineer_notification
	}
}
