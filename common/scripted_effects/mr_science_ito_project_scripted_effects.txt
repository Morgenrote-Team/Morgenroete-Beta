###### Naturalist Project effects ######

### Improve Forestry Project ###
#Button Effect
mr_naturalist_improve_forestry_button_effect = {
	if = {
		limit = {
			ito_is_naturalist_automated_trigger = yes
		}
		clear_variable_list = ito_improve_forestry_buildings_chosen_variable_list
		ordered_scope_building = {
			position = 0 #We only want the first one here
			limit = {
				ito_building_valid_for_forestry_project_no_tooltip_trigger = yes
			}
			order_by = earnings
			root = {
				add_to_variable_list = {
					name = ito_improve_forestry_buildings_chosen_variable_list
					target = prev
				}
			}
		}
		add_journal_entry = {
			type = je_ito_improve_forestry_project
		}
		var:ito_naturalist_var ?= {
			set_character_busy_and_immortal = yes
		}
	}
	else = {
		clear_variable_list = ito_improve_forestry_buildings_variable_list
		ordered_scope_building = {
			limit = {
				ito_building_valid_for_forestry_project_no_tooltip_trigger = yes
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = ito_improve_forestry_buildings_variable_list
					target = prev
				}
			}
		}
		set_variable = dubois_improve_forestry_project_building_choser_var
		show_as_tooltip = {
			add_modifier = {
				name = dubois_naturalist_forestry_cost_modifier
				multiplier = mr_local_project_cost_very_small_value
			}
		}
	}
}
