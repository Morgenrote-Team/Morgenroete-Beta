# Creates a new Biologist without losing the old one
#This one was a little harder as the scope was only valid in the random_list
#Therefore a Variable is set to save the scope again. then gets deleted
mr_create_historical_biologist_for_character_creation_effect = {
	random_list = {
		100 = { #Carl Julius Salomonsen, Denmark
			trigger = {
				NOT = { has_global_variable = mr_salomonsen_carl_julius_created_global_var }
				has_state_in_state_region = STATE_ZEALAND #Studied in Copenhagen
				country_average_cultural_acceptance = {
					target = cu:danish
					value >= acceptance_status_4
				}
				game_date < 1902.12.06 #Age 55
				year >= 1871 #Graduated
			}
			create_character = {
				template = mr_den_salomonsen_carl_julius_character_template

				save_scope_as = theiler_new_biologist_scope
			}
			scope:theiler_new_biologist_scope = {
				set_variable = mr_is_biologist_var
				set_variable = mr_no_headgear_var
				biologist_experience_set_up_effect = yes
				place_character_in_void = 1200	
			}

			set_variable = {
				name = theiler_new_biologist_var
				value = scope:theiler_new_biologist_scope
			}

			set_global_variable = mr_salomonsen_carl_julius_created_global_var
		}
		200 = { #Louis Pasteur, France
			trigger = {
				NOT = { has_global_variable = mr_pasteur_louis_created_global_var }
				has_state_in_state_region = STATE_FRANCHE_COMTE #Born in Dole
				country_average_cultural_acceptance = {
					target = cu:french
					value >= acceptance_status_4
				}
				game_date < 1877.12.27 #Age 55
				year >= 1848 #Became Professor in Strassbourg
			}
			create_character = {
				template = mr_fra_pasteur_louis_character_template

				save_scope_as = theiler_new_biologist_scope
			}
			scope:theiler_new_biologist_scope = {
				set_variable = mr_is_biologist_var
				set_variable = mr_no_headgear_var
				set_variable = mr_female_glasses_var
				biologist_experience_set_up_effect = yes
				place_character_in_void = 1200
			}

			set_variable = {
				name = theiler_new_biologist_var
				value = scope:theiler_new_biologist_scope
			}

			set_global_variable = mr_pasteur_louis_created_global_var
		}
		100 = { #Gerhard Armauer Hansen, Norway
			trigger = {
				NOT = { has_global_variable = mr_hansen_gerhard_armauer_created_global_var }
				has_state_in_state_region = STATE_WESTERN_NORWAY #Born in Bergen
				country_average_cultural_acceptance = {
					target = cu:norwegian
					value >= acceptance_status_4
				}
				game_date < 1896.07.29 #Age 55
				year >= 1866 #25 years old
			}
			create_character = {
				template = mr_nor_hansen_gerhard_armauer_character_template

				save_scope_as = theiler_new_biologist_scope
			}
			scope:theiler_new_biologist_scope = {
				set_variable = mr_is_biologist_var
				set_variable = mr_no_headgear_var
				biologist_experience_set_up_effect = yes
				place_character_in_void = 1200	
			}

			set_variable = {
				name = theiler_new_biologist_var
				value = scope:theiler_new_biologist_scope
			}

			set_global_variable = mr_hansen_gerhard_armauer_created_global_var
		}
	}
}

mr_create_random_biologist_for_character_creation_effect = {
	random_list = {
		1 = {
			create_character = {
				female = yes
				trait_generation = {}
				save_scope_as = theiler_new_biologist_scope
			}
			scope:theiler_new_biologist_scope = {
				add_trait = theiler_biologist
				set_variable = mr_is_biologist_var
				biologist_experience_set_up_effect = yes
				mr_trait_generation_biologist_effect = yes
				place_character_in_void = 1200	
			}
			set_variable = {
				name = theiler_new_biologist_var
				value = scope:theiler_new_biologist_scope
			}

			modifier = {
				if = {
					limit = { has_law = law_type:law_women_own_property }
					add = 4
				}
				if = {
					limit = { has_law = law_type:law_women_in_the_workplace }
					add = 9
				}
				if = {
					limit = { has_law = law_type:law_womens_suffrage }
					add = 19
				}
			}
		}
		99 = {
			create_character = {
				trait_generation = {}
				save_scope_as = theiler_new_biologist_scope
			}
			scope:theiler_new_biologist_scope = {
				add_trait = theiler_biologist
				set_variable = mr_is_biologist_var
				biologist_experience_set_up_effect = yes
				mr_trait_generation_biologist_effect = yes
				place_character_in_void = 1200
			}
			set_variable = {
				name = theiler_new_biologist_var
				value = scope:theiler_new_biologist_scope
			}

			modifier = {
				if = {
					limit = { has_law = law_type:law_women_own_property }
					add = -4
				}
				if = {
					limit = { has_law = law_type:law_women_in_the_workplace }
					add = -9
				}
				if = {
					limit = { has_law = law_type:law_womens_suffrage }
					add = -19
				}
			}
		}
	}
}
