#Handles the Pacific Expedition
ai_pacific_expedition_effect = {
	if = {
		limit = {
			country_rank >= rank_value:major_power
			net_fixed_income > 1000
			gold_reserves > 0
			has_technology_researched = screw_frigate
			has_technology_researched = dialectics
			NOR = {
				has_modifier = rapanui_pacific_expedition_modifier
				has_modifier = rapanui_had_pacific_expedition_modifier
			}
			navy_size >= 15
			any_scope_building = {
				is_building_type = building_port
			}
			any_scope_building = {
				is_building_type = building_university
			}
			any_scope_character = {
				has_role_of_type = admiral
				is_busy = no
			}
			NOT = {
				any_country = {
					OR = {
						has_technology_researched = dreadnought_tech
						has_technology_researched = destroyer
					}
				}
			}
		}

		#Effect
		add_treasury = -25000
		add_modifier = {
			name = rapanui_pacific_expedition_modifier
		}
		trigger_event = {
			id = rapanui.201
			popup = yes
		}
	}
}

#Handles the Mountain Expeditions
ai_mountain_expedition_1_effect = {
	if = {
		limit = {
			douglas_has_active_mountaineer = yes
			gold_reserves > 0
			OR = {
				has_modifier = douglas_mountaineering_equipment_level_1_modifier
				has_modifier = douglas_mountaineering_equipment_level_2_modifier
				has_modifier = douglas_mountaineering_equipment_level_3_modifier
				has_modifier = douglas_mountaineering_equipment_level_4_modifier
				has_modifier = douglas_mountaineering_equipment_level_5_modifier
			}
			OR = {
				douglas_has_unconquered_level_1_summit_trigger = yes
				any_country = {
					douglas_has_unconquered_level_1_summit_trigger = yes
					has_diplomatic_pact = {
						who = ROOT
						type = mountaineering_license
					}
				}
			}
			NOR = {
				has_modifier = douglas_mountaineering_success_modifier
				has_modifier = douglas_mountaineering_disaster_modifier
				has_modifier = douglas_mountaineering_disappointment_modifier
			}
			NAND = {
				has_variable = douglas_alpine_club_board_member_var
				has_journal_entry = je_douglas_international_mountaineering_expedition_preparation
			}
		}

		#Effect
		add_treasury = -10000
		if = {
			limit = {
				has_variable = douglas_mountaineer_var
			}
			var:douglas_mountaineer_var = {
				set_character_busy_and_immortal = yes
			}
		}
		trigger_event = {
			id = douglas.201
			popup = yes
		}
	}
}

#Handles the Arctica Expedition
ai_arctica_expedition_effect = {
	if = {
		limit = {
			has_technology_researched = zeppelins
			NOR = {
				has_strategy = ai_strategy_armed_isolationism
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			net_fixed_income > mr_project_cost_small_value
			gold_reserves > 0
			NOR = {
				has_modifier = curtiss_arctica_expedition_modifier
				has_modifier = curtiss_had_arctica_expedition_modifier
			}
			any_scope_building = {
				is_building_type = building_verrier_observatory
				OR = {
					has_active_production_method = pm_curtiss_weather_station
					has_active_production_method = pm_curtiss_modern_weather_station
				}
			}
			curtiss_has_active_pilot_no_tooltip_trigger = yes
		}

		#Effect
		set_variable = {
			name = curtiss_zeppelin_progress
			value = 0
		}
		set_variable = {
			name = curtiss_zeppelin_science
			value = 0
		}
		set_variable = {
			name = curtiss_zeppelin_danger
			value = 0
		}
		add_modifier = {
			name = curtiss_arctica_expedition_modifier
		}
		add_modifier = {
			name = curtiss_zeppelin_expedition_cost_modifier
			multiplier = mr_project_cost_very_small_value
		}
		trigger_event = {
			id = curtiss.520
			popup = yes
		}
	}
}

#Handles Zoo Expansion for the AI
ai_zoo_expansion_effect = {
	if = {
		limit = {
			any_scope_state = {
				is_capital = yes
				any_scope_building = {
					is_building_type = building_dubois_zoo
					level > 0
					NOT = {
						has_active_production_method = pm_zoo_level_5
					}
				}
			}
		}
		#Zoo Level 2
		if = {
			limit = {
				NOT = {
					has_variable = dubois_zoo_first_extension
				}
				any_scope_state = {
					is_capital = yes
					any_scope_building = {
						is_building_type = building_dubois_zoo
						level > 0
						NOR = {
							has_active_production_method = pm_zoo_level_2
							has_active_production_method = pm_zoo_level_3
							has_active_production_method = pm_zoo_level_4
							has_active_production_method = pm_zoo_level_5
						}
					}
				}
				var:dubois_zoo_animal_total >= 10
				OR = {
					var:dubois_zoo_animal_africa >= 2
					var:dubois_zoo_animal_asia >= 2
					var:dubois_zoo_animal_australia >= 2
				}
				NOT = {
					has_modifier = dubois_zoo_extension_modifier
				}
			}

			#Effect
			set_variable = dubois_zoo_first_extension
			random_scope_state = {
				limit = {
					has_building = building_dubois_zoo
					is_capital = yes
				}
				activate_production_method = {
					building_type = building_dubois_zoo
					production_method = pm_zoo_level_2
				}
			}
			add_modifier = {
				name = dubois_zoo_extension_modifier
				days = 300
			}
		}
		#Zoo Level 3
		if = {
			limit = {
				NOT = {
					has_variable = dubois_zoo_second_extension
				}
				any_scope_state = {
					is_capital = yes
					any_scope_building = {
						is_building_type = building_dubois_zoo
						level > 0
						NOR = {
							has_active_production_method = pm_zoo_level_3
							has_active_production_method = pm_zoo_level_4
							has_active_production_method = pm_zoo_level_5
						}

				}

					var:dubois_zoo_animal_africa >= 5
					var:dubois_zoo_animal_asia >= 5
					var:dubois_zoo_animal_australia >= 5

				}
			}
			add_modifier = {
				name = dubois_zoo_extension_modifier
				days = 400
			}
		}

#Handles the Patron of the Arts for the AI
ai_patron_of_arts_effect = {
	#Adds Patron of the Arts
	if = {
		limit = {
			NOT = {
				has_modifier = artists_patron_of_the_arts_modifier
			}
			any_scope_building = {
				OR = {
					is_building_type = building_art_academy
					is_building_type = building_elgar_opera
				}
			}
			net_fixed_income > mr_project_cost_very_small_value
			gold_reserves > 0
		}

		#Effect
		set_variable = {
            name = patron_of_the_arts_cooldown_var
            months = 12
        }
        if = {
			limit = {
				has_modifier = artists_patron_of_the_arts_modifier
			}
			remove_modifier = artists_patron_of_the_arts_modifier
		}
		add_modifier = {
			name = artists_patron_of_the_arts_modifier
			multiplier = mr_project_cost_very_small_value
		}
	}
	#Removes Patron of the Arts
	if = {
		limit = {
			has_modifier = artists_patron_of_the_arts_modifier
			NOT = {
				has_variable = patron_of_the_arts_cooldown_var
			}
			net_fixed_income < mr_project_cost_very_small_value
			gold_reserves < 0
		}

		#Effect
		remove_modifier = artists_patron_of_the_arts_modifier
	}
}

			mr_gui_has_geology_tech_trigger = yes
			any_scope_character = {
				has_role_of_type = general
				is_busy = no
				OR = {
					has_trait = douglas_mountaineer
					has_trait = douglas_mountaineer_veteran
					has_trait = explorer
					has_trait = mountain_commander
				}
			}
		}

		#Effect
		set_variable = douglas_mountaineering_club_var
		hidden_effect = {
			if = {
				limit = { has_variable = douglas_mountaineering_club_var }
				create_new_character_base_effect = { CHARACTER = mountaineer PROJECT = douglas }
				douglas_set_new_mountaineer_as_mountaineer_effect = yes
			}
		}
		trigger_event = {
			id = douglas.101
		}
	}
	#Alpine Club
	if = {
		limit = {
			NOT = {
				has_global_variable = douglas_alpine_club_founded_global_var
			}
			has_variable = douglas_mountaineering_club_var
			mr_gui_has_organized_sports_tech_trigger = yes
			douglas_has_mountaineer = yes
			var:douglas_mountaineer_var = {
				var:douglas_mountaineer_experience_var >= 50
			}
			var:douglas_num_mountain_ascents_var >= 2
			var:douglas_summits_conquered_var >= 2
			global_var:douglas_num_alpine_club_members_global_var >= 6
		}

		#Effect
		set_global_variable = douglas_alpine_club_founded_global_var
		set_variable = douglas_alpine_club_founding_country_var
		capital = {
			create_building = {
				building = building_douglas_alpine_club
				level = 1
			}
			state_region = {
				set_global_variable = { #This is used in the GUI to display the state
            		name = douglas_alpine_club_seat_global_var
           			value = this
        		}
			}
		}
		country_definition = {
			set_global_variable = {
				name = douglas_alpine_club_president_global_var
				value = ROOT
			}
		}
		every_country = {
			limit = {
				has_variable = douglas_mountaineering_club_var
			}
			trigger_event = {
				id = douglas.102
			}
		}
	}
	#Usurp Alpine Club
	if = {
		limit = {
			has_global_variable = douglas_alpine_club_founded_global_var
			has_variable = douglas_mountaineering_club_var
			NOT = {
				any_country = {
					has_variable = douglas_alpine_club_founding_country_var
				}
			}
			any_scope_state = {
				has_building = building_douglas_alpine_club
			}
			is_at_war = no
		}

		#Effect
		set_variable = douglas_alpine_club_founding_country_var
		country_definition = {
			set_global_variable = {
				name = douglas_alpine_club_president_global_var
				value = ROOT
			}
		}
	}
	#Alpine Club Expedition
	if = {
		limit = {
			has_variable = douglas_alpine_club_founding_country_var
			douglas_has_active_mountaineer = yes
			var:douglas_alpine_heritage_var >= 25
			any_country = {
				has_variable = douglas_alpine_club_founding_country_var
				NOT = {
					has_modifier = douglas_mountaineering_international_expedition_modifier
				}
			}
			any_country = {
				has_variable = douglas_alpine_club_founding_country_var
				OR = {
					has_modifier = douglas_mountaineering_equipment_level_3_modifier
					has_modifier = douglas_mountaineering_equipment_level_4_modifier
					has_modifier = douglas_mountaineering_equipment_level_5_modifier
				}
			}
			any_country = {
				has_variable = douglas_mountaineering_club_var
				OR = {
					douglas_has_unconquered_level_2_summit_trigger = yes
					douglas_has_unconquered_level_3_summit_trigger = yes
					douglas_has_unconquered_level_4_summit_trigger = yes
				}
			}
		}

		#Effect
		douglas_add_x_alpine_heritage_effect = {
			AMOUNT = -25
		}
		if = {
			limit = {
				has_variable = douglas_mountaineer_var
			}
			var:douglas_mountaineer_var = {
				set_character_busy_and_immortal = yes
			}
		}
		add_modifier = {
			name = douglas_mountaineering_international_expedition_modifier
			years = 10
		}
		trigger_event = {
			id = douglas.321
			popup = yes
		}
	}
	#Alpine Club Expedition Proposal
	if = {
		limit = {
			has_variable = douglas_alpine_club_board_member_var
			NOR = {
				has_global_variable = douglas_ongoing_international_mountain_expedition
				has_variable = douglas_alpine_club_waiting_for_response_var
			}
			any_country = {
				has_variable = douglas_alpine_club_founding_country_var
				douglas_has_active_mountaineer = yes
			}
			var:douglas_alpine_heritage_var >= 25
			any_country = {
				has_variable = douglas_alpine_club_founding_country_var
				NOT = {
					has_modifier = douglas_mountaineering_international_expedition_modifier
				}
			}
			any_country = {
				has_variable = douglas_alpine_club_founding_country_var
				OR = {
					has_modifier = douglas_mountaineering_equipment_level_3_modifier
					has_modifier = douglas_mountaineering_equipment_level_4_modifier
					has_modifier = douglas_mountaineering_equipment_level_5_modifier
				}
			}
			any_country = {
				has_variable = douglas_mountaineering_club_var
				OR = {
					douglas_has_unconquered_level_2_summit_trigger = yes
					douglas_has_unconquered_level_3_summit_trigger = yes
					douglas_has_unconquered_level_4_summit_trigger = yes
				}
			}
			NOT = {
				has_modifier = douglas_mountaineering_international_expedition_setback_modifier
			}
		}

		#Effect
		set_variable = {
			name = douglas_alpine_club_waiting_for_response_var
			days = 200
		}
		douglas_add_x_alpine_heritage_effect = {
			AMOUNT = -25
		}
		random_country = {
			limit = {
				has_variable = douglas_alpine_club_founding_country_var
			}
			trigger_event = {
				id = douglas.311
			}
		}
	}
	#Alpine Club Expedition Join
	if = {
		limit = {
			has_variable = douglas_alpine_club_board_member_var
			douglas_has_active_mountaineer = yes
		}

		#Effect
		set_variable = douglas_alpine_club_expedition_member_var
		douglas_add_mountain_ascent_effect = yes
		if = {
			limit = {
				has_variable = douglas_mountaineer_var
			}
			var:douglas_mountaineer_var = {
				set_character_busy_and_immortal = yes
			}
		}
	}
}
