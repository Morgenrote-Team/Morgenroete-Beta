### Volcano Expedition ###
mr_geologist_volcano_expedition_effect = {
	#Effect for AI and Player
	set_variable = agassiz_volcano_expedition_active_var
	if = {
		limit = {
			has_variable = agassiz_geologist_var
		}
		var:agassiz_geologist_var ?= {
			set_character_busy_and_immortal = yes
		}
	}

	#Effect depending on AI/Automation or Player
	if = {
		limit = {
			agassiz_is_geology_automated_trigger = yes
		}
		random_scope_state = {
			limit = {
				agassiz_state_has_unexplored_volcano = yes
			}
			set_variable = agassiz_targeted_volcano_var
		}
		if = {
			limit = {
				is_player = yes
			}
			add_journal_entry = {
				type = je_agassiz_volcano_expedition
			}
		}
		else = {
			add_journal_entry = {
				type = je_agassiz_volcano_expedition_ai
			}
		}
	}
	else = {
		trigger_event = {
			id = agassiz.601
			popup = yes
		}
	}
}

### Improve Mines Project ###
#Button Effect
mr_geologist_improve_mines_button_effect = {
	if = {
		limit = {
			agassiz_is_geology_automated_trigger = yes
		}
		clear_variable_list = agassiz_improve_mine_buildings_chosen_variable_list
		ordered_scope_building = {
			position = 0 #We only want the first one here
			limit = {
				agassiz_mine_valid_for_improvement_trigger = yes
			}
			order_by = earnings
			root = {
				add_to_variable_list = {
					name = agassiz_improve_mine_buildings_chosen_variable_list
					target = prev
				}
			}
		}
		add_journal_entry = {
			type = je_agassiz_improve_mines_project
		}
		var:agassiz_geologist_var ?= {
			set_character_busy_and_immortal = yes
		}
	}
	else = {
		clear_variable_list = agassiz_improve_mine_buildings_variable_list
		ordered_scope_building = {
			limit = {
				agassiz_mine_valid_for_improvement_trigger = yes
			}
			order_by = earnings
			check_range_bounds = no

			root = {
				add_to_variable_list = {
					name = agassiz_improve_mine_buildings_variable_list
					target = prev
				}
			}
		}
		set_variable = agassiz_improve_mine_project_building_choser_var
		show_as_tooltip = {
			add_modifier = {
				name = agassiz_ore_search_cost_modifier
				multiplier = mr_local_project_cost_small_value
			}
		}
	}
}

mr_geologist_tectonics_congress_button_effect = {
	if = {
		limit = {
			agassiz_is_geology_automated_trigger = yes
		}
		set_global_variable = agassiz_tectonic_congress_hosted_var
		var:agassiz_geologist_var ?= {
			set_variable = {
				name = agassiz_geologist_achievement_invented_plate_tectonics_var
				value = flag:agassiz_geologist_achievement_invented_plate_tectonics_loc
			}
		}
		add_modifier = {
			name = agassiz_congress_host_modifier
			years = 5
		}
		set_global_variable = {
			name = agassiz_plate_tectonics_yes
			value = 0
		}
		set_global_variable = {
			name = agassiz_plate_tectonics_no
			value = 0
		}
		var:agassiz_geologist_var ?= {
			set_variable = {
				name = agassiz_geologist_achievement_attended_geologist_congress_var
				value = flag:agassiz_geologist_achievement_attended_geologist_congress_loc
			}
			save_scope_as = agassiz_geologist_scope
		}
		every_country = {
			limit = {
				NOT = {
					this = ROOT
				}
				has_variable = agassiz_geologist_var
			}
			trigger_event = {
				id = agassiz.502
				days = { 7 14 }
			}
		}
		hidden_effect = {
			trigger_event = {
				id = agassiz.504
				days = { 14 28 }
			}
			trigger_event = {
				id = agassiz.509
				days = 180
			}
		}
	}
	else = {
		set_global_variable = agassiz_tectonic_congress_hosted_var
		var:agassiz_geologist_var ?= {
			set_variable = {
				name = agassiz_geologist_achievement_invented_plate_tectonics_var
				value = flag:agassiz_geologist_achievement_invented_plate_tectonics_loc
			}
		}
		trigger_event = {
			id = agassiz.501
			popup = yes
		}
		add_modifier = {
			name = agassiz_congress_host_modifier
			years = 5
		}
	}
}
