#Handles Biology for the AI and player automation

# This effect will choose projects depending on various circumstances
ai_biology_automation_effect = {
	hidden_effect = {
		if = {
			limit = {
				theiler_is_biology_automated_trigger = yes
				theiler_has_active_biologist_no_tooltip_trigger = yes
			}

			#Uses an if else construct to prioritize Projects
			# This allows for the most important Projects to get prioritized
			if = {
				limit = {
					NOT = { has_variable = theiler_biologist_action_active_var }
					OR = {
						net_fixed_income >= mr_small_local_project_cost_weekly_value
						gold_reserves >= mr_small_local_project_cost_yearly_value
					}
					has_technology_researched = theiler_microbiology_tech
					theiler_can_create_vaccine_trigger = yes
					theiler_has_x_microorganisms_no_tooltip_trigger = { AMOUNT = 10000 }
				}

				mr_biologist_create_vaccine_effect = yes
			}
			else_if = {
				limit = {
					OR = {
						net_fixed_income >= mr_very_small_local_project_cost_weekly_value
						gold_reserves >= mr_very_small_local_project_cost_yearly_value
					}
					NOT = { has_variable = theiler_biologist_action_active_var }
					has_technology_researched = theiler_microbiology_tech
					has_technology_researched = aniline
					panum_has_miasma_theory_trigger = no
					OR = {
						theiler_has_achromatic_microscope_trigger = yes
						theiler_has_electron_microscope_trigger = yes
					}
					theiler_can_find_pathogen_trigger = yes
				}

				mr_biologist_find_pathogen_effect = yes
			}
			else_if = {
				limit = {
					NOT = { has_variable = theiler_screwworm_fly_prevention_var }
					any_scope_state = {
						is_in_geographic_region = theiler_geographic_region_screwworm_fly
					}
					has_technology_researched = theiler_cell_culture_tech
					has_technology_researched = verrier_nuclear_physics_tech
					has_technology_researched = curtiss_modern_aviation_tech
				}

				mr_biologist_cochliomyia_eredication_button_effect = yes
			}
			else_if = {
				limit = {
					NOR = {
						has_variable = theiler_codling_moth_prevention_var
						has_variable = theiler_potato_beetle_prevention_var
					}
					has_technology_researched = theiler_early_pesticides_tech
					any_scope_state = {
						OR = {
							is_in_geographic_region = theiler_geographic_region_potato_beetle
							is_in_geographic_region = theiler_geographic_region_codling_moth
						}
					}
					mendelejew_has_chemist_no_tooltip_trigger = yes
				}

				MR_BIOLOGIST_LEAD_ARSENATE_BUTTON_effect = yes
			}
			else_if = {
				limit = {
					theiler_has_locust_swarm_prevention_trigger = no
					any_scope_state = {
						is_vulnerable_to_locust_swarms = yes
					}
					mr_gui_has_modern_pesticides_tech_trigger = yes
					mendelejew_has_chemist_no_tooltip_trigger = yes
				}

				MR_BIOLOGIST_SODIUM_ARSENITE_BUTTON_effect = yes
			}
			else_if = {
				limit = {
					theiler_has_boll_weevil_prevention_trigger = no
					any_scope_state = {
						is_in_geographic_region = theiler_geographic_region_boll_weevil
					}
					mr_gui_has_modern_pesticides_tech_trigger = yes
					mr_gui_has_military_aviation_tech_trigger = yes
					mendelejew_has_chemist_no_tooltip_trigger = yes
				}

				MR_BIOLOGIST_CALCIUM_ARSENATE_BUTTON_effect = yes
			}
			else_if = {
				limit = {
					theiler_has_tea_mosquito_prevention_trigger = no
					any_scope_state = {
						is_in_geographic_region = theiler_geographic_region_tea_mosquito
					}
					mr_gui_has_modern_pesticides_tech_trigger = yes
					mr_gui_has_malaria_prevention_tech_trigger = yes
					mendelejew_has_chemist_no_tooltip_trigger = yes
				}

				MR_BIOLOGIST_NICOTINE_SULFATE_BUTTON_effect = yes
			}
			else_if = {
				limit = {
					theiler_has_grape_phylloxera_prevention_trigger = no
					any_scope_state = {
						is_in_geographic_region = theiler_geographic_region_grape_phylloxera
					}
					mr_gui_has_genetics_tech_trigger = yes
				}

				MR_BIOLOGIST_NICOTINE_SULFATE_BUTTON_effect = yes
			}
			else_if = {
				limit = {
					NOR = {
						has_variable = theiler_improve_agriculture_project_building_chooser_var
						has_journal_entry = je_theiler_improve_agriculture_project
					}
					OR = {
						net_fixed_income >= mr_small_local_project_cost_weekly_value
						gold_reserves >= mr_small_local_project_cost_yearly_value
					}
					theiler_country_has_building_valid_for_agriculture_project_trigger = yes
				}

				mr_biologist_improve_agriculture_button_effect = yes
			}
		}
	}
}
