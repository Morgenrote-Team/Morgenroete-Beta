#Handles Biology for the AI and player automation

# This effect will choose projects depending on various circumstances
ai_biology_automation_effect = {
	if = {
		limit = {
			theiler_is_biology_automated_trigger = yes
			theiler_has_active_biologist_no_tooltip_trigger = yes
		}

		#Uses an if else construct to prioritize Projects
		# This allows for the most important Projects to get prioritized
		if = {
			limit = {
				OR = {
					net_fixed_income >= mr_small_local_project_cost_weekly_value
					gold_reserves >= mr_small_local_project_cost_yearly_value
				}
				NOT = { has_variable = theiler_biologist_action_active_var }
				has_technology_researched = theiler_microbiology_tech
				theiler_can_create_vaccine_trigger = yes
				theiler_has_x_microorganisms_no_tooltip_trigger = { AMOUNT = 10000 }
			}

			mr_biologist_create_vaccine_effect = yes
		}
		else_if = {
			limit = {
				OR = {
					net_fixed_income >= mr_very_small_local_project_cost_weekly_value
					gold_reserves >= mr_very_small_local_project_cost_yearly_value
				}
				NOT = { has_variable = theiler_biologist_action_active_var }
				has_technology_researched = theiler_microbiology_tech
				has_technology_researched = aniline
				panum_has_miasma_theory_trigger = no
				OR = {
					theiler_has_achromatic_microscope_trigger = yes
					theiler_has_electron_microscope_trigger = yes
				}
				theiler_can_find_pathogen_trigger = yes
			}

			mr_biologist_find_pathogen_effect = yes
		}
		else_if = {
			limit = {
				NOT = { has_variable = theiler_cochliomyia_eredication_program_var }
				any_scope_state = {
					has_state_trait = theiler_state_trait_cochliomyia
				}
				has_technology_researched = theiler_cell_culture_tech
				has_technology_researched = verrier_nuclear_physics_tech
				has_technology_researched = curtiss_modern_aviation_tech
			}

			mr_biologist_cochliomyia_eredication_button_effect = yes
		}
		else_if = {
			limit = {
				OR = {
					net_fixed_income >= mr_very_small_local_project_cost_weekly_value
					gold_reserves >= mr_very_small_local_project_cost_yearly_value
				}
				has_variable = theiler_cochliomyia_eredication_program_var
				any_scope_state = {
					has_state_trait = theiler_state_trait_cochliomyia
				}
			}

			mr_biologist_sti_button_button_effect = yes
		}
		else_if = {
			limit = {
				NOT = { has_variable = theiler_cydia_lead_arsenate_var }
				any_scope_state = {
					has_state_trait = theiler_state_trait_cydia
				}
				has_technology_researched = malaria_prevention
				mendelejew_has_chemist_no_tooltip_trigger = yes
			}

			mr_biologist_cydia_pesticide_button_effect = yes
		}
		else_if = {
			limit = {
				OR = {
					net_fixed_income >= mr_small_local_project_cost_weekly_value
					gold_reserves >= mr_small_local_project_cost_yearly_value
				}
				NOT = { has_variable = theiler_improve_agriculture_project_building_chooser_var }
				theiler_country_has_building_valid_for_agriculture_project_trigger = yes
			}

			mr_biologist_improve_agriculture_button_effect = yes
		}
	}
}
