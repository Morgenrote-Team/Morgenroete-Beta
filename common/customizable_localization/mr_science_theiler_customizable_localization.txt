###Microorganism Research Active###
#Displays whether the Biologist is currently researching Microorganisms or not
je_theiler_microorganism_research_active_type = {
	type = country

	text = {
		trigger = {
			theiler_has_active_biologist_no_tooltip_trigger = yes
		}

		localization_key = je_theiler_microorganism_research_active_loc
	}

	text = {
		localization_key = je_theiler_microorganism_research_inactive_loc
	}
}
