### MOUNTAIN WINDOW ###
#These custom locs are used to display whether a mountain was climbed, by whom and when.

# Mount Whitney #
douglas_mount_whitney_year_type = {
	type = country
	
	text = {
		trigger = {
			has_global_variable = douglas_mount_whitney_year_global_var
		}
		localization_key = douglas_mount_whitney_year_loc
	}

	text = {
		localization_key = douglas_mount_whitney_not_yet_loc
	}
}

douglas_mount_whitney_conqueror_type = {
	type = country
	
	text = {
		trigger = {
			has_global_variable = douglas_mount_whitney_year_global_var
			any_country = {
				has_variable = douglas_mount_whitney_conqueror_var
			}
		}
		localization_key = douglas_mount_whitney_conqueror_loc
	}

	text = {
		trigger = {
			has_global_variable = douglas_mount_whitney_year_global_var
		}
		localization_key = douglas_mount_whitney_conqueror_definition_loc
	}
}
