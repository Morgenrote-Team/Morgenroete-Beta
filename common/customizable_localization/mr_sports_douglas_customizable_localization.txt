		trigger = {
			douglas_has_mountain_range_trigger = yes
		}

	}

	text = {
		localization_key = douglas_yearly_alpine_heritage_mountain_range_type_inactive_loc

### MOUNTAIN WINDOW ###
#These custom locs are used to display whether a mountain was climbed, by whom and when.

# Mount McKinley #
douglas_mount_mckinley_year_type = {
	type = country
	
	text = {
		trigger = {
			has_global_variable = douglas_mount_mckinley_year_global_var
		}
		localization_key = douglas_mount_mckinley_year_loc
	}

	text = {
		localization_key = douglas_mount_mckinley_not_yet_loc
	}
}

douglas_mount_mckinley_conqueror_type = {
	type = country
	
	text = {
		trigger = {
			has_global_variable = douglas_mount_mckinley_year_global_var
			any_country = {
				has_variable = douglas_mount_mckinley_conqueror_var
			}
		}
		localization_key = douglas_mount_mckinley_conqueror_loc
	}

	text = {
		trigger = {
			has_global_variable = douglas_mount_mckinley_year_global_var
		}
		localization_key = douglas_mount_mckinley_conqueror_definition_loc
	}
}
