#Replacement Tooltip for the Mountaineer --> Used to show whether you will replace an existing Mountaineer
douglas_replacement_mountaineer_custom_loc = {
	type = country
	random_valid = no #From top to bottom decides which name list is appropriate.

	text = {
		trigger = {
			has_variable = douglas_mountaineer_var
		}
		localization_key = douglas_would_replace_mountaineer_loc
	}

	text = {
		trigger = {
			NOT = {
				has_variable = douglas_mountaineer_var
			}
		}
		localization_key = douglas_will_not_replace_mountaineer_loc
	}
}
