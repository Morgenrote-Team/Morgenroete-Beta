### Achievements of the Architect ###

GUI_gaudi_architect_achievements_type = {
	type = country
	
	text = {
		trigger = {
			has_variable = gaudi_architect_var
		}
		localization_key = GUI_gaudi_architect_achievements_loc
	}

	text = {
		localization_key = GUI_gaudi_architect_achievements_no_architect_loc
	}
}

GUI_gaudi_architect_achievements_experience_tt_type = {
	type = country
	
	text = {
		trigger = {
			var:gaudi_architect_var = {
				has_trait = gaudi_great_architect
			}
		}
		localization_key = GUI_gaudi_architect_achievements_great_architect_tt_loc
	}

	text = {
		localization_key = GUI_gaudi_architect_achievements_architect_tt_loc
	}
}

### Experience of the Architect ###

GUI_gaudi_architect_experience_type = {
	type = country
	random_valid = no

	text = {
		trigger = {
			NOT = { has_variable = gaudi_architect_var }
		}
		localization_key = GUI_gaudi_architect_experience_no_architect_loc
	}

	text = {
		trigger = {
			var:gaudi_architect_var ?= {
				var:gaudi_architect_experience >= 500
			}
		}
		localization_key = GUI_gaudi_architect_famous_loc
	}

	text = {
		localization_key = GUI_gaudi_architect_experience_normal_loc
	}
}

GUI_gaudi_architect_experience_text_type = {
	type = country
	random_valid = no

	text = {
		trigger = {
			NOT = { has_variable = gaudi_architect_var }
		}
		localization_key = GUI_gaudi_architect_experience_text_no_architect_loc
	}

	text = {
		trigger = {
			var:gaudi_architect_var ?= {
				var:gaudi_architect_experience >= 500
			}
		}
		localization_key = GUI_gaudi_architect_experience_text_500_loc
	}

	text = {
		trigger = {
			var:gaudi_architect_var ?= {
				var:gaudi_architect_experience < 500
			}
		}
		localization_key = GUI_gaudi_architect_experience_text_progressing_loc
	}
}
