### Experience of the Architect ###

GUI_gaudi_architect_experience_type = {
	type = country
	random_valid = no

	text = {
		trigger = {
			NOT = { has_variable = gaudi_architect_var }
		}
		localization_key = GUI_gaudi_architect_experience_no_architect_loc
	}

	text = {
		trigger = {
			var:gaudi_architect_var ?= {

	}
