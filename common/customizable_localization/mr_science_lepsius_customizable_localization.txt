#Used to show the experience level of the Archaeologist in the overview
lepsius_archaeologist_experience_level_type = {
	type = character

	text = {
		trigger = {
			OR = {
				has_trait = lepsius_archaeologist_famous
				var:lepsius_archaeologist_experience_var >= 300
			}
		}
		localization_key = lepsius_archaeologist_achievement_famous_level_archaeologist_overview_loc
	}

	text = {
		trigger = {
			OR = {
				has_trait = lepsius_archaeologist_experienced
				AND = {
					var:lepsius_archaeologist_experience_var >= 100
					var:lepsius_archaeologist_experience_var < 300
				}
			}
		}
		localization_key = lepsius_archaeologist_achievement_experienced_level_archaeologist_overview_loc
	}

	text = {
		localization_key = lepsius_archaeologist_achievement_normal_level_archaeologist_overview_loc
	}
}
