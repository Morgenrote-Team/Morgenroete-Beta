#Current Microscope Tooltip
theiler_gui_current_microscope_tooltip_type = {
	type = country

	text = {
		trigger = {
			var:theiler_microscope_var ?= flag:theiler_early_compound_microscope_flag
		}
		localization_key = theiler_gui_current_microscope_compound_tooltip_loc
	}

	text = {
		trigger = {
			var:theiler_microscope_var ?= flag:theiler_achromatic_microscope_flag
		}
		localization_key = theiler_gui_current_microscope_achromatic_tooltip_loc
	}

	text = {
		trigger = {
			var:theiler_microscope_var ?= flag:theiler_electron_microscope_flag
		}
		localization_key = theiler_gui_current_microscope_electron_tooltip_loc
	}
}
