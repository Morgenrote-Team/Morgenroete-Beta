#This is used to have the same scope as in the GUI
andersson_anthropologist_achievements_switch_to_character_scope_type = {
	type = character

	text = {
		localization_key = andersson_anthropologist_achievements_loc
	}
}

#Used to show the experience level of the Anthropologist in the overview
andersson_anthropologist_experience_level_type = {
	type = character

	text = {
		trigger = {
			OR = {
				has_trait = andersson_anthropologist_famous
				var:andersson_anthropologist_experience_var >= 300
			}
		}
		localization_key = andersson_anthropologist_achievement_famous_level_anthropologist_overview_loc
	}

	text = {
		trigger = {
			OR = {
				has_trait = andersson_anthropologist_experienced
				AND = {
					var:andersson_anthropologist_experience_var >= 100
					var:andersson_anthropologist_experience_var < 300
				}
			}
		}
		localization_key = andersson_anthropologist_achievement_experienced_level_anthropologist_overview_loc
	}

	text = {
		localization_key = andersson_anthropologist_achievement_normal_level_anthropologist_overview_loc
	}
}
