#Used to show the experience level of the Anthropologist in the overview
andersson_anthropologist_experience_level_type = {
	type = character

	text = {
		trigger = {
			OR = {
				has_trait = andersson_anthropologist_famous
				var:andersson_anthropologist_experience_var >= 300
			}
		}
		localization_key = andersson_anthropologist_achievement_famous_level_anthropologist_overview_loc
	}

	text = {
		trigger = {
			OR = {
				has_trait = andersson_anthropologist_experienced
				AND = {
					var:andersson_anthropologist_experience_var >= 100
					var:andersson_anthropologist_experience_var < 300
				}
			}
		}
		localization_key = andersson_anthropologist_achievement_experienced_level_anthropologist_overview_loc
	}

	text = {
		localization_key = andersson_anthropologist_achievement_normal_level_anthropologist_overview_loc
	}
}

#Found total number of Neoanthropus Fossils
andersson_anthropologist_achievement_total_neoanthropus_fossils_anthropologist_type = {
	type = character

	text = {
		trigger = {
			has_variable = andersson_neoanthropus_fossils_anthropologist_var
			var:andersson_neoanthropus_fossils_anthropologist_var > 0
			OR = {
				AND = {
					has_variable = andersson_palaeanthropus_fossils_anthropologist_var
					var:andersson_palaeanthropus_fossils_anthropologist_var > 0
				}
				AND = {
					has_variable = andersson_pithecanthropus_fossils_anthropologist_var
					var:andersson_pithecanthropus_fossils_anthropologist_var > 0
				}
				AND = {
					has_variable = andersson_australopithecus_fossils_anthropologist_var
					var:andersson_australopithecus_fossils_anthropologist_var > 0
				}
			}
		}
		localization_key = andersson_anthropologist_achievement_total_neoanthropus_fossils_anthropologist_comma_loc
	}

	text = {
		trigger = {
			has_variable = andersson_neoanthropus_fossils_anthropologist_var
			var:andersson_neoanthropus_fossils_anthropologist_var > 0
		}
		localization_key = andersson_anthropologist_achievement_total_neoanthropus_fossils_anthropologist_no_comma_loc
