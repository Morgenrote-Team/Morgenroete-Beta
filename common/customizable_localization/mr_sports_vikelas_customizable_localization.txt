###Olympic Games Winner(s)###
#This custom loc uses different locs for the amount of countries that won the most gold medals
vikelas_winners_type = {
	type = country

	text = {
		trigger = { #If both the first, second and third country have the same amount of gold medals, this triggers.
			global_var:vikelas_medal_rank_1_this_year_global_var.var:vikelas_gold_medals_this_year_var = global_var:vikelas_medal_rank_2_this_year_global_var.var:vikelas_gold_medals_this_year_var
			global_var:vikelas_medal_rank_1_this_year_global_var.var:vikelas_gold_medals_this_year_var = global_var:vikelas_medal_rank_3_this_year_global_var.var:vikelas_gold_medals_this_year_var
		}
		localization_key = vikelas_3_winners_loc
	}

	text = {
		trigger = { #If both the first and second country have the same amount of gold medals, this triggers.
			global_var:vikelas_medal_rank_1_this_year_global_var.var:vikelas_gold_medals_this_year_var = global_var:vikelas_medal_rank_2_this_year_global_var.var:vikelas_gold_medals_this_year_var
		}
		localization_key = vikelas_2_winners_loc
	}

	text = { #This triggers when none of the other locs are valid.
		localization_key = vikelas_1_winner_loc
	}
}

######### Cricket Club Names ##########

### Historical Cricket Club Names ###
# Access with with e.g. [SCOPE.sState('state_scope').GetCustom('vikelas_historical_cricket_club_names_type')]
vikelas_historical_cricket_club_names_type = {
	type = state

	#Historical
	text = {
		trigger = {
			NOT = { var:vikelas_chosen_cricket_club_var = flag:vikelas_generic_cricket_club_fallback_loc }
		}
		localization_key = vikelas_historical_cricket_club_name_loc
	}

	#If no other loc is possible, this loc is chosen
	text = {
		trigger = {
			always = no
		}
		fallback = yes # Optional. If fallback is enabled, this option will be picked when there are no other valid options, even if the trigger isn't met.
		localization_key = vikelas_generic_cricket_club_fallback_loc
	}
}
