#As all scientists use the same system, we can just use one unified system to choose their title.
#However we have to add a seperate custom_loc for each character role as this is required.
character_role_title_for_texts_custom_loc = {
	type = character
	random_valid = yes

	text = {
		localization_key = mr_prof_for_text_title
		trigger = {
			mr_character_is_sir_trigger = no
			mr_experience_for_professor_title_trigger = yes
		}
	}

	text = {
		localization_key = mr_dr_for_text_title
		trigger = {
			mr_character_is_sir_trigger = no
			mr_experience_for_doctor_title_trigger = yes
		}
	}

	text = {
		localization_key = mr_sir_for_text_title
		trigger = {
			is_female = no
			mr_character_is_sir_trigger = yes
		}
	}

	text = {
		localization_key = mr_dame_for_text_title
		trigger = {
			is_female = yes
			mr_character_is_sir_trigger = yes
		}
	}
}
