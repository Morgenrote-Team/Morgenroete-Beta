	random_valid = yes

		localization_key = mr_prof_for_text_title
		trigger = {
			mr_character_is_sir_trigger = no
			mr_character_is_professor_trigger = yes
		}
	}

	text = {
		localization_key = mr_dr_for_text_title
		trigger = {
			mr_character_is_sir_trigger = no
			mr_character_is_professor_trigger = no
			mr_character_is_doctor_trigger = yes
		}
	}

	text = {
		localization_key = mr_sir_for_text_title
		trigger = {
			is_female = no
			mr_character_is_sir_trigger = yes
		}
	}

	text = {
		localization_key = mr_dame_for_text_title
		trigger = {
			is_female = yes
			mr_character_is_sir_trigger = yes
		}
	}

	text = {
		localization_key = mr_mr_for_text_title
		trigger = {
			is_female = no
			mr_character_is_sir_trigger = no
			mr_character_is_doctor_trigger = no
			mr_character_is_professor_trigger = no
		}
	}

	text = {
		localization_key = mr_ms_for_text_title
		trigger = {
			is_female = yes
			mr_character_is_sir_trigger = no
			mr_character_is_doctor_trigger = no
			mr_character_is_professor_trigger = no
		}
	}
