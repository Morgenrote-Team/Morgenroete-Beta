#Alpine Club President
douglas_alpine_club_president_type = {
	type = country
	random_valid = no #From top to bottom decides which name list is appropriate.

	text = {
		trigger = {
			has_global_variable = douglas_alpine_club_founded_global_var
		}
		localization_key = douglas_has_alpine_club_president_loc
	}

	text = {
		trigger = {
			NOT = {
				has_global_variable = douglas_alpine_club_founded_global_var
			}
		}
		localization_key = douglas_has_no_alpine_club_president_loc
	}
}
