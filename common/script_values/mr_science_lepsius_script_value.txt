#This value determines the rating of a Archaeologist
lepsius_archaeologist_character_rating_value = {
	### Experience: 1 point per 100 Experience ###
	value = var:lepsius_archaeologist_experience_var
	divide = 100
	### Positive Traits ###
	#Experience Trait
	if = {
		limit = {
			has_trait = lepsius_archaeologist_famous
		}
		add = 3
	}
	else_if = {
		limit = {
			has_trait = lepsius_archaeologist_experienced
		}
		add = 2
	}
	else = {
		add = 1
	}
	#Specializations
	if = {
		limit = {
			has_trait = lepsius_egypt_specialist
		}
		add = 3
	}
	if = {
		limit = {
			has_trait = lepsius_mesopotamia_specialist
		}
		add = 3
	}
	if = {
		limit = {
			has_trait = lepsius_aegean_specialist
		}
		add = 3
	}
	if = {
		limit = {
			has_trait = lepsius_precolumbian_specialist
		}
		add = 3
	}
	if = {
		limit = {
			has_trait = lepsius_local_specialist
		}
		add = 3
	}
	#General
	if = {
		limit = {
			OR = {
				has_trait = panum_physician
				has_trait = panum_physician_experienced
				has_trait = panum_physician_famous
				has_trait = panum_epidemiologist
			}
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = explorer
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = persistent
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = ambitious
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = brave
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = surveyor
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = cautious
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = romantic
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = innovative
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = meticulous
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = charismatic
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = tactful
		}
		add = 1
	}
	if = {
		limit = {
			OR = {
				has_trait = basic_diplomat
				has_trait = experienced_diplomat
				has_trait = masterful_diplomat
			}
		}
		add = 1
	}
	### Negative Traits ###
	if = {
		limit = {
			has_trait = reckless
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = wrathful
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = cruel
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = hedonist
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = bigoted
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = arrogant
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = grifter
		}
		subtract = 1
	}
	if = {
		limit = {
			mr_character_has_negative_trait = yes
		}
		subtract = 2
	}
	if = {
		limit = {
			has_trait = senile
		}
		subtract = 2
	}
	### Age Factor, the older the worse for the rating ###
	if = {
		limit = { age > 30 }
		subtract = 1
	}
	if = {
		limit = { age > 40 }
		subtract = 2
	}
	if = {
		limit = { age > 50 }
		subtract = 3
	}
	if = {
		limit = { age > 60 }
		subtract = 4
	}
	if = {
		limit = { age > 70 }
		subtract = 5
	}
}

#This value calculates the number of Egyptian artifacts from Excavation sites
lepsius_yearly_egyptian_artifacts_value = {
	#Valley of the Kings
	if = {
		limit = {
			has_modifier = lepsius_upper_egypt_excavation_modifier
			NOT = {
				s:STATE_UPPER_EGYPT = {
					any_scope_state = {
						has_state_trait = lepsius_state_trait_upper_egypt
						has_modifier = lepsius_excavation_sacked_state_modifier
					}
				}
			}
		}
		#Egyptology (gives +2 instead of +1)
		if = {
			limit = {
				has_technology_researched = lepsius_specialization_egypt_tech
			}
			value = 2
		}
		else = {
			value = 1
		}
		if = {
			limit = {
				has_technology_researched = lepsius_modern_archaeology_tech ##Gives +1 excavation artifacts each year
			}
			add = 1
		}
	}

	#The Pyramids
	if = {
		limit = {
			has_modifier = lepsius_lower_egypt_excavation_modifier
			NOT = {
				s:STATE_LOWER_EGYPT = {
					any_scope_state = {

						has_modifier = lepsius_excavation_sacked_state_modifier
					}
				}
			}
		}
		#Egyptology (gives +2 instead of +1)
		if = {
			limit = {
				has_technology_researched = lepsius_specialization_egypt_tech
			}
			value = 2
		}
		else = {
			value = 1

lepsius_auction_value = {
	#Experienced Archaeologist
	if = {
		limit = {
			has_variable = lepsius_archaeologist_var
		}
		add = {
			value = var:lepsius_archaeologist_var.var:lepsius_archaeologist_experience_var
			divide = 30
		}
	}
	#Artifacts
	if = {
		limit = {
			has_variable = lepsius_total_artifacts
		}
		add = {
			value = var:lepsius_total_artifacts
			divide = 10
		}
	}
	#Great Powers are very important
	if = {
		limit = {
			country_rank = rank_value:great_power
		}
		add = 5
	}

	#Major Powers are important
	if = {
		limit = {
			country_rank = rank_value:major_power
		}
		add = 3
	}
}
