### Hire Academic Values ###

#General cost value
academics_hiring_academic_general_cost_value = {
	value = 1
	#1.2x for foreign character
	if = {
		limit = {
			NOT = {
				owner ?= player
			}
		}
		multiply = {
			value = 1.2
			desc = "academics_hiring_academic_general_cost_foreign_academic_tt"
			format = "academics_hiring_academic_general_breakdown_format"
		}
	}
	#1.3x for no common language
	if = {
		limit = {
			NOT = {
				culture = {
					shares_language_trait_group_with_any_primary_culture = player
				}
			}
		}
		multiply = {
			value = 1.3
			desc = "academics_hiring_academic_general_cost_no_common_language_tt"
			format = "academics_hiring_academic_general_breakdown_format"
		}
	}
	#1.5x for no common culture trait
	if = {
		limit = {
			NOT = {
				culture = {
					shares_heritage_trait_with_any_primary_culture = player
				}
			}
		}
		multiply = {
			value = 1.5
			desc = "academics_hiring_academic_general_cost_no_common_culture_trait_tt"
			format = "academics_hiring_academic_general_breakdown_format"
		}
	}
}
