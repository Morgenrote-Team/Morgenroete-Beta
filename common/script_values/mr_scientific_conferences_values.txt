mr_scientist_votes_value = {
	# Evaluated in the character scope
	value = 0
	if = {
		limit = {
			mr_character_has_famous_trait_trigger = yes
		}
		add = {
			desc = mr_value_from_famous_trait
			value = 5
		}
	}
	else_if = {
		limit = {
			mr_character_has_experienced_trait_trigger = yes
		}
		add = {
			desc = mr_value_from_experienced_trait
			value = 2
		}
	}
	else = {
		add = {
			desc = mr_value_from_no_experience_trait
			value = 1
		}
	}
}

verrier_pro_ceres_stance_value = {
	# Evaluated in the character scope
	value = 0
	
	# Add loyalty if they like their home country
	if = {
		limit = {
			owner ?= {
				var:mr_conference_position_var ?= flag:mr_stance_pro_ceres
			}
			loyalty > define:NCharacters|LOYALTY_UNCERTAIN_THRESHOLD # Loyalty is 25 or below
		}
		add = loyalty
	}
	# Subtract the loyalty gap if they hate their home country; yes scientists are petty
	else_if = {
		limit = {
			owner ?= {
				var:mr_conference_position_var ?= flag:mr_stance_pro_ceres
			}
			loyalty <= define:NCharacters|LOYALTY_UNCERTAIN_THRESHOLD # Loyalty is 25 or below
		}
		subtract = {
			add = 100
			subtract = loyalty
		}
	}
}
