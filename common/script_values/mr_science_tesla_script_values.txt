#This Script value depends on the technological advancement of a country.
tesla_tech_level_multiplier_5_value = {
	value = tesla_percentage_more_advanced_countries_value
	multiply = 5
}

#This Script Value calculates the percentage of more advanced countries that are not decentralized
tesla_percentage_more_advanced_countries_value = {
	if = {
		limit = {
			any_country = {
				techs_researched >= prev.techs_researched
			}
		}
		every_country = {
			limit = {
				NOT = { is_country_type = decentralized }
				techs_researched >= prev.techs_researched
			}
			add = 1
		}
	}
	else = {
		value = 0.01
	}

	divide = tesla_num_countries_not_decentralized_value
}

#This Script Value counts the number of countries that are not decentralized
tesla_num_countries_not_decentralized_value = {
	every_country = {
		limit = {
			NOT = { is_country_type = decentralized }
		}
		add = 1
	}
}

### Civil Engineering Projects ###

#Monthly Civil Engineering Progress
tesla_monthly_civil_engineering_progress_value = {
	value = {
		desc = "tesla_engineering_progress_base_tt"
		value = 1
	}

	if = {
		limit = {
			tesla_has_civil_engineer_no_tooltip_trigger = yes
		}

		#Having a Civil Engineer
		add = {
			desc = "tesla_engineering_progress_civil_engineer_value_tt"
			value = 1
		}

		#Engineer's Civil Engineering Experience divided by 300
		if = {
			limit = {
				var:tesla_engineer_var ?= {
					has_variable = tesla_civil_engineer_experience_var
					var:tesla_civil_engineer_experience_var > 0
				}
			}

			add = {
				desc = "tesla_engineering_progress_civil_experience_value_tt"
				value = tesla_engineering_progress_civil_experience_value
			}
		}

		if = { #Engineer has Civil Engineer Trait
			limit = {
				var:tesla_engineer_var ?= {
					has_trait = tesla_civil_engineer
				}
			}
			add = {
				desc = "tesla_engineering_progress_civil_engineer_trait_tt"
				value = 1
			}
		}
		if = { #Engineer has Meticulous Trait
			limit = {
				var:tesla_engineer_var ?= {
					has_trait = meticulous
				}
			}
			add = {
				desc = "tesla_engineering_progress_meticulous_trait_tt"
				value = 0.1
			}
		}
		if = { #Engineer has Persistent Trait
			limit = {
				var:tesla_engineer_var ?= {
					has_trait = persistent
				}
			}
			add = {
				desc = "tesla_engineering_progress_persistent_trait_tt"
				value = 0.1
			}
		}
		if = { #Engineer has Innovative Trait
			limit = {
				var:tesla_engineer_var ?= {
					has_trait = innovative
				}
			}
			add = {
				desc = "tesla_engineering_progress_innovative_trait_tt"
				value = 0.1
			}
		}
		if = { #Engineer has Surveyor Trait
			limit = {
				var:tesla_engineer_var ?= {
					has_trait = surveyor
				}
			}
			add = {
				desc = "tesla_engineering_progress_surveyor_trait_tt"
				value = 0.1
			}
		}
		if = { #Engineer has Physicist Trait
			limit = {
				var:tesla_engineer_var ?= {
					has_trait = curie_physicist
				}
			}

			add = {

#This script value is used to determine the number of states with the roads modifier
tesla_states_with_roads_value = {
	every_scope_state = {
		limit = { has_modifier = gaudi_modern_roads_modifier }
		add = 1
	}
}

#This script value is used to calculate the number of all agriculture buildings in a state
tesla_state_total_agriculture_level_value = {
    every_scope_building = {
        limit = {
            OR = {
                is_building_group = bg_agriculture
                is_building_group = bg_plantations
            }
			NOT = { is_building_group = bg_subsistence_agriculture }
        }
        add = level
    }
}

#Calculates the monthly mechanical computer progress
tesla_mechanical_computer_monthly_progress_value = {
	value = 1
	if = {
		limit = {
			tesla_has_mechanical_engineer_trigger = yes
		}
		add =  1
		if = {
			limit = {
				var:tesla_engineer_var = {
					has_variable = tesla_mechanical_engineer_experience_var
				}
			}
			add = {
				value = var:tesla_engineer_var.var:tesla_mechanical_engineer_experience_var
				divide = 300
			}
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = tesla_mechanical_engineer
				}
			}
			add = 1
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				# Engineer is mathematician
				var:tesla_engineer_var = {
					has_trait = verrier_mathematician
				}
			}
			add = 1
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = innovative
				}
			}
			add = 0.5
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = curie_physicist
				}
			}
			add = 0.5
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = meticulous
				}
			}
			add = 0.1
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = persistent
				}
			}
			add = 0.1
		}
		if = {
			limit = {
				has_variable = tesla_engineer_var
				var:tesla_engineer_var = {
					OR = {
						has_trait = alcoholic
						has_trait = expensive_tastes
						has_trait = opium_addiction
						has_trait = cocaine_addiction
						has_trait = syphilis
						has_trait = psychological_affliction
					}
				}
			}
			add = -0.2
		}
		if = {
			limit = {
				has_variable = tesla_engineer_var
				var:tesla_engineer_var = {
					has_trait = senile
				}
			}
			add = -3
		}
	}
	if = {
		limit = {
			academics_num_mathematical_universities > 0
		}
		add = academics_mathematical_university_multiplier_tiny
	}
	if = {
		limit = {
			tesla_num_tooling_workshop_levels_value > 0
		}
		add = tesla_num_tooling_workshop_levels_value
	}
}

### Engineer Rating ###
tesla_engineer_rating_value = {
	#Experience depending on specialty
	# Civil Engineer
	if = {
		limit = {
			has_modifier = academics_is_civil_engineer_modifier
		}
		#Experience: 1 point per 50 Experience
		value = var:tesla_civil_engineer_experience_var
		divide = 50
	}
	# Mechanical Engineer
	else_if = {
		limit = {
			has_modifier = academics_is_mechanical_engineer_modifier
		}
		#Experience: 1 point per 50 Experience
		value = var:tesla_mechanical_engineer_experience_var
		divide = 50
	}
	# Militar Engineer
	else_if = {
		limit = {
			has_modifier = academics_is_military_engineer_modifier
		}
		#Experience: 1 point per 50 Experience
		value = var:tesla_military_engineer_experience_var
		divide = 50
	}
	# Electrical Engineer
	else_if = {
		limit = {
			has_modifier = academics_is_electrical_engineer_modifier
		}
		#Experience: 1 point per 50 Experience
		value = var:tesla_electrical_engineer_experience_var
		divide = 50
	}
	# Aeronautical Engineer
	else_if = {
		limit = {
			has_modifier = academics_is_aeronautical_engineer_modifier
		}
		#Experience: 1 point per 50 Experience
		value = var:tesla_aeronautical_engineer_experience_var
		divide = 50
	}
	else = {
		value = 0
	}

	#Specialty Traits
	# Civil Engineer
	if = {
		limit = {
			has_modifier = academics_is_civil_engineer_modifier
			has_trait = tesla_civil_engineer
		}
		add = 3
	}
	# Mechanical Engineer
	else_if = {
		limit = {
			has_modifier = academics_is_mechanical_engineer_modifier
			has_trait = tesla_mechanical_engineer
		}
		add = 3
	}
	# Militar Engineer
	else_if = {
		limit = {
			has_modifier = academics_is_military_engineer_modifier
			has_trait = tesla_military_engineer
		}
		add = 3
	}
	# Electrical Engineer
	else_if = {
		limit = {
			has_modifier = academics_is_electrical_engineer_modifier
			has_trait = tesla_electrical_engineer
		}
		add = 3
	}
	# Aeronautical Engineer
	else_if = {
		limit = {
			has_modifier = academics_is_aeronautical_engineer_modifier
			has_trait = tesla_aeronautical_engineer
		}
		add = 3
	}

	#Positive Traits
	if = {
		limit = {
			has_trait = verrier_mathematician
		}
		add = 3
	}
	if = {
		limit = {
			has_trait = innovative
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = curie_physicist
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = meticulous
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = persistent
		}
		add = 1
	}
	if = {
		limit = {
			has_modifier = academics_is_civil_engineer_modifier
			has_trait = surveyor
		}
		add = 1
	}
	#Negative Traits
	if = {
		limit = {
			OR = {
				has_trait = alcoholic
				has_trait = expensive_tastes
				has_trait = opium_addiction
				has_trait = cocaine_addiction
				has_trait = syphilis
				has_trait = psychological_affliction
			}
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = senile
		}
		subtract = 3
	}
	#Age Factor, the older the worse for the ranking
	if = {
		limit = { age > 30 }
		subtract = 1
	}
	if = {
		limit = { age > 40 }
		subtract = 1
	}
	if = {
		limit = { age > 50 }
		subtract = 2
	}
	if = {
		limit = { age > 60 }
		subtract = 4
	}
	if = {
		limit = { age > 70 }
		subtract = 5
	}
}
