#This Script value depends on the technological advancement of a country.
tesla_tech_level_multiplier_5_value = {
	value = tesla_percentage_more_advanced_countries_value
	multiply = 5
}

#This Script Value calculates the percentage of more advanced countries that are not decentralized
tesla_percentage_more_advanced_countries_value = {
	if = {
		limit = {
			any_country = {
				techs_researched >= root.techs_researched
			}
		}
		every_country = {
			limit = {
				NOT = { is_country_type = decentralized }
				techs_researched >= root.techs_researched
			}
			add = 1
		}
	}
	else = {
		value = 0.01
	}

	divide = tesla_num_countries_not_decentralized_value
}

#This Script Value counts the number of countries that are not decentralized
tesla_num_countries_not_decentralized_value = {
	every_country = {
		limit = {
			NOT = { is_country_type = decentralized }
		}
		add = 1
	}
}

#Monthly Civil Engineering Progress
tesla_monthly_civil_engineering_progress_value = {
	value = 1


	if = {
		limit = {
			tesla_has_civil_engineer_no_tooltip_trigger = yes
		}

		#Engineer's Civil Engineering Experience divided by 300
		if = {
			limit = {
				tesla_engineer_var ?= {
					has_variable = tesla_civil_engineer_experience_var
				}
			}

			add = {
				value = var:tesla_engineer_var.var:tesla_civil_engineer_experience_var
				divide = 300
				max = 1
			}
		}

		if = { #Engineer has Civil Engineer Trait
			limit = {
				var:tesla_engineer_var = {
					has_trait = gaudi_civil_engineer
				}
			}
			add = 1
		}
		if = { #Engineer has Meticulous Trait
			limit = {
				var:tesla_engineer_var = {
					has_trait = meticulous
				}
			}
			add = 0.1
		}
		if = { #Engineer has Persistent Trait
			limit = {
				var:tesla_engineer_var = {
					has_trait = persistent
				}
			}
			add = 0.1
		}
		if = { #Engineer has Innovative Trait
			limit = {
				var:tesla_engineer_var = {
					has_trait = innovative
				}
			}
			add = 0.1
		}
		if = { #Engineer has Surveyor Trait
			limit = {
				var:tesla_engineer_var = {
					has_trait = surveyor
				}
			}
			add = 0.1
		}
		if = { #Engineer has Physicist Trait
			limit = {
				var:tesla_engineer_var = {
					has_trait = curie_physicist
				}
			}
			add = 0.1
		}
		if = {
			limit = {
				has_modifier = gaudi_additional_support_for_civil_engineer_modifier
			}
			add = 0.1
		}

		#Specialized Universities
		add = academics_mathematical_university_multiplier_tiny
	}
}

#Calculates the monthly mechanical computer progress
tesla_mechanical_computer_monthly_progress_value = {
	value = 1
	if = {
		limit = {
			tesla_has_mechanical_engineer_trigger = yes
		}
		add =  1
		if = {
			limit = {
				var:tesla_engineer_var = {
					has_variable = tesla_mechanical_engineer_experience_var
				}
			}
			add = {
				value = var:tesla_engineer_var.var:tesla_mechanical_engineer_experience_var
				divide = 300
			}
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = gaudi_mechanical_engineer
				}
			}
			add = 1
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				# Engineer is mathematician
				var:tesla_engineer_var = {
					has_trait = verrier_mathematician
				}
			}
			add = 1
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = innovative
				}
			}
			add = 0.5
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = curie_physicist
				}
			}
			add = 0.5
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = meticulous
				}
			}
			add = 0.1
		}
		if = {
			limit = {
				tesla_has_mechanical_engineer_trigger = yes
				var:tesla_engineer_var = {
					has_trait = persistent
				}
			}
			add = 0.1
		}
		if = {
			limit = {
				has_variable = tesla_engineer_var
				var:tesla_engineer_var = {
					OR = {
						has_trait = alcoholic
						has_trait = expensive_tastes
						has_trait = opium_addiction
						has_trait = cocaine_addiction
						has_trait = syphilis
						has_trait = psychological_affliction
					}
				}
			}
			add = -0.2
		}
		if = {
			limit = {
				has_variable = tesla_engineer_var
				var:tesla_engineer_var = {
					has_trait = senile
				}
			}
			add = -3
		}
	}
	if = {
		limit = {
			academics_num_mathematical_universities > 0
		}
		add = academics_mathematical_university_multiplier_tiny
	}
	if = {
		limit = {
			tesla_num_tooling_workshop_levels_value > 0
		}
		add = tesla_num_tooling_workshop_levels_value
	}
}
