#This value determines the rating of a chemist
mendelejew_chemist_character_rating_value = {
	### Experience: 1 point per 50 Experience ###
	value = var:mendelejew_chemist_experience_var
	divide = 50
	### Positive Traits ###
	#Experience Trait
	if = {
		limit = {
			has_trait = mendelejew_chemist_famous
		}
		add = 3
	}
	else_if = {
		limit = {
			has_trait = mendelejew_chemist_experienced
		}
		add = 2
	}
	else = {
		add = 1
	}
	#General
	if = {
		limit = {
			has_trait = mendelejew_chemist_theorist
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = mendelejew_chemist_analyst
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = mendelejew_chemist_periodic_table_inventor
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = mendelejew_chemist_inspector
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = engineer
		}
		add = 2
	}
	if = {
		limit = {
			OR = {
				has_trait = verrier_astronomer
				has_trait = verrier_astronomer_experienced
				has_trait = verrier_astronomer_famous
			}
		}
		add = 1
	}
	#if = {
	#	limit = {
	#		OR = {
	#			has_trait = agassiz_geologist
	#			has_trait = agassiz_geologist_experienced
	#			has_trait = agassiz_geologist_famous
	#		}
	#	}
	#	add = 1
	#}
	if = {
		limit = {
			has_trait = innovative
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = meticulous
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = persistent
		}
		add = 1
	}
	### Negative Traits ###
	if = {
		limit = {
			has_trait = alcoholic
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = opium_addiction
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = cocaine_addiction
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = senile
		}
		subtract = 3
	}
	### Age Factor, the older the worse for the rating ###
	if = {
		limit = { age > 30 }
		subtract = 1
	}
	if = {
		limit = { age > 40 }
		subtract = 2
	}
	if = {
		limit = { age > 50 }
		subtract = 3
	}
	if = {
		limit = { age > 60 }
		subtract = 4
	}
	if = {
		limit = { age > 70 }
		subtract = 5
	}
}

### General Mendelejew Progress ###
mendelejew_general_project_progress_value = {

	value = {
		desc = "mendelejew_chemistry_progress_base_tt"
		value = 1
	}

	if = {
		limit = {
			mendelejew_has_chemist_no_tooltip_trigger = yes
		}

		#Chemist's Experience divided by 300
		if = {
			limit = {
				var:mendelejew_chemist_var ?= {
					has_variable = mendelejew_chemist_experience_var
					var:mendelejew_chemist_experience_var > 0
				}
			}

			add = {
				desc = "mendelejew_chemistry_progress_experience_value_tt"
				value = mendelejew_improvement_progress_from_experience_value
			}
		}

		if = { #Chemist has Specialization
			limit = {
				var:mendelejew_chemist_var ?= {
					OR = {
						has_trait = mendelejew_chemist_theorist
						has_trait = mendelejew_chemist_analyst
						has_trait = mendelejew_chemist_inspector
					}
				}
			}
			add = {
				desc = "mendelejew_chemistry_progress_specialization_trait_tt"
				value = 1
			}
		}
		if = { #Chemist has Meticulous Trait
			limit = {
				var:mendelejew_chemist_var ?= {
					has_trait = meticulous
				}
			}
			add = {
				desc = "mendelejew_chemistry_progress_meticulous_trait_tt"
				value = 0.1
			}
		}
		if = { #Chemist has Persistent Trait
			limit = {
				var:mendelejew_chemist_var ?= {
					has_trait = persistent
				}
			}
			add = {
				desc = "mendelejew_chemistry_progress_persistent_trait_tt"
				value = 0.1
			}
		}
		if = { #Chemist has Innovative Trait
			limit = {
				var:mendelejew_chemist_var ?= {
					has_trait = innovative
				}
			}
			add = {
				desc = "mendelejew_chemistry_progress_innovative_trait_tt"
				value = 0.1
			}
		}
		if = { #Chemist has Physicist Trait
			limit = {
				var:mendelejew_chemist_var ?= {
					has_trait = curie_physicist
				}
			}
			add = {
				desc = "mendelejew_chemistry_progress_curie_physicist_trait_tt"
				value = 0.1
			}
		}
		if = { #Chemist has Mathematician Trait
			limit = {
				var:mendelejew_chemist_var ?= {
					has_trait = verrier_mathematician
				}
			}
			add = {
				desc = "mendelejew_chemistry_progress_verrier_mathematician_trait_tt"
				value = 0.1
			}
		}

		#Specialized Universities
		if = {
			limit = {
				academics_mathematical_university_multiplier_tiny > 0
			}
			add = {
				desc = "mendelejew_chemistry_progress_mathematical_universities_tt"
				add = academics_mathematical_university_multiplier_tiny
			}
		}
	}
}
