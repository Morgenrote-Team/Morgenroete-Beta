#This value determines the rating of a Anthropologist
andersson_anthropologist_character_rating_value = {
	### Experience: 1 point per 50 Experience ###
	value = var:andersson_anthropologist_experience_var
	divide = 50
	### Positive Traits ###
	#Experience Trait
	if = {
		limit = {
			has_trait = andersson_anthropologist_famous
		}
		add = 3
	}
	else_if = {
		limit = {
			has_trait = andersson_anthropologist_experienced
		}
		add = 2
	}
	else = {
		add = 1
	}
	#General
	if = {
		limit = {
			has_trait = andersson_speleologist
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = andersson_polygenist
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = andersson_monogenist
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = erudite
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = explorer
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = brave
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = innovative
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = meticulous
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = aesthete
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = persistent
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = ambitious
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = surveyor
		}
		add = 1
	}
	### Negative Traits ###
	if = {
		limit = {
			has_trait = bigoted
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = arrogant
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = wrathful
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = hedonist
		}
		subtract = 1
	}
	if = {
		limit = {
			has_trait = cautious
		}
		subtract = 1
	}
	if = {
		limit = {
			mr_character_has_negative_trait = yes
		}
		subtract = 2
	}
	if = {
		limit = {
			has_trait = senile
		}
		subtract = 2
	}
	### Age Factor, the older the worse for the rating ###
	if = {
		limit = { age > 30 }
		subtract = 1
	}
	if = {
		limit = { age > 40 }
		subtract = 2
	}
	if = {
		limit = { age > 50 }
		subtract = 3
	}
	if = {
		limit = { age > 60 }
		subtract = 4
	}
	if = {
		limit = { age > 70 }
		subtract = 5
	}
}

###Piltdown Man###

andersson_piltdown_attacker_value = {
	value = 0

	if = {
		limit = {
			has_variable = andersson_anthropologist_var
		}
		add = var:andersson_anthropologist_var.var:andersson_anthropologist_experience_var #Experience gives 1-10 points
		divide = 30
		round = yes
		min = 1
	}

	add = { 0 1 } #random success gives 0-1 points

	if = {
		limit = {
			has_variable = mendelejew_chemist_var
			var:mendelejew_chemist_var = {
				has_trait = mendelejew_chemist_famous #Historically, chemistry helped a lot to expose the fraud, so this will tackle the base defend value
			}
		}
		add = 3
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				has_trait = andersson_polygenist #Positive traits will influence the outcome
			}
		}
		add = 3
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				has_trait = firebrand #Positive traits will influence the outcome
			}
		}
		add = 3
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				has_trait = demagogue #Positive traits will influence the outcome
			}
		}
		add = 2
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				has_trait = erudite #Positive traits will influence the outcome
			}
		}
		add = 2
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				has_trait = dubois_paleontologist #Positive traits will influence the outcome
			}
		}
		add = 2
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				has_trait = inspirational_orator #Positive traits will influence the outcome
			}
		}
		add = 1
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				has_trait = charismatic #Positive traits will influence the outcome
			}
		}
		add = 1
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				has_trait = innovative #Positive traits will influence the outcome
			}
		}
		add = 1
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				has_trait = persistent #Positive traits will influence the outcome
			}
		}
		add = 1
	}

	if = {
		limit = {
			var:andersson_anthropologist_var = {
				OR = {
					has_trait = andersson_evolutionist #Positive traits will influence the outcome
					has_trait = andersson_diffusionist #Positive traits will influence the outcome
				}
			}
		}
		add = 1
	}

	if = {
		limit = { #Beeing ill is not good
			var:andersson_anthropologist_var = {
				OR = {
					has_trait = psychological_affliction
					has_trait = panum_cholera
					has_trait = panum_tuberculosis
					has_trait = panum_yellow_fever
				}
			}
		}
		add = -1
	}

	if = {
		limit = { #Addictions are not good
			var:andersson_anthropologist_var = {
				OR = {
					has_trait = alcoholic
					has_trait = opium_addiction
					has_trait = cocaine_addiction
				}
			}
		}
		add = -1
	}

	if = {
		limit = { #Senile is bad
			var:andersson_anthropologist_var = {
				OR = {
					has_trait = senile
				}
			}
		}
		add = -2
	}

	#round = yes
}
