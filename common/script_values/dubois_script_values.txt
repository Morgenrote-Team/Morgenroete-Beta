#This value is used to randomly change the amount of animals in the breeding je
dubois_breeding_multiplier = {
	value = { 0.8 1.2 }
	if = {
		limit = {
			has_technology_researched = dubois_genetics_tech
		}
		add = { 0.1 0.4 }
	}
	if = {
		limit = {
			has_variable = dubois_naturalist_var
		}
		add = 0.02
	}
	if = {
		limit = {
			AND = {
				has_variable = dubois_naturalist_var
				var:dubois_naturalist_var = {
					var:dubois_naturalist_experience >= 100
				}
			}
		}
		add = 0.02
	}
	if = {
		limit = {
			AND = {
				has_variable = dubois_naturalist_var
				var:dubois_naturalist_var = {
					var:dubois_naturalist_experience >= 300
				}
			}
		}
		add = 0.02
	}
	if = {
		limit = {
			AND = {
				has_variable = dubois_naturalist_var
				var:dubois_naturalist_var = {
					has_trait = dubois_zoologist
				}
			}
		}
		add = 0.02
	}
	if = {
		limit = {
			AND = {
				has_variable = dubois_naturalist_var
				var:dubois_naturalist_var = {
					has_trait = innovative
				}
			}
		}
		add = 0.02
	}
	if = {
		limit = {
			any_scope_building = {
				has_active_production_method = pm_zoo_level_4
			}
		}
		add = 0.02
	}
	if = {
		limit = {
			any_scope_building = {
				has_active_production_method = pm_zoo_level_5
			}
		}
		add = 0.04
	}
	add = academics_natural_university_multiplier_for_breeding
}

#Calculates the starting success chance for the Paleontological Expedition
dubois_paleontological_expedition_starting_success_chance_value = {
	value = 0

	#Paleontologist Experience
	if = {
		limit = {
			var:dubois_paleontologist_var.var:dubois_paleontologist_experience >= 300
		}
		add = 3
	}
	else_if = {
		limit = {
			var:dubois_paleontologist_var.var:dubois_paleontologist_experience >= 200
		}
		add = 2
	}
	else_if = {
		limit = {
			var:dubois_paleontologist_var.var:dubois_paleontologist_experience >= 100
		}
		add = 1
	}

	#Paleontologist Traits
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				OR = {
					has_trait = surveyor
					has_trait = dubois_geologist
				}
			}
		}
		add = 1
	}
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				has_trait = meticulous
			}
		}
		add = 1
	}
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				has_trait = persistent
			}
		}
		add = 1
	}
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				OR = {
					has_trait = alcoholic
					has_trait = opium_addiction
					has_trait = cocaine_addiction
					has_trait = psychological_affliction
					has_trait = reckless
				}
			}
		}
		add = -1
	}
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				has_trait = senile
			}
		}
		add = -2
	}

	#Natural Sciences University Levels
	if = {
		limit = {
			academics_num_natural_universities >= 20
		}
		add = 3
	}
	else_if = {
		limit = {
			academics_num_natural_universities >= 10
		}
		add = 2
	}
	else_if = {
		limit = {
			academics_num_natural_universities >= 5
		}
		add = 1
	}
}

#This is used as the goal value for the paleontological excavation progress
dubois_dinosaur_excavation_finding_progress_goal_value = 240

dubois_paleontological_expedition_positive_effect_chance_value = {
	value = 0

	#Paleontologist
	if = {
		limit = {
			has_variable = dubois_paleontologist_var
		}
		add = 0.3
		add = {
			value = {
				value = var:dubois_paleontologist_var.var:dubois_paleontologist_experience
				divide = 1000
				max = 0.3
			}
		}
		### Traits of Paleontologist ###
		#Geologist
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = dubois_geologist
				}
			}
			add = 0.1
		}

		#Surveyor
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = surveyor
				}
			}
			add = 0.05
		}

		#Innovative
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = innovative
				}
			}
			add = 0.05
		}

		#Meticulous
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = meticulous
				}
			}
			add = 0.05
		}

		#Persistent
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = persistent
				}
			}
			add = 0.05
		}

		#Reckless
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = reckless
				}
			}
			add = -0.1
		}

		#Bad Trait
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					OR = {
						has_trait = alcoholic
						has_trait = opium_addiction
						has_trait = cocaine_addiction
						has_trait = psychological_affliction
						has_trait = senile
					}
				}
			}
			add = -0.4
		}	
	}

	#Paleontology tech
	if = {
		limit = {
			has_technology_researched = dubois_paleontology_tech
		}
		add = 0.2
	}

	#Natural History Universities
	if = {
		limit = {
			any_scope_building = {
				level > 0
				has_active_production_method = pm_dubois_university_natural_history
			}
		}
		add = {
			value = academics_num_natural_universities
			multiply = 0.01
		}
	}

	#There is no negative progress
	min = 0.1
	max = 0.9
}

dubois_paleontological_expedition_negative_effect_chance_value = {
	value = 1
	subtract = dubois_paleontological_expedition_positive_effect_chance_value
}

#This value determines the rating of a naturalist
dubois_naturalist_character_rating_value = {
	### Experience: 1 point per 50 Experience ###
	value = var:dubois_naturalist_experience
	divide = 50
	### Positive Traits ###
	#Experience Trait
	if = {
		limit = {
			has_trait = dubois_naturalist_famous
		}
		add = 3
	}
	else_if = {
		limit = {
			has_trait = dubois_naturalist_experienced
		}
		add = 2
	}
	else = {
		add = 1
	}
	#General
	if = {
		limit = {
			has_trait = dubois_zoologist
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = dubois_botanist
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = dubois_entomologist
		}
		add = 2
	}
	if = {
		limit = {
			has_trait = dubois_marine_biologist
		}
		add = 2
	}
	if = {
		limit = {
			OR = {
				has_trait = meticulous
				has_trait = persistent
			}
		}
		add = 1
	}
	if = {
		limit = {
			has_trait = innovative
			owner = {
				has_technology_researched = dubois_nature_protection_tech
			}
		}
		add = 1
	}
	### Negative Traits ###
	if = {
		limit = {
			OR = {
				has_trait = alcoholic
				has_trait = opium_addiction
				has_trait = cocaine_addiction
				has_trait = grifter
				has_trait = expensive_tastes
			}
		}
		subtract = 2
	}
	if = {
		limit = {
			has_trait = senile
		}
		subtract = 5
	}
	### Age Factor, the older the worse for the ranking ###
	if = {
		limit = { age > 30 }
		subtract = 1
	}
	if = {
		limit = { age > 40 }
		subtract = 2
	}
	if = {
		limit = { age > 50 }
		subtract = 3
	}
	if = {
		limit = { age > 60 }
		subtract = 4
	}
	if = {
		limit = { age > 70 }
		subtract = 5
	}
}

#This Script value defines the progress when a naturalist documents an animal
naturalist_native_animal_progress_value = {
	### 1 basic point ###
	value = 1
	### Experience ###
	#Naturalist Experience: Experience divided by 60 = Progress
	if = {
		limit = {
			has_variable = dubois_naturalist_var
		}
		add = {
			value = var:dubois_naturalist_var.var:dubois_naturalist_experience
			divide = 60
			floor = yes
			max = 5
		}
	}
	### Positive Traits ###
	#Naturalist Traits: Zoologist
	if = {
		limit = {
			var:dubois_naturalist_var = {
				has_trait = dubois_zoologist 
			}
		}
		add = 1
	}
	#Personality of Naturalist: Meticulous
	if = {
		limit = {
			var:dubois_naturalist_var = {
				has_trait = meticulous
			}
		}
		add = 1
	}
	### Negative Traits ###
	#Vices: -1 point for each trait
	if = {
		limit = {
			var:dubois_naturalist_var = {
				has_trait = alcoholic
			}
		}
		subtract = 1
	}
	if = {
		limit = {
			var:dubois_naturalist_var = {
				has_trait = opium_addiction
			}
		}
		subtract = 1
	}
	if = {
		limit = {
			var:dubois_naturalist_var = {
				has_trait = cocaine_addiction
			}
		}
		subtract = 1
	}
	if = {
		limit = {
			var:dubois_naturalist_var = {
				has_trait = senile
			}
		}
		subtract = 1
	}
	### University ###
	#Natural History Faculty
	if = {
		limit = {
			any_scope_building = {
				is_building_type = building_university
				has_active_production_method = pm_dubois_university_natural_history
			}
		}
		add = 1
	}
	add = academics_natural_university_multiplier_tiny
}
