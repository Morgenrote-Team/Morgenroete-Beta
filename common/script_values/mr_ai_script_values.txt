#Used because negative probabilities break random_list effects

ai_promote_engineer_probability_value = {
	#Higher base chance
	value = 150
	#Nations historically focused on engineering are more likely to get an engineer
	if = {
		limit = {
			OR = {
				c:PRU ?= ROOT
				c:ITA ?= ROOT
				c:SWI ?= ROOT
				c:GER ?= ROOT
				c:AUS ?= ROOT
				c:SPA ?= ROOT
				c:BRZ ?= ROOT
			}
		}
		add = 25
	}
	#Smaller countries should favor engineers
	if = {
		limit = {
			country_rank < rank_value:major_power
		}
		add = 50
	}
	#Nations without access to formations, rainforests or excavations sites have a higher chance to promote Engineers
	if = {
		limit = {
			NOR = {
				dubois_has_formation = yes
				dubois_can_discover_formation = yes
				any_country = {
					has_diplomatic_pact = {
						who = ROOT
						type = paleontological_license
					}
				}
				dubois_has_rainforest_no_tooltip = yes
				any_country = {
					has_diplomatic_pact = {
						who = ROOT
						type = botanical_license
					}
				}
				lepsius_has_excavation_site = yes
				any_country = {
					has_diplomatic_pact = {
						who = ROOT
						type = excavation_license
					}
				}
			}
		}
		add = 200
	}
	#If the AI has an Engineer already, it will not try to get another one except if the existing one is old.
	if = {
		limit = {
			tesla_has_engineer_no_tooltip_trigger = yes
			var:tesla_engineer_var ?= {
				age < 60
			}
		}
		add = -1000
	}

	#random_list breaks if it has a negative number as a chance.
	min = 0
}

			}

		add = 100

	#Nations with a big enough navy for a Pacific expedition have a higher chance
	if = {

		add = 25
	}
	#Nations with a Zoo that does not yet have a Paleontological Collection should have a higher chance for a Paleontologist
	if = {
		limit = {
			any_scope_building = {
				is_building_type = building_dubois_zoo

				has_active_production_method = pm_no_paleontological_collection
			}
		}
