#Calculates the starting success chance for the Paleontological Expedition
dubois_paleontological_expedition_starting_success_chance_value = {
	value = 0

	#Paleontologist Experience
	if = {
		limit = {
			var:dubois_paleontologist_var.var:dubois_paleontologist_experience_var >= 300
		}
		add = 3
	}
	else_if = {
		limit = {
			var:dubois_paleontologist_var.var:dubois_paleontologist_experience_var >= 200
		}
		add = 2
	}
	else_if = {
		limit = {
			var:dubois_paleontologist_var.var:dubois_paleontologist_experience_var >= 100
		}
		add = 1
	}

	#Paleontologist Traits
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				OR = {
					has_trait = surveyor
					has_trait = agassiz_geologist
				}
			}
		}
		add = 1
	}
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				has_trait = meticulous
			}
		}
		add = 1
	}
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				has_trait = persistent
			}
		}
		add = 1
	}
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				OR = {
					has_trait = alcoholic
					has_trait = opium_addiction
					has_trait = cocaine_addiction
					has_trait = psychological_affliction
					has_trait = reckless
				}
			}
		}
		add = -1
	}
	if = {
		limit = {
			var:dubois_paleontologist_var = {
				has_trait = senile
			}
		}
		add = -2
	}

	#Natural Sciences University Levels
	if = {
		limit = {
			academics_num_natural_universities >= 20
		}
		add = 3
	}
	else_if = {
		limit = {
			academics_num_natural_universities >= 10
		}
		add = 2
	}
	else_if = {
		limit = {
			academics_num_natural_universities >= 5
		}
		add = 1
	}
}

#Monthly change of Progress for Paleontological Expeditions
dubois_paleontology_progress_value = {
	value = 0

	#Paleontologist
	if = {
		limit = {
			has_variable = dubois_paleontologist_var
		}
		add = {
			desc = "dubois_paleontology_progress_has_paleontologist_value_tt"
			value = 0.1
		}
		add = {
			desc = "dubois_paleontology_progress_paleontologist_experience_value_tt"
			value = {
				value = var:dubois_paleontologist_var.var:dubois_paleontologist_experience_var
				divide = 1000
				max = 0.3
			}
		}
		### Traits of Paleontologist ###
		#Geologist
		if = {
			limit = {
				var:dubois_paleontologist_var ?= {
					has_trait = agassiz_geologist
				}
			}
			add = {
				desc = "dubois_paleontology_progress_paleontologist_geologist_trait_value_tt"
				value = 0.1
			}
		}

		#Surveyor
		if = {
			limit = {
				var:dubois_paleontologist_var ?= {
					has_trait = surveyor
				}
			}
			add = {
				desc = "dubois_paleontology_progress_paleontologist_surveyor_trait_value_tt"
				value = 0.05
			}
		}

		#Innovative
		if = {
			limit = {
				var:dubois_paleontologist_var ?= {
					has_trait = innovative
				}
			}
			add = {
				desc = "dubois_paleontology_progress_paleontologist_innovative_trait_value_tt"
				value = 0.05
			}
		}

		#Meticulous
		if = {
			limit = {
				var:dubois_paleontologist_var ?= {
					has_trait = meticulous
				}
			}
			add = {
				desc = "dubois_paleontology_progress_paleontologist_meticulous_trait_value_tt"
				value = 0.05
			}
		}

		#Persistent
		if = {
			limit = {
				var:dubois_paleontologist_var ?= {
					has_trait = persistent
				}
			}
			add = {
				desc = "dubois_paleontology_progress_paleontologist_persistent_trait_value_tt"
				value = 0.05
			}
		}

		#Reckless
		if = {
			limit = {
				var:dubois_paleontologist_var ?= {
					has_trait = reckless
				}
			}
			add = {
				desc = "dubois_paleontology_progress_paleontologist_reckless_trait_value_tt"
				value = -0.1
			}
		}

		#Bad Trait
		if = {
			limit = {
				var:dubois_paleontologist_var ?= {
					OR = {
						has_trait = alcoholic
						has_trait = opium_addiction
						has_trait = cocaine_addiction
						has_trait = psychological_affliction
						has_trait = senile
					}
				}
			}
			add = {
				desc = "dubois_paleontology_progress_paleontologist_bad_trait_value_tt"
				value = -0.4
			}
		}
	}

	#Paleontology tech
	if = {
		limit = {
			has_technology_researched = dubois_paleontology_tech
		}
		add = {
			desc = "dubois_paleontology_progress_paleontology_tech_value_tt"
			value = 0.2
		}
	}

	#Natural History Universities
	if = {
		limit = {
			any_scope_building = {
				level > 0
				has_active_production_method = pm_dubois_university_natural_history
			}
		}
		add = {
			desc = "dubois_paleontology_progress_specialized_universities_value_tt"
			value = academics_num_natural_universities
			multiply = 0.01
		}
	}
}

#Yearly Fossils from general factors (technology, Natural History Universities, etc.)
# This should be the same for every excavation.
dubois_yearly_regular_fossils_from_general_factors_value = {
	#Base fossils
	value = 1

	#Additional yearly fossils
	if = {
		limit = {
			has_technology_researched = dubois_paleontology_tech
		}
		add = 1
	}
	if = {
		limit = {
			any_scope_building = {
				is_building_type = building_university
				has_active_production_method = pm_dubois_university_natural_history
				level > 0
			}
		}
		add = 1
		if = {
			limit = {
				academics_num_natural_universities >= 5
			}
			add = 1
		}
		if = {
			limit = {
				academics_num_natural_universities >= 10
			}
			add = 1
		}
	}
}

#This value is used to determine how well the Dinosaur Skeleton was mounted
dubois_probability_positive_dinosaur_skeleton_reconstruction_value = {
	### Paleontologist ###
	if = {
		limit = {
			dubois_has_paleontologist_no_tooltip_trigger = yes
		}

		### Experience ###
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_variable = dubois_paleontologist_experience_var
				}
			}
			add = dubois_paleontologist_experience_for_probability_positive_dinosaur_skeleton_reconstruction_value
		}
		### Traits ###
		#Persistent
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = persistent
				}
			}

			add = 5
		}
		#Innovative
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = innovative
				}
			}

			add = 5
		}
		#Meticulous
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = meticulous
				}
			}

			add = 5
		}
		#Senile
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					has_trait = senile
				}
			}

			subtract = 40
		}
		#Negative Traits
		if = {
			limit = {
				var:dubois_paleontologist_var = {
					OR = {
						has_trait = alcoholic
						has_trait = opium_addiction
						has_trait = cocaine_addiction
						has_trait = grifter
						has_trait = expensive_tastes
					}
				}
			}
			subtract = 10
		}
	}

	### Technology ###
	if = {
		limit = {
			has_technology_researched = dubois_paleontology_tech
		}
		add = 10
	}

	### Natural History Universities
	#Each Natural History University adds 2%
	add = dubois_academics_num_natural_universities_for_probability_positive_dinosaur_skeleton_reconstruction_value

	### Maximum and Minimum
	max = 99 #There should always be a tiny chance for a negative outcome
	min = 1 #There should always be a tiny chance for a positive outcome
}

dubois_paleontologist_experience_for_probability_positive_dinosaur_skeleton_reconstruction_value = {
	value = var:dubois_paleontologist_var.var:dubois_paleontologist_experience_var
	divide = 10
	multiply = 2
	max = 60
}

dubois_academics_num_natural_universities_for_probability_positive_dinosaur_skeleton_reconstruction_value = {
	value = academics_num_natural_universities
	multiply = 2
}

#Used to display value correctly
dubois_probability_positive_dinosaur_skeleton_reconstruction_percentage_value = {
	value = dubois_probability_positive_dinosaur_skeleton_reconstruction_value
	divide = 100
}

#This value is used to determine the chance of a perfect skeleton
dubois_percentage_perfect_dinosaur_skeleton_reconstruction_value = {
	value = 0.33
	if = {
		limit = {
			var:dubois_paleontologist_var ?= {
				var:dubois_paleontologist_experience_var >= 175
			}
		}
		add = 0.165
	}
	if = {
		limit = {
			var:dubois_paleontologist_var ?= {
				var:dubois_paleontologist_experience_var >= 200
			}
		}
		add = 0.165
	}
	if = {
		limit = {
			var:dubois_paleontologist_var ?= {
				var:dubois_paleontologist_experience_var >= 250
			}
		}
		add = 0.2
	}
}

#This value determines the chance of a imperfect skeleton
# 1 - Chance for Perfect = Chance for Imperfect
dubois_percentage_imperfect_dinosaur_skeleton_reconstruction_value = {
	value = 1
	subtract = dubois_percentage_perfect_dinosaur_skeleton_reconstruction_value
}

#This value is the final chance for a perfect skeleton
dubois_probability_perfect_dinosaur_skeleton_reconstruction_value = {
	value = dubois_probability_positive_dinosaur_skeleton_reconstruction_value
	multiply = dubois_percentage_perfect_dinosaur_skeleton_reconstruction_value
}

#This value is the final chance for a imperfect skeleton
dubois_probability_imperfect_dinosaur_skeleton_reconstruction_value = {
	value = dubois_probability_positive_dinosaur_skeleton_reconstruction_value
	multiply = dubois_percentage_imperfect_dinosaur_skeleton_reconstruction_value
}

#This value is the final chance for a poor skeleton
dubois_probability_poor_dinosaur_skeleton_reconstruction_value = {
	value = 100
	subtract = dubois_probability_positive_dinosaur_skeleton_reconstruction_value
}

#This value determines the rating of a paleontologist

	### Experience: 1 point per 50 Experience ###
	value = var:dubois_paleontologist_experience_var
	divide = 50
	### Positive Traits ###
	#Experience Trait
	if = {
		limit = {
			has_trait = dubois_paleontologist_famous
		}
		add = 3
	}
	else_if = {
