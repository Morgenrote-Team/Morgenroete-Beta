mr_aus_amerling_friedrich_character_template = {
	dna = mr_aus_amerling_friedrich
	first_name = mr_friedrich
	last_name = mr_amerling
	culture = cu:south_german
	role = character_role_klimt_painter
	historical = yes
	interest_group = ig_intelligentsia
	birth_date = 1803.04.14
	home_region = STATE_AUSTRIA
	traits = {
		klimt_painter
		klimt_classic_school
		klimt_romantic_school
		meticulous
		persistent
	}
	on_created = {
		set_variable = mr_no_headgear_var
		mr_make_character_painter_effect = yes
		set_global_variable = mr_amerling_friedrich_created_global_var
		if = {
			limit = { age >= klimt_painter_maturity_age_value }
			add_trait = klimt_romantic_painter
		}
	}
}
