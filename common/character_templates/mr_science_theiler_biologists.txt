	culture = cu:wallonian
	role = character_role_theiler_biologist
	role = character_role_panum_physician
	interest_group = ig_intelligentsia
	birth_date = 1870.06.13
	historical = yes
	home_region = STATE_WALLONIA
	traits = {
		theiler_biologist
		theiler_microbiologist
		panum_physician
		meticulous
		innovative
		erudite
