on_building_built = {
	on_actions = {
		andersson_on_building_built
	}
}

on_acquired_technology = {
	on_actions = {
		andersson_on_acquired_technology
	}
}

on_yearly_pulse_country = {
	on_actions = {
		delay = { days = { 1 360 } }
		andersson_on_yearly_pulse_historical_fossils
	}
}

andersson_on_building_built = {
	effect = {
		if = {
			limit = {
				owner = {
					NOT = {
						has_variable = andersson_institute_inauguration_happened_var
					}
					andersson_has_institute_trigger = yes
				}
			}
			owner = {
				trigger_event = {
					id = andersson.10 #Andersson 10: Institute Inauguration
				}
			}
		}
	}
}

andersson_on_acquired_technology = {
}

andersson_on_yearly_pulse_historical_fossils = {

		### Palaeanthropus "Cave Man" ###

#Local Archaeology Events
andersson_on_yearly_pulse_local_anthropology_events = {
	trigger = { andersson_has_active_anthropologist =  yes }

	random_events = {
		chance_to_happen = 100
		chance_of_no_event = {
			value = 93.3
			if = {
				limit = {
					var:andersson_anthropologist_var.var:andersson_anthropologist_experience_var >= 100
				}
				subtract = 1
			}
			else_if = {
				limit = {
					var:andersson_anthropologist_var.var:andersson_anthropologist_experience_var >= 200
				}
				subtract = 2.3
			}
			else_if = {
				limit = {
					var:andersson_anthropologist_var.var:andersson_anthropologist_experience_var >= 300
				}
				subtract = 4.3
			}
		}

		#Aegean
		2 = andersson.401 #Andersson 401: Hunting camp
		3 = andersson.402 #Andersson 402: Ceramics
		1 = andersson.403 #Andersson 403: Neolithic Village
		3 = andersson.404 #Andersson 404: Human Bones
		2 = andersson.405 #Andersson 405: Megalith
		2 = andersson.406 #Andersson 406: A Prehistoric Bone Flute
		5 = andersson.407 #Andersson 407: Skulls, Skulls and more Skulls!
		3 = andersson.408 #Andersson 408: Bones and Stone Tools recovered
		1 = andersson.409 #Andersson 409: Prehistoric Figurine discovered

		#Special Fossils
		50 = andersson.62 #Andersson 62: Piltdown Man (Dynamic)
		10 = andersson.63 #Andersson 63: Paranthropus
	}
}
