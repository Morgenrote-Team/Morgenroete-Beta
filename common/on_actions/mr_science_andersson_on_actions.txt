on_building_built = {
	on_actions = {
		andersson_on_building_built
	}
}

on_acquired_technology = {
	on_actions = {
		andersson_on_acquired_technology
	}
}

on_yearly_pulse_country = {
	on_actions = {
		delay = { days = { 1 360 } }
		andersson_on_yearly_pulse_historical_fossils
	}
}

andersson_on_building_built = {
	effect = {
		if = {
			limit = {
				owner = {
					NOT = {
						has_variable = andersson_institute_inauguration_happened_var
					}
					andersson_has_institute_trigger = yes
				}
			}
			owner = {
				trigger_event = {
					id = andersson.10 #Andersson 10: Institute Inauguration
				}
			}
		}
	}
}

andersson_on_acquired_technology = {
}

andersson_on_yearly_pulse_historical_fossils = {
	trigger = {
		has_game_rule = andersson_anthropology_historical_rule
	}

	events = {
		### Neoanthropus "True Man" ###

		andersson.21 #Andersson 21: Vero Man
		andersson.22 #Andersson 22: Lagoa Santa Man
		andersson.23 #Andersson 23: Cro-Magnon Man
		andersson.24 #Andersson 24: Boskop Man
		andersson.25 #Andersson 25: Tianyuan Man
		andersson.26 #Andersson 26: Talgai Man

		### Palaeanthropus "Cave Man" ###

		andersson.31 #Andersson 31: Neanderthal Man
		andersson.32 #Andersson 32: Gibraltar Man
		andersson.33 #Andersson 33: Spy Man
		andersson.34 #Andersson 34: Krapina Man
		andersson.35 #Andersson 35: Galilee Man

		### Pithecanthropus "Pre-Man" ###

		andersson.41 #Andersson 41: Heidelberg Man
		andersson.42 #Andersson 42: Rhodesian Man
		andersson.43 #Andersson 43: Peking Man
		andersson.44 #Andersson 44: Java Man

		### Australopithecus ###

		andersson.51 #Andersson 51: Taung Child
	}
}
