on_yearly_pulse_country = { 
	on_actions = {
		douglas_on_yearly_pulse
	}
}

douglas_on_yearly_pulse = {
	effect = {
		#This effect is used to add the alpine heritage to all countries
		douglas_yearly_alpine_heritage_effect = yes

		if = {
			limit = {
				has_variable = douglas_alpine_club_founding_country_var
			}
			trigger_event = {
				id = douglas.301
			}
		}
	}
}
