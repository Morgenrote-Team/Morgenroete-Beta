panum_on_yearly_pulse = {
	effect = {
		######################
		#Smallpox Eredication#
		if = {
			limit = {
				NOR = {
					has_global_variable = panum_eredication_smallpox_global_var
					any_country = {
						panum_has_smallpox_prevention_trigger = no
					}
				}
			}
			every_country = {
				limit = {
					is_player = yes
				}
				trigger_event = {
					id = panum.4003
				}
			}
		}
		######################
		#Polio Epidemic#
		if = {
			limit = {
				NOT = {
					has_global_variable = panum_polio_epidemic_global_var
				}
				year >= 1886 #Historically, polio began to become an epidemic disease around the late 19th century. The date is a little earlier because otherwise it coincides too closely with the plague
				any_country = {
					panum_has_polio_prevention_trigger = no
					has_technology_researched = modern_sewerage
					institution_investment_level = {
						institution	= institution_health_system
						value >= 1
					}
					any_scope_state = {
						average_sol <= 35
						total_urbanization >= 1000
					}
				}
			}

			random_country = {
				limit = {
					panum_has_polio_prevention_trigger = no
					has_technology_researched = modern_sewerage
					institution_investment_level = {
						institution	= institution_health_system
						value >= 1
					}
					any_scope_state = {
						average_sol <= 35
						total_urbanization >= 1000
					}
				}
				set_global_variable = panum_polio_epidemic_global_var
				trigger_event = {
					id = panum.1001 #Panum 1001: Polio becomes Epidemic
					days = { 1 60 }
				}
			}
		}
		######################
		#Cholera - Third Wave#
		if = {
			limit = {
				NOT = { has_global_variable = panum_third_cholera_global_var }
				year >= 1846 #We start with the third wave
			}
			s:STATE_EAST_BENGAL = {
				random_scope_state = {
					owner = {
						trigger_event = panum.101 #Panum 101: Third Cholera Pandemic
					}
				}
			}
		}
		######################
		#Third Plague Wave starts#
		if = {
			limit = {
				NOT = { has_global_variable = panum_third_plague_global_var }
				year >= 1894 #Third Wave started way earlier, but we begin with the Hong Kong outbreak to give enough time to develop antitoxins
			}
			s:STATE_YUNNAN = {
				random_scope_state = {
					owner = {
						trigger_event = panum.800 #Panum 800: Third Plague Wave starts
					}
				}
			}
		}
		######################
		#Spanish Flu starts#
		if = {
			limit = {
				has_game_rule = panum_historical_spanish_flu_rule
				NOT = { has_global_variable = panum_spanish_flu_pandemic_global_var }
				year >= 1918 #Patient Zero; however, Spanish Flu started way earlier
			}
			s:STATE_NEW_CASTILE = { #Spanish Flu did not start in Spain, but we get the name from here
				random_scope_state = {
					owner = {
						trigger_event = panum.1101 #Panum 1101: Spanish Flu Pancemic (historical)
					}
				}
			}
		}
		######################
		#The Concept of Anxiety#
		if = {
			limit = {
				NOT = { has_global_variable = panum_concept_of_anxiety_global_var }
				year >= 1844
				year < 1850
				exists = c:DEN
			}
			c:DEN ?= {
				trigger_event = panum.2621 #Panum 2621: The Concept of Anxiety
			}
		}
		######################
		#Historcial Red Cross#
		if = {
			limit = {
				has_game_rule = panum_historical_and_dynamic_red_cross_rule
				NOT = { has_global_variable = panum_red_cross_global_var }
				year >= 1862
				year < 1870
				exists = c:SWI
			}
			c:SWI ?= {
				trigger_event = panum.3002 #Panum 3002: The Red Cross - Event for Switzerland
			}
		}
		######################
		#Red Cross - First Convention#
		if = {
			limit = {
				has_global_variable = panum_first_convention_proposed_global_var
				NOT = { has_global_variable = panum_first_convention_binding_global_var }
				panum_first_geneva_convention_becomes_binding_trigger = yes
			}
			random_scope_country = {
				limit = {
					any_scope_building = {
						is_building_type = building_panum_red_cross
					}
				}
				trigger_event = panum.3031 #Panum 3031: First Convention becomes a binding Convention
			}
		}
		######################
		#Red Cross - Second Convention#
		if = {
			limit = {
				has_global_variable = panum_second_convention_proposed_global_var
				NOT = { has_global_variable = panum_second_convention_binding_global_var }
				panum_second_geneva_convention_becomes_binding_trigger = yes
			}
			random_scope_country = {
				limit = {
					any_scope_building = {
						is_building_type = building_panum_red_cross
					}
				}
				trigger_event = panum.3032 #Panum 3032: Second Convention becomes a binding Convention
			}
		}
		######################
		#Red Cross - Second Convention#
		if = {
			limit = {
				has_global_variable = panum_third_convention_proposed_global_var
				NOT = { has_global_variable = panum_third_convention_binding_global_var }
				panum_third_geneva_convention_becomes_binding_trigger = yes
			}
			random_scope_country = {
				limit = {
					any_scope_building = {
						is_building_type = building_panum_red_cross
					}
				}
				trigger_event = panum.3033 #Panum 3033: Third Convention becomes a binding Convention
			}
		}
	}
}

panum_on_battle_lost = {

	}
}

panum_on_building_built = {
	effect = {
		if = {
			limit = {
				owner = {
					NOT = {
						has_variable = panum_asylum_inauguration_happened
					}
					any_scope_state = {
						any_scope_building = {
							is_building_type = building_panum_asylum
							level >= 1
						}
					}
				}
			}
			owner = {
				trigger_event = {
					id = panum.2600 #Panum 2600: The Asylum
					popup = yes
				}
			}
		}
	}
}
