on_yearly_pulse_country = { 
	on_actions = {
		delay = { days = { 1 360 } }
		theiler_on_yearly_pulse_country
	}
}

on_acquired_technology = {
	on_actions = {
		theiler_on_acquired_technology
	}
}

on_harvest_condition_started_in_country = {
	on_actions = {
		theiler_on_harvest_condition_started_in_country 
	}
}

theiler_on_yearly_pulse_country = {
	events = {
		theiler.60 #Theiler 60: Boil Weevil
		theiler.61 #Theiler 61: Grape Phylloxera
		theiler.62 #Theiler 62: Tea Mosquito Bug
		theiler.63 #Theiler 63: Potato Beetle
		theiler.64 #Theiler 64: Potato Beetle spreads to Europe
		theiler.65 #Theiler 65: Codling Moth spreads to North America
		theiler.66 #Theiler 66: Codling Moth spreads to Australia
		theiler.67 #Theiler 67: Codling Moth spreads to South Africa
		theiler.68 #Theiler 68: Codling Moth spreads to South America
	}
}


theiler_on_acquired_technology = {
	events = {
		theiler.10 #Theiler 10: The Birth of Microbiology
	}
}

theiler_on_harvest_condition_started_in_country = {
	effect = {
		theiler_pest_send_notification_effect = yes
	}
}
