andersson_andersson = {
	possible = {
		has_game_rule = andersson_anthropology_historical_rule
	}
	happened = {
		has_variable = andersson_andersson_achievement
	}
}

andersson_own = {
	possible = {
		has_game_rule = andersson_anthropology_free_rule
	}
	happened = {
		has_variable = andersson_anthropologist_var
		var:andersson_anthropologist_var ?= {
			OR = {
				has_variable = andersson_anthropologist_achievement_discovered_neoanthropus_var
				has_variable = andersson_anthropologist_achievement_discovered_palaeanthropus_var
				has_variable = andersson_anthropologist_achievement_discovered_pithecanthropus_var
				has_variable = andersson_anthropologist_achievement_discovered_australopithecus_var
			}
		}
	}
}
