mountaineering_license = {
	groups= {
		general
	}

	requires_approval = yes
	can_use_obligations = yes
	show_effect_in_tooltip = yes

	unlocking_technologies = {
		nationalism
	}

	selectable = {
		NOT = { is_country_type = decentralized }
		is_revolutionary = no
	}

	potential = {
		trigger_if = {
			limit = { is_ai = yes }
			country_rank >= rank_value:major_power
			douglas_has_mountaineer_no_tooltip_trigger = yes
		}
		NOT = {
			is_country_type = unrecognized
		}
		scope:target_country = {
			douglas_has_conquerable_summit_trigger = yes
			NOT = {
				has_variable = douglas_foreign_mountaineers_banned_var
			}
		}
		hidden_trigger = {
			OR = {
				is_player = no
				AND = {
					is_player = yes
					has_diplomatic_relevance = scope:target_country
				}
			}
		}
	}

	possible = {
		NOT = {
			has_war_with = scope:target_country
		}
	}

	pact = {
		is_two_sided_pact = yes

		income_transfer = 0.001
		second_country_gets_income_transfer = yes
		income_transfer_based_on_second_country = no

		requirement_to_maintain = {
			trigger = {
				NOT = {
					has_war_with = scope:target_country
					scope:target_country = {
						douglas_has_conquerable_summit_trigger = no
					}
				}
			}
		}
	}

	ai = {
		evaluation_chance = {
			value = 1
		}

		will_break = {
			always = no
		}

		propose_score = {
			value = 0
			add = {
				if = {
					limit = {
						country_rank > rank_value:major_power
						scaled_debt <= 0
					}
					value = 100 #Great and Major Powers more interested than Minor Powers
				}
			}
			add = {
				if = {
					limit = {
						NOT = {
							any_scope_character = {
								has_role_of_type = general
								is_busy = no
								OR = {
									has_trait = douglas_mountaineer
									has_trait = explorer
									has_trait = mountain_commander
									has_trait = surveyor
								}
							}
						}
					}
					value = -1000 #AI will never ask for mountaineering rights without mountaineer
				}
			}
		}

		will_propose = {
		}

		accept_score = {
			value = -25
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_BASE"
				value = 0
			}
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RELATIONS"
				if = {
					limit = {
						"relations(scope:actor)" < relations_threshold:cordial
					}
					value = 0
				}
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:cordial
						"relations(scope:actor)" < relations_threshold:amicable
					}
					value = 25
				}
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:amicable
						"relations(scope:actor)" < relations_threshold:friendly
					}
					value = 35
				}
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:friendly
					}
					value = 50
				}
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_INFAMY"
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:infamous
						"scope:actor.infamy" < infamy_threshold:notorious
					}
					value = -10
				}
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:notorious
						"scope:actor.infamy" < infamy_threshold:pariah
					}
					value = -25
				}
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:pariah
					}
					value = -50
				}
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RANK"

				if = {
					limit = { scope:actor = { country_rank > root.country_rank } }
					value = 20
				}

				if = {
					limit = { scope:actor = { country_rank = root.country_rank } }
					value = -15
				}

				if = {
					limit = { scope:actor = { country_rank < root.country_rank } }
					value = -30
				}
			}

			add = {
				desc = "A Great Power will never give access to its highest mountains."

				if = {
					limit = { scope:target_country = { country_rank = rank_value:great_power } } #Great Power will never give access to their mountains
					value = -1000
				}
			}

			add = {
				desc = "We have a Customs Union"

				if = {
					limit = {
						is_in_customs_union_with = scope:actor
					}
					value = 20
				}
			}

			add = {
				desc = "We have a Defensive Pact"

				if = {
					limit = {
						any_scope_treaty = {
							binds = scope:actor
							any_scope_article = {
								has_type = defensive_pact
							}
						}
					}
					value = 40
				}
			}

			add = {
				desc = "We have an Alliance"

				if = {
					limit = {
						any_scope_treaty = {
							binds = scope:actor
							any_scope_article = {
								has_type = alliance
							}
						}
					}
					value = 60
				}
			}

			add = {
				desc = "They are our Subject"

				if = {
					limit = {
						OR = {
							has_diplomatic_pact = {
								who = scope:actor
								type = puppet
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = vassal
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = dominion
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = personal_union
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = tributary
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = protectorate
							}
						}
					}
					value = 150
				}
			}

			add = {
				desc = "They are in debt"

				if = {
					limit = {
						scope:target_country = {
							scaled_debt > 0
						}
					}
					value = 300
				}
			}
		}
	}
}
