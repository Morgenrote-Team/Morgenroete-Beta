#Excavation License

excavation_license = {
	groups= {
		general
	}
	requires_approval = yes
	can_use_obligations = yes
	should_notify_third_parties = yes
	show_effect_in_tooltip = yes

	unlocking_technologies = {
		lepsius_antiquarianism_tech
	}

	selectable = {
		NOT = { is_country_type = decentralized }
		is_revolutionary = no
	}

	potential = {
		country_rank >= rank_value:major_power
		trigger_if = {
			limit = { is_player = yes }
			has_diplomatic_relevance = scope:target_country
		}
		scope:target_country = {
			lepsius_has_excavation_site_for_licence_trigger = yes
		}
	}

	possible = {
		NOT = {
			has_war_with = scope:target_country
		}
		is_revolutionary = no
	}

	pact = {
		is_two_sided_pact = yes

		income_transfer = 0.005
		second_country_gets_income_transfer = yes
		income_transfer_based_on_second_country = no

		requirement_to_maintain = {
			trigger = {
				NOT = { has_war_with = scope:target_country }
				scope:target_country = {
					lepsius_has_excavation_site = yes
				}
			}
		}

		manual_break_effect = {
			change_relations = { country = scope:target_country value = -10 }
		}

		auto_break_effect = {
		}
	}

	ai = {
		evaluation_chance = {
			value = 1
		}

		will_break = {
			always = no
		}

		propose_score = {
			value = 10
			add = {
				if = {
					limit = {
						scaled_debt <= 0
					}
					value = 30 #AI without debt will be willing to pay
				}
			}
			add = {
				if = {
					limit = {
						country_rank = rank_value:great_power
					}
					value = 60 #Great Powers will try to get Excavation Licenses as soon as possible
				}
			}
			add = {
				if = {
					limit = {
						var:lepsius_excavation_teams = 0
					}
					value = -100 #AI will not ask for excavation rights without a free team
				}
			}
		}

		will_propose = {
		}

		accept_score = {
			value = -25
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_BASE"
				value = 0
			}
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RELATIONS"
				if = {
					limit = {
						"relations(scope:actor)" < relations_threshold:cordial
					}
					value = 0
				}
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:cordial
						"relations(scope:actor)" < relations_threshold:amicable
					}
					value = 25
				}
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:amicable
						"relations(scope:actor)" < relations_threshold:friendly
					}
					value = 35
				}
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:friendly
					}
					value = 50
				}
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_INFAMY"
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:infamous
						"scope:actor.infamy" < infamy_threshold:notorious
					}
					value = -10
				}
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:notorious
						"scope:actor.infamy" < infamy_threshold:pariah
					}
					value = -25
				}
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:pariah
					}
					value = -50
				}
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RANK"

				if = {
					limit = { scope:actor = { country_rank > root.country_rank } }
					value = 20
				}

				if = {
					limit = { scope:actor = { country_rank < root.country_rank } }
					value = -10
				}
			}
			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_CUSTOMS_UNION"

				if = {
					limit = {
						is_in_customs_union_with = scope:actor
					}
					value = 20
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_DEFENSIVE_PACT"

				if = {
					limit = {
						any_scope_treaty = {
							binds = scope:actor
							any_scope_article = {
								has_type = defensive_pact
							}
						}
					}
					value = 40
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_ALLIANCE"

				if = {
					limit = {
						any_scope_treaty = {
							binds = scope:actor
							any_scope_article = {
								has_type = alliance
							}
						}
					}
					value = 60
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_SUBJECT"

				if = {
					limit = {
						OR = {
							has_diplomatic_pact = {
								who = scope:actor
								type = puppet
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = vassal
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = dominion
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = personal_union
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = tributary
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = protectorate
							}
						}
					}
					value = 150
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_IN_DEBT"

				if = {
					limit = {
						scope:target_country = {
							scaled_debt > 0
						}
					}
					value = 300
				}
			}
		}
	}
}
