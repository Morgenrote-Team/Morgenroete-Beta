#Ask for a zoo animal

ask_for_zoo_animal = {
	groups= {
		general
	}

	requires_approval = yes
	can_use_obligations = yes
	show_effect_in_tooltip = yes

	unlocking_technologies = {
		dubois_theory_of_evolution_tech
	}

	selectable = {
		NOT = { is_country_type = decentralized }
		is_revolutionary = no
	}

	potential = {
		trigger_if = {
			limit = { is_ai = yes }
			has_technology_researched = dubois_theory_of_evolution_tech
			gold_reserves > 100000
		}
		NOT = { is_country_type = decentralized }
		has_diplomatic_relevance = scope:target_country
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		scope:target_country = {
			NOT = { 
				has_modifier = dubois_no_animal_export_modifier
			}
		}
		NOT = { has_war_with = scope:target_country }
		is_revolutionary = no
		NOT = {
			has_variable = dubois_buying_zoo_animal_var
		}
	}
	
	possible = {
		NOT = {
			has_modifier = dubois_zoo_transport_modifier
		}
		trigger_if = {
			limit = { is_ai = yes }
			scope:target_country.relations:root >= relations_threshold:cordial
			dubois_target_has_animal = yes
		}
		trigger_else = {
			custom_tooltip = {
				text = DIPLO_ACTION_RELATIONS_CORDIAL_OR_GREATER
				scope:target_country.relations:root >= relations_threshold:cordial
			}
			custom_tooltip = {
				text = dubois_target_has_animal_tt
				dubois_target_has_animal = yes
			}
		}
	}

	accept_effect = {
		set_variable = dubois_buying_zoo_animal_var
		add_modifier = {
			name = dubois_zoo_transport_modifier
		}
		add_treasury = -50000
		scope:target_country = {
			add_treasury = 50000
		}
		ROOT = {
			trigger_event = {
				id = dubois.11
				popup = yes
			}
		}
	}
	
	ai = {
		evaluation_chance = {
			value = 1
		}

		propose_score = {
			value = 100
		}
	
		will_propose = {
			is_revolutionary = no
		}

		accept_score = {
			value = -25
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_BASE"
				value = 0
			}
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RELATIONS"			
				if = {
					limit = {
						"relations(scope:actor)" < relations_threshold:cordial
					}
					value = 0
				}
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:cordial
						"relations(scope:actor)" < relations_threshold:amicable
					}
					value = 25
				}	
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:amicable
						"relations(scope:actor)" < relations_threshold:friendly
					}
					value = 35
				}	
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:friendly
					}
					value = 50
				}				
			}
			
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_INFAMY"
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:infamous
						"scope:actor.infamy" < infamy_threshold:notorious
					}
					value = -10
				}	
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:notorious
						"scope:actor.infamy" < infamy_threshold:pariah
					}
					value = -25
				}
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:pariah
					}
					value = -50
				}					
			}					

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RANK"
	
				if = {
					limit = { scope:actor = { country_rank > root.country_rank } }
					value = 20
				}
				
				if = {
					limit = { scope:actor = { country_rank < root.country_rank } }
					value = -10
				}				
			}
			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_CUSTOMS_UNION"

				if = {
					limit = {
						is_in_customs_union_with = scope:actor
					}
					value = 20
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_DEFENSIVE_PACT"

				if = {
					limit = {
						any_scope_treaty = {
							binds = scope:actor
							any_scope_article = {
								has_type = defensive_pact
							}
						}
					}
					value = 40
				}	
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_ALLIANCE"

				if = {
					limit = {
						any_scope_treaty = {
							binds = scope:target_country
							any_scope_article = {
								has_type = alliance
							}
						}
					}
					value = 60
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_SUBJECT"

				if = {
					limit = {
						OR = {
							has_diplomatic_pact = {
								who = scope:actor
								type = puppet
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = vassal
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = dominion
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = personal_union
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = tributary
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = protectorate
							}
						}			
					}
					value = 150
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_NOT_SUBJECT"

				if = {
					limit = {
						is_subject = yes
						NOT = {
							is_subject_of = scope:actor
						}			
					}
					value = -75
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_IN_DEBT"

				if = {
					limit = {
						scope:target_country = {
							scaled_debt > 0
						}
					}
					value = 300
				}
			}
		}
	}
}

smuggle_zoo_animal = {
	groups= {
		general
	}

	requires_approval = no
	show_effect_in_tooltip = yes

	unlocking_technologies = {
		dubois_theory_of_evolution_tech
	}

	potential = {
		is_player = yes
		has_diplomatic_relevance = scope:target_country
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		is_revolutionary = no
		NOT = {
			has_variable = dubois_smuggle_zoo_animal_var
		}
	}
	
	possible = {
		dubois_has_active_naturalist = yes
		OR = {
			has_building = building_dubois_zoo
			has_building = building_dubois_national_park
		}
		custom_tooltip = {
			text = dubois_target_has_animal_tt
			dubois_target_has_animal = yes
		}
		NOR = {
			has_modifier = dubois_zoo_transport_modifier
			has_modifier = dubois_animal_smuggle_modifier
			has_modifier = dubois_poacher_modifier
		}
	}

	accept_effect = {
		set_variable = dubois_smuggle_zoo_animal_var
		add_treasury = -100000
		ROOT = {
			random_list = {
				2 = {
					modifier = {
						if = {
							limit = {
								var:dubois_naturalist_var = {
									var:dubois_naturalist_experience_var >= 100
								}
							}
							add = 1
						}
						if = {
							limit = {
								var:dubois_naturalist_var = {
									var:dubois_naturalist_experience_var >= 200
								}
							}
							add = 1
						}
						if = {
							limit = {
								trigger_if = {
									limit = { has_variable = dubois_naturalist_var }
									var:dubois_naturalist_var = {
										var:dubois_naturalist_experience_var >= 300
									}
								}
								trigger_else = {
									always = no
								}
							}
							add = 1
						}
					}
					trigger_event = {
						id = dubois.12 #Poachers succeed
						popup = yes
					}
				}
				2 = {
					trigger_event = {
						id = dubois.13 #Poachers failed
						popup = yes
					}
				}
			}
		}
	}

	ai = {
		evaluation_chance = {
			value = 0
		}
	}
}
