anthropological_invitation = {
	groups= {
		general
	}
	
	requires_approval = yes
	show_effect_in_tooltip = yes
	can_select_to_break = no

	unlocking_technologies = {
	}

	selectable = {
		NOT = { is_country_type = decentralized }
		is_revolutionary = no
	}

	potential = {
		scope:target_country = {
			has_technology_researched = lepsius_anthropology_tech
		}
		hidden_trigger = {
			OR = {
				is_player = no
				AND = {
					is_player = yes
					has_diplomatic_relevance = scope:target_country
				}
			}
		}
		NOT = {
			has_modifier = lepsius_foreign_anthropological_examination_modifier
		}
		NOT = {
			has_variable = lepsius_anthropology_invitation_break
		}
	}
	
	possible = {	
		NOT = { 
			has_war_with = scope:target_country 
		}
		is_revolutionary = no
		any_scope_state = {
			lepsius_has_discovered_cave = yes
		}
		NOT = {
			andersson_has_anthropologist = yes
		}
		scope:target_country = {
			lepsius_has_active_anthropologist = yes
		}
		NOT = {
			has_technology_researched = lepsius_anthropology_tech
		}
	}
	
	accept_effect = {
		scope:target_country = {
			lepsius_set_anthropologist_busy_effect = yes
		}
		ROOT = {
			add_modifier = {
				name = lepsius_foreign_anthropological_examination_modifier
			}
			if = {
				limit = {
					is_player = yes
				}
				trigger_event = {
					id = lepsius.1421
					days = 3
					popup = yes
				}
			}
			if = {
				limit = {
					is_player = no
				}
				trigger_event = {
					id = lepsius.1422
					days = 3
					popup = yes
				}
			}
		}
	}

	pact = {	
		is_two_sided_pact = yes
	}
	
	ai = {
		evaluation_chance = {
			value = 1
		}
		
		will_break = {
			always = no
		}

		propose_score = {
			value = 0
			add = {
				if = {
					limit = {
						any_scope_state = {
							OR = {
								has_state_trait = lepsius_state_trait_erectus_cave
								has_state_trait = lepsius_state_trait_australopithecus_cave
								has_state_trait = lepsius_state_trait_paranthropus_cave
							}
						}
					}
					value = 100 #AI will try to get valuable bones
				}
			}
		}
	
		will_propose = {					
		}	
		
		accept_score = {
			value = 25 #will mostly accept
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_BASE"
				value = 0
			}
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RELATIONS"			
				if = {
					limit = {
						"relations(scope:actor)" < relations_threshold:cordial
					}
					value = 0
				}
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:cordial
						"relations(scope:actor)" < relations_threshold:amicable
					}
					value = 25
				}	
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:amicable
						"relations(scope:actor)" < relations_threshold:friendly
					}
					value = 35
				}	
				if = {
					limit = {
						"relations(scope:actor)" >= relations_threshold:friendly
					}
					value = 50
				}				
			}
			
			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_INFAMY"
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:infamous
						"scope:actor.infamy" < infamy_threshold:notorious
					}
					value = -10
				}	
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:notorious
						"scope:actor.infamy" < infamy_threshold:pariah
					}
					value = -25
				}
				if = {
					limit = {
						"scope:actor.infamy" >= infamy_threshold:pariah
					}
					value = -50
				}					
			}					

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_RANK"
	
				if = {
					limit = { scope:actor = { country_rank > root.country_rank } }
					value = 20
				}
				
				if = {
					limit = { scope:actor = { country_rank < root.country_rank } }
					value = -10
				}				
			}
			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_CUSTOMS_UNION"

				if = {
					limit = {
						is_in_customs_union_with = scope:actor
					}
					value = 20
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_DEFENSIVE_PACT"

				if = {
					limit = {
						any_scope_treaty = {
							binds = scope:actor
							any_scope_article = {
								has_type = defensive_pact
							}
						}		
					}
					value = 40
				}	
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_ALLIANCE"

				if = {
					limit = {
						any_scope_treaty = {
							binds = scope:actor
							any_scope_article = {
								has_type = alliance
							}
						}
					}
					value = 60
				}
			}

			add = {
				desc = "MR_DIPLOMATIC_ACCEPTANCE_SUBJECT"

				if = {
					limit = {
						OR = {
							has_diplomatic_pact = {
								who = scope:actor
								type = puppet
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = vassal
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = dominion
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = personal_union
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = tributary
							}
							has_diplomatic_pact = {
								who = scope:actor
								type = protectorate
							}
						}			
					}
					value = 150
				}
			}
		}
	}
}
